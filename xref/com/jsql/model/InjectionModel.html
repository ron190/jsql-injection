49  /**
50   * Model class of MVC pattern for processing SQL injection automatically.<br>
51   * Different views can be attached to this observable, like Swing or command line, in order to separate
52   * the functional job from the graphical processing.<br>
53   * The Model has a specific database engine and strategy which run an automatic injection to get name of
54   * databases, tables, columns and values, and it can also retrieve resources like files and shell.<br>
55   * Tasks are run in multi-threads in general to speed the process.
56   */
57  public class InjectionModel extends AbstractModelObservable implements Serializable {

136     /**
137      * Prepare the injection process, can be interrupted by the user (via shouldStopAll).
138      * Erase all attributes eventually defined in a previous injection.
139      * Run by Scan, Standard and TU.
140      */
141     public void beginInjection() {
142         this.resetModel();
143         try {
144             if (this.mediatorUtils.proxyUtil().isNotLive(ShowOnConsole.YES)) {
145                 return;
146             }
147             LOGGER.log(
148                 LogLevelUtil.CONSOLE_INFORM,
149                 "{}: {}",
150                 () -> I18nUtil.valueByKey("LOG_START_INJECTION"),
151                 () -> this.mediatorUtils.connectionUtil().getUrlByUser()
152             );
153             
154             // Check general integrity if user's parameters
155             this.mediatorUtils.parameterUtil().checkParametersFormat();
156             this.mediatorUtils.connectionUtil().testConnection();
157 
158             // TODO Check all path params URL segments
159             boolean hasFoundInjection = this.mediatorMethod.getQuery().testParameters(false);
160             hasFoundInjection = this.mediatorUtils.multipartUtil().testParameters(hasFoundInjection);
161             hasFoundInjection = this.mediatorUtils.soapUtil().testParameters(hasFoundInjection);
162             hasFoundInjection = this.mediatorMethod.getRequest().testParameters(hasFoundInjection);
163             hasFoundInjection = this.mediatorMethod.getHeader().testParameters(hasFoundInjection);
164             hasFoundInjection = this.mediatorUtils.cookiesUtil().testParameters(hasFoundInjection);
165 
166             if (hasFoundInjection && !this.isScanning) {
167                 if (!this.getMediatorUtils().preferencesUtil().isNotShowingVulnReport()) {
168                     this.sendToViews(new Seal.CreateAnalysisReport(this.analysisReport));
169                 }
170                 if (this.getMediatorUtils().preferencesUtil().isZipStrategy()) {
171                     LOGGER.log(LogLevelUtil.CONSOLE_INFORM, "Using Zip mode for reduced query size");
172                 } else if (this.getMediatorUtils().preferencesUtil().isDiosStrategy()) {
173                     LOGGER.log(LogLevelUtil.CONSOLE_INFORM, "Using Dump In One Shot strategy for single query dump");
174                 }
175                 if (!this.mediatorUtils.preferencesUtil().isNotInjectingMetadata()) {
176                     this.dataAccess.getDatabaseInfos();
177                 }
178                 this.dataAccess.listDatabases();
179             }
180             
181             LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, () -> I18nUtil.valueByKey("LOG_DONE"));
182             this.shouldErasePreviousInjection = true;
183         } catch (InterruptedException e) {
184             LOGGER.log(LogLevelUtil.IGNORE, e, e);
185             Thread.currentThread().interrupt();
186         } catch (JSqlRuntimeException | JSqlException | IOException e) {  // Catch expected exceptions only
187             LOGGER.log(
188                 LogLevelUtil.CONSOLE_ERROR,
189                 "Interruption: {}",
190                 e.getMessage() == null ? InjectionModel.getImplicitReason(e) : e.getMessage()
191             );
192         } finally {
193             this.sendToViews(new Seal.EndPreparation());
194         }
195     }

197     public static String getImplicitReason(Throwable e) {
198         String message = e.getClass().getSimpleName();
199         if (e.getMessage() != null) {
200             message += ": "+ e.getMessage();
201         }
202         if (e.getCause() != null && !e.equals(e.getCause())) {
203             message += " > "+ InjectionModel.getImplicitReason(e.getCause());
204         }
205         return message;
206     }

208     /**
209      * Run an HTTP connection to the web server.
210      * @param dataInjection SQL query
211      * @return source code of current page
212      */
213     @Override
214     public String inject(
215         String dataInjection,
216         boolean isUsingIndex,
217         String metadataInjectionProcess,
218         AbstractCallableBit<?> callableBoolean,
219         boolean isReport
220     ) {
221         // Temporary url, we go from "select 1,2,3,4..." to "select 1,([complex query]),2...", but keep initial url
222         String urlInjection = this.mediatorUtils.connectionUtil().getUrlBase();
223         urlInjection = this.mediatorStrategy.buildPath(urlInjection, isUsingIndex, dataInjection);
224         urlInjection = StringUtil.cleanSql(urlInjection.trim());
225 
226         URL urlObject;
227         String urlInjectionFixed;
228         try {
229             urlInjectionFixed = this.initQueryString(
230                 isUsingIndex,
231                 urlInjection,
232                 dataInjection
233             );
234             urlObject = new URI(urlInjectionFixed).toURL();
235         } catch (MalformedURLException | URISyntaxException e) {
236             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Incorrect Query Url: %s", e.getMessage()));
237             return StringUtils.EMPTY;
238         }
239 
240         String pageSource = StringUtils.EMPTY;
241         
242         // Define the connection
243         try {
244             var httpRequestBuilder = HttpRequest.newBuilder()
245                 .uri(URI.create(urlObject.toString()))
246                 .setHeader(HeaderUtil.CONTENT_TYPE_REQUEST, "text/plain")
247                 .timeout(Duration.ofSeconds(15));
248             
249             this.mediatorUtils.csrfUtil().addHeaderToken(httpRequestBuilder);
250             this.mediatorUtils.digestUtil().addHeaderToken(httpRequestBuilder);
251             this.mediatorUtils.connectionUtil().setCustomUserAgent(httpRequestBuilder);
252 
253             String body = this.initRequest(isUsingIndex, dataInjection, httpRequestBuilder);
254             this.initHeader(isUsingIndex, dataInjection, httpRequestBuilder);
255             
256             var httpRequest = httpRequestBuilder.build();
257             if (isReport) {
258                 Color colorReport = UIManager.getColor("TextArea.inactiveForeground");
259                 String report = InjectionModel.BR + StringUtil.formatReport(colorReport, "Method: ") + httpRequest.method();
260                 report += InjectionModel.BR + StringUtil.formatReport(colorReport, "Path: ") + httpRequest.uri().getPath();
261                 if (httpRequest.uri().getQuery() != null) {
262                     report += InjectionModel.BR + StringUtil.formatReport(colorReport, "Query: ") + httpRequest.uri().getQuery();
263                 }
264                 if (
265                     !(this.mediatorUtils.parameterUtil().getListRequest().isEmpty()
266                     && this.mediatorUtils.csrfUtil().getTokenCsrf() == null)
267                 ) {
268                     report += InjectionModel.BR + StringUtil.formatReport(colorReport, "Body: ") + body;
269                 }
270                 report += InjectionModel.BR 
271                     + StringUtil.formatReport(colorReport, "Header: ") 
272                     + httpRequest.headers().map().entrySet().stream()
273                     .map(entry -> 
274                         String.format("%s: %s", entry.getKey(), 
275                         String.join(StringUtils.EMPTY, entry.getValue()))
276                     )
277                     .collect(Collectors.joining(InjectionModel.BR));
278                 return report;
279             }
280             
281             HttpResponse<String> response = this.getMediatorUtils().connectionUtil().getHttpClient().build().send(
282                 httpRequestBuilder.build(),
283                 BodyHandlers.ofString()
284             );
285             if (this.mediatorUtils.parameterUtil().isRequestSoap()) {
286                 // Invalid XML control chars like \x04 requires urlencoding from server
287                 pageSource = URLDecoder.decode(response.body(), StandardCharsets.UTF_8);
288                 pageSource = StringUtil.fromHtml(pageSource);
289             } else {
290                 pageSource = response.body();
291             }
292 
293             Map<String, String> headersResponse = ConnectionUtil.getHeadersMap(response);
294             int sizeHeaders = headersResponse.keySet()
295                 .stream()
296                 .map(key -> headersResponse.get(key).length() + key.length())
297                 .mapToInt(Integer::intValue)
298                 .sum();
299             float size = (float) (pageSource.length() + sizeHeaders) / 1024;
300             var decimalFormat = new DecimalFormat("0.000");
301 
302             String pageSourceFixed = pageSource
303                 .replaceAll("("+ EngineYaml.CALIBRATOR_SQL +"){60,}", "$1...")  // Remove ranges of # created by calibration
304                 .replaceAll("(jIyM){60,}", "$1...");  // Remove batch of chars created by Dios
305 
306             // Send data to Views
307             this.sendToViews(new Seal.MessageHeader(
308                 urlInjectionFixed,
309                 body,
310                 ConnectionUtil.getHeadersMap(httpRequest.headers()),
311                 headersResponse,
312                 pageSourceFixed,
313                 decimalFormat.format(size),
314                 this.mediatorStrategy.getMeta(),
315                 metadataInjectionProcess,
316                 callableBoolean
317             ));
318         } catch (IOException e) {
319             LOGGER.log(
320                 LogLevelUtil.CONSOLE_ERROR,
321                 String.format("Error during connection: %s", e.getMessage())
322             );
323         } catch (InterruptedException e) {
324             LOGGER.log(LogLevelUtil.IGNORE, e, e);
325             Thread.currentThread().interrupt();
326         }
327 
328         return pageSource;
329     }

331     private String initQueryString(
332         boolean isUsingIndex,
333         String urlInjection,
334         String dataInjection
335     ) throws URISyntaxException, MalformedURLException {
336         String urlInjectionFixed = urlInjection;
337         if (
338             this.mediatorUtils.parameterUtil().getListQueryString().isEmpty()
339             && !this.mediatorUtils.preferencesUtil().isProcessingCsrf()
340         ) {
341             return urlInjectionFixed;
342         }
343             
344         // URL without query string like Request and Header can receive
345         // new params from <form> parsing, in that case add the '?' to URL
346         if (!urlInjectionFixed.contains("?")) {
347             urlInjectionFixed += "?";
348         }
349         urlInjectionFixed += this.buildQuery(
350             this.mediatorMethod.getQuery(),
351             this.mediatorUtils.parameterUtil().getQueryStringFromEntries(),
352             isUsingIndex,
353             dataInjection
354         );
355         return this.mediatorUtils.csrfUtil().addQueryStringToken(urlInjectionFixed);
356     }

358     private void initHeader(
359         boolean isUsingIndex,
360         String dataInjection,
361         Builder httpRequest
362     ) {
363         if (!this.mediatorUtils.parameterUtil().getListHeader().isEmpty()) {
364             Stream.of(
365                 this.buildQuery(
366                     this.mediatorMethod.getHeader(),
367                     this.mediatorUtils.parameterUtil().getHeaderFromEntries(),
368                     isUsingIndex,
369                     dataInjection
370                 )
371                 .split("\\\\r\\\\n")
372             )
373             .forEach(header -> {
374                 if (header.split(":").length == 2) {
375                     try {  // TODO Should not catch, rethrow or use runtime exception
376                         HeaderUtil.sanitizeHeaders(
377                             httpRequest,
378                             new SimpleEntry<>(
379                                 header.split(":")[0],
380                                 header.split(":")[1]
381                             )
382                         );
383                     } catch (JSqlException e) {
384                         LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Headers sanitizing issue caught already during connection, ignoring", e);
385                     }
386                 }
387             });
388         }
389     }

391     private String initRequest(
392         boolean isUsingIndex,
393         String dataInjection,
394         Builder httpRequest
395     ) {
396         if (
397             this.mediatorUtils.parameterUtil().getListRequest().isEmpty()
398             && this.mediatorUtils.csrfUtil().getTokenCsrf() == null
399         ) {
400             return dataInjection;
401         }
402             
403         // Set connection method
404         // Active for query string injection too, in that case inject query string still with altered method
405         
406         if (this.mediatorUtils.parameterUtil().isRequestSoap()) {
407             httpRequest.setHeader(HeaderUtil.CONTENT_TYPE_REQUEST, "text/xml");
408         } else {
409             httpRequest.setHeader(HeaderUtil.CONTENT_TYPE_REQUEST, "application/x-www-form-urlencoded");
410         }
411 
412         var body = new StringBuilder();
413         this.mediatorUtils.csrfUtil().addRequestToken(body);
414             
415         if (this.mediatorUtils.connectionUtil().getTypeRequest().matches("PUT|POST")) {
416             if (this.mediatorUtils.parameterUtil().isRequestSoap()) {
417                 body.append(
418                     this.buildQuery(
419                         this.mediatorMethod.getRequest(),
420                         this.mediatorUtils.parameterUtil().getRawRequest(),
421                         isUsingIndex,
422                         dataInjection
423                     )
424                     // Invalid XML characters in recent Spring version
425                     // Server needs to urldecode, or stop using out of range chars
426                     .replace("\u0001", "&#01;")
427                     .replace("\u0003", "&#03;")
428                     .replace("\u0004", "&#04;")
429                     .replace("\u0005", "&#05;")
430                     .replace("\u0006", "&#06;")
431                     .replace("\u0007", "&#07;")
432                     .replace("+", "%2B")  // Prevent replace '+' into 'space' on server side urldecode
433                 );
434             } else {
435                 body.append(
436                     this.buildQuery(
437                         this.mediatorMethod.getRequest(),
438                         this.mediatorUtils.parameterUtil().getRequestFromEntries(),
439                         isUsingIndex,
440                         dataInjection
441                     )
442                 );
443             }
444         }
445         
446         var bodyPublisher = BodyPublishers.ofString(body.toString());
447         httpRequest.method(
448             this.mediatorUtils.connectionUtil().getTypeRequest(),
449             bodyPublisher
450         );
451         return body.toString();
452     }

454     private String buildQuery(AbstractMethodInjection methodInjection, String paramLead, boolean isUsingIndex, String sqlTrail) {
455         String query;
456         String paramLeadFixed = paramLead.replace(
457             InjectionModel.STAR,
458             TamperingUtil.TAG_OPENED + InjectionModel.STAR + TamperingUtil.TAG_CLOSED
459         );
460         if (
461             // No parameter transformation if method is not selected by user
462             this.mediatorUtils.connectionUtil().getMethodInjection() != methodInjection
463             // No parameter transformation if injection point in URL
464             || this.mediatorUtils.connectionUtil().getUrlBase().contains(InjectionModel.STAR)
465         ) {
466             query = paramLeadFixed;  // Just pass parameters without any transformation
467         } else if (
468             // If method is selected by user and URL does not contain injection point
469             // but parameters contain an injection point
470             // then replace injection point by SQL expression in this parameter
471             paramLeadFixed.contains(InjectionModel.STAR)
472         ) {
473             query = this.initStarInjection(paramLeadFixed, isUsingIndex, sqlTrail);
474         } else {
475             query = this.initRawInjection(paramLeadFixed, isUsingIndex, sqlTrail);
476         }
477         query = this.cleanQuery(methodInjection, query);  // Remove comments except empty /**/
478         // Add empty comments with space=>/**/
479         if (this.mediatorUtils.connectionUtil().getMethodInjection() == methodInjection) {
480             query = this.mediatorUtils.tamperingUtil().tamper(query);
481         }
482         return this.applyEncoding(methodInjection, query);
483     }

485     private String initRawInjection(String paramLead, boolean isUsingIndex, String sqlTrail) {
486         String query;
487         // Method is selected by user and there's no injection point
488         if (!isUsingIndex) {
489             // Several SQL expressions does not use indexes in SELECT,
490             // like Boolean, Error, Shell and search for character insertion,
491             // in that case concat SQL expression to the end of param.
492             query = paramLead + sqlTrail;
493         } else {
494             // Concat indexes found for Union strategy to params
495             // and use visible Index for injection
496             query = paramLead + this.getMediatorStrategy().getSpecificUnion().getIndexesInUrl().replaceAll(
497                 String.format(EngineYaml.FORMAT_INDEX, this.mediatorStrategy.getSpecificUnion().getVisibleIndex()),
498                 // Oracle column often contains $, which is reserved for regex.
499                 // => need to be escape with quoteReplacement()
500                 Matcher.quoteReplacement(sqlTrail)
501             );
502         }
503         // Add ending line comment by engine
504         return query + this.mediatorEngine.getEngine().instance().endingComment();
505     }
506 
507     private String initStarInjection(String paramLead, boolean isUsingIndex, String sqlTrail) {
508         String query;
509         // Several SQL expressions does not use indexes in SELECT,
510         // like Boolean, Error, Shell and search for character insertion,
511         // in that case replace injection point by SQL expression.
512         // Injection point is always at the end?
513         if (!isUsingIndex) {
514             query = paramLead.replace(
515                 InjectionModel.STAR,
516                 sqlTrail + this.mediatorEngine.getEngine().instance().endingComment()
517             );
518         } else {
519             // Replace injection point by indexes found for Union strategy
520             // and use visible Index for injection
521             query = paramLead.replace(
522                 InjectionModel.STAR,
523                 this.mediatorStrategy.getSpecificUnion().getIndexesInUrl().replace(
524                     String.format(EngineYaml.FORMAT_INDEX, this.mediatorStrategy.getSpecificUnion().getVisibleIndex()),
525                     sqlTrail
526                 )
527                 + this.mediatorEngine.getEngine().instance().endingComment()
528             );
529         }
530         return query;
531     }

533     /**
534      * Dependency:
535      * - Tamper space=>comment
536      */
537     private String cleanQuery(AbstractMethodInjection methodInjection, String query) {
538         String queryFixed = query;
539         if (
540             methodInjection == this.mediatorMethod.getRequest()
541             && (
542                 this.mediatorUtils.parameterUtil().isRequestSoap()
543                 || this.mediatorUtils.parameterUtil().isMultipartRequest()
544             )
545         ) {
546             queryFixed = StringUtil.removeSqlComment(queryFixed)
547                 .replace("+", " ")
548                 .replace("%2b", "+")  // Failsafe
549                 .replace("%23", "#");  // End comment
550             if (this.mediatorUtils.parameterUtil().isMultipartRequest()) {
551                 // restore linefeed from textfield
552                 queryFixed = queryFixed.replaceAll("(?s)\\\\n", "\r\n");
553             }
554         } else {
555             queryFixed = StringUtil.cleanSql(queryFixed);
556         }
557         return queryFixed;
558     }

560     private String applyEncoding(AbstractMethodInjection methodInjection, String query) {
561         String queryFixed = query;
562         if (!this.mediatorUtils.parameterUtil().isRequestSoap()) {
563             if (methodInjection == this.mediatorMethod.getQuery()) {
564                 // URL encode each character because no query parameter context
565                 if (!this.mediatorUtils.preferencesUtil().isUrlEncodingDisabled()) {
566                     queryFixed = queryFixed.replace("'", "%27");
567                     queryFixed = queryFixed.replace("(", "%28");
568                     queryFixed = queryFixed.replace(")", "%29");
569                     queryFixed = queryFixed.replace("{", "%7b");
570                     queryFixed = queryFixed.replace("[", "%5b");
571                     queryFixed = queryFixed.replace("]", "%5d");
572                     queryFixed = queryFixed.replace("}", "%7d");
573                     queryFixed = queryFixed.replace(">", "%3e");
574                     queryFixed = queryFixed.replace("<", "%3c");
575                     queryFixed = queryFixed.replace("?", "%3f");
576                     queryFixed = queryFixed.replace("_", "%5f");
577                     queryFixed = queryFixed.replace(",", "%2c");
578                 }
579                 // HTTP forbidden characters
580                 queryFixed = queryFixed.replace(StringUtils.SPACE, "+");
581                 queryFixed = queryFixed.replace("`", "%60");  // from `${database}`.`${table}`
582                 queryFixed = queryFixed.replace("\"", "%22");
583                 queryFixed = queryFixed.replace("|", "%7c");
584                 queryFixed = queryFixed.replace("\\", "%5c");
585             } else if (methodInjection != this.mediatorMethod.getRequest()) {
586                 // For cookies in Spring (confirmed, covered by integration tests)
587                 queryFixed = queryFixed.replace("+", "%20");
588                 queryFixed = queryFixed.replace(",", "%2c");
589                 try {  // fix #95709: IllegalArgumentException on decode()
590                     queryFixed = URLDecoder.decode(queryFixed, StandardCharsets.UTF_8);
591                 } catch (IllegalArgumentException e) {
592                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Incorrect values in [{}], please check the parameters", methodInjection.name());
593                     throw new JSqlRuntimeException(e);
594                 }
595             }
596         }
597         return queryFixed;
598     }
