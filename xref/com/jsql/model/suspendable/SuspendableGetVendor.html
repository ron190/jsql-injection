14  import java.util.concurrent.atomic.AtomicBoolean;
15  
16  public class SuspendableGetVendor extends AbstractSuspendable {
17  
18      private static final Logger LOGGER = LogManager.getRootLogger();
19  
20      public SuspendableGetVendor(InjectionModel injectionModel) {
21          super(injectionModel);
22      }
23  
24      @Override
25      public String run(Object... args) {
26          LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Fingerprinting database with Boolean match...");
27  
28          AtomicBoolean isVendorFound = new AtomicBoolean(false);
29          this.injectionModel.getMediatorVendor().getVendorsForFingerprint()
30          .stream()
31          .filter(vendor -> vendor != this.injectionModel.getMediatorVendor().getAuto())
32          .filter(vendor -> StringUtils.isNotEmpty(
33              vendor.instance().getModelYaml().getStrategy().getConfiguration().getFingerprint().getVendorSpecific()
34          ))
35          .forEach(vendor -> {
36              if (isVendorFound.get()) {
37                  return;
38              }
39              String vendorSpecificWithOperator = this.injectionModel.getMediatorVendor().getVendor().instance().sqlTestBlindWithOperator(
40                  vendor.instance().getModelYaml().getStrategy().getConfiguration().getFingerprint().getVendorSpecific(),
41                  AbstractInjectionBit.BlindOperator.OR  // TODO should also test AND and no mode
42              );
43              try {
44                  var injectionCharInsertion = new InjectionVendor(this.injectionModel, vendorSpecificWithOperator, vendor);
45                  if (injectionCharInsertion.isInjectable(vendorSpecificWithOperator)) {
46                      if (this.isSuspended()) {
47                          throw new StoppedByUserSlidingException();
48                      }
49  
50                      LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Found [{}] using Boolean match", vendor);
51                      this.injectionModel.getMediatorVendor().setVendor(vendor);
52                      isVendorFound.set(true);
53  
54                      this.injectionModel.sendToViews(new Request3.SetVendor(
55                          this.injectionModel.getMediatorUtils().getConnectionUtil().getUrlByUser(),
56                          this.injectionModel.getMediatorVendor().getVendor()
57                      ));
58                  }
59              } catch (StoppedByUserSlidingException e) {
60                  throw new JSqlRuntimeException(e);
61              }
62          });
63          return null;  // unused
64      }
65  }
