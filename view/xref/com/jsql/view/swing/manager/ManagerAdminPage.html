14  import com.jsql.util.LogLevelUtil;

17  import com.jsql.view.swing.util.I18nViewUtil;
18  import com.jsql.view.swing.util.MediatorHelper;
19  import org.apache.commons.lang3.StringUtils;
20  import org.apache.logging.log4j.LogManager;
21  import org.apache.logging.log4j.Logger;
22  
23  import java.awt.*;
24  
25  /**
26   * Manager to display webpages frequently used as backoffice administration.
27   */
28  public class ManagerAdminPage extends AbstractManagerList {
29      
30      private static final Logger LOGGER = LogManager.getRootLogger();
31  
32      /**
33       * Create admin page finder.
34       */
35      public ManagerAdminPage() {
36          super("swing/list/admin-page.txt");
37  
38          this.buildRunButton("ADMIN_PAGE_RUN_BUTTON_LABEL", "ADMIN_PAGE_RUN_BUTTON_TOOLTIP");
39          this.run.setName("runManagerAdminPage");
40          this.run.addActionListener(actionEvent -> this.runSearch());
41          this.listPaths.setName("listManagerAdminPage");  // no tooltip, too annoying
42  
43          this.lastLine.add(this.horizontalGlue);
44          this.lastLine.add(this.progressBar);
45          this.lastLine.add(this.run);
46          this.add(this.lastLine, BorderLayout.SOUTH);
47      }
48  
49      private void runSearch() {
50          if (this.listPaths.getSelectedValuesList().isEmpty()) {
51              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select at least one admin page in the list");
52              return;
53          }
54          
55          String urlAddressBar = MediatorHelper.panelAddressBar().getTextFieldAddress().getText();
56          if (!urlAddressBar.isEmpty() && !urlAddressBar.matches("(?i)^https?://.*")) {
57              if (!urlAddressBar.matches("(?i)^\\w+://.*")) {
58                  LOGGER.log(LogLevelUtil.CONSOLE_INFORM, () -> I18nUtil.valueByKey("LOG_ADMIN_NO_PROTOCOL"));
59                  urlAddressBar = "http://"+ urlAddressBar;
60              } else {
61                  LOGGER.log(LogLevelUtil.CONSOLE_INFORM, () -> I18nUtil.valueByKey("LOG_ADMIN_UNKNOWN_PROTOCOL"));
62                  return;
63              }
64          }
65          
66          String urlFinal = urlAddressBar;
67          new Thread(() -> this.searchAdminPages(urlFinal), "ThreadAdminPage").start();
68      }
69  
70      private void searchAdminPages(String urlAddressBar) {
71          if (this.run.getState() == StateButton.STARTABLE) {
72              if (StringUtils.isEmpty(urlAddressBar)) {
73                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Missing URL in address bar");
74              } else {
75                  LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "{} admin pages...", () -> I18nUtil.valueByKey("LOG_CHECKING"));
76                  this.run.setText(I18nViewUtil.valueByKey("ADMIN_PAGE_RUN_BUTTON_STOP"));
77                  this.run.setState(StateButton.STOPPABLE);
78                  this.progressBar.setVisible(true);
79                  this.horizontalGlue.setVisible(false);
80                  MediatorHelper.model().getResourceAccess().createAdminPages(
81                      urlAddressBar,
82                      this.listPaths.getSelectedValuesList().stream().map(ItemList::toString).toList()
83                  );
84                  this.endProcess();
85              }
86          } else if (this.run.getState() == StateButton.STOPPABLE) {
87              MediatorHelper.model().getResourceAccess().stopSearchAdmin();
88              this.run.setEnabled(false);
89              this.run.setState(StateButton.STOPPING);
90          }
91      }
92  }
