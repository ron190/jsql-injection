21  public class PanelUserAgent extends JPanel {
22      
23      /**
24       * Log4j logger sent to view.
25       */
26      private static final Logger LOGGER = LogManager.getRootLogger();
27  
28      private final JCheckBox checkboxIsCustomUserAgent = new JCheckBox();
29      private final JTextArea textfieldCustomUserAgent = new JPopupTextArea(new JTextAreaPlaceholder("Set User Agent")).getProxy();
30      
31      public PanelUserAgent() {
32          
33          this.setBorder(BorderFactory.createEmptyBorder(10, 0, 0, 0));
34          
35          var jsonScan = new StringBuilder();
36          
37          try (
38              var inputStream = UiUtil.class.getClassLoader().getResourceAsStream("swing/list/user-agent.txt");
39              var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream));
40              var reader = new BufferedReader(inputStreamReader)
41          ) {
42  
43              String line;
44              while ((line = reader.readLine()) != null) {
45  
46                  jsonScan.append(line).append(System.lineSeparator());
47              }
48  
49          } catch (IOException e) {
50              
51              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
52          }
53          
54          this.textfieldCustomUserAgent.getCaret().setBlinkRate(500);
55          this.textfieldCustomUserAgent.setText(jsonScan.toString());
56          MediatorHelper.model().getMediatorUtils().getUserAgentUtil().setCustomUserAgent(jsonScan.toString());
57          this.textfieldCustomUserAgent.getDocument().addDocumentListener(new DocumentListenerEditing() {
58  
59              @Override
60              public void process() {
61                  
62                  MediatorHelper.model().getMediatorUtils().getUserAgentUtil().setCustomUserAgent(
63                      PanelUserAgent.this.textfieldCustomUserAgent.getText()
64                  );
65              }
66          });
67          
68          this.checkboxIsCustomUserAgent.setFocusable(false);
69          var labelIsCheckingUpdate = new JButton("Customize User Agent (randomize multiple agents)");
70          
71          labelIsCheckingUpdate.addActionListener(actionEvent -> {
72              
73              this.checkboxIsCustomUserAgent.setSelected(!this.checkboxIsCustomUserAgent.isSelected());
74              
75              MediatorHelper.model().getMediatorUtils().getUserAgentUtil().setIsCustomUserAgent(
76                  this.checkboxIsCustomUserAgent.isSelected()
77              );
78          });
79          
80          var textAreaIsTamperingEval = new LightScrollPane(this.textfieldCustomUserAgent);
81          textAreaIsTamperingEval.setBorder(UiUtil.BORDER_FOCUS_LOST);
82          this.textfieldCustomUserAgent.setMinimumSize(new Dimension(40000, 100));
83          
84          var emptyLabelSessionManagement = new JLabel();
85          
86          Stream.of(labelIsCheckingUpdate)
87          .forEach(label -> {
88              
89              label.setHorizontalAlignment(SwingConstants.LEFT);
90              label.setBorderPainted(false);
91              label.setContentAreaFilled(false);
92          });
93          
94          var groupLayout = new GroupLayout(this);
95          this.setLayout(groupLayout);
96          
97          groupLayout
98          .setHorizontalGroup(
99              groupLayout
100             .createSequentialGroup()
101             .addGroup(
102                 groupLayout
103                 .createParallelGroup(GroupLayout.Alignment.TRAILING, false)
104                 .addComponent(this.checkboxIsCustomUserAgent)
105                 .addComponent(emptyLabelSessionManagement)
106             )
107             .addGroup(
108                 groupLayout
109                 .createParallelGroup()
110                 .addComponent(labelIsCheckingUpdate)
111                 .addComponent(textAreaIsTamperingEval)
112             )
113         );
114 
115         groupLayout
116         .setVerticalGroup(
117             groupLayout
118             .createSequentialGroup()
119             .addGroup(
120                 groupLayout
121                 .createParallelGroup(GroupLayout.Alignment.BASELINE)
122                 .addComponent(this.checkboxIsCustomUserAgent)
123                 .addComponent(labelIsCheckingUpdate)
124             )
125             .addGroup(
126                 groupLayout
127                 .createParallelGroup(GroupLayout.Alignment.BASELINE)
128                 .addComponent(emptyLabelSessionManagement)
129                 .addComponent(textAreaIsTamperingEval)
130             )
131         );
132     }
