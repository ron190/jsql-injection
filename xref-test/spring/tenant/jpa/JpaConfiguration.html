12  import org.springframework.orm.jpa.JpaVendorAdapter;
13  import org.springframework.orm.jpa.persistenceunit.PersistenceUnitManager;
14  import org.springframework.transaction.PlatformTransactionManager;
15  import org.springframework.transaction.annotation.EnableTransactionManagement;
16  import spring.SpringApp;
17  
18  import javax.sql.DataSource;
19  import java.sql.DriverManager;
20  import java.util.*;
21  
22  @Configuration
23  @EnableTransactionManagement
24  public class JpaConfiguration {
25  
26      @Bean
27      public PlatformTransactionManager transactionManager(EntityManagerFactory emf) {
28          return new JpaTransactionManager(emf);
29      }
30  
31      @Bean
32      public EntityManagerFactoryBuilder entityManagerFactoryBuilder(
33          JpaVendorAdapter jpaVendorAdapter,
34          ObjectProvider<PersistenceUnitManager> persistenceUnitManager
35      ) {  // init entity, required for h2
36          return new EntityManagerFactoryBuilder(
37              jpaVendorAdapter,
38              f -> Map.of(),
39              persistenceUnitManager.getIfAvailable()
40          );
41      }
42  
43      @Bean
44      public DataSource dataSource() {  // unused when hibernate tenant active
45          Map<Object, Object> resolvedDataSources = new HashMap<>();
46  
47          DriverManager.setLogWriter(null);  // remove annoying logs from jdbc driver
48          SpringApp.getPropertiesFilterByProfile().map(AbstractMap.SimpleEntry::getValue).forEach(props -> {
49              DataSource dataSource = DataSourceBuilder.create()
50                  .url(props.getProperty(JdbcSettings.JAKARTA_JDBC_URL))
51                  .username(props.getProperty(JdbcSettings.JAKARTA_JDBC_USER))
52                  .password(props.getProperty(JdbcSettings.JAKARTA_JDBC_PASSWORD))
53                  .driverClassName(props.getProperty(JdbcSettings.JAKARTA_JDBC_DRIVER))
54              .build();
55              resolvedDataSources.put(props.getProperty(SpringApp.JSQL_TENANT), dataSource);
56          });
57  
58          AbstractRoutingDataSource dataSource = new MultitenantDataSource();
59          dataSource.setDefaultTargetDataSource(resolvedDataSources.get(SpringApp.H2));
60          dataSource.setTargetDataSources(resolvedDataSources);
61  
62          dataSource.afterPropertiesSet();
63          return dataSource;
64      }
65  }
