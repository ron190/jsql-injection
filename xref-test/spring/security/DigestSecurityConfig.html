13  import org.springframework.security.web.authentication.www.DigestAuthenticationEntryPoint;
14  import org.springframework.security.web.authentication.www.DigestAuthenticationFilter;
15  import org.springframework.security.web.servlet.util.matcher.PathPatternRequestMatcher;
16  
17  @Configuration
18  @Order(3)
19  public class DigestSecurityConfig {
20      
21      private static final String DIGEST_AUTHORITY = "ADMIN2";
22      private static final String DIGEST_REALM = "Digest Realm";
23      public static final String DIGEST_USERNAME = "login-digest";
24      public static final String DIGEST_PASSWORD = "password-digest";
25      private static final String DIGEST_PASSWORD_ENCODED = DigestAuthUtils.encodePasswordInA1Format(
26          DigestSecurityConfig.DIGEST_USERNAME,
27          DigestSecurityConfig.DIGEST_REALM,
28          DigestSecurityConfig.DIGEST_PASSWORD
29      );
30      public static final CustomFilter FILTER = new CustomFilter("digest");
31  
32      @Bean
33      public SecurityFilterChain filterChainDigest(HttpSecurity http) {
34          return http.securityMatcher("/digest/**")
35              .csrf(Customizer.withDefaults())
36              .authorizeHttpRequests(auth -> auth
37                  .requestMatchers(PathPatternRequestMatcher.withDefaults().matcher("/digest/**"))
38                  .hasAuthority(DigestSecurityConfig.DIGEST_AUTHORITY)
39              )
40              .addFilterAfter(DigestSecurityConfig.FILTER, AuthorizationFilter.class)
41              .addFilter(this.digestAuthenticationFilter())
42              .exceptionHandling(exception -> exception
43                  .defaultAuthenticationEntryPointFor(
44                      this.digestEntryPoint(),
45                      PathPatternRequestMatcher.withDefaults().matcher("/digest/**")
46                  )
47              )
48              .build();
49      }
50  
51      public Filter digestAuthenticationFilter() {
52          UserDetails user = User.builder()
53              .username(DigestSecurityConfig.DIGEST_USERNAME)
54              .password(DigestSecurityConfig.DIGEST_PASSWORD_ENCODED)
55              .authorities(DigestSecurityConfig.DIGEST_AUTHORITY)
56              .build();
57  
58          DigestAuthenticationFilter digestAuthenticationFilter = new DigestAuthenticationFilter();
59          digestAuthenticationFilter.setAuthenticationEntryPoint(this.digestEntryPoint());
60          digestAuthenticationFilter.setUserDetailsService(username -> user);
61          digestAuthenticationFilter.setPasswordAlreadyEncoded(true);
62          digestAuthenticationFilter.setCreateAuthenticatedToken(true);
63  
64          return digestAuthenticationFilter;
65      }
66  
67      @Bean
68      public DigestAuthenticationEntryPoint digestEntryPoint() {
69          DigestAuthenticationEntryPoint digestAuthenticationEntryPoint = new DigestAuthenticationEntryPoint();
70          digestAuthenticationEntryPoint.setKey("mykey");
71          digestAuthenticationEntryPoint.setRealmName(DigestSecurityConfig.DIGEST_REALM);
72          return digestAuthenticationEntryPoint;
73      }
74  }
