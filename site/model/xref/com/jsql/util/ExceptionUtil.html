18  /**
19   * Utility class managing exception reporting to GitHub.
20   */
21  public class ExceptionUtil {
22      
23      private static final Logger LOGGER = LogManager.getRootLogger();
24      
25      private final InjectionModel injectionModel;
26      
27      private final Set<String> exceptionsMd5Cached = new CopyOnWriteArraySet<>();
28      
29      public ExceptionUtil(InjectionModel injectionModel) {
30          this.injectionModel = injectionModel;
31      }
32      
33      /**
34       * Handler class processing errors on top of the JVM in order to send
35       * a report to GitHub automatically.
36       */
37      public class ExceptionHandler implements Thread.UncaughtExceptionHandler {
38  
39          @Override
40          public void uncaughtException(Thread thread, Throwable throwable) {
41              LOGGER.log(
42                  LogLevelUtil.CONSOLE_JAVA,
43                  () -> String.format("Unhandled Exception on %s", thread.getName()),
44                  throwable
45              );
46              LOGGER.log(  // Display to stdout
47                  Level.ERROR,
48                  () -> String.format("Unhandled Exception on %s", thread.getName()),
49                  throwable
50              );
51  
52              // Report #214: ignore if OutOfMemoryError: Java heap space
53              if (
54                  ExceptionUtil.this.injectionModel.getMediatorUtils().getPreferencesUtil().isReportingBugs()
55                  && ExceptionUtils.getStackTrace(throwable).contains("com.jsql")
56                  && !(throwable instanceof OutOfMemoryError)
57                  && !ExceptionUtils.getStackTrace(throwable).contains("OutOfMemoryError")  // when implicit
58              ) {
59                  if (ExceptionUtils.getStackTrace(throwable).contains("Could not initialize class java.awt.Toolkit")) {
60                      LOGGER.log(LogLevelUtil.CONSOLE_JAVA, "System libraries are missing, please use a proper Java runtime instead of headless runtime");
61                      return;
62                  } else if (ExceptionUtils.getStackTrace(throwable).contains("Could not initialize class sun.awt.X11.XToolkit")) {
63                      LOGGER.log(LogLevelUtil.CONSOLE_JAVA, "System libraries are missing or wrong DISPLAY variable, please verify your settings");
64                      return;
65                  }
66  
67                  try {
68                      var messageDigest = MessageDigest.getInstance("Md5");
69  
70                      String stackTrace = ExceptionUtils.getStackTrace(throwable).trim();
71                      var passwordString = String.valueOf(stackTrace.toCharArray());
72  
73                      byte[] passwordByte = passwordString.getBytes(StandardCharsets.UTF_8);
74                      messageDigest.update(passwordByte, 0, passwordByte.length);
75  
76                      byte[] encodedPassword = messageDigest.digest();
77                      var md5Exception = HashUtil.digestToHexString(encodedPassword);
78                      if (!ExceptionUtil.this.exceptionsMd5Cached.contains(md5Exception)) {
79                          ExceptionUtil.this.exceptionsMd5Cached.add(md5Exception);
80                          ExceptionUtil.this.injectionModel.getMediatorUtils().getGitUtil().sendUnhandledException(
81                              thread.getName(),
82                              throwable
83                          );
84                      }
85                  } catch (NoSuchAlgorithmException e) {
86                      LOGGER.log(LogLevelUtil.IGNORE, e);
87                  }
88              }
89          }
90      }
91      
92      /**
93       * Add the error reporting mechanism on top of the JVM in order to
94       * intercept and process the error to GitHub.
95       */
96      public void setUncaughtExceptionHandler() {
97          Thread.setDefaultUncaughtExceptionHandler(new ExceptionHandler());  // Regular Exception
98          try {  // Event dispatching thread Exception
99              SwingUtilities.invokeAndWait(() ->
100                 Thread.currentThread().setUncaughtExceptionHandler(new ExceptionHandler())  // We are in the event dispatching thread
101             );
102         } catch (InvocationTargetException | InterruptedException e) {
103             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
104             Thread.currentThread().interrupt();
105         }
106     }
107 }
