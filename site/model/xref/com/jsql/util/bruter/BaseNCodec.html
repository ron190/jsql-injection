30  /**
31   * Abstract superclass for Base-N encoders and decoders.
32   *
33   * <p>
34   * This class is thread-safe.
35   * </p>
36   *
37   * You can set the decoding behavior when the input bytes contain leftover trailing bits that cannot be created by a valid
38   * encoding. These can be bits that are unused from the final character or entire characters. The default mode is
39   * lenient decoding.
40   * <ul>
41   * <li>Lenient: Any trailing bits are composed into 8-bit bytes where possible. The remainder are discarded.
42   * <li>Strict: The decoding will raise an {@link IllegalArgumentException} if trailing bits are not part of a valid
43   * encoding. Any unused bits from the final character must be zero. Impossible counts of entire final characters are not
44   * allowed.
45   * </ul>
46   * <p>
47   * When strict decoding is enabled it is expected that the decoded bytes will be re-encoded to a byte array that matches
48   * the original, i.e. no changes occur on the final character. This requires that the input bytes use the same padding
49   * and alphabet as the encoder.
50   * </p>
51   */
52  public abstract class BaseNCodec implements BinaryEncoder, BinaryDecoder {

206      * If {@code chunkSeparatorLength} is zero, then chunking is disabled.
207      * @param unencodedBlockSize the size of an unencoded block (e.g. Base64 = 3)
208      * @param encodedBlockSize the size of an encoded block (e.g. Base64 = 4)
209      * @param lineLength if &gt; 0, use chunking with a length {@code lineLength}
210      * @param chunkSeparatorLength the chunk separator length, if relevant
211      * @param pad byte used as padding byte.
212      * @param decodingPolicy Decoding policy.
213      * @since 1.15
214      */
215     protected BaseNCodec(
216         final int unencodedBlockSize,
217         final int encodedBlockSize,
218         final int lineLength,
219         final int chunkSeparatorLength,
220         final byte pad,
221         final CodecPolicy decodingPolicy
222     ) {
223         
224         this.unencodedBlockSize = unencodedBlockSize;
225         this.encodedBlockSize = encodedBlockSize;
226         final boolean useChunking = lineLength > 0 && chunkSeparatorLength > 0;
227         this.lineLength = useChunking ? (lineLength / encodedBlockSize) * encodedBlockSize : 0;
228         this.chunkSeparatorLength = chunkSeparatorLength;
229         this.pad = pad;
230         this.decodingPolicy = Objects.requireNonNull(decodingPolicy, "codecPolicy");
231     }

233     /**
234      * Compares two {@code int} values numerically treating the values
235      * as unsigned. Taken from JDK 1.8.
236      *
237      * @param  x the first {@code int} to compare
238      * @param  y the second {@code int} to compare
239      * @return the value {@code 0} if {@code x == y}; a value less
240      *         than {@code 0} if {@code x < y} as unsigned values; and
241      *         a value greater than {@code 0} if {@code x > y} as
242      *         unsigned values
243      */
244     private static int compareUnsigned(final int x, final int y) {
245         
246         return Integer.compare(x + Integer.MIN_VALUE, y + Integer.MIN_VALUE);
247     }

249     /**
250      * Create a positive capacity at least as large the minimum required capacity.
251      * If the minimum capacity is negative then this throws an OutOfMemoryError as no array
252      * can be allocated.
253      *
254      * @param minCapacity the minimum capacity
255      * @return the capacity
256      * @throws OutOfMemoryError if the {@code minCapacity} is negative
257      */
258     private static int createPositiveCapacity(final int minCapacity) {
259         
260         if (minCapacity < 0) {
261             
262             // overflow
263             throw new OutOfMemoryError("Unable to allocate array size: " + (minCapacity & 0xffffffffL));
264         }
265         
266         // This is called when we require buffer expansion to a very big array.
267         // Use the conservative maximum buffer size if possible, otherwise the biggest required.
268         //
269         // Note: In this situation JDK 1.8 java.util.ArrayList returns Integer.MAX_VALUE.
270         // This excludes some VMs that can exceed MAX_BUFFER_SIZE but not allocate a full
271         // Integer.MAX_VALUE length array.
272         // The result is that we may have to allocate an array of this size more than once if
273         // the capacity must be expanded again.
274         return Math.max(minCapacity, MAX_BUFFER_SIZE);
275     }

304             default :
305                 return false;

309     /**
310      * Increases our buffer by the {@link #DEFAULT_BUFFER_RESIZE_FACTOR}.
311      * @param context the context to be used
312      * @param minCapacity the minimum required capacity
313      * @return the resized byte[] buffer
314      * @throws OutOfMemoryError if the {@code minCapacity} is negative
315      */
316     private static byte[] resizeBuffer(final Context context, final int minCapacity) {
317         
318         // Overflow-conscious code treats the min and new capacity as unsigned.
319         final int oldCapacity = context.buffer.length;
320         int newCapacity = oldCapacity * DEFAULT_BUFFER_RESIZE_FACTOR;
321         
322         if (compareUnsigned(newCapacity, minCapacity) < 0) {
323             
324             newCapacity = minCapacity;
325         }
326         
327         if (compareUnsigned(newCapacity, MAX_BUFFER_SIZE) > 0) {
328             
329             newCapacity = createPositiveCapacity(minCapacity);
330         }
331 
332         final var b = new byte[newCapacity];
333         System.arraycopy(context.buffer, 0, b, 0, context.buffer.length);
334         context.buffer = b;
335         
336         return b;
337     }

378      * Decodes a byte[] containing characters in the Base-N alphabet.
379      *
380      * @param pArray
381      *            A byte array containing Base-N character data
382      * @return a byte array containing binary data
383      */
384     @Override
385     public byte[] decode(final byte[] pArray) {
386         
387         if (pArray == null || pArray.length == 0) {
388             
389             return pArray;
390         }
391         
392         final var context = new Context();
393         this.decode(pArray, 0, pArray.length, context);
394         this.decode(pArray, 0, EOF, context); // Notify decoder of EOF.
395         final var result = new byte[context.pos];
396         this.readResults(result, 0, result.length, context);
397         
398         return result;
399     }

541     /**
542      * Ensure that the buffer has room for {@code size} bytes
543      *
544      * @param size minimum spare space required
545      * @param context the context to be used
546      * @return the buffer
547      */
548     protected byte[] ensureBufferSize(final int size, final Context context) {
549         
550         if (context.buffer == null) {
551             
552             context.buffer = new byte[Math.max(size, this.getDefaultBufferSize())];
553             context.pos = 0;
554             context.readPos = 0;
555 
556             // Overflow-conscious:
557             // x + y > z  ==  x + y - z > 0
558             
559         } else if (context.pos + size - context.buffer.length > 0) {
560             
561             return resizeBuffer(context, context.pos + size);
562         }
563         
564         return context.buffer;
565     }

594     /**
595      * Calculates the amount of space needed to encode the supplied array.
596      *
597      * @param pArray byte[] array which will later be encoded
598      *
599      * @return amount of space needed to encoded the supplied array.
600      * Returns a long since a max-len array will require &gt; Integer.MAX_VALUE
601      */
602     public long getEncodedLength(final byte[] pArray) {
603         
604         // Calculate non-chunked size - rounded up to allow for padding
605         // cast to long is needed to avoid possibility of overflow
606         long len = ((pArray.length + this.unencodedBlockSize-1)  / this.unencodedBlockSize) * (long) this.encodedBlockSize;
607         
608         if (this.lineLength > 0) { // We're using chunking
609             // Round up to nearest multiple
610             len += ((len + this.lineLength-1) / this.lineLength) * this.chunkSeparatorLength;
611         }
612         
613         return len;
614     }

637     /**
638      * Tests a given byte array to see if it contains only valid characters within the alphabet.
639      * The method optionally treats whitespace and pad as valid.
640      *
641      * @param arrayOctet byte array to test
642      * @param allowWSPad if {@code true}, then whitespace and PAD are also allowed
643      *
644      * @return {@code true} if all bytes are valid characters in the alphabet or if the byte array is empty;
645      *         {@code false}, otherwise
646      */
647     public boolean isInAlphabet(final byte[] arrayOctet, final boolean allowWSPad) {
648         
649         for (final byte octet : arrayOctet) {
650             
651             if (
652                 !this.isInAlphabet(octet)
653                 && (!allowWSPad || (octet != this.pad)
654                 && !isWhiteSpace(octet))
655             ) {
656                 return false;
657             }
658         }
659         
660         return true;
661     }

702      * @param bPos
703      *            position in byte[] array to start extraction at.
704      * @param bAvail
705      *            amount of bytes we're allowed to extract. We may extract fewer (if fewer are available).
706      * @param context
707      *            the context to be used
708      * @return The number of bytes successfully extracted into the provided byte[] array.
709      */
710     private int readResults(final byte[] b, final int bPos, final int bAvail, final Context context) {
711         
712         if (context.buffer != null) {
713             
714             final int len = Math.min(this.available(context), bAvail);
715             System.arraycopy(context.buffer, context.readPos, b, bPos, len);
716             context.readPos += len;
717             
718             if (context.readPos >= context.pos) {
719                 
720                 context.buffer = null; // so hasData() will return false, and this method can return -1
721             }
722             
723             return len;
724         }
725         
726         return context.eof ? EOF : 0;
727     }

729     /**
730      * Holds thread context so classes can be thread-safe.
731      *
732      * This class is not itself thread-safe; each thread must allocate its own copy.
733      *
734      * @since 1.7
735      */
736     protected static class Context {
737 
738         /**
739          * Place holder for the bytes we're dealing with for our based logic.
740          * Bitwise operations store and extract the encoding or decoding from this variable.
741          */
742         protected int ibitWorkArea;
743 
744         /**
745          * Place holder for the bytes we're dealing with for our based logic.
746          * Bitwise operations store and extract the encoding or decoding from this variable.
747          */
748         protected long lbitWorkArea;
749 
750         /**
751          * Buffer for streaming.
752          */
753         protected byte[] buffer;
754 
755         /**
756          * Position where next character should be written in the buffer.
757          */
758         protected int pos;
759 
760         /**
761          * Position where next character should be read from the buffer.
762          */
763         protected int readPos;
764 
765         /**
766          * Boolean flag to indicate the EOF has been reached. Once EOF has been reached, this object becomes useless,
767          * and must be thrown away.
768          */
769         protected boolean eof;
770 
771         /**
772          * Variable tracks how many characters have been written to the current line. Only used when encoding. We use
773          * it to make sure each encoded line never goes beyond lineLength (if lineLength &gt; 0).
774          */
775         protected int currentLinePos;
776 
777         /**
778          * Writes to the buffer only occur after every 3/5 reads when encoding, and every 4/8 reads when decoding. This
779          * variable helps track that.
780          */
781         protected int modulus;
782 
783         /**
784          * Returns a String useful for debugging (especially within a debugger.)
785          *
786          * @return a String useful for debugging.
787          */
788         @SuppressWarnings("boxing") // OK to ignore boxing here
789         @Override
790         public String toString() {
791             return
792                 String.format(
793                     "%s[buffer=%s, currentLinePos=%s, eof=%s, ibitWorkArea=%s, lbitWorkArea=%s, modulus=%s, pos=%s, readPos=%s]",
794                     this.getClass().getSimpleName(),
795                     Arrays.toString(this.buffer),
796                     this.currentLinePos,
797                     this.eof,
798                     this.ibitWorkArea,
799                     this.lbitWorkArea,
800                     this.modulus,
801                     this.pos,
802                     this.readPos
803                 );
804         }
805     }
