1   package com.jsql.model.suspendable.callable;
2   
3   import com.jsql.model.InjectionModel;
4   import org.apache.commons.lang3.StringUtils;
5   
6   import java.util.concurrent.Callable;
7   
8   /**
9    * Callable for concurrent HTTP tasks
10   * url: SQL query
11   * content: source code of the web page
12   * tag: store user information (ex. current index)
13   */
14  public class CallablePageSource implements Callable<CallablePageSource> {
15      
16      /**
17       * URL to load.
18       */
19      private final String query;
20      private final String metadataInjectionProcess;
21  
22      /**
23       * Source code for current page.
24       */
25      private String content = StringUtils.EMPTY;
26  
27      /**
28       * Character used for current page.
29       */
30      private String characterInsertion;
31      
32      private final InjectionModel injectionModel;
33      
34      /**
35       * Create a callable to get initial query or insertion character.
36       */
37      public CallablePageSource(String query, InjectionModel injectionModel, String metadataInjectionProcess) {
38          this.query = query;
39          this.injectionModel = injectionModel;
40          this.metadataInjectionProcess = metadataInjectionProcess;
41      }
42  
43      /**
44       * Create callable for current insertion character test.
45       */
46      public CallablePageSource(
47          String query,
48          String characterInsertion,
49          InjectionModel injectionModel,
50          String metadataInjectionProcess
51      ) {
52          this(query, injectionModel, metadataInjectionProcess);
53          this.characterInsertion = characterInsertion;
54      }
55      
56      @Override
57      public CallablePageSource call() {
58          this.content = this.injectionModel.injectWithoutIndex(this.query, this.metadataInjectionProcess);
59          return this;
60      }
61  
62      
63      // Getters
64  
65      public String getQuery() {
66          return this.query;
67      }
68      
69      public String getContent() {
70          return this.content;
71      }
72      
73      public String getCharacterInsertion() {
74          return this.characterInsertion;
75      }
76  }
