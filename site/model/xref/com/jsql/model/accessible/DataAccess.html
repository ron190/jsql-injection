45      /**
46       * Regex characters marking the end of the result of an injection.
47       * Process stops when this schema is encountered:
48       * <pre>SQLix01x03x03x07
49       */
50      public static final String TRAIL_RGX = "\\x01\\x03\\x03\\x07";
51      
52      /**
53       * Regex character used between each table cells.
54       * Expected schema of multiple table cells :
55       * <pre>
56       * x04[table cell]x05[number of occurrences]x04x06x04[table cell]x05[number of occurrences]x04
57       */
58      public static final String SEPARATOR_CELL_RGX = "\\x06";
59      
60      /**
61       * Regex character used between the table cell and the number of occurrence of the cell text.
62       * Expected schema of a table cell data is
63       * <pre>x04[table cell]x05[number of occurrences]x04
64       */
65      public static final String SEPARATOR_QTE_RGX = "\\x05";
66  
67      /**
68       * Regex character enclosing a table cell returned by injection.
69       * It allows to detect the correct end of a table cell data during parsing.
70       * Expected schema of a table cell data is
71       * <pre>x04[table cell]x05[number of occurrences]x04
72       */
73      public static final String ENCLOSE_VALUE_RGX = "\\x04";
74      
75      public static final String LEAD = "SqLi";
76      public static final String SHELL_LEAD = "${shell.lead}";
77      public static final String TRAIL = "iLQS";
78      public static final String SHELL_TRAIL = "${shell.trail}";
79      
80      /**
81       * Regex keywords corresponding to multiline and case insensitive match.
82       */
83      public static final String MODE = "(?si)";
84      
85      /**
86       * Regex schema describing a table cell with firstly the cell content and secondly the number of occurrences
87       * of the cell text, separated by the reserved character x05 in hexadecimal.
88       * The range of characters from x01 to x1F are not printable ASCII characters used to parse the data and exclude
89       * printable characters during parsing.
90       * Expected schema of a table cell data is
91       * <pre>x04[table cell]x05[number of occurrences]x04
92       */
93      public static final String CELL_TABLE = "([^\\x01-\\x09\\x0B-\\x0C\\x0E-\\x1F]*)"+ SEPARATOR_QTE_RGX +"([^\\x01-\\x09\\x0B-\\x0C\\x0E-\\x1F]*)(\\x08)?";

102     /**
103      * Get general database informations.<br>
104      * => version{%}database{%}user{%}CURRENT_USER
105      */
106     public void getDatabaseInfos() {
107         
108         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, () -> I18nUtil.valueByKey("LOG_FETCHING_INFORMATIONS"));
109         
110         var sourcePage = new String[]{ StringUtils.EMPTY };
111 
112         var resultToParse = "";
113         
114         try {
115             resultToParse = new SuspendableGetRows(this.injectionModel).run(
116                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlInfos(),
117                 sourcePage,
118                 false,
119                 0,
120                 null,
121                 "metadata"
122             );
123         
124         } catch (AbstractSlidingException e) {
125 
126             resultToParse = getPartialResultAndLog(e, resultToParse);
127 
128         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
129 
130             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
131         }
132 
133         if (StringUtils.isEmpty(resultToParse)) {
134             
135             this.injectionModel.sendResponseFromSite("Incorrect metadata", sourcePage[0].trim());
136         }
137         
138         try {
139             String versionDatabase = resultToParse.split(ENCLOSE_VALUE_RGX)[0].replaceAll("\\s+", StringUtils.SPACE);
140             String nameDatabase = resultToParse.split(ENCLOSE_VALUE_RGX)[1];
141             String username = resultToParse.split(ENCLOSE_VALUE_RGX)[2];
142             
143             var infos = String.format(
144                 "Database [%s] on %s [%s] for user [%s]",
145                 nameDatabase,
146                 this.injectionModel.getMediatorVendor().getVendor(),
147                 versionDatabase,
148                 username
149             );
150             
151             LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, infos);
152             
153         } catch (ArrayIndexOutOfBoundsException e) {
154 
155             LOGGER.log(
156                 LogLevelUtil.CONSOLE_ERROR,
157                 String.format("%s: %s", I18nUtil.valueByKey("LOG_DB_METADATA_INCORRECT"), resultToParse)
158             );
159             LOGGER.log(LogLevelUtil.CONSOLE_INFORM, I18nUtil.valueByKey("LOG_DB_METADATA_WARN"));
160         }
161     }

163     /**
164      * Get database names and table counts and send them to the view.<br>
165      * Use readable text (not hexa) and parse this pattern:<br>
166      * => hh[database name 1]jj[table count]hhgghh[database name 2]jj[table count]hhggh...hi<br>
167      * Data window can be cut before the end of the request but the process helps to obtain
168      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
169      * @return list of databases found
170      * @throws JSqlException when injection failure or stopped by user
171      */
172     public List<Database> listDatabases() throws JSqlException {
173         
174         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, () -> I18nUtil.valueByKey("LOG_FETCHING_DATABASES"));
175         
176         List<Database> databases = new ArrayList<>();
177         
178         String resultToParse = StringUtils.EMPTY;
179         
180         try {
181             var sourcePage = new String[]{ StringUtils.EMPTY };
182             resultToParse = new SuspendableGetRows(this.injectionModel).run(
183                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlDatabases(),
184                 sourcePage,
185                 true,
186                 0,
187                 null,
188                 "databases"
189             );
190             
191         } catch (AbstractSlidingException e) {
192 
193             resultToParse = getPartialResultAndLog(e, resultToParse);
194 
195         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
196 
197             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
198         }
199 
200         // Parse all data we have retrieved
201         var regexSearch = Pattern.compile(
202                 MODE
203                 + ENCLOSE_VALUE_RGX
204                 + CELL_TABLE
205                 + ENCLOSE_VALUE_RGX
206             )
207             .matcher(resultToParse);
208 
209         if (!regexSearch.find()) {
210             
211             throw new InjectionFailureException("No match while injecting databases");
212         }
213         
214         regexSearch.reset();
215 
216         // Build an array of Database objects from the data we have parsed
217         while (regexSearch.find()) {
218             
219             String databaseName = regexSearch.group(1);
220             String tableCount = regexSearch.group(2);
221 
222             var newDatabase = new Database(databaseName, tableCount);
223             databases.add(newDatabase);
224         }
225 
226         var request = new Request();
227         request.setMessage(Interaction.ADD_DATABASES);
228         request.setParameters(databases);
229         this.injectionModel.sendToViews(request);
230         
231         return databases;
232     }

234     /**
235      * Get tables name and row count and send them to the view.<br>
236      * Use readable text (not hexa) and parse this pattern:<br>
237      * => hh[table name 1]jj[rows count]hhgghh[table name 2]jj[rows count]hhggh...hi<br>
238      * Data window can be cut before the end of the request but the process helps to obtain
239      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
240      * @param database which contains tables to find
241      * @return list of tables found
242      * @throws JSqlException when injection failure or stopped by user
243      */
244     public List<Table> listTables(Database database) throws JSqlException {
245         
246         // Reset stoppedByUser if list of Databases is partial
247         // and some Tables are still reachable
248         this.injectionModel.setIsStoppedByUser(false);
249         
250         // Inform the view that database has just been used
251         var requestStartProgress = new Request();
252         requestStartProgress.setMessage(Interaction.START_PROGRESS);
253         requestStartProgress.setParameters(database);
254         this.injectionModel.sendToViews(requestStartProgress);
255 
256         var tableCount = Integer.toString(database.getChildCount());
257         
258         String resultToParse = StringUtils.EMPTY;
259         
260         try {
261             var pageSource = new String[]{ StringUtils.EMPTY };
262             resultToParse = new SuspendableGetRows(this.injectionModel).run(
263                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlTables(database),
264                 pageSource,
265                 true,
266                 Integer.parseInt(tableCount),
267                 database,
268                 "tables"
269             );
270             
271         } catch (AbstractSlidingException e) {
272 
273             resultToParse = getPartialResultAndLog(e, resultToParse);
274 
275         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
276 
277             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
278         }
279 
280         // Parse all the data we have retrieved
281         var regexSearch = Pattern.compile(
282                 MODE
283                 + ENCLOSE_VALUE_RGX
284                 + CELL_TABLE
285                 + ENCLOSE_VALUE_RGX
286             )
287             .matcher(resultToParse);
288         
289         var requestEndProgress = new Request();
290         requestEndProgress.setMessage(Interaction.END_PROGRESS);
291         requestEndProgress.setParameters(database);
292         this.injectionModel.sendToViews(requestEndProgress);
293         
294         if (!regexSearch.find()) {
295             
296             throw new InjectionFailureException("No match while injecting tables");
297         }
298         
299         regexSearch.reset();
300         
301         List<Table> tables = new ArrayList<>();
302         
303         // Build an array of Table objects from the data we have parsed
304         while (regexSearch.find()) {
305             
306             String tableName = regexSearch.group(1);
307             String rowCount = regexSearch.group(2);
308             
309             var newTable = new Table(tableName, rowCount, database);
310             tables.add(newTable);
311         }
312         
313         var requestAddTables = new Request();
314         requestAddTables.setMessage(Interaction.ADD_TABLES);
315         requestAddTables.setParameters(tables);
316         this.injectionModel.sendToViews(requestAddTables);
317         
318         return tables;
319     }

321     /**
322      * Get column names and send them to the view.<br>
323      * Use readable text (not hexa) and parse this pattern with 2nd member forced to 31 (1 in ascii):<br>
324      * => hh[column name 1]jj[31]hhgghh[column name 2]jj[31]hhggh...hi<br>
325      * Data window can be cut before the end of the request but the process helps to obtain
326      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
327      * @param table which contains columns to find
328      * @return list of columns found
329      * @throws JSqlException when injection failure or stopped by user
330      */
331     public List<Column> listColumns(Table table) throws JSqlException {
332         
333         List<Column> columns = new ArrayList<>();
334         
335         // Inform the view that table has just been used
336         var requestStartProgress = new Request();
337         requestStartProgress.setMessage(Interaction.START_INDETERMINATE_PROGRESS);
338         requestStartProgress.setParameters(table);
339         this.injectionModel.sendToViews(requestStartProgress);
340 
341         String resultToParse = StringUtils.EMPTY;
342         
343         try {
344             var pageSource = new String[]{ StringUtils.EMPTY };
345             resultToParse = new SuspendableGetRows(this.injectionModel).run(
346                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlColumns(table),
347                 pageSource,
348                 true,
349                 0,
350                 table,
351                 "columns"
352             );
353             
354         } catch (AbstractSlidingException e) {
355 
356             resultToParse = getPartialResultAndLog(e, resultToParse);
357 
358         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
359 
360             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
361         }
362 
363         // Build SQLite columns
364         if (this.injectionModel.getMediatorVendor().isSqlite()) {
365             
366             resultToParse = this.injectionModel.getMediatorVendor().getSqlite().transformSqlite(resultToParse);
367         }
368         
369         // Parse all the data we have retrieved
370         var regexSearch = Pattern.compile(
371                 MODE
372                 + ENCLOSE_VALUE_RGX
373                 + CELL_TABLE
374                 + ENCLOSE_VALUE_RGX
375             )
376             .matcher(resultToParse);
377 
378         var requestEndProgress = new Request();
379         requestEndProgress.setMessage(Interaction.END_INDETERMINATE_PROGRESS);
380         requestEndProgress.setParameters(table);
381         this.injectionModel.sendToViews(requestEndProgress);
382 
383         if (!regexSearch.find()) {
384             
385             throw new InjectionFailureException("No match while injecting columns");
386         }
387 
388         regexSearch.reset();
389 
390         // Build an array of Column objects from the data we have parsed
391         while (regexSearch.find()) {
392             
393             String nameColumn = regexSearch.group(1);
394 
395             var column = new Column(nameColumn, table);
396             columns.add(column);
397         }
398 
399         var requestAddColumns = new Request();
400         requestAddColumns.setMessage(Interaction.ADD_COLUMNS);
401         requestAddColumns.setParameters(columns);
402         this.injectionModel.sendToViews(requestAddColumns);
403         
404         return columns;
405     }

407     /**
408      * Get table values and count each occurrences and send them to the view.<br>
409      * Values are on clear text (not hexa) and follows this window pattern<br>
410      * => hh[value 1]jj[count]hhgghh[value 2]jj[count]hhggh...hi<br>
411      * Data window can be cut before the end of the request but the process helps to obtain
412      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
413      * @param columnsBean choice by the user
414      * @return a 2x2 table containing values by columns
415      * @throws JSqlException when injection failure or stopped by user
416      */
417     public String[][] listValues(List<Column> columnsBean) throws JSqlException {
418         
419         var databaseBean = (Database) columnsBean.get(0).getParent().getParent();
420         var tableBean = (Table) columnsBean.get(0).getParent();
421         int rowCount = columnsBean.get(0).getParent().getChildCount();
422 
423         // Inform the view that table has just been used
424         var request = new Request();
425         request.setMessage(Interaction.START_PROGRESS);
426         request.setParameters(tableBean);
427         this.injectionModel.sendToViews(request);
428 
429         // Build an array of column names
430         List<String> columnsName = new ArrayList<>();
431         for (AbstractElementDatabase columnBean: columnsBean) {
432             columnsName.add(columnBean.toString());
433         }
434 
435         // From that array, build the SQL fields nicely
436         // => col1{%}col2...
437         // ==> trim(ifnull(`col1`,0x00)),0x7f,trim(ifnull(`Col2`,0x00))...
438         String[] columns = columnsName.toArray(new String[0]);
439 
440         List<List<String>> listValues = this.getRows(databaseBean, tableBean, rowCount, columns);
441 
442         // Add the default title to the columns: row number, occurrence
443         columnsName.add(0, StringUtils.EMPTY);
444         columnsName.add(0, StringUtils.EMPTY);
445 
446         String[][] table = this.getTable(columnsName, listValues);
447 
448         columns = columnsName.toArray(new String[0]);
449         
450         // Group the columns names, values and Table object in one array
451         var objectData = new Object[]{ columns, table, tableBean };
452 
453         var requestCreateValuesTab = new Request();
454         requestCreateValuesTab.setMessage(Interaction.CREATE_VALUES_TAB);
455         requestCreateValuesTab.setParameters(objectData);
456         this.injectionModel.sendToViews(requestCreateValuesTab);
457 
458         var requestEndProgress = new Request();
459         requestEndProgress.setMessage(Interaction.END_PROGRESS);
460         requestEndProgress.setParameters(tableBean);
461         this.injectionModel.sendToViews(requestEndProgress);
462         
463         return table;
464     }

466     private List<List<String>> getRows(Database database, Table table, int rowCount, String[] columns) throws InjectionFailureException {
467         
468         String resultToParse = StringUtils.EMPTY;
469         
470         try {
471             var pageSource = new String[]{ StringUtils.EMPTY };
472             
473             resultToParse = new SuspendableGetRows(this.injectionModel).run(
474                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlRows(columns, database, table),
475                 pageSource,
476                 true,
477                 rowCount,
478                 table,
479                 "rows"
480             );
481             
482         } catch (AbstractSlidingException e) {  // Catch all exceptions but prevent detecting bug
483 
484             resultToParse = getPartialResultAndLog(e, resultToParse);
485 
486         } catch (Exception e) {
487             
488             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
489         }
490 
491         return SuspendableGetRows.parse(resultToParse);
492     }

494     private static String getPartialResultAndLog(AbstractSlidingException e, String resultToParse) {
495 
496         LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage());
497 
498         // Get pieces of data already retrieved instead of losing them
499         if (StringUtils.isNotEmpty(e.getSlidingWindowAllRows())) {
500 
501             resultToParse = e.getSlidingWindowAllRows();
502 
503         } else if (StringUtils.isNotEmpty(e.getSlidingWindowCurrentRows())) {
504 
505             resultToParse = e.getSlidingWindowCurrentRows();
506         }
507 
508         return resultToParse;
509     }

511     private String[][] getTable(List<String> columnsName, List<List<String>> values) {
512         
513         // Build a proper 2D array from the data
514         var table = new String[values.size()][columnsName.size()];
515         
516         for (var indexRow = 0 ; indexRow < values.size() ; indexRow++) {
517             
518             var isIncomplete = false;
519             
520             for (var indexColumn = 0 ; indexColumn < columnsName.size() ; indexColumn++) {
521                 
522                 try {
523                     table[indexRow][indexColumn] = values.get(indexRow).get(indexColumn);
524                     
525                 } catch (IndexOutOfBoundsException e) {
526                     
527                     isIncomplete = true;
528                     
529                     LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, I18nUtil.valueByKey("LOG_LIST_VALUES_INCOMPLETE"));
530                     
531                     LOGGER.log(LogLevelUtil.IGNORE, e);
532                 }
533             }
534             
535             if (isIncomplete) {
536                 
537                 int logIndexRow = indexRow;
538                 LOGGER.log(
539                     LogLevelUtil.CONSOLE_ERROR,
540                     "{}{}: ",
541                     () -> I18nUtil.valueByKey("LOG_LIST_VALUES_TOO_LONG"),
542                     () -> logIndexRow + 1
543                 );
544                 LOGGER.log(
545                     LogLevelUtil.CONSOLE_ERROR,
546                     () -> String.join(
547                         ", ",
548                         values.get(logIndexRow).toArray(new String[0])
549                     )
550                 );
551             }
552         }
553         
554         return table;
555     }
