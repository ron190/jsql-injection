45      /**
46       * Regex characters marking the end of the result of an injection.
47       * Process stops when this schema is encountered:
48       * <pre>SQLix01x03x03x07
49       */
50      public static final String TRAIL_RGX = "\\x01\\x03\\x03\\x07";
51      
52      /**
53       * Regex character used between each table cells.
54       * Expected schema of multiple table cells :
55       * <pre>
56       * x04[table cell]x05[number of occurrences]x04x06x04[table cell]x05[number of occurrences]x04
57       */
58      public static final String SEPARATOR_CELL_RGX = "\\x06";
59      
60      /**
61       * Regex character used between the table cell and the number of occurrence of the cell text.
62       * Expected schema of a table cell data is
63       * <pre>x04[table cell]x05[number of occurrences]x04
64       */
65      public static final String SEPARATOR_QTE_RGX = "\\x05";
66  
67      /**
68       * Regex character enclosing a table cell returned by injection.
69       * It allows to detect the correct end of a table cell data during parsing.
70       * Expected schema of a table cell data is
71       * <pre>x04[table cell]x05[number of occurrences]x04
72       */
73      public static final String ENCLOSE_VALUE_RGX = "\\x04";
74      
75      public static final String LEAD = "SqLi";
76      public static final String SHELL_LEAD = "${shell.lead}";
77      public static final String TRAIL = "iLQS";
78      public static final String SHELL_TRAIL = "${shell.trail}";
79      
80      /**
81       * Regex keywords corresponding to multiline and case-insensitive match.
82       */
83      public static final String MODE = "(?si)";
84      
85      /**
86       * Regex schema describing a table cell with firstly the cell content and secondly the number of occurrences
87       * of the cell text, separated by the reserved character x05 in hexadecimal.
88       * The range of characters from x01 to x1F are not printable ASCII characters used to parse the data and exclude
89       * printable characters during parsing.
90       * Expected schema of a table cell data is
91       * <pre>x04[table cell]x05[number of occurrences]x04
92       */
93      public static final String CELL_TABLE = "([^\\x01-\\x09\\x0B-\\x0C\\x0E-\\x1F]*)"+ SEPARATOR_QTE_RGX +"([^\\x01-\\x09\\x0B-\\x0C\\x0E-\\x1F]*)(\\x08)?";

101     /**
102      * Get general database information.<br>
103      * => version{%}database{%}user{%}CURRENT_USER
104      */
105     public void getDatabaseInfos() {
106         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, () -> I18nUtil.valueByKey("LOG_FETCHING_INFORMATIONS"));
107         
108         var sourcePage = new String[]{ StringUtils.EMPTY };
109 
110         var resultToParse = "";
111         try {
112             resultToParse = new SuspendableGetRows(this.injectionModel).run(
113                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlInfos(),
114                 sourcePage,
115                 false,
116                 0,
117                 AbstractElementDatabase.MOCK,
118                 "metadata"
119             );
120         } catch (AbstractSlidingException e) {
121             resultToParse = getPartialResultAndLog(e, resultToParse);
122         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
123             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
124         }
125 
126         if (StringUtils.isEmpty(resultToParse)) {
127             this.injectionModel.sendResponseFromSite("Incorrect metadata", sourcePage[0].trim());
128         }
129         
130         try {
131             String versionDatabase = resultToParse.split(ENCLOSE_VALUE_RGX)[0].replaceAll("\\s+", StringUtils.SPACE);
132             String nameDatabase = resultToParse.split(ENCLOSE_VALUE_RGX)[1];
133             String username = resultToParse.split(ENCLOSE_VALUE_RGX)[2];
134             
135             var infos = String.format(
136                 "Database [%s] on %s [%s] for user [%s]",
137                 nameDatabase,
138                 this.injectionModel.getMediatorVendor().getVendor(),
139                 versionDatabase,
140                 username
141             );
142             LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, infos);
143         } catch (ArrayIndexOutOfBoundsException e) {
144             LOGGER.log(
145                 LogLevelUtil.CONSOLE_ERROR,
146                 String.format("%s: %s", I18nUtil.valueByKey("LOG_DB_METADATA_INCORRECT"), resultToParse)
147             );
148             LOGGER.log(LogLevelUtil.CONSOLE_INFORM, I18nUtil.valueByKey("LOG_DB_METADATA_WARN"));
149         }
150     }

152     /**
153      * Get database names and table counts and send them to the view.<br>
154      * Use readable text (not hexa) and parse this pattern:<br>
155      * => hh[database name 1]jj[table count]hhgghh[database name 2]jj[table count]hhggh...hi<br>
156      * Data window can be cut before the end of the request but the process helps to obtain
157      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
158      * @return list of databases found
159      * @throws JSqlException when injection failure or stopped by user
160      */
161     public List<Database> listDatabases() throws JSqlException {
162         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, () -> I18nUtil.valueByKey("LOG_FETCHING_DATABASES"));
163         List<Database> databases = new ArrayList<>();
164         String resultToParse = StringUtils.EMPTY;
165         
166         try {
167             var sourcePage = new String[]{ StringUtils.EMPTY };
168             resultToParse = new SuspendableGetRows(this.injectionModel).run(
169                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlDatabases(),
170                 sourcePage,
171                 true,
172                 0,
173                 AbstractElementDatabase.MOCK,
174                 "databases"
175             );
176         } catch (AbstractSlidingException e) {
177             resultToParse = getPartialResultAndLog(e, resultToParse);
178         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
179             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
180         }
181 
182         // Parse all data we have retrieved
183         var regexSearch = Pattern.compile(
184                 MODE
185                 + ENCLOSE_VALUE_RGX
186                 + CELL_TABLE
187                 + ENCLOSE_VALUE_RGX
188             )
189             .matcher(resultToParse);
190         if (!regexSearch.find()) {
191             throw new InjectionFailureException("No match while injecting databases");
192         }
193         
194         regexSearch.reset();
195         // Build an array of Database objects from the data we have parsed
196         while (regexSearch.find()) {
197             String databaseName = regexSearch.group(1);
198             String tableCount = regexSearch.group(2);
199 
200             var newDatabase = new Database(databaseName, tableCount);
201             databases.add(newDatabase);
202         }
203 
204         var request = new Request();
205         request.setMessage(Interaction.ADD_DATABASES);
206         request.setParameters(databases);
207         this.injectionModel.sendToViews(request);
208         return databases;
209     }

211     /**
212      * Get tables name and row count and send them to the view.<br>
213      * Use readable text (not hexa) and parse this pattern:<br>
214      * => hh[table name 1]jj[rows count]hhgghh[table name 2]jj[rows count]hhggh...hi<br>
215      * Data window can be cut before the end of the request but the process helps to obtain
216      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
217      * @param database which contains tables to find
218      * @return list of tables found
219      * @throws JSqlException when injection failure or stopped by user
220      */
221     public List<Table> listTables(Database database) throws JSqlException {
222         // Reset stoppedByUser if list of Databases is partial
223         // and some Tables are still reachable
224         this.injectionModel.setIsStoppedByUser(false);
225         
226         // Inform the view that database has just been used
227         var requestStartProgress = new Request();
228         requestStartProgress.setMessage(Interaction.START_PROGRESS);
229         requestStartProgress.setParameters(database);
230         this.injectionModel.sendToViews(requestStartProgress);
231 
232         var tableCount = Integer.toString(database.getChildCount());
233         
234         String resultToParse = StringUtils.EMPTY;
235         try {
236             var pageSource = new String[]{ StringUtils.EMPTY };
237             resultToParse = new SuspendableGetRows(this.injectionModel).run(
238                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlTables(database),
239                 pageSource,
240                 true,
241                 Integer.parseInt(tableCount),
242                 database,
243                 "tables"
244             );
245         } catch (AbstractSlidingException e) {
246             resultToParse = getPartialResultAndLog(e, resultToParse);
247         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
248             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
249         }
250 
251         // Parse all the data we have retrieved
252         var regexSearch = Pattern.compile(
253                 MODE
254                 + ENCLOSE_VALUE_RGX
255                 + CELL_TABLE
256                 + ENCLOSE_VALUE_RGX
257             )
258             .matcher(resultToParse);
259         
260         var requestEndProgress = new Request();
261         requestEndProgress.setMessage(Interaction.END_PROGRESS);
262         requestEndProgress.setParameters(database);
263         this.injectionModel.sendToViews(requestEndProgress);
264         
265         if (!regexSearch.find()) {
266             throw new InjectionFailureException("No match while injecting tables");
267         }
268         regexSearch.reset();
269         
270         List<Table> tables = new ArrayList<>();
271         
272         // Build an array of Table objects from the data we have parsed
273         while (regexSearch.find()) {
274             String tableName = regexSearch.group(1);
275             String rowCount = regexSearch.group(2);
276             
277             var newTable = new Table(tableName, rowCount, database);
278             tables.add(newTable);
279         }
280         
281         var requestAddTables = new Request();
282         requestAddTables.setMessage(Interaction.ADD_TABLES);
283         requestAddTables.setParameters(tables);
284         this.injectionModel.sendToViews(requestAddTables);
285         return tables;
286     }

288     /**
289      * Get column names and send them to the view.<br>
290      * Use readable text (not hexa) and parse this pattern with 2nd member forced to 31 (1 in ascii):<br>
291      * => hh[column name 1]jj[31]hhgghh[column name 2]jj[31]hhggh...hi<br>
292      * Data window can be cut before the end of the request but the process helps to obtain
293      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
294      * @param table which contains columns to find
295      * @return list of columns found
296      * @throws JSqlException when injection failure or stopped by user
297      */
298     public List<Column> listColumns(Table table) throws JSqlException {
299         List<Column> columns = new ArrayList<>();
300         
301         // Inform the view that table has just been used
302         var requestStartProgress = new Request();
303         requestStartProgress.setMessage(Interaction.START_INDETERMINATE_PROGRESS);
304         requestStartProgress.setParameters(table);
305         this.injectionModel.sendToViews(requestStartProgress);
306 
307         String resultToParse = StringUtils.EMPTY;
308         
309         try {
310             var pageSource = new String[]{ StringUtils.EMPTY };
311             resultToParse = new SuspendableGetRows(this.injectionModel).run(
312                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlColumns(table),
313                 pageSource,
314                 true,
315                 0,
316                 table,
317                 "columns"
318             );
319         } catch (AbstractSlidingException e) {
320             resultToParse = getPartialResultAndLog(e, resultToParse);
321         } catch (Exception e) {  // Catch all exceptions but prevent detecting bug
322             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
323         }
324 
325         // Build SQLite columns
326         if (this.injectionModel.getMediatorVendor().isSqlite()) {
327             resultToParse = this.injectionModel.getMediatorVendor().getSqlite().transformSqlite(resultToParse);
328         }
329         
330         // Parse all the data we have retrieved
331         var regexSearch = Pattern.compile(
332                 MODE
333                 + ENCLOSE_VALUE_RGX
334                 + CELL_TABLE
335                 + ENCLOSE_VALUE_RGX
336             )
337             .matcher(resultToParse);
338 
339         var requestEndProgress = new Request();
340         requestEndProgress.setMessage(Interaction.END_INDETERMINATE_PROGRESS);
341         requestEndProgress.setParameters(table);
342         this.injectionModel.sendToViews(requestEndProgress);
343 
344         if (!regexSearch.find()) {
345             throw new InjectionFailureException("No match while injecting columns");
346         }
347 
348         regexSearch.reset();
349 
350         // Build an array of Column objects from the data we have parsed
351         while (regexSearch.find()) {
352             String nameColumn = regexSearch.group(1);
353             var column = new Column(nameColumn, table);
354             columns.add(column);
355         }
356 
357         var requestAddColumns = new Request();
358         requestAddColumns.setMessage(Interaction.ADD_COLUMNS);
359         requestAddColumns.setParameters(columns);
360         this.injectionModel.sendToViews(requestAddColumns);
361         return columns;
362     }

364     /**
365      * Get table values and count each occurrence and send it to the view.<br>
366      * Values are on clear text (not hexa) and follows this window pattern<br>
367      * => hh[value 1]jj[count]hhgghh[value 2]jj[count]hhggh...hi<br>
368      * Data window can be cut before the end of the request but the process helps to obtain
369      * the rest of the unreachable data. The process can be interrupted by the user (stop/pause).
370      * @param columnsBean choice by the user
371      * @return a 2x2 table containing values by columns
372      * @throws JSqlException when injection failure or stopped by user
373      */
374     public String[][] listValues(List<Column> columnsBean) throws JSqlException {
375         var databaseBean = (Database) columnsBean.get(0).getParent().getParent();
376         var tableBean = (Table) columnsBean.get(0).getParent();
377         int rowCount = columnsBean.get(0).getParent().getChildCount();
378 
379         // Inform the view that table has just been used
380         var request = new Request();
381         request.setMessage(Interaction.START_PROGRESS);
382         request.setParameters(tableBean);
383         this.injectionModel.sendToViews(request);
384 
385         // Build an array of column names
386         List<String> columnsName = new ArrayList<>();
387         for (AbstractElementDatabase columnBean: columnsBean) {
388             columnsName.add(columnBean.toString());
389         }
390 
391         // From that array, build the SQL fields nicely
392         // => col1{%}col2...
393         // ==> trim(ifnull(`col1`,0x00)),0x7f,trim(ifnull(`Col2`,0x00))...
394         String[] columns = columnsName.toArray(new String[0]);
395 
396         List<List<String>> listValues = this.getRows(databaseBean, tableBean, rowCount, columns);
397 
398         // Add the default title to the columns: row number, occurrence
399         columnsName.add(0, StringUtils.EMPTY);
400         columnsName.add(0, StringUtils.EMPTY);
401 
402         String[][] table = this.getTable(columnsName, listValues);
403 
404         columns = columnsName.toArray(new String[0]);
405         
406         // Group the columns names, values and Table object in one array
407         var objectData = new Object[]{ columns, table, tableBean };
408 
409         var requestCreateValuesTab = new Request();
410         requestCreateValuesTab.setMessage(Interaction.CREATE_VALUES_TAB);
411         requestCreateValuesTab.setParameters(objectData);
412         this.injectionModel.sendToViews(requestCreateValuesTab);
413 
414         var requestEndProgress = new Request();
415         requestEndProgress.setMessage(Interaction.END_PROGRESS);
416         requestEndProgress.setParameters(tableBean);
417         this.injectionModel.sendToViews(requestEndProgress);
418         return table;
419     }

421     private List<List<String>> getRows(Database database, Table table, int rowCount, String[] columns) throws InjectionFailureException {
422         String resultToParse = StringUtils.EMPTY;
423         
424         try {
425             var pageSource = new String[]{ StringUtils.EMPTY };
426             resultToParse = new SuspendableGetRows(this.injectionModel).run(
427                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlRows(columns, database, table),
428                 pageSource,
429                 true,
430                 rowCount,
431                 table,
432                 "rows"
433             );
434         } catch (AbstractSlidingException e) {  // Catch all exceptions but prevent detecting bug
435             resultToParse = getPartialResultAndLog(e, resultToParse);
436         } catch (Exception e) {
437             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage(), e);
438         }
439         return SuspendableGetRows.parse(resultToParse);
440     }

442     private static String getPartialResultAndLog(AbstractSlidingException e, String resultToParse) {
443         LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage());
444 
445         // Get pieces of data already retrieved instead of losing them
446         if (StringUtils.isNotEmpty(e.getSlidingWindowAllRows())) {
447             resultToParse = e.getSlidingWindowAllRows();
448         } else if (StringUtils.isNotEmpty(e.getSlidingWindowCurrentRows())) {
449             resultToParse = e.getSlidingWindowCurrentRows();
450         }
451         return resultToParse;
452     }

454     private String[][] getTable(List<String> columnsName, List<List<String>> values) {
455         // Build a proper 2D array from the data
456         var table = new String[values.size()][columnsName.size()];
457         
458         for (var indexRow = 0 ; indexRow < values.size() ; indexRow++) {
459             var isIncomplete = false;
460             
461             for (var indexColumn = 0 ; indexColumn < columnsName.size() ; indexColumn++) {
462                 try {
463                     table[indexRow][indexColumn] = values.get(indexRow).get(indexColumn);
464                 } catch (IndexOutOfBoundsException e) {
465                     isIncomplete = true;
466                     LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, I18nUtil.valueByKey("LOG_LIST_VALUES_INCOMPLETE"));
467                     LOGGER.log(LogLevelUtil.IGNORE, e);
468                 }
469             }
470             
471             if (isIncomplete) {
472                 int logIndexRow = indexRow;
473                 LOGGER.log(
474                     LogLevelUtil.CONSOLE_ERROR,
475                     "{}{}: ",
476                     () -> I18nUtil.valueByKey("LOG_LIST_VALUES_TOO_LONG"),
477                     () -> logIndexRow + 1
478                 );
479 
480                 LOGGER.log(
481                     LogLevelUtil.CONSOLE_ERROR,
482                     () -> String.join(
483                         ", ",
484                         values.get(logIndexRow).toArray(new String[0])
485                     )
486                 );
487             }
488         }
489         return table;
490     }
