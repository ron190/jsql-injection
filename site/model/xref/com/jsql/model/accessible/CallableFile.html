13  import java.util.concurrent.Callable;
14  
15  /**
16   * Thread unit to read source of a file by SQL injection.
17   * User can interrupt the process and get a partial result of the file content.
18   */
19  public class CallableFile implements Callable<CallableFile> {
20      
21      /**
22       * Log4j logger sent to view.
23       */
24      private static final Logger LOGGER = LogManager.getRootLogger();
25      
26      /**
27       * Path to the file to read.
28       */
29      private final String pathFile;
30  
31      /**
32       * Source of file.
33       */
34      private String sourceFile = StringUtils.EMPTY;
35      
36      /**
37       * Suspendable task that reads lines of the file by injection.
38       */
39      private final SuspendableGetRows suspendableReadFile;
40  
41      private final InjectionModel injectionModel;
42      
43      /**
44       * Create Callable to read a file.
45       */
46      public CallableFile(String pathFile, InjectionModel injectionModel) {
47          this.pathFile = pathFile;
48          this.injectionModel= injectionModel;
49          this.suspendableReadFile = new SuspendableGetRows(injectionModel);
50      }
51      
52      /**
53       * Read a file on the server using SQL injection.
54       * Get partial result if user interrupts the process.
55       */
56      @Override
57      public CallableFile call() throws Exception {
58          var sourcePage = new String[]{ StringUtils.EMPTY };
59  
60          String resultToParse = StringUtils.EMPTY;
61          try {
62              resultToParse = this.suspendableReadFile.run(
63                  this.injectionModel.getMediatorVendor().getVendor().instance().sqlFileRead(this.pathFile),
64                  sourcePage,
65                  false,
66                  1,
67                  MockElement.MOCK,
68                  "file"
69              );
70          } catch (InjectionFailureException e) {
71              // Usually thrown if File does not exist
72              LOGGER.log(LogLevelUtil.IGNORE, e);
73          } catch (StoppedByUserSlidingException e) {
74              // Get partial source
75              if (StringUtils.isNotEmpty(e.getSlidingWindowAllRows())) {
76                  resultToParse = e.getSlidingWindowAllRows();
77              } else if (StringUtils.isNotEmpty(e.getSlidingWindowCurrentRows())) {
78                  resultToParse = e.getSlidingWindowCurrentRows();
79              }
80              LOGGER.log(LogLevelUtil.IGNORE, e);
81          }
82          
83          this.sourceFile = resultToParse;
84          return this;
85      }
86      
87      
88      // Getters
89      
90      public String getPathFile() {
91          return this.pathFile;
92      }
93  
94      public String getSourceFile() {
95          return this.sourceFile;
96      }
97  
98      public SuspendableGetRows getSuspendableReadFile() {
99          return this.suspendableReadFile;
100     }
101 }
