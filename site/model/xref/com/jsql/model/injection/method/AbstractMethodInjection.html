18  public abstract class AbstractMethodInjection implements Serializable {
19      
20      /**
21       * Log4j logger sent to view.
22       */
23      private static final Logger LOGGER = LogManager.getRootLogger();
24      
25      protected final InjectionModel injectionModel;
26      
27      protected AbstractMethodInjection(InjectionModel injectionModel) {
28          this.injectionModel = injectionModel;
29      }
30      
31      public abstract boolean isCheckingAllParam();
32      public abstract String getParamsAsString();
33      public abstract List<SimpleEntry<String, String>> getParams();
34      public abstract String name();
35      
36      public boolean testParameters(boolean hasFoundInjection) throws JSqlException {
37          if (!hasFoundInjection) {
38              LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking {} params...", () -> name().toLowerCase());
39              return this.testParameters();
40          }
41          return true;
42      }
43  
44      /**
45       * Verify if injection works for specific Method using 3 modes: standard (last param), injection point
46       * and full params injection. Special injections like JSON and SOAP are checked.
47       * @return true if injection didn't fail
48       * @throws JSqlException when no params' integrity, process stopped by user, or injection failure
49       */
50      public boolean testParameters() throws JSqlException {
51          var hasFoundInjection = false;
52          
53          // Injects URL, Request or Header params only if user tests every params
54          // or method is selected by user.
55          if (
56              !this.injectionModel.getMediatorUtils().getPreferencesUtil().isCheckingAllParam()
57              && this.injectionModel.getMediatorUtils().getConnectionUtil().getMethodInjection() != this
58          ) {
59              return false;
60          }
61          
62          // Force injection method of model to current running method
63          this.injectionModel.getMediatorUtils().getConnectionUtil().setMethodInjection(this);
64          
65          // Injection by injection point in params or in path
66          if (
67              this.getParamsAsString().contains(InjectionModel.STAR)
68              || this.injectionModel.getMediatorUtils().getConnectionUtil().getUrlBase().contains(InjectionModel.STAR)
69          ) {
70              hasFoundInjection = this.checkParamWithStar();
71          } else if (!this.isCheckingAllParam()) {
72              hasFoundInjection = this.checkLastParam();
73          } else {
74              hasFoundInjection = this.checkAllParams();
75          }
76          return hasFoundInjection;
77      }
78  
79      private boolean checkParamWithStar() throws JSqlException {
80          SimpleEntry<String, String> parameterToInject = this.getParams().stream()
81              .filter(entry -> entry.getValue().contains("*"))
82              .findFirst()
83              .orElse(null);
84          return this.injectionModel.getMediatorStrategy().testStrategies(parameterToInject);
85      }
86  
87      /**
88       *  Default injection: last param tested only
89       */
90      private boolean checkLastParam() throws JSqlException {
91          // Will check param value by user.
92          // Notice options 'Inject each URL params' and 'inject JSON' must be checked both
93          // for JSON injection of last param
94          SimpleEntry<String, String> parameterToInject = this.getParams().stream()
95              .reduce((a, b) -> b)
96              .orElseThrow(NullPointerException::new);
97          return this.injectionModel.getMediatorStrategy().testStrategies(parameterToInject);
98      }
99  
100     /**
101      * Injection of every params: isCheckingAllParam() == true.
102      * Params are tested one by one in two loops:
103      * - inner loop erases * from previous param
104      * - outer loop adds * to current param
105      */
106     private boolean checkAllParams() throws StoppedByUserSlidingException {
107         // This param will be marked by * if injection is found,
108         // inner loop will erase mark * otherwise
109         for (SimpleEntry<String, String> paramBase: this.getParams()) {
110             // This param is the current tested one.
111             // For JSON value attributes are traversed one by one to test every values.
112             // For standard value mark * is simply added to the end of its value.
113             for (SimpleEntry<String, String> paramStar: this.getParams()) {
114                 if (paramStar == paramBase) {
115                     try {
116                         if (this.isParamInjectable(paramStar)) {
117                             return true;
118                         }
119                     } catch (JSONException e) {
120                         LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
121                     }
122                 }
123             }
124         }
125         return false;
126     }
127 
128     private boolean isParamInjectable(SimpleEntry<String, String> paramStar) throws StoppedByUserSlidingException {
129         boolean hasFoundInjection;
130         
131         // Will test if current value is a JSON entity
132         Object jsonEntity = JsonUtil.getJson(paramStar.getValue());
133         
134         // Define a tree of JSON attributes with path as the key: root.a => value of a
135         List<SimpleEntry<String, String>> attributesJson = JsonUtil.createEntries(jsonEntity, "root", null);
136         
137         // When option 'Inject JSON' is selected and there's a JSON entity to inject
138         // then loop through each paths to add * at the end of value and test each strategies.
139         // Marks * are erased between each tests.
140         if (!attributesJson.isEmpty() && this.injectionModel.getMediatorUtils().getPreferencesUtil().isCheckingAllJsonParam()) {
141             hasFoundInjection = this.injectionModel.getMediatorUtils().getJsonUtil().testJsonParam(this, paramStar);
142         } else {
143             hasFoundInjection = this.testJsonlessParam(paramStar);  // Standard non JSON injection
144         }
145         return hasFoundInjection;
146     }
147     
148     public boolean testJsonlessParam(SimpleEntry<String, String> paramStar) throws StoppedByUserSlidingException {
149         var hasFoundInjection = false;
150 
151         paramStar.setValue(paramStar.getValue() + InjectionModel.STAR);
152         
153         try {
154             LOGGER.log(
155                 LogLevelUtil.CONSOLE_INFORM,
156                 "Checking {} parameter {}={}",
157                 this::name,
158                 paramStar::getKey,
159                 () -> paramStar.getValue().replace(InjectionModel.STAR, StringUtils.EMPTY)
160             );
161             
162             // Test current standard value marked with * for injection
163             // Keep original param
164             hasFoundInjection = this.injectionModel.getMediatorStrategy().testStrategies(paramStar);
165             
166         } catch (StoppedByUserSlidingException e) { // Break all params processing in upper methods
167             throw e;
168         } catch (JSqlException e) {  // Injection failure
169             LOGGER.log(
170                 LogLevelUtil.CONSOLE_ERROR,
171                 "No {} injection found for parameter {}={} ({})",
172                 this.name(),
173                 paramStar.getKey(),
174                 paramStar.getValue().replaceAll("\\+.?$|\\" + InjectionModel.STAR, StringUtils.EMPTY),
175                 e.getMessage()
176             );
177         } finally {
178             if (!hasFoundInjection) {  // Erase * from JSON if failure
179                 
180                 // Erase * at the end of each params
181                 this.getParams().forEach(e ->
182                     e.setValue(
183                         e.getValue().replaceAll(Pattern.quote(InjectionModel.STAR) +"$", StringUtils.EMPTY)
184                     )
185                 );
186                 
187                 // TODO It erases STAR from value => * can't be used in parameter
188                 paramStar.setValue(paramStar.getValue().replace(InjectionModel.STAR, StringUtils.EMPTY));
189             }
190         }
191         return hasFoundInjection;
192     }
