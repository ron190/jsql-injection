19  public class StrategyInjectionError extends AbstractStrategy {
20      
21      /**
22       * Log4j logger sent to view.
23       */
24      private static final Logger LOGGER = LogManager.getRootLogger();
25      
26      private String[] tabCapacityMethod;
27      
28      private int indexErrorStrategy = 0;
29  
30      public StrategyInjectionError(InjectionModel injectionModel) {
31          super(injectionModel);
32      }
33  
34      @Override
35      public void checkApplicability() {
36          // Reset applicability of new Vendor
37          this.isApplicable = false;
38          
39          var strategyYaml = this.injectionModel.getMediatorVendor().getVendor().instance().getModelYaml().getStrategy();
40  
41          if (this.injectionModel.getMediatorUtils().getPreferencesUtil().isStrategyErrorDisabled()) {
42              LOGGER.log(LogLevelUtil.CONSOLE_INFORM, AbstractStrategy.FORMAT_SKIP_STRATEGY_DISABLED, getName());
43              return;
44          } else if (strategyYaml.getError().getMethod().isEmpty()) {
45              LOGGER.log(
46                  LogLevelUtil.CONSOLE_ERROR,
47                  AbstractStrategy.FORMAT_STRATEGY_NOT_IMPLEMENTED,
48                  getName(),
49                  this.injectionModel.getMediatorVendor().getVendor()
50              );
51              return;
52          }
53  
54          LOGGER.log(
55              LogLevelUtil.CONSOLE_DEFAULT,
56              AbstractStrategy.FORMAT_CHECKING_STRATEGY,
57              () -> I18nUtil.valueByKey("LOG_CHECKING_STRATEGY"),
58              this::getName
59          );
60          
61          this.tabCapacityMethod = new String[strategyYaml.getError().getMethod().size()];
62          var indexErrorMethod = 0;
63          var errorCapacity = 0;
64          
65          for (Method errorMethod: strategyYaml.getError().getMethod()) {
66              boolean methodIsApplicable = this.isApplicable(errorMethod);
67              if (methodIsApplicable) {
68                  Matcher regexSearch = this.getPerformance(errorMethod);
69                  if (regexSearch.find()) {
70                      errorCapacity = this.getCapacity(indexErrorMethod, errorCapacity, errorMethod, regexSearch);
71                  } else {
72                      LOGGER.log(
73                          LogLevelUtil.CONSOLE_ERROR,
74                          "{} {} but injectable size is incorrect",
75                          () -> I18nUtil.valueByKey("LOG_VULNERABLE"),
76                          errorMethod::getName
77                      );
78                      methodIsApplicable = false;
79                  }
80              }
81              
82              if (methodIsApplicable) {
83                  this.allow(indexErrorMethod);
84              } else {
85                  this.unallow(indexErrorMethod);
86              }
87              indexErrorMethod++;
88          }
89      }
90  
91      private boolean isApplicable(Method errorMethod) {
92          var methodIsApplicable = false;
93  
94          String performanceSourcePage = this.injectionModel.injectWithoutIndex(
95              this.injectionModel.getMediatorVendor().getVendor().instance().sqlErrorIndice(errorMethod),
96              "error#confirm"
97          );
98  
99          var indexZeroToFind = "0";
100         String regexIndexZero = String.format(VendorYaml.FORMAT_INDEX, indexZeroToFind);
101         if (performanceSourcePage.matches("(?s).*"+ regexIndexZero +".*")) {
102             methodIsApplicable = true;
103             this.isApplicable = true;
104         }
105         return methodIsApplicable;
106     }
107 
108     private Matcher getPerformance(Method errorMethod) {
109         String performanceErrorSourcePage = this.injectionModel.injectWithoutIndex(
110             this.injectionModel.getMediatorVendor().getVendor().instance().sqlErrorCalibrator(errorMethod),
111             "error#size"
112         );
113         return Pattern.compile("(?s)"+ DataAccess.LEAD +"(#+)").matcher(performanceErrorSourcePage);
114     }
115 
116     private int getCapacity(int indexErrorMethod, int errorCapacityDefault, Method errorMethod, Matcher regexSearch) {
117         int errorCapacityImproved = errorCapacityDefault;
118         
119         regexSearch.reset();
120         while (regexSearch.find()) {
121             if (errorCapacityImproved < regexSearch.group(1).length()) {
122                 this.indexErrorStrategy = indexErrorMethod;
123             }
124             errorCapacityImproved = regexSearch.group(1).length();
125             this.tabCapacityMethod[indexErrorMethod] = Integer.toString(errorCapacityImproved);
126         }
127         
128         int logErrorCapacityImproved = errorCapacityImproved;
129         LOGGER.log(
130             LogLevelUtil.CONSOLE_SUCCESS,
131             "{} [Error {}] showing [{}] characters",
132             () -> I18nUtil.valueByKey("LOG_VULNERABLE"),
133             errorMethod::getName,
134             () -> Integer.toString(logErrorCapacityImproved)
135         );
136         
137         return errorCapacityImproved;
138     }
139 
140     @Override
141     public void allow(int... indexError) {
142         this.injectionModel.appendAnalysisReport(
143             StringUtil.formatReport(
144                 LogLevelUtil.COLOR_BLU,
145                 "### Strategy: " + getName() + ":" + this.injectionModel.getMediatorVendor().getVendor().instance()
146                 .getModelYaml()
147                 .getStrategy()
148                 .getError()
149                 .getMethod()
150                 .get(indexError[0])
151                 .getName()
152             )
153             + this.injectionModel.getReportWithoutIndex(
154                 this.injectionModel.getMediatorVendor().getVendor().instance().sqlError(
155                     StringUtil.formatReport(LogLevelUtil.COLOR_GREEN, "&lt;query&gt;"),
156                     "0",
157                     indexError[0],
158                     true
159                 ),
160                 "metadataInjectionProcess"
161             )
162         );
163         this.markVulnerability(Interaction.MARK_ERROR_VULNERABLE, indexError[0]);
164     }
165 
166     @Override
167     public void unallow(int... indexError) {
168         this.markVulnerability(Interaction.MARK_ERROR_INVULNERABLE, indexError[0]);
169     }
170 
171     @Override
172     public String inject(String sqlQuery, String startPosition, AbstractSuspendable stoppable, String metadataInjectionProcess) {
173         return this.injectionModel.injectWithoutIndex(
174             this.injectionModel.getMediatorVendor().getVendor().instance().sqlError(sqlQuery, startPosition, this.indexErrorStrategy, false),
175             metadataInjectionProcess
176         );
177     }
178 
179     @Override
180     public void activateWhenApplicable() {
181         if (this.injectionModel.getMediatorStrategy().getStrategy() == null && this.isApplicable()) {
182             LOGGER.log(
183                 LogLevelUtil.CONSOLE_INFORM,
184                 "{} [{} {}]",
185                 () -> I18nUtil.valueByKey("LOG_USING_STRATEGY"),
186                 this::getName,
187                 () -> this.injectionModel.getMediatorVendor().getVendor().instance().getModelYaml().getStrategy()
188                 .getError().getMethod().get(this.indexErrorStrategy).getName()
189             );
190             this.injectionModel.getMediatorStrategy().setStrategy(this.injectionModel.getMediatorStrategy().getError());
191 
192             var request = new Request();
193             request.setMessage(Interaction.MARK_ERROR_STRATEGY);
194             this.injectionModel.sendToViews(request);
195         }
196     }
197     
198     @Override
199     public String getPerformanceLength() {
200         return this.tabCapacityMethod[this.indexErrorStrategy];
201     }
202     
203     @Override
204     public String getName() {
205         return "Error";
206     }
207     
208     public Integer getIndexErrorStrategy() {
209         return this.indexErrorStrategy;
210     }
211     
212     public void setIndexErrorStrategy(int indexErrorStrategy) {
213         this.indexErrorStrategy = indexErrorStrategy;
214     }
