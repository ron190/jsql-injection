79      /**
80       * Process the whole boolean injection, character by character, bit by bit.
81       * @param sqlQuery SQL query
82       * @param suspendable Action a user can stop
83       * @return Final string: SQLiABCDEF...
84       */
85      public String inject(String sqlQuery, AbstractSuspendable suspendable) throws StoppedByUserSlidingException {
86          // List of the characters, each one represented by an array of 8 bits
87          // e.g. SQLi: bytes[0] => 01010011:S, bytes[1] => 01010001:Q ...
88          List<char[]> bytes = new ArrayList<>();
89          var indexChar = new AtomicInteger(0);  // current char position
90  
91          // Concurrent URL requests
92          ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableAbstractBoolean");
93          CompletionService<T> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
94  
95          var countTasksSubmitted = new AtomicInteger(0);
96          var countBadAsciiCode = new AtomicInteger(0);
97  
98          this.initNextChar(sqlQuery, bytes, indexChar, taskCompletionService, countTasksSubmitted, countBadAsciiCode, null);
99  
100         // Process the job until there is no more active task,
101         // in other word until all HTTP requests are done
102         while (countTasksSubmitted.get() > 0) {
103             if (suspendable.isSuspended()) {
104                 String result = this.stop(bytes, taskExecutor);
105                 throw new StoppedByUserSlidingException(result);
106             }
107             
108             try {
109                 var currentCallable = taskCompletionService.take().get();  // URL call done
110                 countTasksSubmitted.decrementAndGet();  // one task just ended
111                 
112                 // If SQL result is not empty, then add a new unknown character and define a new array of 7 undefined bit.
113                 // Then add 7 bits requests for that new character.
114                 var isComplete = this.injectCharacter(bytes, countBadAsciiCode, currentCallable);
115                 if (isComplete || currentCallable.isBinary()) {  // prevents bitwise overload new char init on each bit
116                     this.initNextChar(sqlQuery, bytes, indexChar, taskCompletionService, countTasksSubmitted, countBadAsciiCode, currentCallable);
117                 }
118 
119                 String result = AbstractInjectionBit.convert(bytes);
120                 if (result.matches("(?s).*"+ DataAccess.TRAIL_RGX +".*")) {
121                     countTasksSubmitted.set(0);
122                     break;
123                 }
124             } catch (InterruptedException e) {
125                 LOGGER.log(LogLevelUtil.IGNORE, e, e);
126                 Thread.currentThread().interrupt();
127             } catch (ExecutionException e) {
128                 LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
129             } catch (InjectionFailureException e) {
130                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e.getMessage());
131                 break;
132             }
133         }
134         return this.stop(bytes, taskExecutor);
135     }

137     private static String convert(List<char[]> bytes) {
138         var result = new StringBuilder();
139         for (char[] c: bytes) {
140             try {
141                 var charCode = Integer.parseInt(new String(c), 2);
142                 var str = Character.toString((char) charCode);
143                 result.append(str);
144             } catch (NumberFormatException err) {
145                 // Ignore, byte string not fully constructed (0x1x010x)
146             }
147         }
148         return result.toString();
149     }

151     protected boolean injectCharacter(
152         List<char[]> bytes,
153         AtomicInteger countBadAsciiCode,
154         T currentCallable
155     ) throws InjectionFailureException {
156         // Process url that has just checked one bit, convert bits to chars,
157         // and change current bit from undefined to 0 or 1
158         char[] maskAsciiChar = this.initMaskAsciiChar(bytes, currentCallable);
159         var asciiCodeBit = new String(maskAsciiChar);
160         var isComplete = false;
161 
162         // Inform the View if bits array is complete, else nothing #Need fix
163         if (asciiCodeBit.matches("^[01]{8}$")) {
164             var asciiCode = Integer.parseInt(asciiCodeBit, 2);
165             if (asciiCode == 127 || asciiCode == 0) {  // Stop if many 11111111, 01111111 or 00000000
166                 if (countBadAsciiCode.get() > 15) {
167                     throw new InjectionFailureException("Boolean false positive, stopping...");
168                 }
169                 countBadAsciiCode.incrementAndGet();
170             }
171 
172             currentCallable.setCharText(Character.toString((char) asciiCode));
173             
174             var interaction = new Request();
175             interaction.setMessage(Interaction.MESSAGE_BINARY);
176             interaction.setParameters(
177                 asciiCodeBit
178                 + "="
179                 + currentCallable.getCharText()
180                 .replace("\\n", "\\\\\\n")
181                 .replace("\\r", "\\\\\\r")
182                 .replace("\\t", "\\\\\\t")
183             );
184             this.injectionModel.sendToViews(interaction);
185             isComplete = true;
186         }
187         return isComplete;
188     }

190     private String stop(List<char[]> bytes, ExecutorService taskExecutor) {
191         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
192 
193         // Get current progress and display
194         var result = new StringBuilder();
195         
196         for (char[] c: bytes) {
197             try {
198                 var charCode = Integer.parseInt(new String(c), 2);
199                 var str = Character.toString((char) charCode);
200                 result.append(str);
201             } catch (NumberFormatException err) {
202                 // Byte string not fully constructed : 0x1x010x
203                 // Ignore
204             }
205         }
206         return result.toString();
207     }
