15  public class MasterService {
16  
17      private final HashMap<String, ConnectionProvider> hashMap = new HashMap<>();
18  
19      public MasterService() {
20          DriverManager.setLogWriter(null);  // remove annoying logs from jdbc driver
21  
22          SpringApp.getPropertiesFilterByProfile().map(AbstractMap.SimpleEntry::getKey).forEach(props -> {
23              DatasourceConnectionProviderImpl connectionProvider = new DatasourceConnectionProviderImpl();
24  
25              DriverManagerDataSource dataSource = new DriverManagerDataSource();
26              dataSource.setUrl(props.getProperty(JdbcSettings.JAKARTA_JDBC_URL));
27              dataSource.setUsername(props.getProperty(JdbcSettings.JAKARTA_JDBC_USER));
28              dataSource.setPassword(props.getProperty(JdbcSettings.JAKARTA_JDBC_PASSWORD));
29  
30              connectionProvider.configure(Map.of(
31                  Environment.DATASOURCE, dataSource
32              ));
33  
34              this.hashMap.put(props.getProperty("jsql.tenant"), connectionProvider);
35          });
36      }
37  
38      public HashMap<String, ConnectionProvider> getDataSourceHashMap() {
39          return this.hashMap;
40      }
