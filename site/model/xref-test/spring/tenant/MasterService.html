1   package spring.tenant;

3   import org.hibernate.cfg.Environment;
4   import org.hibernate.cfg.JdbcSettings;
5   import org.hibernate.engine.jdbc.connections.internal.DatasourceConnectionProviderImpl;
6   import org.hibernate.engine.jdbc.connections.spi.ConnectionProvider;
7   import org.springframework.jdbc.datasource.DriverManagerDataSource;
8   import spring.SpringTargetApplication;
9   
10  import java.sql.DriverManager;
11  import java.util.AbstractMap;
12  import java.util.HashMap;
13  import java.util.Map;
14  
15  public class MasterService {
16      
17      private final HashMap<String, ConnectionProvider> hashMap = new HashMap<>();
18      
19      public MasterService() {
20          
21          DriverManager.setLogWriter(null);  // remove annoying logs from jdbc driver
22          
23          SpringTargetApplication.getPropertiesFilterByProfile().map(AbstractMap.SimpleEntry::getKey).forEach(props -> {
24              
25              DatasourceConnectionProviderImpl connectionProvider = new DatasourceConnectionProviderImpl();
26              
27              DriverManagerDataSource dataSource = new DriverManagerDataSource();
28              dataSource.setUrl(props.getProperty(JdbcSettings.JAKARTA_JDBC_URL));
29              dataSource.setUsername(props.getProperty(JdbcSettings.JAKARTA_JDBC_USER));
30              dataSource.setPassword(props.getProperty(JdbcSettings.JAKARTA_JDBC_PASSWORD));
31              
32              connectionProvider.configure(Map.of(
33                  Environment.DATASOURCE, dataSource
34              ));
35  
36              this.hashMap.put(props.getProperty("jsql.tenant"), connectionProvider);
37          });
38      }
39      
40      public HashMap<String, ConnectionProvider> getDataSourceHashMap() {
41          return this.hashMap;
42      }
43  }
