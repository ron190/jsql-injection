1   package spring.tenant;

3   import org.hibernate.cfg.Environment;
4   import org.hibernate.engine.jdbc.connections.internal.DatasourceConnectionProviderImpl;
5   import org.hibernate.engine.jdbc.connections.spi.ConnectionProvider;
6   import org.springframework.jdbc.datasource.DriverManagerDataSource;
7   import spring.SpringTargetApplication;
8   
9   import java.sql.DriverManager;
10  import java.util.AbstractMap;
11  import java.util.HashMap;
12  import java.util.Map;
13  
14  public class MasterService {
15      
16      private final HashMap<String, ConnectionProvider> hashMap = new HashMap<>();
17      
18      public MasterService() {
19          
20          // Remove annoying logs from jdbc driver
21          DriverManager.setLogWriter(null);
22          
23          SpringTargetApplication.propertiesByEngine.stream()
24          .filter(propertyByEngine -> System.getProperty("profileId", "tests").equals(
25              propertyByEngine.getKey().getProperty("jsql.profile", "tests")  // undefined by default
26          ))
27          .map(AbstractMap.SimpleEntry::getKey).forEach(props -> {
28              
29              DatasourceConnectionProviderImpl connectionProvider = new DatasourceConnectionProviderImpl();
30              
31              DriverManagerDataSource dataSource = new DriverManagerDataSource();
32              dataSource.setUrl(props.getProperty(Environment.URL));
33              dataSource.setUsername(props.getProperty(Environment.USER));
34              dataSource.setPassword(props.getProperty(Environment.PASS));
35              
36              connectionProvider.configure(Map.of(
37                  Environment.DATASOURCE, dataSource
38              ));
39  
40              this.hashMap.put(props.getProperty("jsql.tenant"), connectionProvider);
41          });
42      }
43      
44      public HashMap<String, ConnectionProvider> getDataSourceHashMap() {
45          return this.hashMap;
46      }
47  }
