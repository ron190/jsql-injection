107     public static void initializeDatabases() throws Exception {
108 
109         if (System.getProperty("profileId") == null || "tests".equals(System.getProperty("profileId"))) {
110 
111             initializeHsqldb();
112             initializeH2();
113             initializeNeo4j();
114             initializeDerby();
115             initializeMckoi();
116         }
117 
118         getPropertiesFilterByProfile()
119         .forEach(propertyByEngine -> {
120             
121             Configuration configuration = new Configuration();
122             configuration.addProperties(propertyByEngine.getKey()).configure("hibernate/hibernate.cfg.xml");
123             configuration.addAnnotatedClass(Student.class);
124             
125             StandardServiceRegistryBuilder builder = new StandardServiceRegistryBuilder().applySettings(configuration.getProperties());
126             
127             try (
128                 SessionFactory factory = configuration.buildSessionFactory(builder.build());
129                 Session session = factory.openSession()
130             ) {
131                 Transaction transaction = session.beginTransaction();
132                 Student student = new Student();
133                 student.setAge(1);
134                 student.setFirstName("firstName");
135                 student.setClassName("className");
136                 student.setLastName("lastName");
137                 student.setRollNo("rollNo");
138                 session.persist(student);
139                 transaction.commit();
140                 
141             } catch (Exception e) {
142                 LOGGER.error(e);
143             }
144         });
145     }

163     private static void initializeMckoi() throws SQLException, IOException {
164 
165         DefaultDBConfig config = new DefaultDBConfig(File.createTempFile("mckoi.config", null));
166         config.setMinimumDebugLevel(Lvl.ERROR);
167 
168         DBSystem database;
169         DBController controller = DBController.getDefault();
170         boolean isDatabaseExists = controller.databaseExists(config);
171         if (!isDatabaseExists) {
172             database = controller.createDatabase(config, "user", "password");
173         } else {
174             database = controller.startDatabase(config);
175         }
176 
177         serverMckoi = new TCPJDBCServer(database);
178         serverMckoi.start();
179 
180         if (!isDatabaseExists) {
181             try (
182                 Connection con = DriverManager.getConnection("jdbc:mckoi://127.0.0.1/APP", "user", "password");
183                 PreparedStatement pstmt = con.prepareStatement("CREATE TABLE APP.pwn (dataz text)");
184                 PreparedStatement pstmtinsert = con.prepareStatement("INSERT INTO APP.pwn (dataz) VALUES ('jsql data')")
185             ) {
186                 pstmt.execute();
187                 pstmtinsert.executeUpdate();
188             }
189         }
190     }

198     private static void initializeNeo4j() throws IOException {
199         
200         String graphMovie;
201         try (InputStream stream = new ClassPathResource("neo4j/movie-graph.txt").getInputStream()) {
202             graphMovie = new String(stream.readAllBytes(), StandardCharsets.UTF_8);
203         }
204         
205         Driver driver = GraphDatabase.driver("bolt://jsql-neo4j:7687", AuthTokens.basic("neo4j", "test"));
206         
207         try (org.neo4j.driver.Session session = driver.session()) {
208             
209             session.run("MATCH (n) DETACH DELETE n");
210             
211             Result result = session.run(graphMovie);
212             result.forEachRemaining(LOGGER::info);
213             
214         } catch (Exception e) {
215             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
216         }
217         
218         driver.close();
219     }

221     public static Stream<SimpleEntry<Properties, String>> getPropertiesFilterByProfile() {
222         return SpringTargetApplication.propertiesByEngine.parallelStream().filter(propertyByEngine ->
223             System.getProperty("profileId") == null
224             || propertyByEngine.getKey().getProperty("jsql.profile").equals(System.getProperty("profileId"))
225         );
