75              try (InputStream inputStream = classloader.getResourceAsStream(simpleEntry.getValue())) {

83      public static void initializeDatabases() throws Exception {
84          
85          initializeHsqldb();
86          initializeH2();
87          initializeNeo4j();
88          initializeDerby();
89          
90          ArrayList<Properties> properties = new ArrayList<>(
91              Arrays.asList(
92                  SpringTargetApplication.propsH2,
93                  SpringTargetApplication.propsMysql,
94                  SpringTargetApplication.propsMysqlError,
95                  SpringTargetApplication.propsPostgreSql,
96                  SpringTargetApplication.propsSqlServer,
97                  SpringTargetApplication.propsCubrid,
98                  SpringTargetApplication.propsSqlite,
99                  SpringTargetApplication.propsDb2,
100                 SpringTargetApplication.propsHsqldb,
101                 SpringTargetApplication.propsDerby,
102                 SpringTargetApplication.propsFirebird
103             )
104         );
105         
106         properties
107         .parallelStream()
108         .forEach(props -> {
109             
110             Configuration configuration = new Configuration();
111             configuration.addProperties(props).configure("hibernate/hibernate.cfg.xml");
112             configuration.addAnnotatedClass(Student.class);
113             
114             StandardServiceRegistryBuilder builder = new StandardServiceRegistryBuilder().applySettings(configuration.getProperties());
115             
116             try (
117                 SessionFactory factory = configuration.buildSessionFactory(builder.build());
118                 Session session = factory.openSession()
119             ) {
120                 Transaction transaction = session.beginTransaction();
121                 Student student = new Student();
122                 student.setAge(1);
123                 student.setFirstName("firstName");
124                 student.setClassName("className");
125                 student.setLastName("lastName");
126                 student.setRollNo("rollNo");
127                 session.persist(student);
128                 transaction.commit();
129                 
130             } catch (Exception e) {
131                 LOGGER.error(e);
132             }
133         });
134     }

158     private static void initializeNeo4j() throws IOException {
159         
160         String graphMovie;
161         try (InputStream stream = new ClassPathResource("neo4j/movie-graph.txt").getInputStream()) {
162             graphMovie = new String(stream.readAllBytes(), StandardCharsets.UTF_8);
163         }
164         
165         Driver driver = GraphDatabase.driver("bolt://jsql-neo4j:7687", AuthTokens.basic("neo4j", "test"));
166         
167         try (org.neo4j.driver.Session session = driver.session()) {
168             
169             session.run("MATCH (n) DETACH DELETE n");
170             
171             Result result = session.run(graphMovie);
172             result.forEachRemaining(LOGGER::info);
173             
174         } catch (Exception e) {
175             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
176         }
177         
178         driver.close();
179     }
