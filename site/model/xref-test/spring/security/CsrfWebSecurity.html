1   package spring.security;

3   import jakarta.servlet.http.HttpServletRequest;
4   import org.springframework.context.annotation.Bean;
5   import org.springframework.context.annotation.Configuration;
6   import org.springframework.core.annotation.Order;
7   import org.springframework.security.config.Customizer;
8   import org.springframework.security.config.annotation.web.builders.HttpSecurity;
9   import org.springframework.security.web.SecurityFilterChain;
10  import org.springframework.security.web.access.intercept.AuthorizationFilter;
11  import org.springframework.security.web.csrf.CookieCsrfTokenRepository;
12  import org.springframework.security.web.csrf.CsrfTokenRequestAttributeHandler;
13  import org.springframework.security.web.util.matcher.AndRequestMatcher;
14  import org.springframework.security.web.util.matcher.AntPathRequestMatcher;
15  import org.springframework.security.web.util.matcher.RegexRequestMatcher;
16  import org.springframework.security.web.util.matcher.RequestMatcher;
17  
18  @Configuration
19  @Order(2)
20  public class CsrfWebSecurity {
21  
22      public static final CustomFilter FILTER = new CustomFilter("csrf");
23  
24      @Bean
25      public SecurityFilterChain filterChainCsrf(HttpSecurity http) throws Exception {
26  
27          CookieCsrfTokenRepository tokenRepository = CookieCsrfTokenRepository.withHttpOnlyFalse();
28          CsrfTokenRequestAttributeHandler requestHandler = new CsrfTokenRequestAttributeHandler();
29          requestHandler.setCsrfRequestAttributeName("_csrf");
30  
31          return http.securityMatcher("/csrf/**")
32              .authorizeHttpRequests(auth -> auth
33                  .requestMatchers(AntPathRequestMatcher.antMatcher("/csrf/**")).permitAll()
34              )
35              .csrf(csrf -> csrf
36                  .csrfTokenRepository(tokenRepository)
37                  .csrfTokenRequestHandler(requestHandler)
38                  .requireCsrfProtectionMatcher(
39                      new AndRequestMatcher(
40                          new DefaultRequiresCsrfMatcher(),
41                          new RegexRequestMatcher("/csrf.*", null)
42                      )
43                  )
44              )
45              .addFilterAfter(CsrfWebSecurity.FILTER, AuthorizationFilter.class)
46              .exceptionHandling(Customizer.withDefaults())
47              .build();
48      }
49  
50      /**
51       * Allow Csrf also for GET.
52       * The default is to ignore GET, HEAD, TRACE, OPTIONS and process all other requests.
53       */
54      private static final class DefaultRequiresCsrfMatcher implements RequestMatcher {
55          @Override
56          public boolean matches(HttpServletRequest request) {
57              return true;
58          }
59      }
60  }
