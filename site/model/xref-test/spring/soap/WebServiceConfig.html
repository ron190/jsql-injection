3   import org.springframework.boot.web.servlet.ServletRegistrationBean;
4   import org.springframework.context.ApplicationContext;
5   import org.springframework.context.annotation.Bean;
6   import org.springframework.context.annotation.Configuration;
7   import org.springframework.core.io.ClassPathResource;
8   import org.springframework.ws.config.annotation.EnableWs;
9   import org.springframework.ws.config.annotation.WsConfigurerAdapter;
10  import org.springframework.ws.transport.http.MessageDispatcherServlet;
11  import org.springframework.ws.wsdl.wsdl11.DefaultWsdl11Definition;
12  import org.springframework.xml.xsd.SimpleXsdSchema;
13  import org.springframework.xml.xsd.XsdSchema;
14  
15  @EnableWs
16  @Configuration
17  public class WebServiceConfig extends WsConfigurerAdapter {
18  
19      @Bean
20      public ServletRegistrationBean<MessageDispatcherServlet> messageDispatcherServlet(ApplicationContext applicationContext) {
21          
22          MessageDispatcherServlet servlet = new MessageDispatcherServlet();
23          servlet.setApplicationContext(applicationContext);
24          servlet.setTransformWsdlLocations(true);
25          
26          return new ServletRegistrationBean<>(servlet, "/ws/*");
27      }
28  
29      @Bean(name = "countries")
30      public DefaultWsdl11Definition defaultWsdl11Definition(XsdSchema countriesSchema) {
31          
32          DefaultWsdl11Definition wsdl11Definition = new DefaultWsdl11Definition();
33          wsdl11Definition.setPortTypeName("CountriesPort");
34          wsdl11Definition.setLocationUri("/ws");
35          wsdl11Definition.setTargetNamespace("http://www.baeldung.com/springsoap/gen");
36          wsdl11Definition.setSchema(countriesSchema);
37          
38          return wsdl11Definition;
39      }
40  
41      @Bean
42      public XsdSchema countriesSchema() {
43          
44          return new SimpleXsdSchema(new ClassPathResource("countries.xsd"));
45      }
46  }
