17  import java.net.URLDecoder;
18  import java.net.URLEncoder;
19  import java.nio.charset.StandardCharsets;
20  import java.util.List;
21  
22  @Component
23  public class CountryRepository {
24      
25      private static final Logger LOGGER = LogManager.getRootLogger();
26  
27      @Autowired
28      private SessionFactory sessionFactory;
29      
30      private static final String TEMPLATE = "Hello, s!";
31      private final ObjectMapper objectMapper = new ObjectMapper();
32  
33      @Transactional
34  	public Country findCountry(String name) throws JsonProcessingException {
35  	    
36          Country country = new Country();
37  
38          Session session = this.sessionFactory.getCurrentSession();
39  
40          String nameUrlDecoded = URLDecoder.decode(name, StandardCharsets.UTF_8);
41  
42          try {
43              NativeQuery<Object> query = session.createNativeQuery(
44                  "select 1,2,3,4,First_Name,5,6,7,8 from Student where '1' = '" + nameUrlDecoded + "'",
45                  Object.class
46              );
47              
48              List<Object> results = query.getResultList();
49  
50              country.setName(URLEncoder.encode(
51                  TEMPLATE + StringEscapeUtils.unescapeJava(this.objectMapper.writeValueAsString(results)),
52                  StandardCharsets.UTF_8
53              ));
54          } catch (Exception e) {
55              
56              // Required by multiple columns
57              country.setName(URLEncoder.encode(this.initializeErrorMessage(e).getContent(), StandardCharsets.UTF_8));
58              
59              // Required by transaction rollback
60              throw e;
61          }
62  
63          return country;
64  	}
65  
66      private Greeting initializeErrorMessage(Exception e) {
67          
68          String stacktrace = ExceptionUtils.getStackTrace(e);
69          
70          LOGGER.debug(stacktrace);
71  
72          return new Greeting(TEMPLATE + "#" + StringEscapeUtils.unescapeJava(stacktrace));
73      }
74  }
