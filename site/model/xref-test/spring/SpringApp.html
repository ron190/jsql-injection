107     public static void initDatabases() throws Exception {
108         if (System.getProperty("profileId") == null || "tests".equals(System.getProperty("profileId"))) {
109             SpringApp.initHsqldb();
110             SpringApp.initH2();
111             SpringApp.initNeo4j();
112             SpringApp.initDerby();
113             SpringApp.initMckoi();
114         }
115 
116         SpringApp.getPropertiesFilterByProfile().forEach(propertyByEngine -> {
117             Configuration configuration = new Configuration();
118             configuration.addProperties(propertyByEngine.getKey()).configure("hibernate/hibernate.cfg.xml");
119             configuration.addAnnotatedClass(Student.class);
120             configuration.addAnnotatedClass(StudentForDelete.class);
121 
122             StandardServiceRegistryBuilder builder = new StandardServiceRegistryBuilder().applySettings(configuration.getProperties());
123             try (
124                 SessionFactory factory = configuration.buildSessionFactory(builder.build());
125                 Session session = factory.openSession()
126             ) {
127                 Transaction transaction = session.beginTransaction();
128                 Student student = new Student();
129                 student.setAge(1);
130                 student.setFirstName("firstName");
131                 student.setClassName("className");
132                 student.setLastName("lastName");
133                 student.setRollNo("rollNo");
134                 session.persist(student);
135                 session.persist(new StudentForDelete());
136                 transaction.commit();
137             } catch (Exception e) {
138                 LOGGER.error(e);
139             }
140         });
141     }

157     private static void initMckoi() throws SQLException, IOException {
158         DefaultDBConfig config = new DefaultDBConfig(File.createTempFile("mckoi.config", null));
159         config.setMinimumDebugLevel(Lvl.ERROR);
160 
161         DBSystem database;
162         DBController controller = DBController.getDefault();
163         boolean isDatabaseExists = controller.databaseExists(config);
164         if (!isDatabaseExists) {
165             database = controller.createDatabase(config, "user", "password");
166         } else {
167             database = controller.startDatabase(config);
168         }
169 
170         SpringApp.serverMckoi = new TCPJDBCServer(database);
171         SpringApp.serverMckoi.start();
172 
173         if (!isDatabaseExists) {
174             try (
175                 Connection con = DriverManager.getConnection("jdbc:mckoi://127.0.0.1/APP", "user", "password");
176                 PreparedStatement pstmt = con.prepareStatement("CREATE TABLE APP.pwn (dataz text)");
177                 PreparedStatement pstmtinsert = con.prepareStatement("INSERT INTO APP.pwn (dataz) VALUES ('jsql data')")
178             ) {
179                 pstmt.execute();
180                 pstmtinsert.executeUpdate();
181             }
182         }
183     }

190     private static void initNeo4j() throws IOException {
191         String graphMovie;
192         try (InputStream stream = new ClassPathResource("neo4j/movie-graph.txt").getInputStream()) {
193             graphMovie = new String(stream.readAllBytes(), StandardCharsets.UTF_8);
194         }
195         
196         Driver driver = GraphDatabase.driver("bolt://jsql-neo4j:7687", AuthTokens.basic("neo4j", "test"));
197         try (org.neo4j.driver.Session session = driver.session()) {
198             session.run("MATCH (n) DETACH DELETE n");
199             
200             Result result = session.run(graphMovie);
201             result.forEachRemaining(LOGGER::info);
202         } catch (Exception e) {
203             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
204         }
205         driver.close();
206     }
