47      private Greeting getResponse(
48          String name,
49          String sqlQuery,
50          boolean isError,
51          boolean isUpdate,
52          boolean isVisible,
53          boolean isOracle,
54          boolean isBoolean,
55          boolean isStack
56      ) {
57          if (name == null) {  // Empty when scanning
58              return null;
59          }
60  
61          String inject = isOracle ? name : name.replace(":", "\\:");
62  
63          try {
64              Session session = this.sessionFactory.getCurrentSession();
65              Query<Object> query = session.createNativeQuery(String.format(sqlQuery, inject), Object.class);
66              if (isUpdate) {
67                  query.executeUpdate();
68              } else {
69                  if (isStack) {
70                      try (Connection connection = ((SessionImpl) session).getJdbcConnectionAccess().obtainConnection()) {
71                          Statement stmt = connection.createStatement();
72                          boolean hasMoreResultSets = stmt.execute(String.format(sqlQuery, inject));
73                          StringBuilder results = new StringBuilder();
74                          while (hasMoreResultSets) {
75                              ResultSet rs = stmt.getResultSet();
76                              ResultSetMetaData metaData = rs.getMetaData();
77                              int columnCount = metaData.getColumnCount();
78                              while (rs.next()) {
79                                  for (int columnNumber = 1; columnNumber <= columnCount; columnNumber++) {
80                                      results.append(rs.getString(columnNumber));
81                                  }
82                              }
83                              hasMoreResultSets = stmt.getMoreResults();
84                          }
85                          return new Greeting(
86                              HibernateRestController.TEMPLATE + StringEscapeUtils.unescapeJava(this.objectMapper.writeValueAsString(results))
87                          );
88                      }
89                  } else {
90                      List<Object> results = query.getResultList();
91                      if (isVisible) {
92                          return new Greeting(
93                              isBoolean
94                              ? results.isEmpty() ? "true" : "false"
95                              : HibernateRestController.TEMPLATE + StringEscapeUtils.unescapeJava(this.objectMapper.writeValueAsString(results))
96                          );
97                      }
98                  }
99              }
100         } catch (Exception e) {
101             if (isError) {  // Hide useless SQL error messages except for error base injection
102                 return this.initErrorMessage(e);
103             }
104         }
105 
106         return null;
107     }
108 
109     private Greeting initErrorMessage(Exception e) {
110         String stacktrace = ExceptionUtils.getStackTrace(e);
111         LOGGER.debug(stacktrace);
112         return new Greeting(HibernateRestController.TEMPLATE + "#" + StringEscapeUtils.unescapeJava(stacktrace));
113     }

131     @RequestMapping("/stack2")
132     public Greeting endpointStack2(@RequestParam(value="name", defaultValue="World") String name, @RequestHeader Map<String, String> headers) {
133         Greeting greeting = new Greeting(null);
134         var result = new StringBuilder();
135         Arrays.stream(name.split(";")).map(String::trim).forEach(query -> {
136             query = query +";";
137             Greeting g = this.getResponse(query, "select First_Name from Student where '1' = '%s'", true, false, true, false, false, false);
138             if (g != null) result.append(g.getContent());
139         });
140         greeting.setContent(result.toString());
141         return greeting;
142     }

225         String inject = request.getParameterMap().get("name")[0];
226         inject = inject.replaceAll("\\\\:", ":");
227         try {
228             new JSONObject(inject);
229         } catch (Exception e) {
230             return null;
231         }
232         inject = new JSONObject(inject).getJSONObject("b").getJSONArray("b").getJSONObject(3).getJSONObject("a").getString("a");
233         inject = inject.replaceAll(":", "\\\\:");
234         inject = inject.replace(":", "\\:");
235 
236         return this.getResponse(inject, "select First_Name from Student where '1' = '%s'", true, false, true, true, false, false);
