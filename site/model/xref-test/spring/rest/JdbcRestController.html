359     @RequestMapping("/neo4j")
360     public Greeting greetingNeo4j(@RequestParam(value="name", defaultValue="World") String name) {
361         
362         Greeting greeting;
363         
364         try (org.neo4j.driver.Session session = this.driver.session()) {
365             Result result = session.run("MATCH (n:Person) where 1="+ name +" RETURN n.name, n.from, n.title, n.hobby");
366             
367             String a = result.stream().map(driverRecord ->
368             
369                 driverRecord
370                 .keys()
371                 .stream()
372                 .map(key -> key + "=" + driverRecord.get(key))
373                 .collect(Collectors.joining(", ", "{", "}"))
374                 
375             ).collect(Collectors.joining());
376             
377             greeting = new Greeting(template+ StringEscapeUtils.unescapeJava(this.objectMapper.writeValueAsString(a)));
378             
379         } catch (Exception e) {
380             
381             greeting = this.initializeErrorMessage(e);
382         }
383         
384         return greeting;
385     }
