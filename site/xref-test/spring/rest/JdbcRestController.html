57      @RequestMapping("/neo4j")
58      public Greeting greetingNeo4j(@RequestParam(value="name", defaultValue="World") String name) {
59          Greeting greeting;
60  
61          try (org.neo4j.driver.Session session = this.driver.session()) {
62              Result result = session.run("MATCH (n:Person) where 1="+ name +" RETURN n.name, n.from, n.title, n.hobby");
63  
64              String a = result.stream().map(driverRecord -> driverRecord
65                      .keys()
66                      .stream()
67                      .map(key -> key + "=" + driverRecord.get(key))
68                      .collect(Collectors.joining(", ", "{", "}"))
69                  ).collect(Collectors.joining());
70  
71              greeting = new Greeting(JdbcRestController.TEMPLATE + StringEscapeUtils.unescapeJava(this.objectMapper.writeValueAsString(a)));
72          } catch (Exception e) {
73              greeting = this.initErrorMessage(e);
74          }
75  
76          return greeting;
77      }
