10      public String resolveCurrentTenantIdentifier() {
11          
12          ServletRequestAttributes attributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
13          String tenant;  // Undefined used by SOAP
14          
15          if (attributes != null) {
16              
17              tenant = attributes.getRequest().getParameter("tenant");
18              
19              if (tenant == null) {
20                  tenant = "mysql";
21              }
22          } else {
23              tenant = "mysql";
24          }
25          
26          return tenant;
