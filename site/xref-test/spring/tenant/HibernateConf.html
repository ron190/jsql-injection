9   import org.springframework.orm.hibernate5.LocalSessionFactoryBean;
10  import org.springframework.security.web.firewall.StrictHttpFirewall;
11  import org.springframework.transaction.PlatformTransactionManager;
12  import org.springframework.transaction.annotation.EnableTransactionManagement;
13  import org.springframework.web.bind.annotation.RequestMethod;
14  
15  import java.util.Arrays;
16  import java.util.List;
17  import java.util.Properties;
18  import java.util.stream.Collectors;
19  import java.util.stream.Stream;
20  
21  @Configuration
22  @EnableTransactionManagement
23  public class HibernateConf {
24  
25      @Bean
26      public PlatformTransactionManager hibernateTransactionManager() {
27          HibernateTransactionManager transactionManager = new HibernateTransactionManager();
28          transactionManager.setSessionFactory(this.sessionFactory().getObject());
29          return transactionManager;
30      }
31  
32      @Bean
33      public LocalSessionFactoryBean sessionFactory() {
34          Properties hibernateProperties = new Properties();
35          hibernateProperties.setProperty(Environment.DIALECT, "org.hibernate.dialect.MySQLDialect");
36          hibernateProperties.setProperty(AvailableSettings.MULTI_TENANT_CONNECTION_PROVIDER, "spring.tenant.MultiTenantConnectionProviderImpl");
37          hibernateProperties.setProperty(AvailableSettings.MULTI_TENANT_IDENTIFIER_RESOLVER, "spring.tenant.CurrentTenantIdentifierResolverImpl");
38          hibernateProperties.setProperty("hibernate.multiTenancy", "DATABASE");
39  
40          LocalSessionFactoryBean sessionFactory = new LocalSessionFactoryBean();
41          sessionFactory.setHibernateProperties(hibernateProperties);
42          return sessionFactory;
43      }
44  
45      @Bean
46      public StrictHttpFirewall httpFirewall() {
47          StrictHttpFirewall firewall = new StrictHttpFirewall();
48  
49          List<String> httpMethods = Stream.concat(
50                  Stream.of(CustomMethodSuiteIT.CUSTOM_METHOD),
51                  Arrays.stream(RequestMethod.values()).map(Enum::name)
52              )
53              .collect(Collectors.toList());
54  
55          firewall.setAllowedHttpMethods(httpMethods);
56          firewall.setUnsafeAllowAnyHttpMethod(true);
57          return firewall;
58      }
59  }
