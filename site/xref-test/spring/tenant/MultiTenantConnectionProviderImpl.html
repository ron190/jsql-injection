6   public class MultiTenantConnectionProviderImpl extends AbstractMultiTenantConnectionProvider<String> {
7   
8       private static final MasterService MASTER_SERVICE = new MasterService();
9       
10      @Override
11      protected ConnectionProvider getAnyConnectionProvider() {
12          return MASTER_SERVICE.getDataSourceHashMap().get("h2");
13      }
14  
15      @Override
16      protected ConnectionProvider selectConnectionProvider(String tenant) {
17          return MASTER_SERVICE.getDataSourceHashMap().getOrDefault(
18              tenant,
19              // Provide default connection to prevent error logs when tenant is undefined
20              getAnyConnectionProvider()
21          );
22      }
23  }
