12  import org.springframework.security.web.access.intercept.AuthorizationFilter;
13  import org.springframework.security.web.authentication.www.DigestAuthenticationEntryPoint;
14  import org.springframework.security.web.authentication.www.DigestAuthenticationFilter;
15  import org.springframework.security.web.util.matcher.AntPathRequestMatcher;
16  
17  @Configuration
18  @Order(3)
19  public class DigestSecurityConfig {
20      
21      private static final String DIGEST_AUTHORITY = "ADMIN2";
22      private static final String DIGEST_REALM = "Digest Realm";
23      public static final String DIGEST_USERNAME = "login-digest";
24      public static final String DIGEST_PASSWORD = "password-digest";
25      private static final String DIGEST_PASSWORD_ENCODED = DigestAuthUtils.encodePasswordInA1Format(DigestSecurityConfig.DIGEST_USERNAME, DigestSecurityConfig.DIGEST_REALM, DigestSecurityConfig.DIGEST_PASSWORD);
26      public static final CustomFilter FILTER = new CustomFilter("digest");
27  
28      @Bean
29      public SecurityFilterChain filterChainDigest(HttpSecurity http) throws Exception {
30  
31          return http.securityMatcher("/digest/**")
32              .csrf(Customizer.withDefaults())
33              .authorizeHttpRequests(auth -> auth
34                  .requestMatchers(AntPathRequestMatcher.antMatcher("/digest/**")).hasAuthority(DigestSecurityConfig.DIGEST_AUTHORITY)
35              )
36              .addFilterAfter(DigestSecurityConfig.FILTER, AuthorizationFilter.class)
37              .addFilter(this.digestAuthenticationFilter())
38              .exceptionHandling()
39              // Deprecated but no compatible entry point setting available
40              .defaultAuthenticationEntryPointFor(this.digestEntryPoint(), new AntPathRequestMatcher("/digest/**"))
41              .and()
42              .build();
43      }
44  
45      public Filter digestAuthenticationFilter() {
46  
47          UserDetails user = User.builder()
48              .username(DigestSecurityConfig.DIGEST_USERNAME)
49              .password(DigestSecurityConfig.DIGEST_PASSWORD_ENCODED)
50              .authorities(DigestSecurityConfig.DIGEST_AUTHORITY)
51              .build();
52  
53          DigestAuthenticationFilter digestAuthenticationFilter = new DigestAuthenticationFilter();
54          digestAuthenticationFilter.setAuthenticationEntryPoint(this.digestEntryPoint());
55          digestAuthenticationFilter.setUserDetailsService(username -> user);
56          digestAuthenticationFilter.setPasswordAlreadyEncoded(true);
57          digestAuthenticationFilter.setCreateAuthenticatedToken(true);
58  
59          return digestAuthenticationFilter;
60      }
61  
62      @Bean
63      public DigestAuthenticationEntryPoint digestEntryPoint() {
64  
65          DigestAuthenticationEntryPoint digestAuthenticationEntryPoint = new DigestAuthenticationEntryPoint();
66          digestAuthenticationEntryPoint.setKey("mykey");
67          digestAuthenticationEntryPoint.setRealmName(DigestSecurityConfig.DIGEST_REALM);
68  
69          return digestAuthenticationEntryPoint;
70      }
71  }
