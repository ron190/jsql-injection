23       * &lt; WWW-Authenticate: Digest realm="Digest Realm", qop="auth", nonce="MTY5MDA4NjM0MDUxMzozZTljM2M5NjU1NTgzZWQwMDI2OGVhNjc4ZTNhNTgxZA=="

27       * > Authorization: Digest username="login-digest",realm="Digest Realm",nonce="MTY5MDA4NjM0MDUxMzozZTljM2M5NjU1NTgzZWQwMDI2OGVhNjc4ZTNhNTgxZA==",uri="/digest?tenant=mysql&amp;name=",cnonce="2ecb0e39da79fcb5aa6ffb1bd45cb3bb",nc=00000001,response="e5703ca7433d944525a40f49c5501155",qop="auth"

30      @Override
31      public void setupInjection() throws Exception {
32          InjectionModel model = new InjectionModel();
33          this.injectionModel = model;
34  
35          model.subscribe(new SystemOutTerminal());
36  
37          model.getMediatorUtils().getParameterUtil().initQueryString(
38              "http://localhost:8080/digest?tenant=mysql-error&name="
39          );
40  
41          model
42          .getMediatorUtils()
43          .getPreferencesUtil()
44          .withIsStrategyBlindDisabled(true)
45          .withIsStrategyTimeDisabled(true);
46  
47          model.setIsScanning(true);
48  
49          model
50          .getMediatorUtils()
51          .getAuthenticationUtil()
52          .set(
53              false,
54              DigestSecurityConfig.DIGEST_USERNAME,
55              DigestSecurityConfig.DIGEST_PASSWORD,
56              false,
57              StringUtils.EMPTY,
58              StringUtils.EMPTY
59          );
60          
61          model
62          .getMediatorUtils()
63          .getConnectionUtil()
64          .withMethodInjection(model.getMediatorMethod().getQuery())
65          .withTypeRequest("GET");
66          
67          model.beginInjection();
68      }
