65      @BeforeAll
66      public synchronized void initializeBackend() throws Exception {
67          
68          if (AbstractTestSuite.isSetupStarted.compareAndSet(false, true)) {
69              
70              LOGGER.info("@BeforeClass: loading Hibernate and Spring...");
71              SpringApp.initializeDatabases();
72              SpringApplication.run(SpringApp.class);
73              
74              AbstractTestSuite.isSetupDone.set(true);
75          }
76              
77          LOGGER.info("@BeforeAll: backend is setting up...");
78          Awaitility.await().atMost(Duration.ofMinutes(2)).until(AbstractTestSuite.isSetupDone::get);
79  
80          if (this.injectionModel == null) {
81              
82              this.requestJdbc();
83              this.setupInjection();
84          }
85      }

93      public void requestJdbc() {
94          try (
95              Connection connection = DriverManager.getConnection(this.jdbcURL, this.jdbcUser, this.jdbcPass);
96                  
97              Statement statementDatabase = connection.createStatement();
98              ResultSet resultSetDatabase = statementDatabase.executeQuery(this.jdbcQueryForDatabaseNames);
99                  
100             Statement statementTable = connection.createStatement();
101             ResultSet resultSetTable = statementTable.executeQuery(this.jdbcQueryForTableNames);
102                 
103             Statement statementColumn = connection.createStatement();
104             ResultSet resultSetColumn = statementColumn.executeQuery(this.jdbcQueryForColumnNames);
105                 
106             Statement statementValues = connection.createStatement();
107             ResultSet resultSetValues = statementValues.executeQuery(this.jdbcQueryForValues)
108         ) {
109             while (resultSetDatabase.next()) {
110                 
111                 String dbName = resultSetDatabase.getString(this.jdbcColumnForDatabaseName);
112                 this.databasesFromJdbc.add(dbName);
113             }
114             while (resultSetTable.next()) {
115                 
116                 String tableName = resultSetTable.getString(this.jdbcColumnForTableName);
117                 this.tablesFromJdbc.add(tableName);
118             }
119             while (resultSetColumn.next()) {
120                 
121                 String colName = resultSetColumn.getString(this.jdbcColumnForColumnName);
122                 this.columnsFromJdbc.add(colName);
123             }
124             while (resultSetValues.next()) {
125                 
126                 String value = resultSetValues.getString(this.jsqlColumnName);
127                 this.valuesFromJdbc.add(value);
128             }
129         } catch (SQLException e) {
130             LOGGER.error(e, e);
131         }
132     }

134     @Ignore("Enabled on inherit")
135     public void listDatabases() throws JSqlException {
136         
137         Set<String> setValuesFromInjection = new HashSet<>();
138         Set<String> setValuesFromJdbc = new HashSet<>();
139         
140         try {
141             List<String> databases = this.injectionModel.getDataAccess()
142                 .listDatabases()
143                 .stream()
144                 .map(Database::toString)
145                 .collect(Collectors.toList());
146 
147             setValuesFromInjection.addAll(databases);
148             setValuesFromJdbc.addAll(this.databasesFromJdbc);
149 
150             LOGGER.info("ListDatabases: found {}, to find {}", setValuesFromInjection, setValuesFromJdbc);
151 
152             Assertions.assertTrue(
153                 !setValuesFromInjection.isEmpty()
154                 && !setValuesFromJdbc.isEmpty()
155                 && setValuesFromInjection.containsAll(setValuesFromJdbc)
156             );
157         } catch (AssertionError e) {
158             
159             Set<String> tablesUnkown = Stream.concat(
160                 setValuesFromInjection.stream().filter(value -> !setValuesFromJdbc.contains(value)),
161                 setValuesFromJdbc.stream().filter(value -> !setValuesFromInjection.contains(value))
162             )
163             .collect(Collectors.toCollection(TreeSet::new));
164             
165             throw new AssertionError(String.format("Unknown databases: %s%n%s", tablesUnkown, e));
166         }
167     }

249     @Ignore("Enabled on inherit")
250     public void listValues() throws JSqlException {
251         
252         Set<String> setValuesFromInjection = new TreeSet<>();
253         Set<String> setValuesFromJdbc = new TreeSet<>();
254 
255         try {
256             String[][] rows = this.injectionModel.getDataAccess().listValues(List.of(
257                 new Column(this.jsqlColumnName,
258                     new Table(this.jsqlTableName, "0",
259                         new Database(this.jsqlDatabaseName, "0")
260                     )
261                 )
262             ));
263             
264             List<String> valuesFound = Arrays.stream(rows)
265                 // => row number, occurrence, value1, value2...
266                 .map(row -> row[2].replaceAll("\r\n", "\n"))
267                 .collect(Collectors.toList());
268 
269             setValuesFromInjection.addAll(valuesFound);
270             setValuesFromJdbc.addAll(this.valuesFromJdbc);
271 
272             String logValuesFromInjection = setValuesFromInjection.toString()
273                 .replaceAll("\n", "[n]")
274                 .replaceAll("\r", "[r]");
275             
276             String logValuesFromJdbc = setValuesFromJdbc.toString()
277                 .replaceAll("\n", "[n]")
278                 .replaceAll("\r", "[r]");
279             
280             LOGGER.info("Values: found {}, to find {}", logValuesFromInjection, logValuesFromJdbc);
281 
282             Assertions.assertTrue(
283                 !setValuesFromInjection.isEmpty()
284                 && !setValuesFromJdbc.isEmpty()
285                 // TODO update and delete injection prevent exact matching => create a specific table
286                 && setValuesFromInjection.containsAll(setValuesFromJdbc)
287             );
288         } catch (AssertionError e) {
289             
290             Set<String> valuesUnknown = Stream.concat(
291                 setValuesFromInjection.stream().filter(value -> !setValuesFromJdbc.contains(value)),
292                 setValuesFromJdbc.stream().filter(value -> !setValuesFromInjection.contains(value))
293             )
294             .collect(Collectors.toCollection(TreeSet::new));
295             
296             throw new AssertionError(String.format("Unknown values: %s%n%s", valuesUnknown, e));
297         }
