3   import javax.swing.*;
4   import javax.swing.text.BadLocationException;
5   import java.awt.*;
6   import java.awt.event.*;
7   
8   public class BlockSelectionTextArea extends JTextArea {
9   
10      private int startRow = -1, startCol = -1;
11      private int endRow = -1, endCol = -1;
12  
13      public BlockSelectionTextArea() {
14          // Mouse listener to initiate and handle the selection
15          this.addMouseListener(new MouseAdapter() {
16              @Override
17              public void mousePressed(MouseEvent e) {
18                  // Record the initial mouse press position (row, col)
19                  int row = BlockSelectionTextArea.this.getRowAtPosition(e.getPoint());
20                  int col = BlockSelectionTextArea.this.getColumnAtPosition(e.getPoint());
21                  BlockSelectionTextArea.this.startRow = row;
22                  BlockSelectionTextArea.this.startCol = col;
23              }
24  
25              @Override
26              public void mouseReleased(MouseEvent e) {
27                  // Finalize the block selection on mouse release
28                  int row = BlockSelectionTextArea.this.getRowAtPosition(e.getPoint());
29                  int col = BlockSelectionTextArea.this.getColumnAtPosition(e.getPoint());
30                  BlockSelectionTextArea.this.endRow = row;
31                  BlockSelectionTextArea.this.endCol = col;
32  
33                  // Update the selection based on the rectangular block
34                  try {
35                      BlockSelectionTextArea.this.setBlockSelection(BlockSelectionTextArea.this.startRow, BlockSelectionTextArea.this.startCol, BlockSelectionTextArea.this.endRow, BlockSelectionTextArea.this.endCol);
36                  } catch (BadLocationException ex) {
37                      throw new RuntimeException(ex);
38                  }
39              }
40          });
41  
42          // Mouse motion listener for drag behavior
43          this.addMouseMotionListener(new MouseMotionAdapter() {
44              @Override
45              public void mouseDragged(MouseEvent e) {
46                  // Update the block selection as the mouse is dragged
47                  int row = BlockSelectionTextArea.this.getRowAtPosition(e.getPoint());
48                  int col = BlockSelectionTextArea.this.getColumnAtPosition(e.getPoint());
49                  BlockSelectionTextArea.this.endRow = row;
50                  BlockSelectionTextArea.this.endCol = col;
51  
52                  // Update the selection during dragging
53                  try {
54                      BlockSelectionTextArea.this.setBlockSelection(BlockSelectionTextArea.this.startRow, BlockSelectionTextArea.this.startCol, BlockSelectionTextArea.this.endRow, BlockSelectionTextArea.this.endCol);
55                  } catch (BadLocationException ex) {
56                      throw new RuntimeException(ex);
57                  }
58              }
59          });
60      }
61  
62      // Converts the mouse position to the corresponding row
63      private int getRowAtPosition(Point p) {
64          int offset = this.viewToModel(p);
65          try {
66              return this.getLineOfOffset(offset);
67          } catch (BadLocationException e) {
68              return 0;
69          }
70      }
71  
72      // Converts the mouse position to the corresponding column
73      private int getColumnAtPosition(Point p) {
74          int offset = this.viewToModel(p);
75          try {
76              int row = this.getLineOfOffset(offset);
77              return offset - this.getLineStartOffset(row);
78          } catch (BadLocationException e) {
79              return 0;
80          }
81      }
82  
83      // Set the block selection between start and end coordinates (row, col)
84      private void setBlockSelection(int startRow, int startCol, int endRow, int endCol) throws BadLocationException {
85          // Normalize the selection direction (so it always goes from top-left to bottom-right)
86          if (startRow > endRow || (startRow == endRow && startCol > endCol)) {
87              int tempRow = startRow;
88              int tempCol = startCol;
89              startRow = endRow;
90              startCol = endCol;
91              endRow = tempRow;
92              endCol = tempCol;
93          }
94  
95          // Convert row/column to character offsets
96          int startOffset = this.getLineStartOffset(startRow) + startCol;
97          int endOffset = this.getLineStartOffset(endRow) + endCol;
98  
99          // Adjust the selection range to reflect the rectangular block
100         this.setSelectionStart(startOffset);
101         this.setSelectionEnd(endOffset);
102     }

104     public static void main(String[] args) {
105         // Create a frame with block selection text area
106         JFrame frame = new JFrame("Block Selection Example");
107         BlockSelectionTextArea textArea = new BlockSelectionTextArea();
108         textArea.setText("This is a test.\nThis is another test.\nBlock selection here.");
109         textArea.setFont(new Font("Monospaced", Font.PLAIN, 16));
110         frame.add(new JScrollPane(textArea));
111         frame.setSize(400, 300);
112         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
113         frame.setVisible(true);
114     }
