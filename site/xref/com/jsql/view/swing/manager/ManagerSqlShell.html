11  package com.jsql.view.swing.manager;
12  
13  import com.jsql.model.exception.JSqlException;
14  import com.jsql.util.I18nUtil;
15  import com.jsql.view.swing.text.JPopupTextField;
16  import com.jsql.view.swing.util.MediatorHelper;
17  import com.jsql.view.swing.util.UiUtil;
18  
19  import javax.swing.*;
20  import java.awt.*;
21  
22  /**
23   * Manager for uploading PHP SQL shell to the host and send queries.
24   */
25  public class ManagerSqlShell extends AbstractManagerShell {
26      
27      private final JTextField username = new JPopupTextField(I18nUtil.valueByKey("SQL_SHELL_USERNAME_LABEL")).getProxy();
28      
29      private final JTextField password = new JPopupTextField(I18nUtil.valueByKey("SQL_SHELL_PASSWORD_LABEL")).getProxy();
30      
31      /**
32       * Build the manager panel.
33       */
34      public ManagerSqlShell() {
35  
36          this.run.setText("Create SQL shell(s)");
37          
38          var userPassPanel = new JPanel();
39          
40          var layout = new GroupLayout(userPassPanel);
41          userPassPanel.setLayout(layout);
42          userPassPanel.setAlignmentX(Component.LEFT_ALIGNMENT);
43          userPassPanel.setBorder(BorderFactory.createEmptyBorder(1, 1, 1, 1));
44          
45          this.username.setToolTipText(I18nUtil.valueByKey("SQL_SHELL_USERNAME_TOOLTIP"));
46          this.password.setToolTipText(I18nUtil.valueByKey("SQL_SHELL_PASSWORD_TOOLTIP"));
47          
48          this.username.setBorder(UiUtil.BORDER_BLU);
49          this.password.setBorder(UiUtil.BORDER_BLU);
50          
51          var panelPassword = new JPanel(new BorderLayout());
52          panelPassword.setBorder(BorderFactory.createEmptyBorder(1, 0, 0, 0));
53          panelPassword.add(this.password);
54          
55          layout
56          .setHorizontalGroup(
57              layout
58              .createSequentialGroup()
59              .addGroup(
60                  layout
61                  .createParallelGroup(GroupLayout.Alignment.TRAILING, false)
62              )
63              .addGroup(
64                  layout.createParallelGroup()
65                  .addComponent(this.username)
66                  .addComponent(panelPassword)
67              )
68          );
69  
70          layout
71          .setVerticalGroup(
72              layout
73              .createSequentialGroup()
74              .addGroup(
75                  layout
76                  .createParallelGroup(GroupLayout.Alignment.BASELINE)
77                  .addComponent(this.username)
78              )
79              .addGroup(
80                  layout
81                  .createParallelGroup(GroupLayout.Alignment.BASELINE)
82                  .addComponent(panelPassword)
83              )
84          );
85          
86          this.add(userPassPanel, BorderLayout.NORTH);
87      }
88  
89      @Override
90      protected void createPayload(String shellPath, String shellURL) throws JSqlException, InterruptedException {
91          
92          MediatorHelper.model().getResourceAccess().createSqlShell(shellPath, shellURL, this.username.getText(), this.password.getText());
93      }
94  }
