40  /**
41   * Manager for uploading PHP webshell to the host and send system commands.
42   */
43  public abstract class AbstractManagerShell extends AbstractManagerList {
44      
45      /**
46       * Log4j logger sent to view.
47       */
48      private static final Logger LOGGER = LogManager.getRootLogger();
49  
50      private final JTextField textfieldUrlShell = new JPopupTextField(I18nUtil.valueByKey("SHELL_URL_LABEL")).getProxy();
51      
52      /**
53       * Build the manager panel.
54       */
55      protected AbstractManagerShell() {
56          this.setLayout(new BorderLayout());
57          
58          List<ItemList> itemsList = new ArrayList<>();
59          try (
60              var inputStream = UiUtil.class.getClassLoader().getResourceAsStream(UiUtil.PATH_WEB_FOLDERS);
61              var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream), StandardCharsets.UTF_8);
62              var reader = new BufferedReader(inputStreamReader)
63          ) {
64              String line;
65              while ((line = reader.readLine()) != null) {
66                  itemsList.add(new ItemList(line));
67              }
68          } catch (IOException e) {
69              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
70          }
71  
72          this.listPaths = new DnDList(itemsList);
73          this.add(new JScrollPane(this.listPaths), BorderLayout.CENTER);
74  
75          String urlTooltip = I18nUtil.valueByKey("SHELL_URL_TOOLTIP");
76          this.textfieldUrlShell.setToolTipText(urlTooltip);
77  
78          JPanel lastLine = this.initializeRunButtonPanel();
79          var southPanel = new JPanel();
80          southPanel.setLayout(new BoxLayout(southPanel, BoxLayout.Y_AXIS));
81          southPanel.add(this.textfieldUrlShell);
82          southPanel.add(lastLine);
83          this.add(southPanel, BorderLayout.SOUTH);
84      }
85      
86      protected abstract void createPayload(String pathShell, String urlShell) throws JSqlException;
87  
88      private JPanel initializeRunButtonPanel() {
89          this.defaultText = "SHELL_RUN_BUTTON_LABEL";
90          
91          var lastLine = new JPanel();
92          lastLine.setLayout(new BoxLayout(lastLine, BoxLayout.X_AXIS));
93          
94          this.run = new JButtonStateful(this.defaultText);
95          I18nViewUtil.addComponentForKey(this.defaultText, this.run);
96          this.run.setToolTipText(I18nUtil.valueByKey("SHELL_RUN_BUTTON_TOOLTIP"));
97          this.run.setEnabled(false);
98          this.run.addActionListener(new ActionCreationShell());
99  
100         this.privilege = new JLabel(I18nUtil.valueByKey("PRIVILEGE_LABEL"), UiUtil.SQUARE.icon, SwingConstants.LEFT);
101         I18nViewUtil.addComponentForKey("PRIVILEGE_LABEL", this.privilege);
102         this.privilege.setToolTipText(I18nUtil.valueByKey("PRIVILEGE_TOOLTIP"));
103 
104         lastLine.add(Box.createHorizontalStrut(5));
105         lastLine.add(this.privilege);
106         lastLine.add(Box.createHorizontalStrut(5));
107         lastLine.add(Box.createHorizontalGlue());
108         lastLine.add(this.run);
109         return lastLine;
110     }
111     
112     private class ActionCreationShell implements ActionListener {
113         @Override
114         public void actionPerformed(ActionEvent evt) {
115             if (AbstractManagerShell.this.listPaths.getSelectedValuesList().isEmpty()) {
116                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select at least one directory in the list");
117                 return;
118             }
119 
120             String urlShell = AbstractManagerShell.this.textfieldUrlShell.getText();
121             if (!urlShell.isEmpty() && !urlShell.matches("(?i)^https?://.*")) {
122                 if (!urlShell.matches("(?i)^\\w+://.*")) {
123                     LOGGER.log(LogLevelUtil.CONSOLE_INFORM, "Undefined shell URL protocol, forcing to [http://]");
124                     urlShell = "http://"+ urlShell;
125                 } else {
126                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Unknown URL protocol");
127                     return;
128                 }
129             }
130             
131             if (StringUtils.isNotEmpty(urlShell)) {
132                 try {
133                     new URI(urlShell);
134                 } catch (URISyntaxException e) {
135                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Incorrect URL: %s", e.getMessage()));
136                     return;
137                 }
138             }
139             
140             String urlShellFinal = urlShell;
141             AbstractManagerShell.this.listPaths
142                 .getSelectedValuesList()
143                 .forEach(pathShell -> new Thread(
144                     () -> {
145                         try {
146                             AbstractManagerShell.this.createPayload(pathShell.toString(), urlShellFinal);
147                         } catch (JSqlException e) {
148                             LOGGER.log(
149                                 LogLevelUtil.CONSOLE_ERROR,
150                                 String.format("Payload creation error: %s", e.getMessage())
151                             );
152                         }
153                     },
154                     "ThreadGetShell"
155                 ).start());
156         }
157     }
