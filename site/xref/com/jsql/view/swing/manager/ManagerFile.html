11  package com.jsql.view.swing.manager;
12  
13  import com.jsql.util.LogLevelUtil;
14  import com.jsql.view.swing.list.ItemList;
15  import com.jsql.view.swing.manager.util.StateButton;
16  import com.jsql.view.swing.util.I18nViewUtil;
17  import com.jsql.view.swing.util.MediatorHelper;
18  import org.apache.logging.log4j.LogManager;
19  import org.apache.logging.log4j.Logger;
20  
21  import javax.swing.*;
22  import java.awt.*;
23  import java.awt.event.ActionEvent;
24  import java.awt.event.ActionListener;
25  import java.util.List;
26  import java.util.stream.Collectors;
27  
28  /**
29   * Manager to read a file from the host.
30   */
31  public class ManagerFile extends AbstractManagerList {
32  
33      /**
34       * Log4j logger sent to view.
35       */
36      private static final Logger LOGGER = LogManager.getRootLogger();
37  
38      /**
39       * Create the manager panel to read a file.
40       */
41      public ManagerFile() {
42          super("swing/list/file.txt");
43          this.buildRunButton("FILE_RUN_BUTTON_LABEL", "FILE_RUN_BUTTON_TOOLTIP");
44          this.run.setEnabled(false);
45          this.run.addActionListener(new ActionFile());
46          this.buildPrivilege();
47          this.add(this.lastLine, BorderLayout.SOUTH);
48      }
49  
50      private class ActionFile implements ActionListener {
51          @Override
52          public void actionPerformed(ActionEvent e) {
53              if (ManagerFile.this.listPaths.getSelectedValuesList().isEmpty()) {
54                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select at least one file to read in the list");
55                  return;
56              }
57              new SwingWorker<>() {
58                  @Override
59                  protected Object doInBackground() {
60                      Thread.currentThread().setName("SwingWorkerManagerFile");
61                      if (ManagerFile.this.run.getState() == StateButton.STARTABLE) {
62                          ManagerFile.this.run.setText(I18nViewUtil.valueByKey("FILE_RUN_BUTTON_STOP"));
63                          ManagerFile.this.run.setState(StateButton.STOPPABLE);
64                          ManagerFile.this.horizontalGlue.setVisible(false);
65                          ManagerFile.this.progressBar.setVisible(true);
66                          try {
67                              List<String> filePaths = ManagerFile.this.listPaths.getSelectedValuesList().stream().map(ItemList::toString).collect(Collectors.toList());
68                              MediatorHelper.model().getResourceAccess().readFile(filePaths);
69                          } catch (InterruptedException e) {
70                              LOGGER.log(LogLevelUtil.IGNORE, e, e);
71                              Thread.currentThread().interrupt();
72                          } catch (Exception e) {
73                              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, e, e);
74                          }
75                          ManagerFile.this.endProcess();
76                      } else {
77                          MediatorHelper.model().getResourceAccess().stopSearchFile();
78                          ManagerFile.this.run.setEnabled(false);
79                          ManagerFile.this.run.setState(StateButton.STOPPING);
80                      }
81                      return null;
82                  }
83              }.doInBackground();
84          }
85      }
86  }
