38  /**
39   * Manager to upload files to the host.
40   */
41  public class ManagerUpload extends AbstractManagerList {
42      
43      /**
44       * Log4j logger sent to view.
45       */
46      private static final Logger LOGGER = LogManager.getRootLogger();

48      /**
49       * Build the manager panel.
50       */
51      public ManagerUpload() {
52          
53          this.setLayout(new BorderLayout());
54  
55          this.defaultText = "UPLOAD_RUN_BUTTON_LABEL";
56  
57          List<ItemList> pathsList = new ArrayList<>();
58          
59          try (
60              var inputStream = UiUtil.class.getClassLoader().getResourceAsStream(UiUtil.PATH_WEB_FOLDERS);
61              var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream), StandardCharsets.UTF_8);
62              var reader = new BufferedReader(inputStreamReader)
63          ) {
64              String line;
65              
66              while ((line = reader.readLine()) != null) {
67                  
68                  pathsList.add(new ItemList(line));
69              }
70              
71          } catch (IOException e) {
72              
73              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
74          }
75  
76          this.listPaths = new DnDList(pathsList);
77          
78          this.getListPaths().setBorder(BorderFactory.createEmptyBorder(0, 0, LightScrollPane.THUMB_SIZE, 0));
79          this.add(new LightScrollPane(0, 0, 0, 0, this.getListPaths()), BorderLayout.CENTER);
80          
81          var southPanel = new JPanel();
82          southPanel.setLayout(new BoxLayout(southPanel, BoxLayout.Y_AXIS));
83  
84          final JTextField shellURL = new JPopupTextField(I18nUtil.valueByKey("UPLOAD_URL_LABEL")).getProxy();
85          String urlTooltip = I18nUtil.valueByKey("UPLOAD_URL_TOOLTIP");
86          
87          shellURL.setToolTipText(urlTooltip);
88          shellURL.setBorder(
89              BorderFactory.createCompoundBorder(
90                  BorderFactory.createCompoundBorder(
91                      BorderFactory.createMatteBorder(0, 0, 0, 0, UiUtil.COLOR_COMPONENT_BORDER),
92                      BorderFactory.createMatteBorder(1, 1, 0, 1, UiUtil.COLOR_DEFAULT_BACKGROUND)
93                  ),
94                  UiUtil.BORDER_BLU
95              )
96          );
97  
98          var lastLine = new JPanel();
99          lastLine.setLayout(new BoxLayout(lastLine, BoxLayout.X_AXIS));
100         lastLine.setBorder(
101             BorderFactory.createCompoundBorder(
102                 BorderFactory.createMatteBorder(0, 0, 0, 0, UiUtil.COLOR_COMPONENT_BORDER),
103                 BorderFactory.createEmptyBorder(1, 0, 1, 1)
104             )
105         );
106 
107         this.initializeRunButton(shellURL);
108 
109         this.privilege = new JLabel(I18nUtil.valueByKey("PRIVILEGE_LABEL"), UiUtil.ICON_SQUARE_GREY, SwingConstants.LEFT);
110         I18nViewUtil.addComponentForKey("PRIVILEGE_LABEL", this.privilege);
111         this.privilege.setBorder(BorderFactory.createMatteBorder(2, 0, 0, 0, UiUtil.COLOR_DEFAULT_BACKGROUND));
112         this.privilege.setToolTipText(I18nUtil.valueByKey("PRIVILEGE_TOOLTIP"));
113 
114         this.loader.setVisible(false);
115 
116         lastLine.add(this.privilege);
117         lastLine.add(Box.createHorizontalGlue());
118         lastLine.add(this.run);
119 
120         southPanel.add(shellURL);
121         southPanel.add(lastLine);
122         this.add(southPanel, BorderLayout.SOUTH);
123     }
124 
125     private void initializeRunButton(final JTextField shellURL) {
126         
127         this.run = new JButtonStateful(this.defaultText);
128         I18nViewUtil.addComponentForKey(this.defaultText, this.run);
129         this.run.setToolTipText(I18nUtil.valueByKey("UPLOAD_RUN_BUTTON_TOOLTIP"));
130         this.run.setEnabled(false);
131         
132         this.run.setContentAreaFilled(false);
133         this.run.setBorder(BorderFactory.createEmptyBorder(4, 8, 4, 8));
134         this.run.setBackground(UiUtil.COLOR_FOCUS_GAINED);
135         
136         this.run.addMouseListener(new FlatButtonMouseAdapter(this.run));
137         
138         this.run.addActionListener(actionEvent -> this.initializeRunAction(shellURL));
139     }
140 
141     private void initializeRunAction(final JTextField shellURL) {
142         
143         if (ManagerUpload.this.getListPaths().getSelectedValuesList().isEmpty()) {
144             
145             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select directory(ies) to upload a file into");
146             
147             return;
148         }
149 
150         final var filechooser = new JFileChooser(MediatorHelper.model().getMediatorUtils().getPreferencesUtil().getPathFile());
151         filechooser.setDialogTitle(I18nUtil.valueByKey("UPLOAD_DIALOG_TEXT"));
152         
153         // Fix #2402: NullPointerException on showOpenDialog()
154         // Fix #40547: ClassCastException on showOpenDialog()
155         try {
156             int returnVal = filechooser.showOpenDialog(MediatorHelper.frame());
157             
158             if (returnVal != JFileChooser.APPROVE_OPTION) {
159                 
160                 return;
161             }
162                 
163             this.uploadFiles(shellURL, filechooser);
164             
165         } catch (NullPointerException | ClassCastException e) {
166             
167             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
168         }
169     }
170 
171     private void uploadFiles(final JTextField shellURL, final JFileChooser filechooser) {
172         
173         for (final Object path: ManagerUpload.this.getListPaths().getSelectedValuesList()) {
174             
175             new Thread(
176                 () -> {
177                 
178                     var file = filechooser.getSelectedFile();
179                     
180                     try {
181                         ManagerUpload.this.loader.setVisible(true);
182                         MediatorHelper.model().getResourceAccess().uploadFile(path.toString(), shellURL.getText(), file);
183                         
184                     } catch (JSqlException e) {
185                         
186                         LOGGER.log(
187                             LogLevelUtil.CONSOLE_ERROR,
188                             String.format("Payload creation error: %s", e.getMessage())
189                         );
190                         
191                     } catch (IOException e) {
192                         
193                         LOGGER.log(
194                             LogLevelUtil.CONSOLE_ERROR,
195                             String.format("Posting file failed: %s", e.getMessage()),
196                             e
197                         );
198                         
199                     } catch (InterruptedException e) {
200 
201                         LOGGER.log(LogLevelUtil.IGNORE, e, e);
202                         Thread.currentThread().interrupt();
203                     }
204                 },
205                 "ThreadUpload"
206             ).start();
207         }
208     }
