28  /**
29   * Run a brute force attack.
30   */
31  public class ActionBruteForce implements ActionListener, Runnable {
32      
33      /**
34       * Log4j logger sent to view.
35       */
36      private static final Logger LOGGER = LogManager.getRootLogger();
37  
38      private final ManagerBruteForce bruteForceManager;
39      
40      private boolean isStopped = false;
41      
42      public ActionBruteForce(ManagerBruteForce bruteForceManager) {
43          this.bruteForceManager = bruteForceManager;
44      }
45  
46      @Override
47      public void actionPerformed(ActionEvent actionEvent) {
48          if (this.bruteForceManager.getRun().getState() == StateButton.STOPPABLE) {
49              this.bruteForceManager.getRun().setEnabled(false);
50              this.isStopped = true;
51          } else {
52              if (StringUtils.isEmpty(this.bruteForceManager.getHash().getText())) {
53                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> I18nUtil.valueByKey("BRUTEFORCE_EMPTY_HASH"));
54                  return;
55              } else if (this.isRangeNotSelected()) {
56                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> I18nUtil.valueByKey("BRUTEFORCE_CHARACTER_RANGE"));
57                  return;
58              } else if (this.isLengthNotValid()) {
59                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> I18nUtil.valueByKey("BRUTEFORCE_INCORRECT_MIN_MAX_LENGTH"));
60                  return;
61              }
62              new Thread(this, "ThreadDisplayBruteForce").start();
63          }
64      }
65  
66      private boolean isLengthNotValid() {
67          return Integer.parseInt(this.bruteForceManager.getMaximumLength().getValue().toString())
68              < Integer.parseInt(this.bruteForceManager.getMinimumLength().getValue().toString());
69      }
70  
71      private boolean isRangeNotSelected() {
72          return !this.bruteForceManager.getSpecialCharacters().isSelected()
73              && !this.bruteForceManager.getUpperCaseCharacters().isSelected()
74              && !this.bruteForceManager.getLowerCaseCharacters().isSelected()
75              && !this.bruteForceManager.getNumericCharacters().isSelected();
76      }
77  
78      @Override
79      public void run() {
80          // Reset the panel
81          this.bruteForceManager.getRun().setText(I18nViewUtil.valueByKey("BRUTEFORCE_STOP"));
82          this.bruteForceManager.getRun().setState(StateButton.STOPPABLE);
83          this.bruteForceManager.showLoader(true);
84          this.bruteForceManager.getResult().setText(null);
85  
86          final var hashBruter = new HashBruter();
87          this.initializeBruter(hashBruter);
88  
89          // Begin the reverse hashing process
90          new Thread(hashBruter::tryBruteForce, "ThreadRunBruteForce").start();
91          while (!hashBruter.isDone() && !hashBruter.isFound() && !this.isStopped) {
92              
93              hashBruter.setEndtime(System.nanoTime());
94  
95              try {
96                  Thread.sleep(1000);  // delay to update result panel
97              } catch (InterruptedException e) {
98                  LOGGER.log(LogLevelUtil.IGNORE, e, e);
99                  Thread.currentThread().interrupt();
100             }
101             
102             int selectionStart = this.bruteForceManager.getResult().getSelectionStart();
103             int selectionEnd = this.bruteForceManager.getResult().getSelectionEnd();
104             
105             this.updateResult(hashBruter);
106 
107             this.bruteForceManager.getResult().setSelectionStart(selectionStart);
108             this.bruteForceManager.getResult().setSelectionEnd(selectionEnd);
109             
110             if (this.isStopped) {
111                 hashBruter.setIsDone(true);
112                 hashBruter.setFound(true);
113                 break;
114             }
115         }
116 
117         this.displayResult(hashBruter);
118 
119         this.isStopped = false;
120         this.bruteForceManager.showLoader(false);
121         this.bruteForceManager.getRun().setText(I18nViewUtil.valueByKey("BRUTEFORCE_RUN_BUTTON_LABEL"));
122         this.bruteForceManager.getRun().setEnabled(true);
123         this.bruteForceManager.getRun().setState(StateButton.STARTABLE);
124     }
125 
126     private void updateResult(final HashBruter hashBruter) {
127         this.bruteForceManager.getResult().setText(I18nUtil.valueByKey("BRUTEFORCE_CURRENT_STRING") + ": " + hashBruter.getPassword());
128         this.append(this.bruteForceManager.getResult(), I18nUtil.valueByKey("BRUTEFORCE_CURRENT_HASH") + ": " + hashBruter.getGeneratedHash() + "\n");
129         this.append(this.bruteForceManager.getResult(), I18nUtil.valueByKey("BRUTEFORCE_POSSIBILITIES") + ": " + hashBruter.getNumberOfPossibilities());
130         this.append(this.bruteForceManager.getResult(), I18nUtil.valueByKey("BRUTEFORCE_CHECKED_HASHES") + ": " + hashBruter.getCounter());
131         this.append(this.bruteForceManager.getResult(), I18nUtil.valueByKey("BRUTEFORCE_ESTIMATED") + ": " + hashBruter.getRemainder());
132         this.append(this.bruteForceManager.getResult(), I18nUtil.valueByKey("BRUTEFORCE_PERSECOND") + ": " + hashBruter.getPerSecond() + "\n");
133         this.append(this.bruteForceManager.getResult(), hashBruter.calculateTimeElapsed());
134 
135         if (hashBruter.getPerSecond() != 0) {
136             float remainingDuration = Float.parseFloat(Long.toString(hashBruter.getRemainder())) / hashBruter.getPerSecond();
137             this.append(this.bruteForceManager.getResult(), (
138                 I18nUtil.valueByKey("BRUTEFORCE_TRAVERSING_REMAINING") + ": "
139                 + Math.round(Math.floor(remainingDuration / 60f / 60.0f / 24f)) + I18nUtil.valueByKey("BRUTEFORCE_DAYS") + StringUtils.SPACE
140                 + Math.round(Math.floor(remainingDuration / 60f / 60f % 24))    + I18nUtil.valueByKey("BRUTEFORCE_HOURS") + StringUtils.SPACE
141                 + Math.round(Math.floor(remainingDuration / 60f % 60))          + I18nUtil.valueByKey("BRUTEFORCE_MINUTES") + StringUtils.SPACE
142                 + Math.round(remainingDuration % 60)                            + I18nUtil.valueByKey("BRUTEFORCE_SECONDS")
143             ));
144         }
145 
146         this.append(
147             this.bruteForceManager.getResult(),
148             String.format(
149                 "%s: %s%%",
150                 I18nUtil.valueByKey("BRUTEFORCE_PERCENT_DONE"),
151                 100 * (float) hashBruter.getCounter() / hashBruter.getNumberOfPossibilities()
152             )
153         );
154     }
155 
156     private void initializeBruter(final HashBruter hashBruter) {
157         hashBruter.setMinLength(Integer.parseInt(this.bruteForceManager.getMinimumLength().getValue().toString()));
158         hashBruter.setMaxLength(Integer.parseInt(this.bruteForceManager.getMaximumLength().getValue().toString()));
159 
160         if (this.bruteForceManager.getSpecialCharacters().isSelected()) {
161             hashBruter.addSpecialCharacters();
162         }
163         if (this.bruteForceManager.getUpperCaseCharacters().isSelected()) {
164             hashBruter.addUpperCaseLetters();
165         }
166         if (this.bruteForceManager.getLowerCaseCharacters().isSelected()) {
167             hashBruter.addLowerCaseLetters();
168         }
169         if (this.bruteForceManager.getNumericCharacters().isSelected()) {
170             hashBruter.addDigits();
171         }
172         if (StringUtils.isNotEmpty(this.bruteForceManager.getExclude().getText())) {
173             hashBruter.excludeChars(this.bruteForceManager.getExclude().getText());
174         }
175 
176         hashBruter.setType((String) this.bruteForceManager.getHashTypes().getSelectedItem());
177         hashBruter.setHash(
178             this.bruteForceManager.getHash().getText()
179             .toUpperCase(Locale.ROOT)
180             .replaceAll("[^a-zA-Z0-9]", StringUtils.EMPTY)
181             .trim()
182         );
183     }
184 
185     private void displayResult(final HashBruter hashBruter) {
186         // Display the result
187         if (this.isStopped) {
188             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> I18nUtil.valueByKey("BRUTEFORCE_ABORTED"));
189         } else if (hashBruter.isFound()) {
190             this.append(
191                 this.bruteForceManager.getResult(),
192                 String.format(
193                     "%n%s:%n%s => %s",
194                     I18nUtil.valueByKey("BRUTEFORCE_FOUND_HASH"),
195                     hashBruter.getGeneratedHash(),
196                     hashBruter.getPassword()
197                 )
198             );
199             LOGGER.log(
200                 LogLevelUtil.CONSOLE_SUCCESS,
201                 "{}: {} => {}",
202                 () -> I18nUtil.valueByKey("BRUTEFORCE_FOUND_HASH"),
203                 hashBruter::getGeneratedHash,
204                 hashBruter::getPassword
205             );
206         } else if (hashBruter.isDone()) {
207             this.append(this.bruteForceManager.getResult(), "\n"+ I18nUtil.valueByKey("BRUTEFORCE_HASH_NOT_FOUND"));
208             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> I18nUtil.valueByKey("BRUTEFORCE_HASH_NOT_FOUND"));
209         }
210     }
211     
212     public void append(JTextPane textPane, String text) {
213         try {
214             textPane.getDocument().insertString(
215                 textPane.getDocument().getLength(),
216                 (textPane.getDocument().getLength() == 0 ? StringUtils.EMPTY : "\n") + text,
217                 null
218             );
219         } catch (BadLocationException e) {
220             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
221         }
222     }
