30  /**
31   * Manager to display webpages frequently used as backoffice administration.
32   */
33  public class ManagerAdminPage extends AbstractManagerList {
34      
35      /**
36       * Log4j logger sent to view.
37       */
38      private static final Logger LOGGER = LogManager.getRootLogger();
39      
40      /**
41       * Create admin page finder.
42       */
43      public ManagerAdminPage() {
44          
45          super("swing/list/admin-page.txt");
46  
47          this.initializeRunButton();
48          
49          this.listFile.setName("listManagerAdminPage");
50          this.run.setName("runManagerAdminPage");
51          
52          this.lastLine.setLayout(new BorderLayout());
53          this.lastLine.setPreferredSize(new Dimension(0, 26));
54          
55          var panelRunButton = new JPanel();
56          panelRunButton.setLayout(new BoxLayout(panelRunButton, BoxLayout.X_AXIS));
57          
58          panelRunButton.add(Box.createHorizontalGlue());
59          panelRunButton.add(this.loader);
60          panelRunButton.add(Box.createRigidArea(new Dimension(5, 0)));
61          panelRunButton.add(this.run);
62          
63          this.lastLine.add(panelRunButton, BorderLayout.LINE_END);
64          
65          this.add(this.lastLine, BorderLayout.SOUTH);
66      }
67  
68      private void initializeRunButton() {
69          
70          this.defaultText = "ADMIN_PAGE_RUN_BUTTON_LABEL";
71          this.run = new JButtonStateful(this.defaultText);
72          I18nViewUtil.addComponentForKey("ADMIN_PAGE_RUN_BUTTON_LABEL", this.run);
73          this.run.setToolTipText(I18nUtil.valueByKey("ADMIN_PAGE_RUN_BUTTON_TOOLTIP"));
74          
75          this.run.setContentAreaFilled(false);
76          this.run.setBorder(BorderFactory.createEmptyBorder(4, 8, 4, 8));
77          this.run.setBackground(UiUtil.COLOR_FOCUS_GAINED);
78          
79          this.run.addMouseListener(new FlatButtonMouseAdapter(this.run));
80  
81          this.run.addActionListener(actionEvent -> this.runSearch());
82  
83          this.loader.setVisible(false);
84      }
85  
86      private void runSearch() {
87          
88          if (this.listFile.getSelectedValuesList().isEmpty()) {
89              
90              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select at least one admin page in the list");
91              return;
92          }
93          
94          String urlAddressBar = MediatorHelper.panelAddressBar().getTextFieldAddress().getText();
95          
96          if (!urlAddressBar.isEmpty() && !urlAddressBar.matches("(?i)^https?://.*")) {
97              
98              if (!urlAddressBar.matches("(?i)^\\w+://.*")) {
99                  
100                 LOGGER.log(LogLevelUtil.CONSOLE_INFORM, () -> I18nUtil.valueByKey("LOG_ADMIN_NO_PROTOCOL"));
101                 urlAddressBar = "http://"+ urlAddressBar;
102                 
103             } else {
104                 
105                 LOGGER.log(LogLevelUtil.CONSOLE_INFORM, () -> I18nUtil.valueByKey("LOG_ADMIN_UNKNOWN_PROTOCOL"));
106                 return;
107             }
108         }
109         
110         String urlFinal = urlAddressBar;
111         
112         new Thread(() -> this.searchAdminPages(urlFinal), "ThreadAdminPage").start();
113     }
114 
115     private void searchAdminPages(String urlAddressBar) {
116         
117         if (ManagerAdminPage.this.run.getState() == StateButton.STARTABLE) {
118             
119             if (StringUtils.isEmpty(urlAddressBar)) {
120                 
121                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Missing URL in address bar");
122                 
123             } else {
124                 
125                 LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking admin page(s)...");
126                 ManagerAdminPage.this.run.setText(I18nViewUtil.valueByKey("ADMIN_PAGE_RUN_BUTTON_STOP"));
127                 ManagerAdminPage.this.run.setState(StateButton.STOPPABLE);
128                 ManagerAdminPage.this.loader.setVisible(true);
129                 
130                 try {
131                     MediatorHelper.model().getResourceAccess().createAdminPages(
132                         urlAddressBar,
133                         this.listFile.getSelectedValuesList().stream().map(ItemList::toString).collect(Collectors.toList())
134                     );
135                     
136                 } catch (InterruptedException e) {
137                     
138                     LOGGER.log(LogLevelUtil.IGNORE, e, e);
139                     Thread.currentThread().interrupt();
140                 }
141             }
142             
143         } else if (this.run.getState() == StateButton.STOPPABLE) {
144             
145             MediatorHelper.model().getResourceAccess().setSearchAdminStopped(true);
146             ManagerAdminPage.this.run.setEnabled(false);
147             ManagerAdminPage.this.run.setState(StateButton.STOPPING);
148         }
149     }
