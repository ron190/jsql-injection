57      public ManagerScan() {
58          super("swing/list/scan-page.json");
59  
60          this.listPaths.setTransferHandler(null);
61          this.listPaths.setTransferHandler(new ListTransfertHandlerScan());
62          this.listPaths.setCellRenderer(new DefaultListCellRenderer() {
63              @Override
64              public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
65                  JLabel label = (JLabel) super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
66                  ItemList itemList = (ItemList) value;
67                  if (itemList.isVulnerable()) {
68                      label.setIcon(UiUtil.TICK_GREEN.getIcon());
69                  }
70                  return label;
71              }
72          });
73          
74          this.listPaths.setName(ManagerScan.NAME);
75  
76          JPanel lastLine = this.getLastLinePanel();
77          this.add(lastLine, BorderLayout.SOUTH);
78          
79          this.listPaths.addListSelectionListener(e -> {
80              if (this.listPaths.getSelectedValue() == null) {
81                  return;
82              }
83              
84              var beanInjection = ((ItemListScan) this.listPaths.getSelectedValue()).getBeanInjection();
85              MediatorHelper.panelAddressBar().getTextFieldAddress().setText(beanInjection.getUrl());
86              MediatorHelper.panelAddressBar().getTextFieldHeader().setText(beanInjection.getHeader());
87              MediatorHelper.panelAddressBar().getTextFieldRequest().setText(beanInjection.getRequest());
88              
89              String requestType = beanInjection.getRequestType();
90              if (requestType != null && !requestType.isEmpty()) {
91                  MediatorHelper.panelAddressBar().getAtomicRadioMethod().setText(requestType);
92              } else {
93                  MediatorHelper.panelAddressBar().getAtomicRadioMethod().setText(StringUtil.GET);
94              }
95              
96              AbstractMethodInjection method = beanInjection.getMethodInstance();
97              if (method == MediatorHelper.model().getMediatorMethod().getHeader()) {
98                  MediatorHelper.panelAddressBar().getAtomicRadioHeader().setSelected(true);
99              } else if (method == MediatorHelper.model().getMediatorMethod().getRequest()) {
100                 MediatorHelper.panelAddressBar().getAtomicRadioMethod().setSelected(true);
101             } else {
102                 MediatorHelper.panelAddressBar().getAtomicRadioRequest().setSelected(true);
103             }
104         });
105     }

107     @Override
108     public void buildList(String nameFile) {
109         var jsonScan = new StringBuilder();
110         try (
111             var inputStream = UiUtil.class.getClassLoader().getResourceAsStream(nameFile);
112             var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream), StandardCharsets.UTF_8);
113             var reader = new BufferedReader(inputStreamReader)
114         ) {
115             String line;
116             while ((line = reader.readLine()) != null) {
117                 jsonScan.append(line);
118             }
119             var jsonArrayScan = new JSONArray(jsonScan.toString());
120             for (var i = 0 ; i < jsonArrayScan.length() ; i++) {
121                 this.itemsList.add(new ItemListScan(jsonArrayScan.getJSONObject(i)));
122             }
123             this.listPaths = new DnDListScan(this.itemsList);
124         } catch (JSONException | IOException e) {
125             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
126         }
127     }

129     private JPanel getLastLinePanel() {
130         var lastLine = new JPanel();
131         lastLine.setOpaque(false);
132         lastLine.setLayout(new BoxLayout(lastLine, BoxLayout.X_AXIS));
133 
134         this.buildRunButton("SCAN_RUN_BUTTON_LABEL", "SCAN_RUN_BUTTON_TOOLTIP");
135         this.run.addActionListener(new ActionScan());
136 
137         lastLine.add(this.horizontalGlue);
138         lastLine.add(this.progressBar);
139         lastLine.add(this.run);
140         return lastLine;
141     }

143     private class ActionScan implements ActionListener {
144         @Override
145         public void actionPerformed(ActionEvent e) {
146             if (ManagerScan.this.listPaths.getSelectedValuesList().isEmpty()) {
147                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select URL(s) to scan");
148                 return;
149             }
150             new Thread(() -> {
151                 if (ManagerScan.this.run.getState() == StateButton.STARTABLE) {
152                     ManagerScan.this.run.setText(I18nViewUtil.valueByKey("SCAN_RUN_BUTTON_STOP"));
153                     ManagerScan.this.run.setState(StateButton.STOPPABLE);
154                     ManagerScan.this.progressBar.setVisible(true);
155                     ManagerScan.this.horizontalGlue.setVisible(false);
156                     DefaultListModel<ItemList> listModel = (DefaultListModel<ItemList>) ManagerScan.this.listPaths.getModel();
157                     for (var i = 0 ; i < listModel.getSize() ; i++) {
158                         listModel.get(i).reset();
159                     }
160                     ManagerScan.this.scan(ManagerScan.this.listPaths.getSelectedValuesList());
161                 } else {
162                     MediatorHelper.model().getResourceAccess().setScanStopped(true);
163                     MediatorHelper.model().setIsStoppedByUser(true);
164                     ManagerScan.this.run.setEnabled(false);
165                     ManagerScan.this.run.setState(StateButton.STOPPING);
166                 }
167             }, "ThreadScan").start();
168         }
169     }

171     /**
172      * Start fast scan of URLs in sequence and display result.
173      * Unplug any existing view and plug a console-like view in order to
174      * respond appropriately to GUI message with simple text result instead of
175      * build complex graphical components during the multi website injections.
176      * At the end of the scan it plugs again the default view.
177      * @param urlsItemList contains a list of String URL
178      */
179     public void scan(List<ItemList> urlsItemList) {
180         MediatorHelper.frame().resetInterface();  // Erase everything in the view from a previous injection
181         
182         // wait for ending of ongoing interaction between two injections
183         try {
184             Thread.sleep(500);
185         } catch (InterruptedException e) {
186             LOGGER.log(LogLevelUtil.IGNORE, e, e);
187             Thread.currentThread().interrupt();
188         }
189 
190         // Display result only in console
191         var requestUnsubscribe = new Request();
192         requestUnsubscribe.setMessage(Interaction.UNSUBSCRIBE);
193         MediatorHelper.model().sendToViews(requestUnsubscribe);
194         MediatorHelper.model().subscribe(new ScanListTerminal());
195         
196         MediatorHelper.model().setIsScanning(true);
197         MediatorHelper.model().getResourceAccess().setScanStopped(false);
198         
199         for (ItemList urlItemList: urlsItemList) {
200             if (  // detect interrupt by user to end intermediate scan
201                 MediatorHelper.model().isStoppedByUser()
202                 || MediatorHelper.model().getResourceAccess().isScanStopped()
203             ) {
204                 break;
205             }
206 
207             var urlItemListScan = (ItemListScan) urlItemList;
208             LOGGER.log(LogLevelUtil.CONSOLE_INFORM, "Scanning {}", urlItemListScan.getBeanInjection().getUrl());
209 
210             Optional<Vendor> optionalVendor = MediatorHelper.model().getMediatorVendor().getVendors()
211                 .stream()
212                 .filter(vendor -> vendor.toString().equalsIgnoreCase(urlItemListScan.getBeanInjection().getVendor()))
213                 .findAny();
214 
215             MediatorHelper.model().getMediatorVendor().setVendorByUser(
216                 optionalVendor.orElse(MediatorHelper.model().getMediatorVendor().getAuto())
217             );
218             MediatorHelper.model().getMediatorUtils().getParameterUtil().controlInput(
219                 urlItemListScan.getBeanInjection().getUrl(),
220                 urlItemListScan.getBeanInjection().getRequest(),
221                 urlItemListScan.getBeanInjection().getHeader(),
222                 urlItemListScan.getBeanInjection().getMethodInstance(),
223                 urlItemListScan.getBeanInjection().getRequestType(),
224                 true
225             );
226             
227             try {
228                 Thread.sleep(500);
229             } catch (InterruptedException e) {
230                 LOGGER.log(LogLevelUtil.IGNORE, e, e);
231                 Thread.currentThread().interrupt();
232             }
233         }
234         
235         // Get back the default view
236         MediatorHelper.model().sendToViews(requestUnsubscribe);
237         MediatorHelper.model().subscribe(MediatorHelper.frame().getSubscriber());
238         
239         MediatorHelper.model().setIsScanning(false);
240         MediatorHelper.model().setIsStoppedByUser(false);
241         MediatorHelper.model().getResourceAccess().setScanStopped(false);
242 
243         this.endProcess();
244     }
