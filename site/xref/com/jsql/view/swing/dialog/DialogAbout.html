36  /**
37   * A dialog displaying information about jSQL.
38   */
39  public class DialogAbout extends JDialog {
40      
41      /**
42       * Log4j logger sent to view.
43       */
44      private static final Logger LOGGER = LogManager.getRootLogger();
45  
46      /**
47       * Button receiving focus.
48       */
49      private JButton buttonClose = null;
50  
51      /**
52       * Create a dialog about project general information.
53       */
54      public DialogAbout() {
55          super(MediatorHelper.frame(), I18nUtil.valueByKey("ABOUT_WINDOW_TITLE") +" "+ StringUtil.APP_NAME, Dialog.ModalityType.MODELESS);
56          I18nViewUtil.addComponentForKey("ABOUT_WINDOW_TITLE", this);
57          this.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
58          this.setIconImages(UiUtil.getIcons());  // Define a small and large app icon
59  
60          ActionListener escapeListener = actionEvent -> this.dispose();  // Action for ESCAPE key
61          this.getRootPane().registerKeyboardAction(
62              escapeListener,
63              KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0),
64              JComponent.WHEN_IN_FOCUSED_WINDOW
65          );
66          
67          this.setLayout(new BorderLayout());
68  
69          Container dialogPane = this.getContentPane();
70          JPanel lastLine = this.initializeLastLine(escapeListener);
71  
72          var labelIcon = new JLabel(UiUtil.APP_ABOUT.icon);
73          labelIcon.setBorder(BorderFactory.createEmptyBorder(10,10,10,10));
74          dialogPane.add(labelIcon, BorderLayout.WEST);
75          dialogPane.add(lastLine, BorderLayout.SOUTH);
76  
77          final JEditorPane text = this.initializeEditorPane();  // Contact info, use HTML text
78          dialogPane.add(new JScrollPane(text), BorderLayout.CENTER);
79  
80          this.initializeDialog();
81      }
82  
83      private JPanel initializeLastLine(ActionListener escapeListener) {
84          final var buttonWebpage = new JButton(I18nUtil.valueByKey("ABOUT_WEBPAGE"));
85          I18nViewUtil.addComponentForKey("ABOUT_WEBPAGE", buttonWebpage);
86          buttonWebpage.addActionListener(ev -> {
87              try {
88                  Desktop.getDesktop().browse(new URI(MediatorHelper.model().getMediatorUtils().getPropertiesUtil().getProperties().getProperty("github.url")));
89              } catch (IOException | URISyntaxException | UnsupportedOperationException e) {
90                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Browsing to Url failed", e);
91              }
92          });
93  
94          this.buttonClose = new JButton(I18nUtil.valueByKey("ABOUT_CLOSE"));
95          I18nViewUtil.addComponentForKey("ABOUT_CLOSE", this.buttonClose);
96          this.buttonClose.addActionListener(escapeListener);
97  
98          var lastLine = new JPanel();
99          lastLine.setLayout(new BoxLayout(lastLine, BoxLayout.LINE_AXIS));
100         lastLine.setBorder(UiUtil.BORDER_5PX);
101         lastLine.add(buttonWebpage);
102         lastLine.add(Box.createGlue());
103         lastLine.add(this.buttonClose);
104         return lastLine;
105     }
106 
107     private JEditorPane initializeEditorPane() {
108         var editorPane = new JEditorPane();
109         
110         // Fix #82540: NoClassDefFoundError on setText()
111         try (
112             InputStream inputStream = DialogAbout.class.getClassLoader().getResourceAsStream("swing/about.htm");
113             var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream), StandardCharsets.UTF_8);
114             var reader = new BufferedReader(inputStreamReader)
115         ) {
116             editorPane.setContentType("text/html");
117             var result = new StringBuilder();
118             String line;
119             while ((line = reader.readLine()) != null) {
120                 result.append(line);
121             }
122             editorPane.setText(result.toString().replace(
123                 "%JSQLVERSION%",
124                 MediatorHelper.model().getPropertiesUtil().getVersionJsql()
125             ));
126         } catch (NoClassDefFoundError | IOException e) {
127             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
128         }
129 
130         editorPane.addMouseListener(new MouseAdapter() {
131             @Override
132             public void mousePressed(MouseEvent e) {
133                 super.mousePressed(e);
134                 editorPane.requestFocusInWindow();
135             }
136         });
137         editorPane.addFocusListener(new FocusAdapter() {
138             @Override
139             public void focusGained(FocusEvent focusEvent) {
140                 editorPane.getCaret().setVisible(true);
141                 editorPane.getCaret().setSelectionVisible(true);
142             }
143         });
144 
145         editorPane.setDragEnabled(true);
146         editorPane.setEditable(false);
147         editorPane.getCaret().setBlinkRate(0);
148         editorPane.setCaretPosition(0);
149         editorPane.setComponentPopupMenu(new JPopupMenuText(editorPane));
150         editorPane.addHyperlinkListener(linkEvent -> {
151             if (HyperlinkEvent.EventType.ACTIVATED.equals(linkEvent.getEventType())) {
152                 try {
153                     Desktop.getDesktop().browse(linkEvent.getURL().toURI());
154                 } catch (IOException | URISyntaxException | UnsupportedOperationException e) {
155                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Browsing to Url failed", e);
156                 }
157             }
158         });
159         
160         return editorPane;
161     }
162 
163     /**
164      * Set back default setting for About frame.
165      */
166     public final void initializeDialog() {
167         this.setSize(533, 400);
168         this.setLocationRelativeTo(MediatorHelper.frame());
169         this.buttonClose.requestFocusInWindow();
170         this.getRootPane().setDefaultButton(this.buttonClose);
171     }

174         this.buttonClose.requestFocusInWindow();
175     }
