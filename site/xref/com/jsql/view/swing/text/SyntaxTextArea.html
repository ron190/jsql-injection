18  public class SyntaxTextArea extends RSyntaxTextArea {
19  
20      /**
21       * Log4j logger sent to view.
22       */
23      private static final Logger LOGGER = LogManager.getRootLogger();
24  
25      private String placeholderText;
26  
27      public SyntaxTextArea() {
28          this(StringUtils.EMPTY);
29      }
30  
31      public SyntaxTextArea(String text) {
32          this.placeholderText = text;
33  
34          this.setPopupMenu(new JPopupMenuComponent(this));
35          this.getCaret().setBlinkRate(0);
36          this.addFocusListener(new FocusAdapter() {
37              @Override
38              public void focusGained(FocusEvent focusEvent) {
39                  SyntaxTextArea.this.getCaret().setVisible(true);
40                  SyntaxTextArea.this.getCaret().setSelectionVisible(true);
41              }
42          });
43          this.setCursor(Cursor.getPredefinedCursor(Cursor.TEXT_CURSOR));
44          this.setSyntaxEditingStyle(SyntaxConstants.SYNTAX_STYLE_SQL);
45          this.setMarkOccurrences(true);
46          this.setMarkOccurrencesDelay(200);
47      }
48  
49      @Override
50      public void paint(Graphics g) {
51          // Fix #6350: ArrayIndexOutOfBoundsException on paint()
52          // Fix #90822: IllegalArgumentException on paint()
53          // Fix #90761: StateInvariantError on paint()
54          // StateInvariantError possible on jdk 8 when WrappedPlainView.drawLine in paint()
55          try {
56              super.paint(g);
57              if (StringUtils.isEmpty(this.getText()) && StringUtils.isNotEmpty(this.placeholderText)) {
58                  UiUtil.drawPlaceholder(this, g, this.placeholderText);
59              }
60          } catch (IllegalArgumentException | NullPointerException | ArrayIndexOutOfBoundsException e) {
61              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
62          }
63      }
64  
65      @Override
66      public Font getFont() {
67          return UiUtil.FONT_MONO_NON_ASIAN;
68      }
69      @Override
70      public Font getFontForToken(Token token) {
71          return UiUtil.FONT_MONO_NON_ASIAN;
72      }
73      @Override
74      public Font getFontForTokenType(int type) {
75          return UiUtil.FONT_MONO_NON_ASIAN;
76      }
77  
78      public void setPlaceholderText(String placeholderText) {
79          this.placeholderText = placeholderText;
80      }
81  }
