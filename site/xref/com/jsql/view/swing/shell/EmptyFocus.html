1   package com.jsql.view.swing.shell;

3   import java.awt.event.MouseEvent;
4   import java.awt.event.MouseListener;
5   
6   /**
7    * Cancel every mouse click, only gives focus.
8    */
9   public class EmptyFocus implements MouseListener {
10      
11      private final AbstractShell abstractShell;
12      
13      public EmptyFocus(AbstractShell abstractShell) {
14          this.abstractShell = abstractShell;
15      }
16  
17      @Override
18      public void mousePressed(MouseEvent e) {
19          e.consume();
20          this.abstractShell.requestFocusInWindow();
21          this.abstractShell.setCaretPosition(this.abstractShell.getDocument().getLength());
22      }
23      
24      @Override
25      public void mouseReleased(MouseEvent e) {
26          e.consume();
27      }
28      
29      @Override
30      public void mouseExited(MouseEvent e) {
31          e.consume();
32      }
33      
34      @Override
35      public void mouseEntered(MouseEvent e) {
36          e.consume();
37      }
38      
39      @Override
40      public void mouseClicked(MouseEvent e) {
41          e.consume();
42      }
43  }
