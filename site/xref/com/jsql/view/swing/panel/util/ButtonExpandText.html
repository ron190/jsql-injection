11  package com.jsql.view.swing.panel.util;
12  
13  import com.jsql.view.swing.text.JPopupTextArea;
14  import com.jsql.view.swing.text.JTextAreaPlaceholder;
15  import com.jsql.view.swing.util.MediatorHelper;
16  import com.jsql.view.swing.util.UiUtil;
17  
18  import javax.swing.*;
19  import java.awt.*;
20  import java.awt.event.*;
21  
22  /**
23   * A button displayed in address.
24   */
25  public class ButtonExpandText extends JButton {
26  
27      /**
28       * Create a button in address bar.
29       */
30      public ButtonExpandText(String titleFrame, JTextField textFieldHeader) {
31          this.setPreferredSize(new Dimension(16, 16));
32          this.setContentAreaFilled(false);
33  
34          this.setIcon(UiUtil.EXPAND.icon);
35          this.setRolloverIcon(UiUtil.EXPAND_HOVER.icon);
36          this.setPressedIcon(UiUtil.EXPAND_PRESSED.icon);
37  
38          JTextArea textArea = new JPopupTextArea(new JTextAreaPlaceholder("Multiline text")).getProxy();
39          textArea.getCaret().setBlinkRate(500);
40  
41          final JDialog dialogWithTextarea = new JDialog(MediatorHelper.frame(), titleFrame, true);
42          dialogWithTextarea.getContentPane().add(new JScrollPane(textArea));
43          dialogWithTextarea.pack();
44          dialogWithTextarea.setSize(400, 300);
45          dialogWithTextarea.setLocationRelativeTo(null);
46          dialogWithTextarea.addWindowListener(new WindowAdapter() {
47              @Override
48              public void windowClosing(WindowEvent e) {
49                  textFieldHeader.setText(textArea.getText().replace("\n", "\\n").replace("\r", "\\r"));
50                  super.windowClosing(e);
51              }
52          });
53  
54          this.addMouseListener(new MouseAdapter() {
55              @Override
56              public void mouseClicked(MouseEvent e) {
57                  textArea.setText(textFieldHeader.getText().replace("\\n", "\n").replace("\\r", "\r"));
58                  dialogWithTextarea.setVisible(!dialogWithTextarea.isVisible());
59              }
60          });
61  
62          dialogWithTextarea.getRootPane()
63              .getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW)
64              .put(
65                  KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0),
66                  "Cancel"
67              );
68          dialogWithTextarea.getRootPane()
69              .getActionMap()
70              .put("Cancel", new AbstractAction() {
71                  public void actionPerformed(ActionEvent e) {
72                      dialogWithTextarea.dispose();
73                  }
74              });
75      }
76  }
