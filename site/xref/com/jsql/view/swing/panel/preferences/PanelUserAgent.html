28      private final JCheckBox checkboxIsCustomUserAgent = new JCheckBox("Randomize agent with the following list :", MediatorHelper.model().getMediatorUtils().getPreferencesUtil().isUserAgentRandom());
29  
30      public PanelUserAgent(PanelPreferences panelPreferences) {
31          var userAgents = new StringBuilder();
32          try (
33              var inputStream = UiUtil.class.getClassLoader().getResourceAsStream("swing/list/user-agent.txt");
34              var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream), StandardCharsets.UTF_8);
35              var reader = new BufferedReader(inputStreamReader)
36          ) {
37              String line;
38              while ((line = reader.readLine()) != null) {
39                  userAgents.append(line).append(System.lineSeparator());
40              }
41          } catch (IOException e) {
42              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
43          }
44  
45          this.checkboxIsCustomUserAgent.addActionListener(panelPreferences.getActionListenerSave());
46          JTextArea textfieldCustomUserAgent = new JPopupTextArea(new JTextAreaPlaceholder("User agent list")).getProxy();
47          textfieldCustomUserAgent.setMinimumSize(new Dimension(40000, 100));
48          textfieldCustomUserAgent.getCaret().setBlinkRate(500);
49          textfieldCustomUserAgent.setText(userAgents.toString());
50          MediatorHelper.model().getMediatorUtils().getUserAgentUtil().setCustomUserAgent(userAgents.toString());
51          textfieldCustomUserAgent.getDocument().addDocumentListener(new DocumentListenerEditing() {
52              @Override
53              public void process() {
54                  MediatorHelper.model().getMediatorUtils().getUserAgentUtil().setCustomUserAgent(
55                      textfieldCustomUserAgent.getText()
56                  );
57              }
58          });
59          var scrollPane = new JScrollPane(textfieldCustomUserAgent);
60  
61          var labelOrigin = new JLabel("<html><b>Connection user agent</b></html>");
62          labelOrigin.setBorder(PanelGeneral.MARGIN);
63  
64          var groupLayout = new GroupLayout(this);
65          this.setLayout(groupLayout);
66  
67          groupLayout.setHorizontalGroup(
68              groupLayout
69              .createSequentialGroup()
70              .addGroup(
71                  groupLayout
72                  .createParallelGroup(GroupLayout.Alignment.LEADING)
73                  .addComponent(labelOrigin)
74                  .addComponent(this.checkboxIsCustomUserAgent)
75                  .addComponent(scrollPane)
76              )
77          );
78  
79          groupLayout.setVerticalGroup(
80              groupLayout
81              .createSequentialGroup()
82              .addGroup(
83                  groupLayout
84                  .createParallelGroup(GroupLayout.Alignment.BASELINE)
85                  .addComponent(labelOrigin)
86              )
87              .addGroup(
88                  groupLayout
89                  .createParallelGroup(GroupLayout.Alignment.BASELINE)
90                  .addComponent(this.checkboxIsCustomUserAgent)
91              )
92              .addGroup(
93                  groupLayout
94                  .createParallelGroup(GroupLayout.Alignment.BASELINE)
95                  .addComponent(scrollPane)
96              )
97          );
98      }
