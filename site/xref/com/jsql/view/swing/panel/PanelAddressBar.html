166     private void initLayout() {
167         final JLabel advancedButton = this.initAdvancedButton();
168         
169         this.setLayout(new BoxLayout(this, BoxLayout.PAGE_AXIS));
170         
171         // First panel at the top, contains text components
172         var panelTextFields = new JPanel();
173         var groupLayout = new GroupLayout(panelTextFields);
174         panelTextFields.setLayout(groupLayout);
175         panelTextFields.setBorder(BorderFactory.createEmptyBorder(2, 2, 2, 0));
176         this.add(panelTextFields);
177 
178         final var popup = new JPopupMenu();
179         final var buttonGroupMethod = new ButtonGroup();
180 
181         for (String method: new String[]{"DELETE", StringUtil.GET, "HEAD", "OPTIONS", StringUtil.POST, "PUT", "TRACE"}) {
182             final JMenuItem newMenuItem = new JRadioButtonMenuItem(method, StringUtil.GET.equals(method));
183             newMenuItem.addActionListener(actionEvent -> {
184                 this.typeRequest = newMenuItem.getText();
185                 this.atomicRadioMethod.get().setText(this.typeRequest);
186                 this.atomicRadioMethod.get().requestFocusInWindow();  // required to set proper focus
187             });
188             popup.add(newMenuItem);
189             buttonGroupMethod.add(newMenuItem);
190         }
191 
192         var tooltipPanel = new AtomicReference<>(new JToolTipI18n(I18nUtil.valueByKey("METHOD_CUSTOM_TOOLTIP")));
193         var panelCustomMethod = new JPanel(new BorderLayout()) {
194             @Override
195             public JToolTip createToolTip() {
196                 return tooltipPanel.get();
197             }
198         };
199         I18nViewUtil.addComponentForKey("METHOD_CUSTOM_TOOLTIP", tooltipPanel.get());
200         Supplier<Color> colorBackground = () -> UIManager.getColor("MenuItem.background");  // adapt to current theme
201         Supplier<Color> colorSelectionBackground = () -> UIManager.getColor("MenuItem.selectionBackground");  // adapt to current theme
202         panelCustomMethod.setBackground(colorBackground.get());  // required for correct color
203 
204         final var radioCustomMethod = new JRadioButton() {
205             @Override
206             public JToolTip createToolTip() {
207                 return tooltipPanel.get();
208             }
209         };
210         radioCustomMethod.setBorder(BorderFactory.createEmptyBorder(0, 6, 0, 0));
211         radioCustomMethod.setIcon(new FlatRadioButtonMenuItemIcon());
212         radioCustomMethod.setBackground(colorBackground.get());  // required for correct color
213         buttonGroupMethod.add(radioCustomMethod);
214 
215         final JTextField inputCustomMethod = new JPopupTextField("CUSTOM"){
216             @Override
217             public JToolTip createToolTip() {
218                 return tooltipPanel.get();
219             }
220         }.getProxy();
221         inputCustomMethod.addMouseListener(new MouseAdapter() {
222             @Override
223             public void mouseClicked(MouseEvent e) {
224                 radioCustomMethod.setSelected(!radioCustomMethod.isSelected());
225             }
226         });
227         inputCustomMethod.getDocument().addDocumentListener(new DocumentListenerEditing() {
228             @Override
229             public void process() {
230                 PanelAddressBar.this.validate(inputCustomMethod);
231             }
232         });
233         radioCustomMethod.addActionListener(actionEvent -> this.validate(inputCustomMethod));
234 
235         var tooltipCustomMethod = "<html>Set user defined HTTP method.<br/>" +
236             "A valid method is limited to chars:<br>" +
237             "!#$%&'*+-.^_`|~0123456789<br>" +
238             "abcdefghijklmnopqrstuvwxyz<br>" +
239             "ABCDEFGHIJKLMNOPQRSTUVWXYZ" +
240         "</html>";
241         MouseAdapter mouseAdapterSetBackground = new MouseAdapter() {
242             @Override
243             public void mouseEntered(MouseEvent e) {
244                 super.mouseEntered(e);
245                 panelCustomMethod.setBackground(colorSelectionBackground.get());
246                 radioCustomMethod.setBackground(colorSelectionBackground.get());
247             }
248             @Override
249             public void mouseExited(MouseEvent e) {
250                 super.mouseExited(e);
251                 panelCustomMethod.setBackground(colorBackground.get());
252                 radioCustomMethod.setBackground(colorBackground.get());
253             }
254         };
255         Arrays.asList(radioCustomMethod, inputCustomMethod, panelCustomMethod).forEach(component -> {
256             component.addMouseListener(mouseAdapterSetBackground);
257             component.setToolTipText(tooltipCustomMethod);
258         });
259 
260         panelCustomMethod.add(radioCustomMethod, BorderLayout.LINE_START);
261         panelCustomMethod.add(inputCustomMethod, BorderLayout.CENTER);
262         popup.insert(panelCustomMethod, popup.getComponentCount());
263 
264         this.atomicRadioMethod.get().addMouseListener(new MouseAdapter() {
265             @Override
266             public void mousePressed(MouseEvent e) {
267                 Arrays.stream(popup.getComponents()).map(a -> (JComponent) a).forEach(JComponent::updateUI);  // required: incorrect when dark/light mode switch
268                 radioCustomMethod.setIcon(new FlatRadioButtonMenuItemIcon());
269                 radioCustomMethod.updateUI();  // required: incorrect when dark/light mode switch
270                 inputCustomMethod.updateUI();  // required: incorrect when dark/light mode switch
271                 popup.updateUI();  // required: incorrect when dark/light mode switch
272                 popup.applyComponentOrientation(ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale()));
273 
274                 if (ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale()))) {
275                     radioCustomMethod.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 6));
276                 } else {
277                     radioCustomMethod.setBorder(BorderFactory.createEmptyBorder(0, 6, 0, 0));
278                 }
279 
280                 // Fix #96032: NullPointerException on show()
281                 try {
282                     popup.show(
283                         e.getComponent(),
284                         ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale()))
285                             ? e.getComponent().getX() - e.getComponent().getWidth() - popup.getWidth()
286                             : e.getComponent().getX(),
287                         e.getComponent().getY() + e.getComponent().getHeight()
288                     );
289                     popup.setLocation(  // required for proper location
290                         ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale()))
291                             ? e.getComponent().getLocationOnScreen().x + e.getComponent().getWidth() - popup.getWidth()
292                             : e.getComponent().getLocationOnScreen().x,
293                         e.getComponent().getLocationOnScreen().y + e.getComponent().getHeight()
294                     );
295                 } catch (NullPointerException ex) {
296                     LOGGER.log(LogLevelUtil.CONSOLE_JAVA, ex);
297                 }
298             }
299         });

350     private void validate(JTextField inputCustomMethod) {
351         if (StringUtils.isEmpty(inputCustomMethod.getText())) {
352             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Missing custom method label");
353         } else if (ParameterUtil.isInvalidName(inputCustomMethod.getText())) {
354             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> String.format("Illegal method: \"%s\"", inputCustomMethod.getText()));
355         } else {
356             this.typeRequest = inputCustomMethod.getText();
357             this.atomicRadioMethod.get().setText(this.typeRequest);
358         }
359     }

361     private JLabel initAdvancedButton() {
362         var tooltip = new AtomicReference<>(new JToolTipI18n(I18nUtil.valueByKey(PanelAddressBar.BUTTON_ADVANCED)));
363         var advancedButton = new JLabel(UiUtil.ARROW_DOWN.getIcon()) {
364             @Override
365             public JToolTip createToolTip() {
366                 return tooltip.get();
367             }
368         };
369         advancedButton.setName("advancedButton");
370         advancedButton.setToolTipText(I18nUtil.valueByKey(PanelAddressBar.BUTTON_ADVANCED));
371         I18nViewUtil.addComponentForKey(PanelAddressBar.BUTTON_ADVANCED, tooltip.get());
372         advancedButton.addMouseListener(new MouseAdapter() {
373             @Override
374             public void mouseClicked(MouseEvent e) {
375                 boolean isVisible = advancedButton.getIcon() == UiUtil.ARROW_DOWN.getIcon();
376                 PanelAddressBar.this.atomicTextFieldRequest.get().setVisible(isVisible);
377                 PanelAddressBar.this.atomicTextFieldHeader.get().setVisible(isVisible);
378                 PanelAddressBar.this.atomicRadioRequest.get().setVisible(isVisible);
379                 PanelAddressBar.this.atomicRadioMethod.get().setVisible(isVisible);
380                 PanelAddressBar.this.atomicRadioHeader.get().setVisible(isVisible);
381                 PanelAddressBar.this.isAdvanceActivated = isVisible;
382                 MediatorHelper.menubar().setVisible(isVisible);
383                 advancedButton.setIcon(isVisible ? UiUtil.ARROW_UP.getIcon() : UiUtil.ARROW_DOWN.getIcon());
384             }
385         });
386         return advancedButton;
387     }
