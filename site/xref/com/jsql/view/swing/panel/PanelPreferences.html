27      private enum CategoryPreference {
28          INJECTION,
29          TAMPERING,
30          CONNECTION,
31          STRATEGIES,
32          AUTHENTICATION,
33          USER_AGENT,
34          PROXY,
35          GENERAL;
36  
37          @Override
38          public String toString() {
39              return "  "+ WordUtils.capitalizeFully(this.name()).replace('_', ' ') +"  ";
40          }
41      }
42      
43      public PanelPreferences() {
44          this.setLayout(new BorderLayout());
45          this.setBorder(UiUtil.BORDER_5PX);
46  
47          var cards = new JPanel(new CardLayout());
48          cards.setMinimumSize(new Dimension(0, 0));  // required
49  
50          JList<CategoryPreference> categories = PanelPreferences.getCategories(cards);
51          this.add(categories, BorderLayout.LINE_START);
52  
53          this.addToCard(cards, this.panelInjection, CategoryPreference.INJECTION);
54          this.addToCard(cards, this.panelTampering, CategoryPreference.TAMPERING);
55          this.addToCard(cards, this.panelConnection, CategoryPreference.CONNECTION);
56          this.addToCard(cards, this.panelStrategies, CategoryPreference.STRATEGIES);
57          this.addToCard(cards, this.panelAuthentication, CategoryPreference.AUTHENTICATION);
58          this.addToCard(cards, this.panelUserAgent, CategoryPreference.USER_AGENT);
59          this.addToCard(cards, this.panelProxy, CategoryPreference.PROXY);
60          this.addToCard(cards, this.panelGeneral, CategoryPreference.GENERAL);
61          this.add(cards, BorderLayout.CENTER);
62  
63          SwingUtilities.invokeLater(() -> MediatorHelper.menubar().switchLocale(I18nUtil.getCurrentLocale()));  // required for arabic
64      }
65  
66      private static JList<CategoryPreference> getCategories(JPanel cards) {
67          JList<CategoryPreference> categories = new JList<>(CategoryPreference.values());
68          categories.setMinimumSize(new Dimension(0, 0));
69          categories.setName("listCategoriesPreference");
70          categories.setSelectionMode(ListSelectionModel.SINGLE_SELECTION);
71          categories.setSelectedIndex(0);
72          categories.addListSelectionListener(e -> {
73              CardLayout cardLayout = (CardLayout) cards.getLayout();
74              cardLayout.show(cards, categories.getSelectedValue().name());
75          });
76          return categories;
77      }
78  
79      private void addToCard(JPanel cards, JPanel panel, CategoryPreference category) {
80          panel.setBorder(BorderFactory.createEmptyBorder(10, 15, 10, 15));
81          var scrollPane = new JScrollPane(panel);
82          scrollPane.setBorder(BorderFactory.createEmptyBorder());  // required to hide border
83          scrollPane.getVerticalScrollBar().setUnitIncrement(16);
84          scrollPane.getHorizontalScrollBar().setUnitIncrement(16);
85          cards.add(scrollPane, category.name());
86      }
