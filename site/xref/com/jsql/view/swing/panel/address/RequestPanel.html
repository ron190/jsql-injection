8   import org.apache.commons.lang3.StringUtils;
9   import org.apache.logging.log4j.LogManager;
10  import org.apache.logging.log4j.Logger;
11  
12  import javax.swing.*;
13  import javax.swing.plaf.basic.BasicArrowButton;
14  import javax.swing.plaf.basic.BasicRadioButtonMenuItemUI;
15  import java.awt.*;
16  import java.awt.event.MouseAdapter;
17  import java.awt.event.MouseEvent;
18  import java.util.Collections;
19  
20  public class RequestPanel extends JPanel {
21      
22      /**
23       * Log4j logger sent to view.
24       */
25      private static final Logger LOGGER = LogManager.getRootLogger();
26  
27      private String typeRequest = "GET";
28  
29      public RequestPanel(PanelAddressBar panelAddressBar) {
30          
31          this.setLayout(new BoxLayout(this, BoxLayout.LINE_AXIS));
32          this.setMaximumSize(new Dimension(Integer.MAX_VALUE, 16));
33          this.setBorder(null);
34          
35          JButton buttonRequestMethod = new BasicArrowButton(SwingConstants.SOUTH);
36          buttonRequestMethod.setBorderPainted(false);
37          buttonRequestMethod.setOpaque(false);
38          
39          this.add(buttonRequestMethod);
40          this.add(panelAddressBar.getRadioRequest());
41          
42          final var popup = new JPopupMenu();
43          final var buttonGroup = new ButtonGroup();
44          
45          for (String protocol: new String[]{"DELETE", "GET", "HEAD", "OPTIONS", "POST", "PUT", "TRACE"}) {
46              
47              final JMenuItem newMenuItem = new JRadioButtonMenuItem(protocol, "GET".equals(protocol));
48              
49              newMenuItem.addActionListener(actionEvent -> {
50                  
51                  this.typeRequest = (newMenuItem.getText());
52                  panelAddressBar.getRadioRequest().setText(this.typeRequest);
53              });
54              
55              popup.add(newMenuItem);
56              buttonGroup.add(newMenuItem);
57          }
58          
59          for (AbstractButton radioButton: Collections.list(buttonGroup.getElements())) {
60              
61              radioButton.setUI(
62                  new BasicRadioButtonMenuItemUI() {
63                      
64                      @Override
65                      protected void doClick(MenuSelectionManager msm) {
66                          
67                          this.menuItem.doClick(0);
68                      }
69                  }
70              );
71          }
72          
73          var panelCustomMethod = new JPanel(new BorderLayout());
74          final JTextField inputCustomMethod = new JPopupTextField("CUSTOM").getProxy();
75  
76          final var radioCustomMethod = new JRadioButton();
77          radioCustomMethod.setBorder(BorderFactory.createEmptyBorder(0, 6, 0, 0));
78          radioCustomMethod.setIcon(new RadioMenuItemIconCustom());
79          
80          buttonGroup.add(radioCustomMethod);
81          
82          radioCustomMethod.addActionListener(actionEvent -> {
83              
84              if (StringUtils.isNotEmpty(inputCustomMethod.getText())) {
85                  
86                  this.typeRequest = (inputCustomMethod.getText());
87                  panelAddressBar.getRadioRequest().setText(this.typeRequest);
88                  
89              } else {
90                  
91                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Define label of Custom request method");
92              }
93          });
94        
95          panelCustomMethod.add(radioCustomMethod, BorderLayout.LINE_START);
96          panelCustomMethod.add(inputCustomMethod, BorderLayout.CENTER);
97          popup.insert(panelCustomMethod, popup.getComponentCount());
98          
99          buttonRequestMethod.addMouseListener(new MouseAdapter() {
100             
101             @Override
102             public void mousePressed(MouseEvent e) {
103                 
104                 popup.applyComponentOrientation(ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault()));
105                 
106                 if (ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault()))) {
107                     
108                     radioCustomMethod.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 6));
109                     
110                 } else {
111                     
112                     radioCustomMethod.setBorder(BorderFactory.createEmptyBorder(0, 6, 0, 0));
113                 }
114                 
115                 popup.show(
116                     e.getComponent(),
117                     ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault()))
118                     ? e.getComponent().getX() - e.getComponent().getWidth() - popup.getWidth()
119                     : e.getComponent().getX(),
120                     e.getComponent().getY() + e.getComponent().getWidth()
121                 );
122                 
123                 popup.setLocation(
124                     ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault()))
125                     ? e.getComponent().getLocationOnScreen().x + e.getComponent().getWidth() - popup.getWidth()
126                     : e.getComponent().getLocationOnScreen().x,
127                     e.getComponent().getLocationOnScreen().y + e.getComponent().getWidth()
128                 );
129             }
130         });
131     }
132 
133     public String getTypeRequest() {
134         return this.typeRequest;
135     }
136 
137     public void setTypeRequest(String typeRequest) {
138         this.typeRequest = typeRequest;
139     }
140 }
