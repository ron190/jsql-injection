46      public PanelTrailingAddress(PanelAddressBar panelAddressBar) {
47          this.buttonStart.addActionListener(new ActionStart(panelAddressBar));
48          this.setOpaque(false);
49          this.setBorder(null);
50          this.labelStrategy.setText("Strategy auto");
51          this.labelStrategy.setName("menuStrategy");
52  
53          for (final AbstractStrategy strategy: MediatorHelper.model().getMediatorStrategy().getStrategies()) {
54              var nameStrategy = strategy.getName().toUpperCase(Locale.ROOT);
55              JMenuItem itemRadioStrategy;
56              if (strategy == MediatorHelper.model().getMediatorStrategy().getError()) {
57                  itemRadioStrategy = new JMenu(strategy.toString());
58                  this.itemRadioStrategyError = (JMenu) itemRadioStrategy;
59                  itemRadioStrategy.getComponent().setName("itemRadioStrategyError");
60              } else {
61                  var atomicTooltip = new AtomicReference<>(new JToolTipI18n(
62                      I18nUtil.valueByKey(String.format(I18N_TOOLTIP_STRATEGY, nameStrategy))
63                  ));
64                  itemRadioStrategy = new JRadioButtonMenuItem(strategy.toString()) {
65                      @Override
66                      public JToolTip createToolTip() {
67                          atomicTooltip.set(new JToolTipI18n(
68                              I18nUtil.valueByKey(
69                                  String.format(I18N_TOOLTIP_STRATEGY, nameStrategy)
70                              )
71                          ));
72                          return atomicTooltip.get();
73                      }
74                  };
75                  I18nViewUtil.addComponentForKey(
76                      String.format(I18N_TOOLTIP_STRATEGY, nameStrategy),
77                      atomicTooltip.get()
78                  );
79                  itemRadioStrategy.getComponent().setName("itemRadioStrategy" + strategy);
80                  itemRadioStrategy.addActionListener(actionEvent -> {
81                      this.labelStrategy.setText(strategy.toString());
82                      MediatorHelper.model().getMediatorStrategy().setStrategy(strategy);
83                  });
84                  this.groupStrategy.add(itemRadioStrategy);
85              }
86  
87              this.popupMenuStrategies.add(itemRadioStrategy);
88              itemRadioStrategy.setToolTipText(
89                  I18nUtil.valueByKey(String.format(I18N_TOOLTIP_STRATEGY, nameStrategy))
90              );
91              ((JComponent) itemRadioStrategy).setEnabled(false);
92          }
93  
94          this.labelVendor.setText(MediatorHelper.model().getMediatorVendor().getAuto().toString());
95          this.labelVendor.setName("menuVendor");
96          this.popupMenuVendors.setLayout(UiUtil.getColumnLayout(MediatorHelper.model().getMediatorVendor().getVendors().size()));
97          var groupVendor = new ButtonGroup();
98          for (final Vendor vendor: MediatorHelper.model().getMediatorVendor().getVendors()) {
99              JMenuItem itemRadioVendor = new JRadioButtonMenuItem(vendor.toString(), vendor == MediatorHelper.model().getMediatorVendor().getAuto());
100             itemRadioVendor.setName("itemRadioVendor"+ vendor);
101             itemRadioVendor.addActionListener(actionEvent -> {
102                 this.labelVendor.setText(vendor.toString());
103                 MediatorHelper.model().getMediatorVendor().setVendorByUser(vendor);
104             });
105             popupMenuVendors.add(itemRadioVendor);
106             groupVendor.add(itemRadioVendor);
107         }
108 
109         this.loader = new JProgressBar();
110         var dimension = UIManager.getDimension("ProgressBar.horizontalSize");
111         this.loader.setPreferredSize(new Dimension(32, dimension.height));
112         this.loader.setIndeterminate(true);
113         this.add(this.loader);
114 
115         labelVendor.addMouseListener(new MouseAdapter() {
116             @Override
117             public void mousePressed(MouseEvent e) {
118                 Arrays.stream(popupMenuVendors.getComponents())
119                     .map(component -> (JComponent) component)
120                     .forEach(JComponent::updateUI);  // required: incorrect when dark/light mode switch
121                 popupMenuVendors.updateUI();  // required: incorrect when dark/light mode switch
122                 popupMenuVendors.show(e.getComponent(), e.getComponent().getX(),5 + e.getComponent().getY() + e.getComponent().getHeight());
123                 popupMenuVendors.setLocation(e.getComponent().getLocationOnScreen().x,5 + e.getComponent().getLocationOnScreen().y + e.getComponent().getHeight());
124             }
125         });
126         labelStrategy.addMouseListener(new MouseAdapter() {
127             @Override
128             public void mousePressed(MouseEvent e) {
129                 Arrays.stream(popupMenuStrategies.getComponents()).map(a -> (JComponent) a).forEach(JComponent::updateUI);  // required: incorrect when dark/light mode switch
130                 for (var i = 0 ; i < getMenuError().getItemCount() ; i++) {
131                     getMenuError().getItem(i).updateUI();  // required: incorrect when dark/light mode switch
132                 }
133                 popupMenuStrategies.updateUI();  // required: incorrect when dark/light mode switch
134                 popupMenuStrategies.show(e.getComponent(), e.getComponent().getX(),5 + e.getComponent().getY() + e.getComponent().getHeight());
135                 popupMenuStrategies.setLocation(e.getComponent().getLocationOnScreen().x,5 + e.getComponent().getLocationOnScreen().y + e.getComponent().getHeight());
136             }
137         });
138 
139         this.add(this.labelVendor);
140         this.add(this.labelStrategy);
141         this.add(this.buttonStart);
142         this.setCursor(Cursor.getDefaultCursor());
143         this.loader.setVisible(false);
144     }

152     public void reset() {
153         this.labelStrategy.setText("Strategy auto");
154         if (MediatorHelper.model().getMediatorVendor().getVendorByUser() == MediatorHelper.model().getMediatorVendor().getAuto()) {
155             this.labelVendor.setText(MediatorHelper.model().getMediatorVendor().getAuto().toString());
156         }
157         Arrays.stream(this.popupMenuStrategies.getComponents())
158             .forEach(component -> component.setEnabled(false));
159         this.getMenuError().removeAll();
160 
161         this.groupStrategy.clearSelection();
162         Iterable<AbstractButton> iterable = () -> this.groupStrategy.getElements().asIterator();
163         StreamSupport.stream(iterable.spliterator(), false)
164             .filter(abstractButton -> abstractButton.getName().startsWith(PanelTrailingAddress.PREFIX_NAME_ERROR))
165             .forEach(this.groupStrategy::remove);
166     }

168     public void setVendor(Vendor vendor) {
169         this.labelVendor.setText(vendor.toString());
170         this.itemRadioStrategyError.removeAll();
171         var indexError = 0;
172         if (
173             vendor != MediatorHelper.model().getMediatorVendor().getAuto()
174             && vendor.instance().getModelYaml().getStrategy().getError() != null
175         ) {
176             for (Method methodError: vendor.instance().getModelYaml().getStrategy().getError().getMethod()) {
177                 JMenuItem itemRadioError = new JRadioButtonMenuItem(methodError.getName());
178                 itemRadioError.setEnabled(false);
179                 itemRadioError.setName(PanelTrailingAddress.PREFIX_NAME_ERROR + methodError.getName());
180                 this.itemRadioStrategyError.add(itemRadioError);
181                 this.groupStrategy.add(itemRadioError);
182                 int indexErrorFinal = indexError;
183                 itemRadioError.addActionListener(actionEvent -> {
184                     this.labelStrategy.setText(methodError.getName());
185                     MediatorHelper.model().getMediatorStrategy().setStrategy(MediatorHelper.model().getMediatorStrategy().getError());
186                     MediatorHelper.model().getMediatorStrategy().getError().setIndexErrorStrategy(indexErrorFinal);
187                 });
188                 indexError++;
189             }
190         }
191     }

217     public void markError() {
218         StrategyInjectionError strategy = MediatorHelper.model().getMediatorStrategy().getError();
219         this.labelStrategy.setText(strategy.toString());
220         int indexError = strategy.getIndexErrorStrategy();
221         String nameError = MediatorHelper.model().getMediatorVendor().getVendor().instance().getModelYaml().getStrategy().getError().getMethod().get(indexError).getName();
222 
223         Arrays.stream(this.getMenuError().getMenuComponents())
224             .map(component -> (JRadioButtonMenuItem) component)
225             .filter(component -> component.getText().equals(nameError))
226             .forEach(jRadioButtonMenuItem -> {
227                 jRadioButtonMenuItem.setSelected(true);
228                 this.labelStrategy.setText(nameError);
229             });
230     }

232     private JMenu getMenuError() {
233         var nameError = MediatorHelper.model().getMediatorStrategy().getError().getName();
234         return (JMenu) Arrays.stream(this.popupMenuStrategies.getComponents())
235             .map(component -> (JMenuItem) component)
236             .filter(jMenuItem -> jMenuItem.getText().equalsIgnoreCase(nameError))
237             .findFirst()
238             .orElse(new JMenuItem("Mock"));
239     }

241     public void markErrorVulnerable(int indexMethodError) {
242         AbstractStrategy strategy = MediatorHelper.model().getMediatorStrategy().getError();
243         Arrays.stream(this.popupMenuStrategies.getComponents())
244             .map(component -> (JMenuItem) component)
245             .filter(jMenuItem -> jMenuItem.getText().equals(strategy.toString()))
246             .map(component -> (JMenu) component)
247             .forEach(jMenuItem -> {
248                 jMenuItem.setEnabled(true);
249                 // Fix #46578: ArrayIndexOutOfBoundsException on getItem()
250                 if (0 <= indexMethodError && indexMethodError < jMenuItem.getItemCount()) {
251                     jMenuItem.getItem(indexMethodError).setEnabled(true);
252                 }
253             });
254     }
