48      public AddressMenuBar(PanelAddressBar panelAddressBar) {
49          
50          this.buttonInUrl.setName("buttonInUrl");
51          this.buttonInUrl.setToolTipText(I18nUtil.valueByKey("BUTTON_START_TOOLTIP"));
52          this.buttonInUrl.addActionListener(new ActionStart(panelAddressBar));
53  
54          var buttonInTextfield = new ComponentBorder(this.buttonInUrl, 17, 0);
55          buttonInTextfield.install(panelAddressBar.getTextFieldAddress());
56          
57          this.setOpaque(false);
58          this.setBorder(null);
59  
60          this.menuStrategy = new ComboMenu("Strategy auto");
61          this.menuStrategy.setName("menuStrategy");
62          
63          final var patternKeyTooltipStrategy = "STRATEGY_%s_TOOLTIP";
64  
65          for (final AbstractStrategy strategy: MediatorHelper.model().getMediatorStrategy().getStrategies()) {
66              
67              JMenuItem itemRadioStrategy;
68  
69              if (strategy == MediatorHelper.model().getMediatorStrategy().getError()) {
70                  
71                  itemRadioStrategy = new JMenu(strategy.toString());
72                  this.itemRadioStrategyError = (JMenu) itemRadioStrategy;
73                  itemRadioStrategy.getComponent().setName("itemRadioStrategyError");
74                  
75              } else {
76                  
77                  final var refTooltip = new JToolTipI18n[] {
78                          
79                      new JToolTipI18n(
80                          I18nUtil.valueByKey(
81                              String.format(
82                                  patternKeyTooltipStrategy,
83                                  strategy.getName().toUpperCase(Locale.ROOT)
84                              )
85                          )
86                      )
87                  };
88                  
89                  itemRadioStrategy = new JRadioButtonMenuItem(strategy.toString()) {
90                      
91                      @Override
92                      public JToolTip createToolTip() {
93                          
94                          JToolTipI18n tipI18n = new JToolTipI18n(
95                              I18nUtil.valueByKey(
96                                  String.format(
97                                      patternKeyTooltipStrategy,
98                                      strategy.getName().toUpperCase(Locale.ROOT)
99                                  )
100                             )
101                         );
102                         
103                         refTooltip[0] = tipI18n;
104                         
105                         return tipI18n;
106                     }
107                 };
108                 
109                 I18nViewUtil.addComponentForKey(
110                     String.format(
111                         patternKeyTooltipStrategy,
112                         strategy.getName().toUpperCase(Locale.ROOT)
113                     ),
114                     refTooltip[0]
115                 );
116 
117                 itemRadioStrategy.getComponent().setName("itemRadioStrategy"+ strategy);
118                 
119                 itemRadioStrategy.addActionListener(actionEvent -> {
120                     
121                     this.menuStrategy.setText(strategy.toString());
122                     MediatorHelper.model().getMediatorStrategy().setStrategy(strategy);
123                 });
124                 
125                 this.groupStrategy.add(itemRadioStrategy);
126             }
127 
128             this.menuStrategy.add(itemRadioStrategy);
129             
130             itemRadioStrategy.setToolTipText(
131                 I18nUtil.valueByKey(
132                     String.format(
133                         patternKeyTooltipStrategy,
134                         strategy.getName().toUpperCase(Locale.ROOT)
135                     )
136                 )
137             );
138             ((JComponent) itemRadioStrategy).setEnabled(false);
139         }
140 
141         this.menuVendor = new ComboMenu(MediatorHelper.model().getMediatorVendor().getAuto().toString());
142         this.menuVendor.setName("menuVendor");
143         this.menuVendor.getPopupMenu().setLayout(new GridLayout((MediatorHelper.model().getMediatorVendor().getVendors().size() + 1) / 2, 2));
144 
145         var groupVendor = new ButtonGroup();
146 
147         for (final Vendor vendor: MediatorHelper.model().getMediatorVendor().getVendors()) {
148             
149             JMenuItem itemRadioVendor = new JRadioButtonMenuItem(vendor.toString(), vendor == MediatorHelper.model().getMediatorVendor().getAuto());
150             itemRadioVendor.setName("itemRadioVendor"+ vendor);
151             itemRadioVendor.addActionListener(actionEvent -> {
152                 
153                 this.menuVendor.setText(vendor.toString());
154                 MediatorHelper.model().getMediatorVendor().setVendorByUser(vendor);
155             });
156             
157             this.menuVendor.add(itemRadioVendor);
158             groupVendor.add(itemRadioVendor);
159         }
160 
161         this.add(Box.createHorizontalGlue());
162         this.add(this.loader);
163         this.add(Box.createHorizontalStrut(5));
164         this.add(this.menuVendor);
165         this.add(this.menuStrategy);
166 
167         this.loader.setVisible(false);
168         this.loader.setCursor(Cursor.getPredefinedCursor(Cursor.TEXT_CURSOR));
169     }

178     public void initErrorMethods(Vendor vendor) {
179         
180         this.itemRadioStrategyError.removeAll();
181 
182         var indexError = 0;
183         
184         if (
185             vendor != MediatorHelper.model().getMediatorVendor().getAuto()
186             && vendor.instance().getModelYaml().getStrategy().getError() != null
187         ) {
188             
189             for (Method methodError: vendor.instance().getModelYaml().getStrategy().getError().getMethod()) {
190                 
191                 JMenuItem itemRadioVendor = new JRadioButtonMenuItem(methodError.getName());
192                 itemRadioVendor.setEnabled(false);
193                 itemRadioVendor.setName("itemRadioVendor"+ methodError.getName());
194                 
195                 this.itemRadioStrategyError.add(itemRadioVendor);
196                 this.groupStrategy.add(itemRadioVendor);
197 
198                 int indexErrorFinal = indexError;
199                 
200                 itemRadioVendor.addActionListener(actionEvent -> {
201                     
202                     this.menuStrategy.setText(methodError.getName());
203                     MediatorHelper.model().getMediatorStrategy().setStrategy(MediatorHelper.model().getMediatorStrategy().getError());
204                     MediatorHelper.model().getMediatorStrategy().getError().setIndexErrorStrategy(indexErrorFinal);
205                 });
206 
207                 indexError++;
208             }
209         }
210     }

215             this.menuVendor.setText(MediatorHelper.model().getMediatorVendor().getAuto().toString());
216         }

218         this.menuStrategy.setText("Strategy auto");
219         
220         for (var i = 0 ; i < this.menuStrategy.getItemCount() ; i++) {
221             this.menuStrategy.getItem(i).setEnabled(false);
222         }
223         
224         this.getMenuError().removeAll();
225         this.groupStrategy.clearSelection();
226     }

284     public void markError() {
285 
286         StrategyInjectionError strategy = MediatorHelper.model().getMediatorStrategy().getError();
287         this.menuStrategy.setText(strategy.toString());
288         
289         JMenu menuError = this.getMenuError();
290         
291         int indexError = strategy.getIndexErrorStrategy();
292         String nameError = MediatorHelper.model().getMediatorVendor().getVendor().instance().getModelYaml().getStrategy().getError().getMethod().get(indexError).getName();
293         
294         for (var i = 0 ; i < menuError.getItemCount() ; i++) {
295             // Fix #44635: ArrayIndexOutOfBoundsException on getItem()
296             try {
297                 if (menuError.getItem(i).getText().equals(nameError)) {
298                     
299                     menuError.getItem(i).setSelected(true);
300                     this.menuStrategy.setText(nameError);
301                     
302                     break;
303                 }
304             } catch (NullPointerException | ArrayIndexOutOfBoundsException e) {
305                 LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
306             }
307         }
308     }

310     private JMenu getMenuError() {
311         
312         return (JMenu) Arrays.stream(this.menuStrategy.getMenuComponents())
313             .filter(c -> ((JMenuItem) c).getText().equalsIgnoreCase(MediatorHelper.model().getMediatorStrategy().getError().getName()))
314             .findFirst()
315             .orElse(new JMenuItem("Mock"));

318     public void markErrorVulnerable(int indexMethodError) {
319         
320         AbstractStrategy strategy = MediatorHelper.model().getMediatorStrategy().getError();
321         
322         for (var i = 0 ; i < this.menuStrategy.getItemCount() ; i++) {
323             
324             JMenuItem menuItemStrategy = this.menuStrategy.getItem(i);
325             
326             if (menuItemStrategy.getText().equals(strategy.toString())) {
327                 
328                 JMenu menuError = (JMenu) menuItemStrategy;
329                 menuError.setEnabled(true);
330                 
331                 // Fix #46578: ArrayIndexOutOfBoundsException on getItem()
332                 if (0 <= indexMethodError && indexMethodError < menuError.getItemCount()) {
333                     menuError.getItem(indexMethodError).setEnabled(true);
334                 }
335                 
336                 break;
337             }
338         }
339     }
