46          if (leaf) {

48          } else {
49              return UiUtil.TABLE_BOLD.icon;

53      @Override
54      protected void displayProgress(PanelNode panelNode, DefaultMutableTreeNode currentNode) {
55          if ("information_schema".equals(this.getParent().toString())) {
56              panelNode.showLoader();
57              AbstractSuspendable suspendableTask = MediatorHelper.model().getMediatorUtils().getThreadUtil().get(this.getElementDatabase());
58              if (suspendableTask != null && suspendableTask.isPaused()) {
59                  panelNode.setLoaderIcon(new ImageOverlap(UiUtil.HOURGLASS.icon, UiUtil.PATH_PAUSE));
60              }
61          } else {
62              super.displayProgress(panelNode, currentNode);
63          }
64      }
65  
66      @Override
67      public void runAction() {
68          if (this.isRunning()) {  // Prevent double thread run
69              return;
70          }
71              
72          DefaultMutableTreeNode treeNode = MediatorHelper.treeDatabase().getTreeNodeModels().get(this.getElementDatabase());
73          treeNode.removeAllChildren();
74          
75          DefaultTreeModel treeModel = (DefaultTreeModel) MediatorHelper.treeDatabase().getModel();
76          treeModel.reload(treeNode);
77  
78          new SwingWorker<>() {
79              @Override
80              protected Object doInBackground() throws Exception {
81                  Thread.currentThread().setName("SwingWorkerNodeModelTable");
82                  var selectedTable = (Table) NodeModelTable.this.getElementDatabase();
83                  return MediatorHelper.model().getDataAccess().listColumns(selectedTable);
84              }
85          }.execute();
86          
87          this.setRunning(true);
88      }

185     private void addCheckUncheckItems(JPopupMenuCustomExtract tablePopupMenu, final TreePath path) {
186         JMenuItem menuItemCheckAll = new JMenuItem(I18nViewUtil.valueByKey("COLUMNS_CHECK_ALL"), 'C');
187         I18nViewUtil.addComponentForKey("COLUMNS_CHECK_ALL", menuItemCheckAll);
188         
189         JMenuItem menuItemUncheckAll = new JMenuItem(I18nViewUtil.valueByKey("COLUMNS_UNCHECK_ALL"), 'U');
190         I18nViewUtil.addComponentForKey("COLUMNS_UNCHECK_ALL", menuItemUncheckAll);
191 
192         if (!this.isLoaded()) {
193             menuItemCheckAll.setEnabled(false);
194             menuItemUncheckAll.setEnabled(false);
195         }
196 
197         menuItemCheckAll.addActionListener(new ActionCheckAll(true, path));
198         menuItemUncheckAll.addActionListener(new ActionCheckAll(false, path));
199 
200         tablePopupMenu.add(new JSeparator());
201         tablePopupMenu.add(menuItemCheckAll);
202         tablePopupMenu.add(menuItemUncheckAll);
203     }
