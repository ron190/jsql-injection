11  package com.jsql.view.swing.interaction;
12  
13  import com.jsql.model.bean.util.Header;
14  import com.jsql.model.bean.util.HttpHeader;
15  import com.jsql.model.injection.strategy.blind.AbstractCallableBoolean;
16  import com.jsql.util.LogLevelUtil;
17  import com.jsql.view.interaction.InteractionCommand;
18  import com.jsql.view.swing.scrollpane.JScrollIndicator;
19  import com.jsql.view.swing.util.MediatorHelper;
20  import org.apache.logging.log4j.LogManager;
21  import org.apache.logging.log4j.Logger;
22  
23  import javax.swing.*;
24  import javax.swing.table.DefaultTableModel;
25  import java.awt.*;
26  import java.util.Arrays;
27  import java.util.Map;
28  
29  /**
30   * Append a text to the tab Header.
31   */
32  public class MessageHeader implements InteractionCommand {
33      
34      /**
35       * Log4j logger sent to view.
36       */
37      private static final Logger LOGGER = LogManager.getRootLogger();
38      
39      // The text to append to the tab
40      private final String url;
41      private final String post;
42      private final Map<String, String> header;
43      private final Map<String, String> response;
44      private final String source;
45      private final String size;
46      private final String metadataProcess;
47      private final String metadataStrategy;
48      private final AbstractCallableBoolean<?> metadataBoolean;
49  
50      /**
51       * @param interactionParams Text to append
52       */
53      @SuppressWarnings("unchecked")
54      public MessageHeader(Object[] interactionParams) {
55          
56          Map<Header, Object> params = (Map<Header, Object>) interactionParams[0];
57          
58          this.url = (String) params.get(Header.URL);
59          this.post = (String) params.get(Header.POST);
60          this.header = (Map<String, String>) params.get(Header.HEADER);
61          this.response = (Map<String, String>) params.get(Header.RESPONSE);
62          this.source = (String) params.get(Header.SOURCE);
63          this.size = (String) params.get(Header.PAGE_SIZE);
64          this.metadataProcess = (String) params.get(Header.METADATA_PROCESS);
65          this.metadataStrategy = (String) params.get(Header.METADATA_STRATEGY);
66          this.metadataBoolean = (AbstractCallableBoolean<?>) params.get(Header.METADATA_BOOLEAN);
67      }
68  
69      @Override
70      public void execute() {
71          
72          MediatorHelper.panelConsoles().getNetworkTable().addHeader(new HttpHeader(this.url, this.post, this.header, this.response, this.source));
73          
74          JViewport viewport = ((JScrollIndicator) MediatorHelper.panelConsoles().getNetworkSplitPane().getLeftComponent()).getScrollPane().getViewport();
75          JTable table = (JTable) viewport.getView();
76          
77          DefaultTableModel model = (DefaultTableModel) table.getModel();
78          
79          try {
80              model.addRow(new Object[] {
81                      
82                  this.url,
83                  this.size,
84                  this.metadataStrategy,
85                  Arrays.asList(this.metadataProcess, this.metadataBoolean)
86              });
87              
88              Rectangle rect = table.getCellRect(table.getRowCount() - 1, 0, true);
89              Point pt = viewport.getViewPosition();
90              rect.translate(-pt.x, -pt.y);
91              viewport.scrollRectToVisible(rect);
92              
93              MediatorHelper.tabConsoles().highlightTab("Network");
94              
95          } catch(NullPointerException | IndexOutOfBoundsException e) {
96              
97              // Fix #4658, #2224, #1797 on model.addRow()
98              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
99          }
100     }
101 }
