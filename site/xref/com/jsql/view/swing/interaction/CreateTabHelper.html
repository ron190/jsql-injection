9   /**
10   * Set result tab panel orientation according to locale when first
11   * tab is inserted.
12   */
13  public class CreateTabHelper {
14  
15      protected CreateTabHelper() {
16          
17          CreateTabHelper.initializeSplitOrientation();
18      }
19      
20      /**
21       * Switch left component with right component when locale orientation requires this.
22       */
23      public static void initializeSplitOrientation() {
24          
25          if (MediatorHelper.tabResults().getTabCount() == 0) {
26              
27              JSplitPane splitPaneLeftRight = MediatorHelper.frame().getSplitHorizontalTopBottom().getSplitVerticalLeftRight();
28              int dividerLocation = splitPaneLeftRight.getDividerLocation();
29              
30              if (ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault()))) {
31                  
32                  splitPaneLeftRight.setLeftComponent(MediatorHelper.tabResults());
33                  
34              } else {
35                  
36                  splitPaneLeftRight.setRightComponent(MediatorHelper.tabResults());
37              }
38              
39              splitPaneLeftRight.setDividerLocation(dividerLocation);
40          }
41      }
42  }
