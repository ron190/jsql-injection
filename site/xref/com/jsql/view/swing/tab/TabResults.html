56      /**
57       * Create the panel containing injection results.
58       */
59      public TabResults() {
60          this.setName("tabResults");
61          this.setTabLayoutPolicy(JTabbedPane.SCROLL_TAB_LAYOUT);
62          this.setTransferHandler(new TabTransferHandler());
63          this.putClientProperty("JTabbedPane.tabClosable", true);
64          this.putClientProperty("JTabbedPane.tabCloseCallback", (IntConsumer) ActionCloseTabResult::perform);
65          UIManager.put("TabbedPane.closeHoverForeground", LogLevelUtil.COLOR_RED);
66          HotkeyUtil.addShortcut(this);  // Add hotkeys to root-pane ctrl-tab, ctrl-shift-tab, ctrl-w
67          this.addMouseWheelListener(new TabbedPaneMouseWheelListener());
68          MediatorHelper.register(this);
69      }

71      public void addFileTab(String label, String content, String path) {
72          JTextArea fileText = new JPopupTextArea().getProxy();
73          fileText.setText(content);
74          fileText.setFont(new Font(UiUtil.FONT_NAME_MONO_NON_ASIAN, Font.PLAIN, 14));
75          fileText.setCaretPosition(0);
76          this.addTextTab(label, path, fileText, UiUtil.DOWNLOAD.icon);
77          MediatorHelper.tabManagersCards().addToLists(path, label);
78      }

80      public void addReportTab(String content) {
81          JEditorPane editorPane = new JEditorPane();
82          editorPane.setContentType("text/html");
83          editorPane.setText("<html><span style=\"white-space: nowrap; font-family:'"+ UiUtil.FONT_NAME_MONO_NON_ASIAN +"'\">" + content + "</span></html>");
84          editorPane.setFont(UIManager.getFont("TextArea.font"));  // required to increase text size
85          editorPane.setDragEnabled(true);
86          editorPane.setEditable(false);
87          editorPane.setCaretPosition(0);
88          editorPane.getCaret().setBlinkRate(0);
89          editorPane.setCursor(Cursor.getPredefinedCursor(Cursor.TEXT_CURSOR));
90          editorPane.setComponentPopupMenu(new JPopupMenuText(editorPane));
91          editorPane.addHyperlinkListener(linkEvent -> {
92              if (HyperlinkEvent.EventType.ACTIVATED.equals(linkEvent.getEventType())) {
93                  try {
94                      Desktop.getDesktop().browse(linkEvent.getURL().toURI());
95                  } catch (IOException | URISyntaxException | UnsupportedOperationException e) {
96                      LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Failing to browse Url", e);
97                  }
98              }
99          });
100         editorPane.addFocusListener(new FocusAdapter() {
101             @Override
102             public void focusGained(FocusEvent focusEvent) {
103                 editorPane.getCaret().setVisible(true);
104                 editorPane.getCaret().setSelectionVisible(true);
105                 editorPane.getCaret().setBlinkRate(0);
106             }
107         });
108         UiUtil.initialize(editorPane);  // silent delete
109 
110         this.addTextTab("Vulnerability report", "Analysis report with all payloads detected", editorPane, UiUtil.REPORT.icon);
111     }

113     public void addTextTab(String label, String toolTipText, JComponent componentText, FlatSVGIcon icon) {
114         var scroller = new JScrollPane(componentText);
115         this.addTab(label + StringUtils.SPACE, scroller);
116         this.setSelectedComponent(scroller);  // Focus on the new tab
117         this.setToolTipTextAt(this.indexOfComponent(scroller), toolTipText);
118         var header = new TabHeader(label, icon);
119         this.setTabComponentAt(this.indexOfComponent(scroller), header);
120 
121         this.updateUI();  // required: light, open/close prefs, dark => light artifacts
122     }

124     public void addTabExploitWeb(String url) {
125         try {
126             var terminalID = UUID.randomUUID();
127             var terminal = new ExploitWeb(terminalID, url);
128             MediatorHelper.frame().getMapUuidShell().put(terminalID, terminal);
129 
130             JScrollPane scroller = new JScrollPane(terminal);
131             this.addTab("Web shell", scroller);
132             this.setSelectedComponent(scroller);  // Focus on the new tab
133 
134             var header = new TabHeader("Web shell", UiUtil.TERMINAL.icon);
135             this.setTabComponentAt(this.indexOfComponent(scroller), header);
136             terminal.requestFocusInWindow();
137 
138             this.updateUI();  // required: light, open/close prefs, dark => light artifacts
139         } catch (MalformedURLException | URISyntaxException e) {
140             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Tab exploit failure: incorrect URL", e);
141         }
142     }

184     public void addTabValues(String[][] data, String[] columnNames, AbstractElementDatabase table) {
185         var panelTable = new PanelTable(data, columnNames);
186         
187         this.addTab(StringUtil.detectUtf8(table.toString()), panelTable);
188         panelTable.setComponentOrientation(ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale()));
189         
190         this.setSelectedComponent(panelTable);  // Focus on the new tab
191 
192         var header = new TabHeader(UiStringUtil.detectUtf8Html(table.toString()), UiUtil.TABLE_BOLD.icon);
193         this.setTabComponentAt(this.indexOfComponent(panelTable), header);
194 
195         this.updateUI();  // required: light, open/close prefs, dark => light artifacts
196     }
