11  package com.jsql.view.swing.tab;
12  
13  import com.jsql.view.swing.manager.*;
14  import com.jsql.view.swing.util.MediatorHelper;
15  import com.jsql.view.swing.util.UiUtil;
16  import org.apache.commons.lang3.StringUtils;
17  
18  import javax.swing.*;
19  import java.awt.*;
20  import java.util.Arrays;
21  import java.util.List;
22  
23  /**
24   * Panel on the left with functionalities like webshell, file reading and admin page finder.
25   */
26  public class TabManagersCards extends JPanel {
27      
28      private final ManagerWebShell managerWebShell = new ManagerWebShell();
29      private final ManagerFile managerFile = new ManagerFile();
30      private final ManagerUpload managerUpload = new ManagerUpload();
31      private final ManagerSqlShell managerSqlShell = new ManagerSqlShell();
32      
33      /**
34       * Create manager panel.
35       */
36      public TabManagersCards() {
37          
38          this.setName("tabManagersProxy");
39          this.setLayout(new CardLayout());
40          this.setMinimumSize(new Dimension(100, 0));  // allow proper minimize
41  
42          var managerScanList = new ManagerScan();
43          var managerDatabase = new ManagerDatabase();
44          var managerAdminPage = new ManagerAdminPage();
45          var managerBruteForce = new ManagerBruteForce();
46  
47          MediatorHelper.register(this.managerWebShell);
48          MediatorHelper.register(this.managerFile);
49          MediatorHelper.register(this.managerUpload);
50          MediatorHelper.register(this.managerSqlShell);
51          MediatorHelper.register(managerScanList);
52          MediatorHelper.register(managerAdminPage);
53          MediatorHelper.register(managerBruteForce);
54  
55          this.buildI18nTab("DATABASE_TAB", managerDatabase);
56          this.buildI18nTab("ADMINPAGE_TAB", managerAdminPage);
57          this.buildI18nTab("FILE_TAB", this.managerFile);
58          this.buildI18nTab("WEBSHELL_TAB", this.managerWebShell);
59          this.buildI18nTab("SQLSHELL_TAB", this.managerSqlShell);
60          this.buildI18nTab("UPLOAD_TAB", this.managerUpload);
61          this.buildI18nTab("BRUTEFORCE_TAB", managerBruteForce);
62          this.buildI18nTab("CODER_TAB", new ManagerCoder());
63          this.buildI18nTab("SCANLIST_TAB", managerScanList);
64  
65          MediatorHelper.register(this);
66      }
67  
68      public void addToLists(String path, String name) {
69          Arrays.asList(this.managerWebShell, this.managerSqlShell, this.managerUpload)
70              .forEach(manager -> manager.addToList(path.replace(name, StringUtils.EMPTY)));
71      }
72      
73      public void markFileSystemInvulnerable() {
74          this.getManagers().forEach(manager -> {
75              manager.changePrivilegeIcon(UiUtil.CROSS_RED.icon);
76              manager.endProcess();
77          });
78      }
79      
80      public void endPreparation() {
81          this.getManagers().forEach(manager -> manager.setButtonEnable(true));
82      }
83      
84      public void markFileSystemVulnerable() {
85          this.getManagers().forEach(manager -> manager.changePrivilegeIcon(UiUtil.TICK_GREEN.icon));
86      }
87  
88      private void buildI18nTab(String keyLabel, Component manager) {
89          manager.setName(keyLabel);
90          this.add(manager, keyLabel);
91      }
92  
93      private List<AbstractManagerList> getManagers() {
94          return Arrays.asList(this.managerFile, this.managerWebShell, this.managerSqlShell, this.managerUpload);
95      }
96  }
