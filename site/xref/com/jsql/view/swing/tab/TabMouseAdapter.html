10  import javax.swing.*;
11  import java.awt.*;
12  import java.awt.event.MouseAdapter;
13  import java.awt.event.MouseEvent;
14  
15  /**
16   * Display popupmenu on right click.
17   * Used on manager tabs.
18   */
19  public class TabMouseAdapter extends MouseAdapter {
20      
21      /**
22       * Log4j logger sent to view.
23       */
24      private static final Logger LOGGER = LogManager.getRootLogger();
25      
26      private final TabbedPaneWheeled tabbedPaneWheeled;
27      
28      public TabMouseAdapter(TabbedPaneWheeled tabbedPaneWheeled) {
29          this.tabbedPaneWheeled = tabbedPaneWheeled;
30      }
31  
32      @Override
33      public void mouseClicked(MouseEvent event) {
34          if (!SwingUtilities.isRightMouseButton(event)) {
35              return;
36          }
37  
38          // Copy menu items from menubar
39          var menu = new JPopupMenu();
40          for (var i = 0 ; i < MediatorHelper.menubar().getMenuView().getMenuComponentCount() ; i++) {
41              final int positionFinal = i;
42              // Fix #35348: SerializationException on clone()
43              try {
44                  // clone and setAction() not efficient (performance and i18n)
45                  var itemOrigin = (JMenuItem) MediatorHelper.menubar().getMenuView().getMenuComponent(i);
46                  JMenuItem itemMenu = new JMenuItem(itemOrigin.getText(), itemOrigin.getIcon());
47                  itemMenu.addActionListener(actionEvent -> this.tabbedPaneWheeled.setSelectedIndex(positionFinal));
48                  menu.add(itemMenu);
49              } catch (SerializationException e) {
50                  LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
51              }
52          }
53  
54          var componentSource = (Component) event.getSource();
55          menu.show(componentSource, event.getX(), event.getY());
56          menu.setLocation(
57              ComponentOrientation.RIGHT_TO_LEFT.equals(ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale()))
58              ? event.getXOnScreen() - menu.getWidth()
59              : event.getXOnScreen(),
60              event.getYOnScreen()
61          );
62      }
63  }
