18  public class TabTransferHandler extends TransferHandler {
19      
20      /**
21       * Log4j logger sent to view.
22       */
23      private static final Logger LOGGER = LogManager.getRootLogger();
24      
25      protected final DataFlavor localObjectFlavor;
26      
27      protected DnDTabbedPane source;
28  
29      public TabTransferHandler() {
30          
31          super();
32          
33          this.localObjectFlavor = new DataFlavor(DnDTabData.class, "DnDTabData");
34      }
35      
36      @Override
37      protected Transferable createTransferable(JComponent c) {
38          
39          if (c instanceof DnDTabbedPane) {
40              
41              this.source = (DnDTabbedPane) c;
42          }
43  
44          return new Transferable() {
45              
46              @Override
47              public DataFlavor[] getTransferDataFlavors() {
48                  
49                  return new DataFlavor[] {TabTransferHandler.this.localObjectFlavor};
50              }
51              
52              @Override
53              public boolean isDataFlavorSupported(DataFlavor flavor) {
54                  
55                  return Objects.equals(TabTransferHandler.this.localObjectFlavor, flavor);
56              }
57              
58              @Override
59              public Object getTransferData(DataFlavor flavor) throws UnsupportedFlavorException {
60                  
61                  if (this.isDataFlavorSupported(flavor)) {
62                      
63                      return new DnDTabData(TabTransferHandler.this.source);
64                      
65                  } else {
66                      
67                      throw new UnsupportedFlavorException(flavor);
68                  }
69              }
70          };
71      }
72      
73      @Override
74      public boolean canImport(TransferHandler.TransferSupport support) {
75          
76          if (!support.isDrop() || !support.isDataFlavorSupported(this.localObjectFlavor)) {
77              return false;
78          }
79          
80          support.setDropAction(TransferHandler.MOVE);
81          var tdl = support.getDropLocation();
82          var pt = tdl.getDropPoint();
83          
84          DnDTabbedPane target = (DnDTabbedPane) support.getComponent();
85          
86          target.autoScrollTest(pt);
87          DnDTabbedPane.DnDDropLocation dl = target.dropLocationForPointDnD(pt);
88          int idx = dl.getIndex();
89  
90          var isDroppable = false;
91          boolean isAreaContains = target.getTabAreaBounds().contains(pt) && idx >= 0;
92          
93          if (target.equals(this.source)) {
94              
95              isDroppable = isAreaContains && idx != target.dragTabIndex && idx != target.dragTabIndex + 1;
96              
97          } else {
98              
99              isDroppable = Optional.ofNullable(this.source).map(c -> !c.isAncestorOf(target)).orElse(false) && isAreaContains;
100         }
101 
102         // [JDK-6700748] Cursor flickering during D&D when using CellRendererPane with validation - Java Bug System
103         // https://bugs.openjdk.java.net/browse/JDK-6700748
104         Cursor cursor = isDroppable ? DragSource.DefaultMoveDrop : DragSource.DefaultMoveNoDrop;
105         Component glassPane = target.getRootPane().getGlassPane();
106         glassPane.setCursor(cursor);
107         target.setCursor(cursor);
108 
109         support.setShowDropLocation(isDroppable);
110         dl.setDroppable(isDroppable);
111         target.setDropLocation(dl, isDroppable);
112         return isDroppable;
113     }
114     
115     private BufferedImage makeDragTabImage(DnDTabbedPane tabbedPane) {
116         
117         Rectangle rect = tabbedPane.getBoundsAt(tabbedPane.dragTabIndex);
118         var image = new BufferedImage(tabbedPane.getWidth(), tabbedPane.getHeight(), BufferedImage.TYPE_INT_ARGB);
119         Graphics g2 = image.createGraphics();
120         tabbedPane.paint(g2);
121         g2.dispose();
122         
123         if (rect.x < 0) {
124             
125             rect.translate(-rect.x, 0);
126         }
127         
128         if (rect.y < 0) {
129             
130             rect.translate(0, -rect.y);
131         }
132         
133         if (rect.x + rect.width > image.getWidth()) {
134             
135             rect.width = image.getWidth() - rect.x;
136         }
137         
138         if (rect.y + rect.height > image.getHeight()) {
139             
140             rect.height = image.getHeight() - rect.y;
141         }
142         
143         return image.getSubimage(rect.x, rect.y, rect.width, rect.height);
144     }
145     
146     @Override
147     public int getSourceActions(JComponent c) {
148         
149         if (c instanceof DnDTabbedPane) {
150             
151             DnDTabbedPane src = (DnDTabbedPane) c;
152             c.getRootPane().setGlassPane(new GhostGlassPane(src));
153             
154             if (src.dragTabIndex < 0) {
155                 
156                 return TransferHandler.NONE;
157             }
158             
159             this.setDragImage(this.makeDragTabImage(src));
160             c.getRootPane().getGlassPane().setVisible(true);
161             
162             return TransferHandler.MOVE;
163         }
164         
165         return TransferHandler.NONE;
166     }
167     
168     @Override
169     public boolean importData(TransferHandler.TransferSupport support) {
170         
171         if (!this.canImport(support)) {
172             
173             return false;
174         }
175 
176         DnDTabbedPane target = (DnDTabbedPane) support.getComponent();
177         DnDTabbedPane.DnDDropLocation dl = target.getDropLocation();
178         
179         try {
180             DnDTabData data = (DnDTabData) support.getTransferable().getTransferData(this.localObjectFlavor);
181             DnDTabbedPane src = data.tabbedPane;
182             int index = dl.getIndex();
183             
184             if (target.equals(src)) {
185                 
186                 src.convertTab(src.dragTabIndex, index);
187                 
188             } else {
189                 
190                 src.exportTab(src.dragTabIndex, target, index);
191             }
192             
193             return true;
194             
195         } catch (UnsupportedFlavorException | IOException e) {
196             
197             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
198         }
199         
200         return false;
201     }
202     
203     @Override
204     protected void exportDone(JComponent c, Transferable data, int action) {
205         
206         DnDTabbedPane src = (DnDTabbedPane) c;
207         
208         src.getRootPane().getGlassPane().setVisible(false);
209         src.setDropLocation(null, false);
210         src.repaint();
211         src.setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
212     }
213 }
