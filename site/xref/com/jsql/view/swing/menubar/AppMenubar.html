102     private JMenu initMenuEdit() {
103         var menuEdit = new JMenu(I18nUtil.valueByKey("MENUBAR_EDIT"));
104         I18nViewUtil.addComponentForKey("MENUBAR_EDIT", menuEdit);
105         menuEdit.setMnemonic('E');
106 
107         JMenuItem itemCopy = new JMenuItem(I18nUtil.valueByKey("CONTEXT_MENU_COPY"), 'C');
108         I18nViewUtil.addComponentForKey("CONTEXT_MENU_COPY", itemCopy);
109         itemCopy.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_C, InputEvent.CTRL_DOWN_MASK));
110         itemCopy.addActionListener(actionEvent -> {
111             if (MediatorHelper.tabResults().getSelectedComponent() instanceof PanelTable) {
112                 ((PanelTable) MediatorHelper.tabResults().getSelectedComponent()).copyTable();
113             } else if (MediatorHelper.tabResults().getSelectedComponent() instanceof JScrollPane) {
114                 ((JTextComponent) ((JScrollPane) MediatorHelper.tabResults().getSelectedComponent()).getViewport().getView()).copy();
115             }
116         });
117 
118         JMenuItem itemSelectAll = new JMenuItem(I18nUtil.valueByKey("CONTEXT_MENU_SELECT_ALL"), 'A');
119         I18nViewUtil.addComponentForKey("CONTEXT_MENU_SELECT_ALL", itemSelectAll);
120         itemSelectAll.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_A, InputEvent.CTRL_DOWN_MASK));
121         itemSelectAll.addActionListener(actionEvent -> {
122             if (MediatorHelper.tabResults().getSelectedComponent() instanceof PanelTable) {
123                 ((PanelTable) MediatorHelper.tabResults().getSelectedComponent()).selectTable();
124             } else if (MediatorHelper.tabResults().getSelectedComponent() instanceof JScrollPane) {
125                 // Textarea need focus to select all
126                 ((JScrollPane) MediatorHelper.tabResults().getSelectedComponent()).getViewport().getView().requestFocusInWindow();
127                 ((JTextComponent) ((JScrollPane) MediatorHelper.tabResults().getSelectedComponent()).getViewport().getView()).selectAll();
128             }
129         });
130 
131         menuEdit.add(itemCopy);
132         menuEdit.add(new JSeparator());
133         menuEdit.add(itemSelectAll);
134         return menuEdit;
135     }

155     private JMenu initMenuI18nContribution() {
156         var menuI18nContribution = new JMenu(I18nUtil.valueByKey("MENUBAR_COMMUNITY_HELPTRANSLATE"));
157         I18nViewUtil.addComponentForKey("MENUBAR_COMMUNITY_HELPTRANSLATE", menuI18nContribution);
158 
159         final var dialogTranslate = new DialogTranslate();  // Render the About dialog behind scene
160         AppMenubar.MODELS_ITEM_INTO.forEach(model -> {
161             model.setMenuItem(new JMenuItem(model.getLanguage().getMenuItemLabel(), model.getLanguage().getFlag()));
162             model.getMenuItem().addActionListener(new ActionTranslate(dialogTranslate, model.getLanguage()));
163             menuI18nContribution.add(model.getMenuItem());
164         });
165 
166         var itemIntoOther = new JMenuItem(I18nUtil.valueByKey("MENUBAR_COMMUNITY_ANOTHERLANGUAGE"));
167         I18nViewUtil.addComponentForKey("MENUBAR_COMMUNITY_ANOTHERLANGUAGE", itemIntoOther);
168 
169         AppMenubar.MODELS_ITEM_INTO.stream().filter(model -> model.getLanguage() == Language.AR)
170         .forEach(modelItemTranslate -> modelItemTranslate.getMenuItem().setComponentOrientation(ComponentOrientation.RIGHT_TO_LEFT));
171         AppMenubar.MODELS_ITEM_INTO.stream().filter(model -> model.getLanguage() == Language.FR)
172         .forEach(modelItemTranslate -> modelItemTranslate.getMenuItem().setName("itemIntoFrench"));
173 
174         menuI18nContribution.add(new JSeparator());
175         menuI18nContribution.add(itemIntoOther);
176         itemIntoOther.addActionListener(new ActionTranslate(dialogTranslate, Language.OT));
177 
178         return menuI18nContribution;
179     }

181     private JMenuItem initItemReportIssue() {
182         JMenuItem itemReportIssue = new JMenuItem(I18nUtil.valueByKey("MENUBAR_COMMUNITY_REPORTISSUE"), 'R');
183         I18nViewUtil.addComponentForKey("MENUBAR_COMMUNITY_REPORTISSUE", itemReportIssue);
184 
185         itemReportIssue.addActionListener(actionEvent -> {
186 
187             var panel = new JPanel(new BorderLayout());
188             final JTextArea textarea = new JPopupTextArea(new JTextArea()).getProxy();
189             textarea.setFont(new Font(
190                 UiUtil.FONT_NAME_MONOSPACED,
191                 Font.PLAIN,
192                 UIManager.getDefaults().getFont("TextField.font").getSize()
193             ));
194             textarea.setText(
195                 "## What's the expected behavior?\n\n"
196                 + "## What's the actual behavior?\n\n"
197                 + "## Any other detailed information on the Issue?\n\n"
198                 + "## Steps to reproduce the problem\n\n"
199                 + "  1. ...\n"
200                 + "  2. ...\n\n"
201                 + "## [Community] Request for new feature\n\n"
202             );
203             panel.add(new JLabel("Describe your bug or issue:"), BorderLayout.NORTH);
204             panel.add(new JScrollPane(textarea));
205             panel.setPreferredSize(new Dimension(500, 350));
206             panel.setMinimumSize(new Dimension(500, 350));
207 
208             textarea.addMouseListener(new MouseAdapter() {
209                 @Override
210                 public void mousePressed(MouseEvent e) {
211                     super.mousePressed(e);
212                     textarea.requestFocusInWindow();
213                 }
214             });
215 
216             int result = JOptionPane.showOptionDialog(
217                 MediatorHelper.frame(),
218                 panel,
219                 "Report an issue or a bug",
220                 JOptionPane.OK_CANCEL_OPTION,
221                 JOptionPane.QUESTION_MESSAGE,
222                 null,
223                 new String[] {
224                     "Send",
225                     I18nUtil.valueByKey("LIST_ADD_VALUE_CANCEL")
226                 },
227                 I18nUtil.valueByKey("LIST_ADD_VALUE_CANCEL")
228             );
229 
230             if (StringUtils.isNotEmpty(textarea.getText()) && result == JOptionPane.YES_OPTION) {
231                 MediatorHelper.model().getMediatorUtils().getGitUtil().sendReport(textarea.getText(), ShowOnConsole.YES, "Report");
232             }
233         });
234 
235         return itemReportIssue;
236     }

299     public void switchLocale(Locale oldLocale, Locale newLocale, boolean isStartup) {
300         I18nUtil.setCurrentBundle(newLocale);
301 
302         Stream.of(
303             JTextPaneAppender.ATTRIBUTE_WARN,
304             JTextPaneAppender.ATTRIBUTE_INFORM,
305             JTextPaneAppender.ATTRIBUTE_SUCCESS,
306             JTextPaneAppender.ATTRIBUTE_ALL
307         )
308         .forEach(attribute -> {
309             StyleConstants.setFontFamily(attribute, I18nViewUtil.isNonUbuntu(newLocale) ? UiUtil.FONT_NAME_MONO_ASIAN : UiUtil.FONT_NAME_MONO_NON_ASIAN);
310             StyleConstants.setFontSize(attribute, I18nViewUtil.isNonUbuntu(newLocale) ? UiUtil.FONT_SIZE_MONO_ASIAN : UiUtil.FONT_SIZE_MONO_NON_ASIAN);
311         });
312 
313         MediatorHelper.managerBruteForce().getResult().setFont(I18nViewUtil.isNonUbuntu(newLocale) ? UiUtil.FONT_MONO_ASIAN : UiUtil.FONT_MONO_NON_ASIAN);
314 
315         this.switchNetworkTable();
316         I18nViewUtil.switchI18nComponents();
317         this.switchOrientation(oldLocale, newLocale, isStartup);
318         this.switchMenuItems();  // required to restore proper language orientation
319         
320         MediatorHelper.treeDatabase().reloadNodes();
321 
322         // Fix #92981: IllegalArgumentException on revalidate()
323         // Fix #96185: NullPointerException on revalidate()
324         // Fix #96226: ArrayIndexOutOfBoundsException on revalidate()
325         try {
326             MediatorHelper.frame().revalidate();  // Fix glitches on Linux
327         } catch (NullPointerException | IllegalArgumentException | ArrayIndexOutOfBoundsException e) {
328             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
329         }
330     }

332     private void switchOrientation(Locale oldLocale, Locale newLocale, boolean isStartup) {
333         var componentOrientation = ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale());
334         MediatorHelper.frame().applyComponentOrientation(componentOrientation);
335         
336         if (!ComponentOrientation.getOrientation(oldLocale).equals(ComponentOrientation.getOrientation(newLocale))) {
337             
338             JSplitPane splitPaneLeftRight = MediatorHelper.frame().getSplitNS().getSplitEW();
339             var componentLeft = splitPaneLeftRight.getLeftComponent();
340             var componentRight = splitPaneLeftRight.getRightComponent();
341 
342             // Reset components
343             splitPaneLeftRight.setLeftComponent(null);
344             splitPaneLeftRight.setRightComponent(null);
345             splitPaneLeftRight.setLeftComponent(componentRight);
346             splitPaneLeftRight.setRightComponent(componentLeft);
347 
348             if (isStartup) {
349                 // TODO unclear and not working properly when starting as locale arabic
350                 splitPaneLeftRight.setDividerLocation(
351                     splitPaneLeftRight.getDividerLocation()
352                 );
353             } else {  // required as switch to arabic uses reversed location
354                 splitPaneLeftRight.setDividerLocation(
355                     splitPaneLeftRight.getWidth() -
356                     splitPaneLeftRight.getDividerLocation()
357                 );
358             }
359         }
360         
361         MediatorHelper.tabResults().setComponentOrientation(ComponentOrientation.getOrientation(newLocale));
362     }

365         Stream.concat(AppMenubar.MODELS_ITEM.stream(), AppMenubar.MODELS_ITEM_INTO.stream())
366         .forEach(model -> model.getMenuItem().setComponentOrientation(
367             model.getLanguage().isRightToLeft()

369             : ComponentOrientation.LEFT_TO_RIGHT
370         ));

373     private void switchNetworkTable() {
374         JTableHeader header = MediatorHelper.panelConsoles().getNetworkTable().getTableHeader();
375         TableColumnModel columnModel = header.getColumnModel();
376         columnModel.getColumn(0).setHeaderValue(I18nUtil.valueByKey("NETWORK_TAB_URL_COLUMN"));
377         columnModel.getColumn(1).setHeaderValue(I18nUtil.valueByKey("NETWORK_TAB_SIZE_COLUMN") +" (KB)");
378         columnModel.getColumn(2).setHeaderValue(I18nUtil.valueByKey("SQLENGINE_STRATEGY"));
379         columnModel.getColumn(3).setHeaderValue(I18nUtil.valueByKey("SQLENGINE_METADATA"));
380         header.repaint();
381     }
