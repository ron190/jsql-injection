202     private JMenuItem getMenuItemSqlEngine() {
203         var itemSqlEngine = new JMenuItem(I18nUtil.valueByKey(MenuWindows.I18N_SQL_ENGINE));
204         I18nViewUtil.addComponentForKey(MenuWindows.I18N_SQL_ENGINE, itemSqlEngine);
205         itemSqlEngine.setName("itemSqlEngine");
206         itemSqlEngine.setMnemonic('S');
207 
208         // Render the SQL Engine dialog behind scene
209         var titleTabSqlEngine = "SQL Engine";
210 
211         itemSqlEngine.addActionListener(actionEvent -> {
212             for (var i = 0 ; i < MediatorHelper.tabResults().getTabCount() ; i++) {
213                 if (titleTabSqlEngine.equals(MediatorHelper.tabResults().getTitleAt(i))) {
214                     MediatorHelper.tabResults().setSelectedIndex(i);
215                     return;
216                 }
217             }
218 
219             MediatorHelper.frame().getSplitNS().initSplitOrientation();
220 
221             var panelSqlEngine = new SqlEngine();
222             MediatorHelper.tabResults().addTab(titleTabSqlEngine, panelSqlEngine);
223             MediatorHelper.tabResults().setSelectedComponent(panelSqlEngine);  // Focus on the new tab
224 
225             // Create a custom tab header
226             var header = new TabHeader(I18nViewUtil.valueByKey(MenuWindows.I18N_SQL_ENGINE), UiUtil.COG.getIcon());
227             I18nViewUtil.addComponentForKey(MenuWindows.I18N_SQL_ENGINE, header.getTabLabel());
228 
229             // Apply the custom header to the tab
230             MediatorHelper.tabResults().setTabComponentAt(MediatorHelper.tabResults().indexOfComponent(panelSqlEngine), header);
231             MediatorHelper.tabResults().updateUI();  // required: light, open/close prefs, dark => light artifacts
232         });
233 
234         return itemSqlEngine;
235     }

272     private JMenu initMenuTranslation() {
273         var menuTranslation = new JMenu(I18nUtil.valueByKey("MENUBAR_LANGUAGE"));
274         I18nViewUtil.addComponentForKey("MENUBAR_LANGUAGE", menuTranslation);
275         menuTranslation.setName("menuTranslation");
276         menuTranslation.setMnemonic('L');
277 
278         var groupRadioLanguage = new ButtonGroup();
279         var atomicIsAnySelected = new AtomicBoolean(false);
280         AppMenubar.MODELS_ITEM.forEach(model -> {
281             atomicIsAnySelected.set(atomicIsAnySelected.get() || model.getLanguage().isCurrentLanguage());
282             model.setMenuItem(new RadioItemPreventClose(
283                 model.getLanguage().getMenuItemLabel(),
284                 model.getLanguage().getFlag(),
285                 model.getLanguage().isCurrentLanguage()
286             ));
287             model.getMenuItem().addActionListener(actionEvent -> {
288                 this.appMenubar.switchLocale(
289                     model.getLanguage() == Language.EN
290                     ? Locale.ROOT  // required as no bundle 'en'
291                     : Locale.forLanguageTag(model.getLanguage().getLanguageTag())
292                 );
293                 MediatorHelper.model().getMediatorUtils().getPreferencesUtil().withLanguageTag(model.getLanguage().getLanguageTag()).persist();
294             });
295             menuTranslation.add(model.getMenuItem());
296             groupRadioLanguage.add(model.getMenuItem());
297         });
298 
299         AppMenubar.MODELS_ITEM.stream().filter(model -> model.getLanguage() == Language.EN)
300         .forEach(modelItem -> {
301             modelItem.getMenuItem().setSelected(!atomicIsAnySelected.get());
302             modelItem.getMenuItem().setName("itemEnglish");
303         });
304         AppMenubar.MODELS_ITEM.stream().filter(model -> model.getLanguage() == Language.RU)
305         .forEach(modelItem -> modelItem.getMenuItem().setName("itemRussian"));
306         AppMenubar.MODELS_ITEM.stream().filter(model -> model.getLanguage() == Language.AR)
307         .forEach(modelItem -> modelItem.getMenuItem().setComponentOrientation(ComponentOrientation.RIGHT_TO_LEFT));
308 
309         return menuTranslation;
310     }

312     public void switchLocaleFromPreferences() {
313         AppMenubar.MODELS_ITEM.stream()
314         .filter(model -> model.getLanguage().getLanguageTag().equals(
315             MediatorHelper.model().getMediatorUtils().getPreferencesUtil().getLanguageTag()
316         ))
317         .forEach(modelItem -> modelItem.getMenuItem().doClick());
318     }
