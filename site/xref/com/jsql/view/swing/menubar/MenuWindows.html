192     private JMenuItem getMenuItemSqlEngine() {
193         var itemSqlEngine = new JMenuItem(I18nUtil.valueByKey(KEY_MENU_SQL_ENGINE));
194         I18nViewUtil.addComponentForKey(KEY_MENU_SQL_ENGINE, itemSqlEngine);
195         itemSqlEngine.setName("itemSqlEngine");
196         itemSqlEngine.setMnemonic('S');
197 
198         // Render the SQL Engine dialog behind scene
199         var titleTabSqlEngine = "SQL Engine";
200 
201         itemSqlEngine.addActionListener(actionEvent -> {
202             for (var i = 0; i < MediatorHelper.tabResults().getTabCount() ; i++) {
203                 if (titleTabSqlEngine.equals(MediatorHelper.tabResults().getTitleAt(i))) {
204                     MediatorHelper.tabResults().setSelectedIndex(i);
205                     return;
206                 }
207             }
208 
209             CreateTabHelper.initializeSplitOrientation();
210 
211             var panelSqlEngine = new SqlEngine();
212             MediatorHelper.tabResults().addTab(titleTabSqlEngine, panelSqlEngine);
213             MediatorHelper.tabResults().setSelectedComponent(panelSqlEngine);  // Focus on the new tab
214 
215             // Create a custom tab header
216             var header = new TabHeader(I18nViewUtil.valueByKey(KEY_MENU_SQL_ENGINE), UiUtil.COG.icon);
217             I18nViewUtil.addComponentForKey(KEY_MENU_SQL_ENGINE, header.getTabLabel());
218 
219             // Apply the custom header to the tab
220             MediatorHelper.tabResults().setTabComponentAt(MediatorHelper.tabResults().indexOfComponent(panelSqlEngine), header);
221 
222             FlatLaf.updateUI();  // required: light, open/close prefs, dark => light artifacts
223             MediatorHelper.frame().revalidate();
224             MediatorHelper.frame().repaint();
225         });
226 
227         return itemSqlEngine;
228     }

267     private JMenu initializeMenuTranslation() {
268         var menuTranslation = new JMenu(I18nUtil.valueByKey("MENUBAR_LANGUAGE"));
269         I18nViewUtil.addComponentForKey("MENUBAR_LANGUAGE", menuTranslation);
270         menuTranslation.setName("menuTranslation");
271         menuTranslation.setMnemonic('L');
272 
273         var groupRadioLanguage = new ButtonGroup();
274         var atomicIsAnySelected = new AtomicBoolean(false);
275         this.appMenubar.modelsItem.forEach(model -> {
276             atomicIsAnySelected.set(atomicIsAnySelected.get() || model.language.isCurrentLanguage());
277             model.menuItem = new JRadioButtonMenuItem(
278                 model.language.getMenuItemLabel(),
279                 model.language.getFlag(),
280                 model.language.isCurrentLanguage()
281             );
282             model.menuItem.addActionListener(actionEvent -> {
283                 this.appMenubar.switchLocale(
284                     model.language == Language.EN
285                     ? Locale.ROOT  // required as no bundle 'en'
286                     : Locale.forLanguageTag(model.language.getLanguageTag())
287                 );
288                 MediatorHelper.model().getMediatorUtils().getPreferencesUtil().withLanguageTag(model.language.getLanguageTag()).persist();
289             });
290             menuTranslation.add(model.menuItem);
291             groupRadioLanguage.add(model.menuItem);
292         });
293 
294         this.appMenubar.modelsItem.stream().filter(modelItem -> modelItem.language == Language.EN)
295         .forEach(modelItem -> {
296             modelItem.menuItem.setSelected(!atomicIsAnySelected.get());
297             modelItem.menuItem.setName("itemEnglish");
298         });
299         this.appMenubar.modelsItem.stream().filter(modelItem -> modelItem.language == Language.RU)
300         .forEach(modelItem -> modelItem.menuItem.setName("itemRussian"));
301         this.appMenubar.modelsItem.stream().filter(modelItem -> modelItem.language == Language.AR)
302         .forEach(modelItem -> modelItem.menuItem.setComponentOrientation(ComponentOrientation.RIGHT_TO_LEFT));
303 
304         return menuTranslation;
305     }

307     public void switchLocaleFromPreferences() {
308         this.appMenubar.modelsItem.stream()
309         .filter(modelItem -> modelItem.language.getLanguageTag().equals(MediatorHelper.model().getMediatorUtils().getPreferencesUtil().getLanguageTag()))
310         .forEach(modelItem -> modelItem.menuItem.doClick());
311     }
