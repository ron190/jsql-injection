44  /**
45   * Does all the magic for getting popups with drop shadows.
46   * It adds the drop shadow border to the Popup,
47   * in {@code #show} it snapshots the screen background as needed,
48   * and in {@code #hide} it cleans up all changes made before.
49   *
50   * @author Karsten Lentzsch
51   * @version $Revision: 1.12 $
52   *
53   * see com.jgoodies.looks.common.ShadowPopupBorder
54   * see com.jgoodies.looks.common.ShadowPopupFactory
55   */
56  public final class ShadowPopup extends Popup {

58      /**
59       * Log4j logger sent to view.
60       */
61      private static final Logger LOGGER = LogManager.getRootLogger();

64       * Max number of items to store in the cache.

66      private static final int MAX_CACHE_SIZE = 5;

68      /**
69       * The cache to use for ShadowPopups.

74       * The singleton instance used to draw all borders.

76      private static final Border SHADOW_BORDER = ShadowPopupBorder.getInstance();

78      /**
79       * The size of the drop shadow.

134     /**
135      * Returns a previously used {@code ShadowPopup}, or a new one
136      * if none of the popups have been recycled.
137      */
138     public static Popup getInstance(Component owner, Component contents, int x, int y, Popup delegate) {
139         
140         ShadowPopup result;
141         
142         synchronized (ShadowPopup.class) {
143             
144             if (cache == null) {
145                 
146                 cache = new ArrayList<>(MAX_CACHE_SIZE);
147             }
148             
149             if (!cache.isEmpty()) {
150                 
151                 result = cache.remove(0);
152                 
153             } else {
154                 
155                 result = new ShadowPopup();
156             }
157         }
158         
159         result.reset(owner, contents, x, y, delegate);
160         
161         return result;
162     }
163 
164     /**
165      * Recycles the ShadowPopup.
166      */
167     private static void recycle(ShadowPopup popup) {
168         
169         synchronized (ShadowPopup.class) {
170             
171             if (cache.size() < MAX_CACHE_SIZE) {
172                 
173                 cache.add(popup);
174             }
175         }
176     }

179         
180         return this.canSnapshot;

183     /**
184      * Hides and disposes of the {@code Popup}. Once a {@code Popup}
185      * has been disposed you should no longer invoke methods on it. A
186      * {@code dispose}d {@code Popup} may be reclaimed and later used
187      * based on the {@code PopupFactory}. As such, if you invoke methods
188      * on a {@code disposed} {@code Popup}, indeterminate
189      * behavior will result.<p>
190      *
191      * In addition to the superclass behavior, we reset the stored
192      * horizontal and vertical drop shadows - if any.
193      */
194     @Override
195     public void hide() {
196         
197         if (this.contents == null) {
198             
199             return;
200         }
201 
202         JComponent parent = (JComponent) this.contents.getParent();
203         this.popup.hide();
204         
205         if (parent != null && SHADOW_BORDER.equals(parent.getBorder())) {
206             
207             parent.setBorder(this.oldBorder);
208             parent.setOpaque(this.oldOpaque);
209             this.oldBorder = null;
210             
211             if (this.heavyWeightContainer != null) {
212                 
213                 parent.putClientProperty(ShadowPopupFactory.PROP_HORIZONTAL_BACKGROUND, null);
214                 parent.putClientProperty(ShadowPopupFactory.PROP_VERTICAL_BACKGROUND, null);
215                 this.heavyWeightContainer = null;
216             }
217         }
218         
219         this.owner = null;
220         this.contents = null;
221         this.popup = null;
222         
223         recycle(this);
224     }
225 
226     /**
227      * Makes the {@code Popup} visible. If the popup has a
228      * heavy-weight container, we try to snapshot the background.
229      * If the {@code Popup} is currently visible, it remains visible.
230      */
231     @Override
232     public void show() {
233         
234         if (this.heavyWeightContainer != null) {
235             
236             this.snapshot();
237         }
238 
239         // Fix #95477: IllegalArgumentException on show()
240         try {
241             this.popup.show();
242         } catch (IllegalArgumentException e) {
243             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e.getMessage(), e);
244         }
245     }

247     /**
248      * Reinitializes this ShadowPopup using the given parameters.
249      *
250      * @param owner component mouse coordinates are relative to, may be null
251      * @param contents the contents of the popup
252      * @param x the desired x location of the popup
253      * @param y the desired y location of the popup
254      * @param popup the popup to wrap
255      */
256     private void reset(
257         Component owner, Component contents, int x, int y,
258         Popup popup
259     ) {
260         
261         this.owner = owner;
262         this.contents = contents;
263         this.popup = popup;
264         this.x = x;
265         this.y = y;
266         
267         // Do not install the shadow border when the contents
268         // has a preferred size less than or equal to 0.
269         // We can't use the size, because it is(0, 0) for new popups.
270         var contentsPrefSize = new Dimension();
271         
272         // Fix #4172: NullPointerException on getPreferredSize()
273         // Implementation by javax.swing.plaf.metal.MetalToolTipUI.getPreferredSize()
274         try {
275             contentsPrefSize = contents.getPreferredSize();
276             
277         } catch(NullPointerException e) {
278             
279             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
280         }
281         
282         if (contentsPrefSize.width <= 0 || contentsPrefSize.height <= 0) {
283             
284             return;
285         }
286         
287         for (Container p = contents.getParent() ; p != null ; p = p.getParent()) {
288             
289             if (p instanceof JWindow || p instanceof Panel) {
290                 
291                 // Workaround for the gray rect problem.
292                 p.setBackground(contents.getBackground());
293                 this.heavyWeightContainer = p;
294                 
295                 break;
296             }
297         }
298         
299         JComponent parent = (JComponent) contents.getParent();
300         this.oldOpaque = parent.isOpaque();
301         this.oldBorder = parent.getBorder();
302         parent.setOpaque(false);
303         parent.setBorder(SHADOW_BORDER);
304         
305         // Pack it because we have changed the border.
306         if (this.heavyWeightContainer != null) {
307             
308             this.heavyWeightContainer.setSize(this.heavyWeightContainer.getPreferredSize());
309             
310         } else {
311             
312             parent.setSize(parent.getPreferredSize());
313         }
314     }

316     /**
317      * Snapshots the background. The snapshots are stored as client
318      * properties of the contents' parent. The next time the border is drawn,
319      * this background will be used.<p>
320      *
321      * Uses a robot on the default screen device to capture the screen
322      * region under the drop shadow. Does <em>not</em> use the window's
323      * device, because that may be an outdated device (due to popup reuse)
324      * and the robot's origin seems to be adjusted with the default screen
325      * device.
326      *
327      * @see #show()
328      * @see com.jgoodies.looks.common.ShadowPopupBorder
329      * @see Robot#createScreenCapture(Rectangle)
330      */
331     private void snapshot() {
332         
333         try {
334             Dimension size = this.heavyWeightContainer.getPreferredSize();
335             int width = size.width;
336             int height = size.height;
337 
338             // Avoid unnecessary and illegal screen captures
339             // for degenerated popups.
340             if (width <= 0 || height <= SHADOW_SIZE) {
341                 
342                 return;
343             }
344 
345             var robot = new Robot(); // uses the default screen device
346 
347             RECT.setBounds(this.x, this.y + height - SHADOW_SIZE, width, SHADOW_SIZE);
348             BufferedImage hShadowBg = robot.createScreenCapture(RECT);
349 
350             RECT.setBounds(this.x + width - SHADOW_SIZE, this.y, SHADOW_SIZE, height - SHADOW_SIZE);
351             BufferedImage vShadowBg = robot.createScreenCapture(RECT);
352 
353             JComponent parent = (JComponent) this.contents.getParent();
354             parent.putClientProperty(ShadowPopupFactory.PROP_HORIZONTAL_BACKGROUND, hShadowBg);
355             parent.putClientProperty(ShadowPopupFactory.PROP_VERTICAL_BACKGROUND, vShadowBg);
356 
357             Container layeredPane = this.getLayeredPane();
358             if (layeredPane == null) {
359                 
360                 // This could happen if owner is null.
361                 return;
362             }
363 
364             int layeredPaneWidth = layeredPane.getWidth();
365             int layeredPaneHeight = layeredPane.getHeight();
366 
367             POINT.x = this.x;
368             POINT.y = this.y;
369             SwingUtilities.convertPointFromScreen(POINT, layeredPane);
370 
371             this.paintHorizontalSnapshot(width, height, hShadowBg, layeredPane, layeredPaneWidth, layeredPaneHeight);
372 
373             this.paintVerticalSnapshot(width, height, vShadowBg, layeredPane, layeredPaneWidth, layeredPaneHeight);
374             
375         } catch (AWTException | SecurityException | IllegalArgumentException e) {
376             
377             this.canSnapshot = false;
378             
379             LOGGER.log(LogLevelUtil.IGNORE, e);
380         }
381     }

383     private void paintVerticalSnapshot(int width, int height, BufferedImage vShadowBg, Container layeredPane, int layeredPaneWidth, int layeredPaneHeight) {
384         
385         // If needed paint dirty region of the vertical snapshot.
386         RECT.x = POINT.x + width - SHADOW_SIZE;
387         RECT.y = POINT.y;
388         RECT.width = SHADOW_SIZE;
389         RECT.height = height - SHADOW_SIZE;
390 
391         this.extracted(vShadowBg, layeredPane, layeredPaneWidth, layeredPaneHeight);
392     }
393 
394     private void paintHorizontalSnapshot(int width, int height, BufferedImage hShadowBg, Container layeredPane, int layeredPaneWidth, int layeredPaneHeight) {
395         
396         // If needed paint dirty region of the horizontal snapshot.
397         RECT.x = POINT.x;
398         RECT.y = POINT.y + height - SHADOW_SIZE;
399         RECT.width = width;
400         RECT.height = SHADOW_SIZE;
401 
402         this.extracted(hShadowBg, layeredPane, layeredPaneWidth, layeredPaneHeight);
403     }
404 
405     private void extracted(BufferedImage shadowBg, Container layeredPane, int layeredPaneWidth, int layeredPaneHeight) {
406         
407         if ((RECT.x + RECT.width) > layeredPaneWidth) {
408             
409             RECT.width = layeredPaneWidth - RECT.x;
410         }
411         
412         if ((RECT.y + RECT.height) > layeredPaneHeight) {
413             
414             RECT.height = layeredPaneHeight - RECT.y;
415         }
416         
417         if (!RECT.isEmpty()) {
418             
419             Graphics g = shadowBg.createGraphics();
420             g.translate(-RECT.x, -RECT.y);
421             g.setClip(RECT);
422             
423             if (layeredPane instanceof JComponent) {
424                 
425                 JComponent c = (JComponent) layeredPane;
426                 boolean doubleBuffered = c.isDoubleBuffered();
427                 c.setDoubleBuffered(false);
428                 ShadowPopup.paintAll(c, g);
429                 c.setDoubleBuffered(doubleBuffered);
430                 
431             } else {
432                 
433                 layeredPane.paintAll(g);
434             }
435             
436             g.dispose();
437         }
438     }
439     
440     private static void paintAll(JComponent c, Graphics g) {
441         
442         // Fix #3127, Fix #6772, Fix #48907: Multiple Exceptions on paintAll()
443         try {
444             c.paintAll(g);
445             
446         } catch (IllegalArgumentException | ArrayIndexOutOfBoundsException | NullPointerException e) {
447             
448             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e.getMessage(), e);
449         }
450     }

452     /**
453      * @return the top level layered pane which contains the owner.
454      */
455     private Container getLayeredPane() {
456         
457         // The code below is copied from PopupFactory#LightWeightPopup#show()
458         Container parent = null;
459         
460         if (this.owner != null) {
461             
462             parent = this.owner instanceof Container
463                 ? (Container) this.owner
464                 : this.owner.getParent();
465         }
466         
467         // Try to find a JLayeredPane and Window to add
468         for (Container p = parent; p != null; p = p.getParent()) {
469             
470             if (p instanceof JRootPane) {
471                 
472                 if (!(p.getParent() instanceof JInternalFrame)) {
473                     
474                     parent = ((JRootPane) p).getLayeredPane();
475                 }
476                 // Continue, so that if there is a higher JRootPane, we'll
477                 // pick it up.
478                 
479             } else if (p instanceof Window) {
480                 
481                 if (parent == null) {
482                     parent = p;
483                 }
484                 
485                 break;
486             }
487         }
488         
489         return parent;
490     }
