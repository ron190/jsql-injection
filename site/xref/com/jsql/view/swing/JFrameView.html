38  /**
39   * View in the MVC pattern, defines all the components
40   * and process actions sent by the model.<br>
41   * Main groups of components:<br>
42   * - at the top: textfield inputs,<br>
43   * - at the center: tree on the left, table on the right,<br>
44   * - at the bottom: information labels.
45   */
46  public class JFrameView extends JFrame {

50      /**
51       * Map of terminal by unique identifier.
52       */
53      private final Map<UUID, AbstractExploit> mapUuidShell = new HashMap<>();
54      private final transient SubscriberInteraction subscriber = new SubscriberInteraction("com.jsql.view.swing.interaction");
55      private TabManagers tabManagers;
56      private boolean isMaximized = false;
57      private final InjectionModel injectionModel;
58      private SplitNS splitNS;  // main
59  
60      public JFrameView(InjectionModel injectionModel) {  // Build the GUI: add app icon, tree icons, the 3 main panels
61          super(StringUtil.APP_NAME);
62          this.injectionModel = injectionModel;
63          MediatorHelper.register(this);
64          UiUtil.prepareGUI();  // Load UI before any component
65          this.initPaneComponents();
66          this.initWindow();
67          this.initShortcuts();
68          this.displayVersion();
69          I18nUtil.checkCurrentLanguage();
70          this.check4K();
71  
72          SwingUtilities.invokeLater(() -> {  // paint native blu svg in theme color behind scene
73              AppMenubar.applyTheme(injectionModel.getMediatorUtils().getPreferencesUtil().getThemeFlatLafName());  // refresh missing components
74              if (injectionModel.getMediatorUtils().getProxyUtil().isNotLive(GitUtil.ShowOnConsole.YES)) {  // network access
75                  return;
76              }
77              if (injectionModel.getMediatorUtils().getPreferencesUtil().isCheckingUpdate()) {
78                  injectionModel.getMediatorUtils().getGitUtil().checkUpdate(GitUtil.ShowOnConsole.NO);
79              }
80              injectionModel.getMediatorUtils().getGitUtil().showNews();
81              this.setVisible(true);
82              MediatorHelper.panelAddressBar().getTextFieldAddress().requestFocusInWindow();  // required here to get focus
83          });
84      }

86      private void initPaneComponents() {
87          // Define the default panel: each component on a vertical line
88          this.getContentPane().setLayout(new BoxLayout(this.getContentPane(), BoxLayout.PAGE_AXIS));
89  
90          this.tabManagers = new TabManagers();  // Tab manager linked to cards
91          this.add(this.tabManagers);
92  
93          var menubar = new AppMenubar();
94          this.setJMenuBar(menubar);
95          MediatorHelper.register(menubar);
96  
97          var panelAddressBar = new PanelAddressBar();  // Textfield at the top
98          MediatorHelper.register(panelAddressBar);
99          this.add(panelAddressBar);
100 
101         var mainPanel = new JPanel(new BorderLayout());  // Main panel for tree and tables in the middle
102         this.splitNS = new SplitNS();
103         mainPanel.add(this.splitNS);
104         this.add(mainPanel);
105 
106         menubar.getMenuWindows().switchLocaleFromPreferences();
107     }

109     private void initWindow() {
110         this.setIconImages(UiUtil.getIcons());  // define small and large app icons
111         var preferences = Preferences.userRoot().node(InjectionModel.class.getName());
112         this.addWindowStateListener(e -> this.isMaximized = (e.getNewState() & Frame.MAXIMIZED_BOTH) == Frame.MAXIMIZED_BOTH);
113         this.addWindowListener(new WindowAdapter() {
114             @Override
115             public void windowOpened(WindowEvent event) {
116                 double nsProportion = preferences.getDouble(PreferencesUtil.NS_SPLIT, 0.75);
117                 if (preferences.getBoolean(PreferencesUtil.IS_MAXIMIZED, false)) {
118                     JFrameView.this.setExtendedState(Frame.MAXIMIZED_BOTH);
119                 }
120                 // Proportion means different location relative to whether frame is maximized or not
121                 // Maximizing must wait AWT events to reflect the new divider location proportion
122                 SwingUtilities.invokeLater(() -> JFrameView.this.splitNS.setDividerLocation(
123                     Math.max(0.0, Math.min(1.0, nsProportion))
124                 ));
125             }
126 
127             @Override
128             public void windowClosing(WindowEvent e) {
129                 preferences.putBoolean(PreferencesUtil.IS_MAXIMIZED, JFrameView.this.isMaximized);
130                 preferences.putInt(
131                     PreferencesUtil.EW_SPLIT,
132                     JFrameView.this.splitNS.getSplitEW().getDividerLocation()  // TODO not compatible arabic location
133                 );
134 
135                 int nsDividerLocation = JFrameView.this.splitNS.getDividerLocation();
136                 int nsHeight = JFrameView.this.splitNS.getHeight();
137                 double nsProportion = (100.0 * nsDividerLocation) / nsHeight;
138                 double nsLocationProportionCapped = Math.max(0.0, Math.min(1.0,
139                     nsProportion / 100.0
140                 ));
141 
142                 // Divider location changes when window is maximized, instead stores location percentage between 0.0 and 1.0
143                 preferences.putDouble(PreferencesUtil.NS_SPLIT, nsLocationProportionCapped);
144 
145                 preferences.putBoolean(PreferencesUtil.BINARY_VISIBLE, false);
146                 preferences.putBoolean(PreferencesUtil.CHUNK_VISIBLE, false);
147                 preferences.putBoolean(PreferencesUtil.NETWORK_VISIBLE, false);
148                 preferences.putBoolean(PreferencesUtil.JAVA_VISIBLE, false);
149                 for (var i = 0 ; i < MediatorHelper.tabConsoles().getTabCount() ; i++) {
150                     if ("CONSOLE_BINARY_LABEL".equals(MediatorHelper.tabConsoles().getTabComponentAt(i).getName())) {
151                         preferences.putBoolean(PreferencesUtil.BINARY_VISIBLE, true);
152                     } else if ("CONSOLE_CHUNK_LABEL".equals(MediatorHelper.tabConsoles().getTabComponentAt(i).getName())) {
153                         preferences.putBoolean(PreferencesUtil.CHUNK_VISIBLE, true);
154                     } else if ("CONSOLE_NETWORK_LABEL".equals(MediatorHelper.tabConsoles().getTabComponentAt(i).getName())) {
155                         preferences.putBoolean(PreferencesUtil.NETWORK_VISIBLE, true);
156                     } else if ("CONSOLE_JAVA_LABEL".equals(MediatorHelper.tabConsoles().getTabComponentAt(i).getName())) {
157                         preferences.putBoolean(PreferencesUtil.JAVA_VISIBLE, true);
158                     }
159                 }
160             }
161         });
162         
163         this.setSize(1024, 768);
164         this.setLocationRelativeTo(null);  // center the window
165         this.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
166     }
167 
168     private void initShortcuts() {
169         HotkeyUtil.addShortcut(this.getRootPane(), MediatorHelper.tabResults());
170         HotkeyUtil.addTextFieldShortcutSelectAll();
171     }
172 
173     public void resetInterface() {  // Empty the interface
174         this.mapUuidShell.clear();
175         MediatorHelper.panelAddressBar().getPanelTrailingAddress().reset();
176         MediatorHelper.panelConsoles().reset();
177         MediatorHelper.treeDatabase().reset();
178         
179         for (var i = 0 ; i < MediatorHelper.tabConsoles().getTabCount() ; i++) {
180             var tabComponent = MediatorHelper.tabConsoles().getTabComponentAt(i);
181             if (tabComponent != null) {
182                 tabComponent.setFont(tabComponent.getFont().deriveFont(Font.PLAIN));
183             }
184         }
185         
186         Stream.of(MediatorHelper.managerFile(), MediatorHelper.managerExploit()).forEach(managerList -> {
187             managerList.setButtonEnable(false);
188             managerList.changePrivilegeIcon(UiUtil.SQUARE.getIcon());
189         });
190     }
191 
192     private void displayVersion() {
193         LOGGER.log(
194             LogLevelUtil.CONSOLE_DEFAULT,
195             "{} v{} on Java {}-{}-{}",
196             () -> StringUtil.APP_NAME,
197             () -> this.injectionModel.getPropertiesUtil().getVersionJsql(),
198             () -> SystemUtils.JAVA_VERSION,
199             () -> SystemUtils.OS_ARCH,
200             () -> SystemUtils.USER_LANGUAGE
201         );
202     }
203 
204     private void check4K() {
205         Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
206         int width = (int) screenSize.getWidth();
207         if (width >= 3840 && !this.injectionModel.getMediatorUtils().getPreferencesUtil().is4K()) {
208             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Screen compatible with resolution 4K, enable high-definition in Preferences");
209         }
210     }
