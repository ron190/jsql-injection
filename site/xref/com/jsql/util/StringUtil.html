44      // Define the schema of conversion to html entities
45      private static final CharEncoder DECIMAL_HTML_ENCODER = new CharEncoder("&#", ";", 10);

47      /**
48       * This utility class defines a schema used to encode a text into a specialized
49       * representation
50       */
51      private static class CharEncoder {
52          
53          private final String prefix;
54          private final String suffix;
55          private final int radix;
56          
57          public CharEncoder(String prefix, String suffix, int radix) {
58              this.prefix = prefix;
59              this.suffix = suffix;
60              this.radix = radix;
61          }
62          
63          protected void encode(char c, StringBuilder buff) {
64              buff
65              .append(this.prefix)
66              .append(Integer.toString(c, this.radix))
67              .append(this.suffix);
68          }
69      }

75      /**
76       * Convert special characters like Chinese and Arabic letters to the corresponding html entities.
77       * @param text string to encode
78       * @return string encoded in html entities
79       */
80      public static String decimalHtmlEncode(String text) {
81          return decimalHtmlEncode(text, false);
82      }

84      public static String decimalHtmlEncode(String text, boolean isRaw) {
85          var result = StringUtil.encode(text);
86          if (isRaw) {
87              return result
88                  .replace("<", "&lt;")
89                  .replace(">", "&gt;")
90                  .replace("&", "&amp;");
91          } else {
92              return result;
93          }
94      }

96      /**
97       * Non-trivial methods to convert unicode characters to html entities.
98       *
99       * @param text string to encode
100      * @return string representation using the encoder schema
101      */
102     private static String encode(String text) {
103         var buff = new StringBuilder();
104         for (var i = 0 ; i < text.length() ; i++) {
105             if (text.charAt(i) > 128) {
106                 StringUtil.DECIMAL_HTML_ENCODER.encode(text.charAt(i), buff);
107             } else {
108                 buff.append(text.charAt(i));
109             }
110         }
111         return buff.toString();
112     }

114     /**
115      * Convert a hexadecimal String to String.
116      * @param hex Hexadecimal String to convert
117      * @return The string converted from hex
118      */
119     public static String hexstr(String hex) {
120         var bytes = new byte[hex.length() / 2];
121         for (var i = 0 ; i < bytes.length ; i++) {
122             bytes[i] = (byte) Integer.parseInt(hex.substring(2 * i, 2 * i + 2), 16);
123         }
124         return new String(bytes, StandardCharsets.UTF_8);
125     }

127     public static boolean isUtf8(String text) {
128         if (text == null) {
129             return false;
130         }
131         
132         var detector = new UniversalDetector(null);
133         detector.handleData(text.getBytes(StandardCharsets.UTF_8), 0, text.length() - 1);
134         detector.dataEnd();
135         String encoding = detector.getDetectedCharset();
136         return encoding != null;
137     }
138     
139     public static String detectUtf8(String text) {
140         if (text == null) {
141             return org.apache.commons.lang3.StringUtils.EMPTY;
142         }
143         
144         String encoding = null;
145         
146         // ArrayIndexOutOfBoundsException on handleData()
147         try {
148             var detector = new UniversalDetector(null);
149             detector.handleData(text.getBytes(StandardCharsets.UTF_8), 0, text.length() - 1);
150             detector.dataEnd();
151             encoding = detector.getDetectedCharset();
152             
153         } catch (ArrayIndexOutOfBoundsException e) {
154             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
155         }
156         
157         String result = text;
158         if (encoding != null) {
159             result = new String(text.getBytes(StandardCharsets.UTF_8), StandardCharsets.UTF_8);
160         }
161         return result;
162     }

214     /**
215      * Zip a string.
216      * @param str Text to zip
217      * @return Zipped string
218      */
219     public static String compress(String str) throws IOException {
220         if (org.apache.commons.lang3.StringUtils.isEmpty(str)) {
221             return str;
222         }
223         
224         var out = new ByteArrayOutputStream();
225         var gzip = new GZIPOutputStream(out);
226         gzip.write(str.getBytes(StandardCharsets.UTF_8));
227         gzip.close();
228         
229         return out.toString(StandardCharsets.ISO_8859_1);
230     }

232     /**
233      * Unzip a String encoded from base64 or hexadecimal.
234      * @param str String to unzip
235      * @return String unzipped
236      */
237     public static String decompress(String str) throws IOException {
238         if (org.apache.commons.lang3.StringUtils.isEmpty(str)) {
239             return str;
240         }
241         
242         final var encode = "ISO-8859-1";
243         var gis = new GZIPInputStream(new ByteArrayInputStream(str.getBytes(encode)));
244         var bf = new BufferedReader(new InputStreamReader(gis, encode));
245 
246         var buff = new char[1024];
247         int read;
248         var response = new StringBuilder();
249         
250         while ((read = bf.read(buff)) != -1) {
251             response.append(buff, 0, read);
252         }
253         return response.toString();
254     }

305     public static String cleanSql(String query) {
306         return removeSqlComment(query)
307             .replaceAll("(?s)([^\\s\\w])(\\s+)", "$1")  // Remove spaces after a word
308             .replaceAll("(?s)(\\s+)([^\\s\\w])", "$2")  // Remove spaces before a word
309             .replaceAll("(?s)\\s+", " ")  // Replace spaces
310             .trim();
311     }

313     /**
314      * Remove SQL comments except tamper /**\/ /*!...*\/
315      * Negative lookahead: don't match tamper empty comment /**\/ or version comment /*!...*\/
316      * JavaScript: (?!\/\*!.*\*\/|\/\*\*\/)\/\*.*\*\/
317      */
318     public static String removeSqlComment(String query) {
319         return query.replaceAll(
320             "(?s)(?!/\\*\\*/|/\\*!.*\\*/)/\\*.*?\\*/",
321             org.apache.commons.lang3.StringUtils.EMPTY
322         );
323     }
