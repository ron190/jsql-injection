15  /**
16   * CRC64 checksum calculator based on the polynom specified in ISO 3309. The
17   * implementation is based on the following publications:
18   * 
19   * <ul>
20   * <li>http://en.wikipedia.org/wiki/Cyclic_redundancy_check</li>
21   * <li>http://www.geocities.com/SiliconValley/Pines/8659/crc.htm</li>
22   * </ul>
23   */
24  public class Crc64Helper {
25      
26      private static final long POLY64REV = 0xd800000000000000L;
27      
28      private static final long[] LOOKUPTABLE;
29      
30      private Crc64Helper() {
31          // Util class
32      }
33  
34      static {
35          LOOKUPTABLE = new long[0x100];
36          for (int i = 0; i < 0x100; i++) {
37              long v = i;
38              for (int j = 0; j < 8; j++) {
39                  if ((v & 1) == 1) {
40                      v = (v >>> 1) ^ POLY64REV;
41                  } else {
42                      v = v >>> 1;
43                  }
44              }
45              LOOKUPTABLE[i] = v;
46          }
47      }
48  
49      /**
50       * Calculates the CRC64 checksum for the given data array.
51       * 
52       * @param data
53       *            data to calculate checksum for
54       * @return checksum value
55       */
56      public static String generateCRC64(final byte[] data) {
57          long sum = 0;
58          for (byte element : data) {
59              final int lookupidx = ((int) sum ^ element) & 0xff;
60              sum = (sum >>> 8) ^ LOOKUPTABLE[lookupidx];
61          }
62          return String.valueOf(sum);
63      }
64  }
