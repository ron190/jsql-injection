52      /**
53       * Verify if application is up-to-date against the version on GitHub.
54       * @param displayUpdateMessage YES for manual update verification, hidden otherwise
55       */
56      public void checkUpdate(ShowOnConsole displayUpdateMessage) {
57          if (displayUpdateMessage == ShowOnConsole.YES) {
58              LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, () -> I18nUtil.valueByKey("UPDATE_LOADING"));
59          }
60          try {
61              var versionGit = Float.parseFloat(this.callService().getString("version"));
62              if (versionGit > Float.parseFloat(this.injectionModel.getPropertiesUtil().getVersionJsql())) {
63                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, () -> I18nUtil.valueByKey("UPDATE_NEW_VERSION"));
64              } else if (displayUpdateMessage == ShowOnConsole.YES) {
65                  LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, () -> I18nUtil.valueByKey("UPDATE_UPTODATE"));
66              }
67          } catch (NumberFormatException | JSONException e) {
68              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, I18nUtil.valueByKey("UPDATE_EXCEPTION"));
69          }
70      }

72      /**
73       * Define the body of an issue to send to GitHub for an unhandled exception.
74       * It adds different system data to the body and remove sensible data like
75       * injection URL.
76       * @param threadName name of thread where the exception occurred
77       * @param throwable unhandled exception to report to GitHub
78       */
79      public void sendUnhandledException(String threadName, Throwable throwable) {
80          Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
81          int width = (int) screenSize.getWidth();
82          int height = (int) screenSize.getHeight();
83  
84          var osMetadata = String.join(
85              "\n",
86              String.format(
87                  "jSQL: v%s",
88                  this.injectionModel.getPropertiesUtil().getVersionJsql()
89              ),
90              String.format(
91                  "Java: v%s-%s-%s on %s%s",
92                  SystemUtils.JAVA_VERSION,
93                  SystemUtils.OS_ARCH,
94                  SystemUtils.USER_LANGUAGE,
95                  SystemUtils.JAVA_VENDOR,
96                  SystemUtils.JAVA_RUNTIME_NAME
97              ),
98              String.format(
99                  "OS: %s (v%s)",
100                 SystemUtils.OS_NAME, SystemUtils.OS_VERSION
101             ),
102             String.format(
103                 "Display: %s (%sx%s, %s)",
104                 UIManager.getLookAndFeel().getName(),
105                 width,
106                 height,
107                 Locale.getDefault().getLanguage()
108             ),
109             String.format(
110                 "Strategy: %s",
111                 this.injectionModel.getMediatorStrategy().getStrategy() != null
112                 ? this.injectionModel.getMediatorStrategy().getStrategy().getName()
113                 : "undefined"
114             ),
115             String.format(
116                 "Db engine: %s",
117                 this.injectionModel.getMediatorVendor().getVendor().toString()
118             )
119         );
120         
121         var exceptionText = String.format("Exception on %s%n%s%n", threadName, ExceptionUtils.getStackTrace(throwable).trim());
122         var clientDescription = String.format("```yaml%n%s%n```%n```java%n%s```", osMetadata, exceptionText);
123         
124         clientDescription = clientDescription.replaceAll("(https?://[.a-zA-Z_0-9]*)+", StringUtils.EMPTY);
125         this.sendReport(clientDescription, ShowOnConsole.NO, "Unhandled "+ throwable.getClass().getSimpleName());
126     }

128     /**
129      * Connect to GitHub webservices and create an Issue on the repository.
130      * Used by translation protocol, unhandled exception detection and manual Issue reporting.
131      * @param reportBody text of the Issue
132      * @param showOnConsole in case of manual Issue reporting. Hidden in case of automatic reporting of unhandled exception.
133      * @param reportTitle title of the Issue
134      */
135     public void sendReport(String reportBody, ShowOnConsole showOnConsole, String reportTitle) {
136         if (this.injectionModel.getMediatorUtils().getProxyUtil().isNotLive(showOnConsole)) {
137             return;
138         }
139 
140         var httpRequest = HttpRequest.newBuilder()
141             .uri(URI.create(this.injectionModel.getMediatorUtils().getPropertiesUtil().getProperties().getProperty("github.issues.url")))
142             .setHeader(
143                 "Authorization",
144                 "token "
145                 + StringUtil.base64Decode(
146                     this.injectionModel.getMediatorUtils().getPropertiesUtil().getProperties().getProperty("github.token")
147                 )
148             )
149             .POST(BodyPublishers.ofString(
150                 new JSONObject()
151                 .put("title", reportTitle)
152                 .put("body", reportBody)
153                 .toString()
154             ))
155             .timeout(Duration.ofSeconds(15))
156             .build();
157             
158         try {
159             HttpResponse<String> response = this.injectionModel.getMediatorUtils().getConnectionUtil().getHttpClient().send(httpRequest, BodyHandlers.ofString());
160             this.readGithubResponse(response, showOnConsole);
161         } catch (InterruptedException | IOException e) {
162             if (showOnConsole == ShowOnConsole.YES) {
163                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Error during GitHub report connection: %s", e.getMessage()));
164             }
165             if (e instanceof InterruptedException) {
166                 Thread.currentThread().interrupt();
167             }
168         }
169     }
170     
171     private void readGithubResponse(HttpResponse<String> response, ShowOnConsole showOnConsole) throws IOException {
172         try {
173             if (showOnConsole == ShowOnConsole.YES) {
174                 var jsonObjectResponse = new JSONObject(response.body());
175                 var urlIssue = jsonObjectResponse.getString("html_url");
176                 LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Sent to GitHub: {}", urlIssue);
177             }
178         } catch (Exception e) {
179             throw new IOException("Connection to the GitHub API failed, check your connection or update jSQL");
180         }
181     }

183     /**
184      * Displays news information on the console from GitHub web service.
185      * Infos concern the general roadmap for the application, current development status
186      * and other useful statements for the community.
187      */
188     public void showNews() {
189         try {
190             var news = this.callService().getJSONArray("news");
191             for (var index = 0 ; index < news.length() ; index++) {
192                 LOGGER.log(LogLevelUtil.CONSOLE_INFORM, news.get(index));
193             }
194         } catch (JSONException e) {
195             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Connection to the GitHub API failed");
196         }
197     }

199     /**
200      * Instantiate the jsonObject from json data if not already set.
201      * @return jsonObject describing json data
202      */
203     public JSONObject callService() {
204         if (this.jsonObject == null) {
205             String json = this.injectionModel.getMediatorUtils().getConnectionUtil().getSource(
206                 this.injectionModel.getMediatorUtils().getPropertiesUtil().getProperties().getProperty("github.webservice.url")
207             );
208             // Fix #45349: JSONException on new JSONObject(json)
209             try {
210                 this.jsonObject = new JSONObject(json);
211             } catch (JSONException e) {
212                 try {
213                     this.jsonObject = new JSONObject("{\"version\": \"0\", \"news\": []}");
214                 } catch (JSONException eInner) {
215                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Fetching default JSON failed", eInner);
216                 }
217                 LOGGER.log(
218                     LogLevelUtil.CONSOLE_ERROR,
219                     "Fetching configuration from GitHub failed. Wait for service to be available, check your connection or update jSQL"
220                 );
221             }
222         }
223         return this.jsonObject;
224     }
