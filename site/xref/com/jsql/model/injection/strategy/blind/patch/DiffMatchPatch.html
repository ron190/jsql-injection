156     /**
157      * Find the differences between two texts.  Simplifies the problem by
158      * stripping any common prefix or suffix off the texts before diffing.
159      * @param valueText1 Old string to be diffed.
160      * @param valueText2 New string to be diffed.
161      * @param checklines Speedup flag.  If false, then don't run a
162      *     line-level diff first to identify the changed areas.
163      *     If true, then run a faster slightly less optimal diff.
164      * @param deadline Time when the diff should be complete by.  Used
165      *     internally for recursive calls.  Users should set DiffTimeout instead.
166      * @return Linked List of Diff objects.
167      */
168     private LinkedList<Diff> diffMain(String valueText1, String valueText2, boolean checklines, long deadline) {
169 
170         String text1 = valueText1;
171         String text2 = valueText2;
172         
173         // Check for null inputs.
174         if (text1 == null || text2 == null) {
175             throw new IllegalArgumentException("Null inputs. (diff_main)");
176         }
177 
178         // Check for equality (speedup).
179         LinkedList<Diff> diffs;
180         if (text1.equals(text2)) {
181             diffs = new LinkedList<>();
182             if (!text1.isEmpty()) {
183                 diffs.add(new Diff(Operation.EQUAL, text1));
184             }
185             return diffs;
186         }
187 
188         // Trim off common prefix (speedup).
189         int commonlength = this.diffCommonPrefix(text1, text2);
190         String commonprefix = text1.substring(0, commonlength);
191         text1 = text1.substring(commonlength);
192         text2 = text2.substring(commonlength);
193 
194         // Trim off common suffix (speedup).
195         commonlength = this.diffCommonSuffix(text1, text2);
196         String commonsuffix = text1.substring(text1.length() - commonlength);
197         text1 = text1.substring(0, text1.length() - commonlength);
198         text2 = text2.substring(0, text2.length() - commonlength);
199 
200         // Compute the diff on the middle block.
201         diffs = this.diffCompute(text1, text2, checklines, deadline);
202 
203         // Restore the prefix and suffix.
204         if (!commonprefix.isEmpty()) {
205             diffs.addFirst(new Diff(Operation.EQUAL, commonprefix));
206         }
207         if (!commonsuffix.isEmpty()) {
208             diffs.addLast(new Diff(Operation.EQUAL, commonsuffix));
209         }
210 
211         this.diffCleanupMerge(diffs);
212         return diffs;
213     }

215     /**
216      * Find the differences between two texts.  Assumes that the texts do not
217      * have any common prefix or suffix.
218      * @param text1 Old string to be diffed.
219      * @param text2 New string to be diffed.
220      * @param checklines Speedup flag.  If false, then don't run a
221      *     line-level diff first to identify the changed areas.
222      *     If true, then run a faster slightly less optimal diff.
223      * @param deadline Time when the diff should be complete by.
224      * @return Linked List of Diff objects.
225      */
226     private LinkedList<Diff> diffCompute(String text1, String text2, boolean checklines, long deadline) {
227 
228         LinkedList<Diff> diffs = new LinkedList<>();
229 
230         if (text1.isEmpty()) {
231             // Just add some text (speedup).
232             diffs.add(new Diff(Operation.INSERT, text2));
233             return diffs;
234         }
235 
236         if (text2.isEmpty()) {
237             // Just delete some text (speedup).
238             diffs.add(new Diff(Operation.DELETE, text1));
239             return diffs;
240         }
241 
242         {
243             // New scope to garbage collect longtext and shorttext.
244             String longtext = text1.length() > text2.length() ? text1 : text2;
245             String shorttext = text1.length() > text2.length() ? text2 : text1;
246             int i = longtext.indexOf(shorttext);
247             if (i != -1) {
248                 // Shorter text is inside the longer text (speedup).
249                 Operation op = (text1.length() > text2.length()) ?
250                         Operation.DELETE : Operation.INSERT;
251                 diffs.add(new Diff(op, longtext.substring(0, i)));
252                 diffs.add(new Diff(Operation.EQUAL, shorttext));
253                 diffs.add(new Diff(op, longtext.substring(i + shorttext.length())));
254                 return diffs;
255             }
256 
257             if (shorttext.length() == 1) {
258                 // Single character string.
259                 // After the previous speedup, the character can't be an equality.
260                 diffs.add(new Diff(Operation.DELETE, text1));
261                 diffs.add(new Diff(Operation.INSERT, text2));
262                 return diffs;
263             }
264         }
265 
266         // Check to see if the problem can be split in two.
267         String[] hm = this.diffHalfMatch(text1, text2);
268         if (hm != null) {
269             // A half-match was found, sort out the return data.
270             String text1A = hm[0];
271             String text1B = hm[1];
272             String text2A = hm[2];
273             String text2B = hm[3];
274             String midCommon = hm[4];
275             // Send both pairs off for separate processing.
276             LinkedList<Diff> diffsA = this.diffMain(text1A, text2A, checklines, deadline);
277             List<Diff> diffsB = this.diffMain(text1B, text2B, checklines, deadline);
278             // Merge the results.
279             diffs = diffsA;
280             diffs.add(new Diff(Operation.EQUAL, midCommon));
281             diffs.addAll(diffsB);
282             return diffs;
283         }
284 
285         if (checklines && text1.length() > 100 && text2.length() > 100) {
286             return this.diffLineMode(text1, text2, deadline);
287         }
288 
289         return this.diffBisect(text1, text2, deadline);
290     }

292     /**
293      * Do a quick line-level diff on both strings, then rediff the parts for
294      * greater accuracy.
295      * This speedup can produce non-minimal diffs.
296      * @param valueText1 Old string to be diffed.
297      * @param valueText2 New string to be diffed.
298      * @param deadline Time when the diff should be complete by.
299      * @return Linked List of Diff objects.
300      */
301     private LinkedList<Diff> diffLineMode(String valueText1, String valueText2, long deadline) {
302 
303         // Scan the text on a line-by-line basis first.
304         LinesToCharsResult b = this.diffLinesToChars(valueText1, valueText2);
305         String text1 = b.chars1;
306         String text2 = b.chars2;
307         List<String> linearray = b.lineArray;
308 
309         LinkedList<Diff> diffs = this.diffMain(text1, text2, false, deadline);
310 
311         // Convert the diff back to original text.
312         this.diffCharsToLines(diffs, linearray);
313         // Eliminate freak matches (e.g. blank lines)
314         this.diffCleanupSemantic(diffs);
315 
316         // Rediff any replacement blocks, this time character-by-character.
317         // Add a dummy entry at the end.
318         diffs.add(new Diff(Operation.EQUAL, ""));
319         int countDelete = 0;
320         int countInsert = 0;
321         StringBuilder textDelete = new StringBuilder();
322         StringBuilder textInsert = new StringBuilder();
323         ListIterator<Diff> pointer = diffs.listIterator();
324         Diff thisDiff = pointer.next();
325 
326         while (thisDiff != null) {
327             switch (thisDiff.getOperation()) {
328             case INSERT:
329                 countInsert++;
330                 textInsert.append(thisDiff.getText());
331                 break;
332             case DELETE:
333                 countDelete++;
334                 textDelete.append(thisDiff.getText());
335                 break;
336             case EQUAL:
337                 // Upon reaching an equality, check for prior redundancies.
338                 if (countDelete >= 1 && countInsert >= 1) {
339                     // Delete the offending records and add the merged ones.
340                     pointer.previous();
341                     for (int j = 0; j < countDelete + countInsert; j++) {
342                         pointer.previous();
343                         pointer.remove();
344                     }
345                     for (Diff newDiff : this.diffMain(textDelete.toString(), textInsert.toString(), false, deadline)) {
346                         pointer.add(newDiff);
347                     }
348                 }
349                 countInsert = 0;
350                 countDelete = 0;
351                 textDelete.setLength(0);
352                 textInsert.setLength(0);
353                 break;
354             }
355             thisDiff = pointer.hasNext() ? pointer.next() : null;
356         }
357         diffs.removeLast();  // Remove the dummy entry at the end.
358 
359         return diffs;
360     }

362     /**
363      * Find the 'middle snake' of a diff, split the problem in two
364      * and return the recursively constructed diff.
365      * See Myers 1986 paper: An O(ND) Difference Algorithm and Its Variations.
366      * @param text1 Old string to be diffed.
367      * @param text2 New string to be diffed.
368      * @param deadline Time at which to bail if not yet complete.
369      * @return LinkedList of Diff objects.
370      */
371     protected LinkedList<Diff> diffBisect(String text1, String text2, long deadline) {
372 
373         // Cache the text lengths to prevent multiple calls.
374         int text1Length = text1.length();
375         int text2Length = text2.length();
376         int maxD = (text1Length + text2Length + 1) / 2;
377         int vLength = 2 * maxD;
378         int[] v1 = new int[vLength];
379         int[] v2 = new int[vLength];
380         for (int x = 0; x < vLength; x++) {
381             v1[x] = -1;
382             v2[x] = -1;
383         }
384         v1[maxD + 1] = 0;
385         v2[maxD + 1] = 0;
386         int delta = text1Length - text2Length;
387         // If the total number of characters is odd, then the front path will
388         // collide with the reverse path.
389         boolean front = delta % 2 != 0;
390         // Offsets for start and end of k loop.
391         // Prevents mapping of space beyond the grid.
392         int k1start = 0;
393         int k1end = 0;
394         int k2start = 0;
395         int k2end = 0;
396 
397         for (int d = 0; d < maxD; d++) {
398             // Bail out if deadline is reached.
399             if (System.currentTimeMillis() > deadline) {
400                 break;
401             }
402 
403             // Walk the front path one step.
404             for (int k1 = -d + k1start; k1 <= d - k1end; k1 += 2) {
405                 int k1Offset = maxD + k1;
406                 int x1;
407                 if (k1 == -d || (k1 != d && v1[k1Offset - 1] < v1[k1Offset + 1])) {
408                     x1 = v1[k1Offset + 1];
409                 } else {
410                     x1 = v1[k1Offset - 1] + 1;
411                 }
412                 int y1 = x1 - k1;
413                 while (x1 < text1Length && y1 < text2Length
414                         && text1.charAt(x1) == text2.charAt(y1)) {
415                     x1++;
416                     y1++;
417                 }
418                 v1[k1Offset] = x1;
419                 if (x1 > text1Length) {
420                     // Ran off the right of the graph.
421                     k1end += 2;
422                 } else if (y1 > text2Length) {
423                     // Ran off the bottom of the graph.
424                     k1start += 2;
425                 } else if (front) {
426                     int k2Offset = maxD + delta - k1;
427                     if (k2Offset >= 0 && k2Offset < vLength && v2[k2Offset] != -1) {
428                         // Mirror x2 onto top-left coordinate system.
429                         int x2 = text1Length - v2[k2Offset];
430                         if (x1 >= x2) {
431                             // Overlap detected.
432                             return this.diffBisectSplit(text1, text2, x1, y1, deadline);
433                         }
434                     }
435                 }
436             }
437 
438             // Walk the reverse path one step.
439             for (int k2 = -d + k2start; k2 <= d - k2end; k2 += 2) {
440                 int k2Offset = maxD + k2;
441                 int x2;
442                 if (k2 == -d || (k2 != d && v2[k2Offset - 1] < v2[k2Offset + 1])) {
443                     x2 = v2[k2Offset + 1];
444                 } else {
445                     x2 = v2[k2Offset - 1] + 1;
446                 }
447                 int y2 = x2 - k2;
448                 while (x2 < text1Length && y2 < text2Length
449                         && text1.charAt(text1Length - x2 - 1)
450                         == text2.charAt(text2Length - y2 - 1)) {
451                     x2++;
452                     y2++;
453                 }
454                 v2[k2Offset] = x2;
455                 if (x2 > text1Length) {
456                     // Ran off the left of the graph.
457                     k2end += 2;
458                 } else if (y2 > text2Length) {
459                     // Ran off the top of the graph.
460                     k2start += 2;
461                 } else if (!front) {
462                     int k1Offset = maxD + delta - k2;
463                     if (k1Offset >= 0 && k1Offset < vLength && v1[k1Offset] != -1) {
464                         int x1 = v1[k1Offset];
465                         int y1 = maxD + x1 - k1Offset;
466                         // Mirror x2 onto top-left coordinate system.
467                         x2 = text1Length - x2;
468                         if (x1 >= x2) {
469                             // Overlap detected.
470                             return this.diffBisectSplit(text1, text2, x1, y1, deadline);
471                         }
472                     }
473                 }
474             }
475         }
476         // Diff took too long and hit the deadline or
477         // number of diffs equals number of characters, no commonality at all.
478         LinkedList<Diff> diffs = new LinkedList<>();
479         diffs.add(new Diff(Operation.DELETE, text1));
480         diffs.add(new Diff(Operation.INSERT, text2));
481         return diffs;
482     }

484     /**
485      * Given the location of the 'middle snake', split the diff in two parts
486      * and recurse.
487      * @param text1 Old string to be diffed.
488      * @param text2 New string to be diffed.
489      * @param x Index of split point in text1.
490      * @param y Index of split point in text2.
491      * @param deadline Time at which to bail if not yet complete.
492      * @return LinkedList of Diff objects.
493      */
494     private LinkedList<Diff> diffBisectSplit(String text1, String text2, int x, int y, long deadline) {
495 
496         String text1a = text1.substring(0, x);
497         String text2a = text2.substring(0, y);
498         String text1b = text1.substring(x);
499         String text2b = text2.substring(y);
500 
501         // Compute both diffs serially.
502         LinkedList<Diff> diffs = this.diffMain(text1a, text2a, false, deadline);
503         List<Diff> diffsb = this.diffMain(text1b, text2b, false, deadline);
504 
505         diffs.addAll(diffsb);
506         return diffs;
507     }

509     /**
510      * Split two texts into a list of strings.  Reduce the texts to a string of
511      * hashes where each Unicode character represents one line.
512      * @param text1 First string.
513      * @param text2 Second string.
514      * @return An object containing the encoded text1, the encoded text2 and
515      *     the List of unique strings.  The zeroth element of the List of
516      *     unique strings is intentionally blank.
517      */
518     protected LinesToCharsResult diffLinesToChars(String text1, String text2) {
519 
520         List<String> lineArray = new ArrayList<>();
521         Map<String, Integer> lineHash = new HashMap<>();
522         // e.g. linearray[4] == "Hello\n"
523         // e.g. linehash.get("Hello\n") == 4
524 
525         // "\x00" is a valid character, but various debuggers don't like it.
526         // So we'll insert a junk entry to avoid generating a null character.
527         lineArray.add("");
528 
529         String chars1 = this.diffLinesToCharsMunge(text1, lineArray, lineHash);
530         String chars2 = this.diffLinesToCharsMunge(text2, lineArray, lineHash);
531         return new LinesToCharsResult(chars1, chars2, lineArray);
532     }

534     /**
535      * Split a text into a list of strings.  Reduce the texts to a string of
536      * hashes where each Unicode character represents one line.
537      * @param text String to encode.
538      * @param lineArray List of unique strings.
539      * @param lineHash Map of strings to indices.
540      * @return Encoded string.
541      */
542     private String diffLinesToCharsMunge(String text, List<String> lineArray,
543             Map<String, Integer> lineHash) {
544 
545         int lineStart = 0;
546         int lineEnd = -1;
547         String line;
548         StringBuilder chars = new StringBuilder();
549         // Walk the text, pulling out a substring for each line.
550         // text.split('\n') would would temporarily double our memory footprint.
551         // Modifying text would create many large strings to garbage collect.
552         while (lineEnd < text.length() - 1) {
553             lineEnd = text.indexOf('\n', lineStart);
554             if (lineEnd == -1) {
555                 lineEnd = text.length() - 1;
556             }
557             line = text.substring(lineStart, lineEnd + 1);
558             lineStart = lineEnd + 1;
559 
560             if (lineHash.containsKey(line)) {
561                 chars.append((char) (int) lineHash.get(line));
562             } else {
563                 lineArray.add(line);
564                 lineHash.put(line, lineArray.size() - 1);
565                 chars.append((char) (lineArray.size() - 1));
566             }
567         }
568         return chars.toString();
569     }

571     /**
572      * Rehydrate the text in a diff from a string of line hashes to real lines of
573      * text.
574      * @param diffs LinkedList of Diff objects.
575      * @param lineArray List of unique strings.
576      */
577     protected void diffCharsToLines(List<Diff> diffs, List<String> lineArray) {
578 
579         StringBuilder text;
580         for (Diff diff : diffs) {
581             text = new StringBuilder();
582             for (int y = 0; y < diff.getText().length(); y++) {
583                 text.append(lineArray.get(diff.getText().charAt(y)));
584             }
585             diff.setText(text.toString());
586         }
587     }

589     /**
590      * Determine the common prefix of two strings
591      * @param text1 First string.
592      * @param text2 Second string.
593      * @return The number of characters common to the start of each string.
594      */
595     public int diffCommonPrefix(String text1, String text2) {
596 
597         // Performance analysis: http://neil.fraser.name/news/2007/10/09/
598         int n = Math.min(text1.length(), text2.length());
599         for (int i = 0; i < n; i++) {
600             if (text1.charAt(i) != text2.charAt(i)) {
601                 return i;
602             }
603         }
604         return n;
605     }

607     /**
608      * Determine the common suffix of two strings
609      * @param text1 First string.
610      * @param text2 Second string.
611      * @return The number of characters common to the end of each string.
612      */
613     public int diffCommonSuffix(String text1, String text2) {
614 
615         // Performance analysis: http://neil.fraser.name/news/2007/10/09/
616         int text1Length = text1.length();
617         int text2Length = text2.length();
618         int n = Math.min(text1Length, text2Length);
619         for (int i = 1; i <= n; i++) {
620             if (text1.charAt(text1Length - i) != text2.charAt(text2Length - i)) {
621                 return i - 1;
622             }
623         }
624         return n;
625     }

627     /**
628      * Determine if the suffix of one string is the prefix of another.
629      * @param valueText1 First string.
630      * @param valueText2 Second string.
631      * @return The number of characters common to the end of the first
632      *     string and the start of the second string.
633      */
634     protected int diffCommonOverlap(String valueText1, String valueText2) {
635 
636         String text1 = valueText1;
637         String text2 = valueText2;
638         
639         // Cache the text lengths to prevent multiple calls.
640         int text1Length = text1.length();
641         int text2Length = text2.length();
642         // Eliminate the null case.
643         if (text1Length == 0 || text2Length == 0) {
644             return 0;
645         }
646         // Truncate the longer string.
647         if (text1Length > text2Length) {
648             text1 = text1.substring(text1Length - text2Length);
649         } else if (text1Length < text2Length) {
650             text2 = text2.substring(0, text1Length);
651         }
652         int textLength = Math.min(text1Length, text2Length);
653         // Quick check for the worst case.
654         if (text1.equals(text2)) {
655             return textLength;
656         }
657 
658         // Start by looking for a single character match
659         // and increase length until no match is found.
660         // Performance analysis: http://neil.fraser.name/news/2010/11/04/
661         int best = 0;
662         int length = 1;
663         while (true) {
664             String pattern = text1.substring(textLength - length);
665             int found = text2.indexOf(pattern);
666             if (found == -1) {
667                 return best;
668             }
669             length += found;
670             if (found == 0 || text1.substring(textLength - length).equals(
671                     text2.substring(0, length))) {
672                 best = length;
673                 length++;
674             }
675         }
676     }

678     /**
679      * Do the two texts share a substring which is at least half the length of
680      * the longer text?
681      * This speedup can produce non-minimal diffs.
682      * @param text1 First string.
683      * @param text2 Second string.
684      * @return Five element String array, containing the prefix of text1, the
685      *     suffix of text1, the prefix of text2, the suffix of text2 and the
686      *     common middle.  Or null if there was no match.
687      */
688     protected String[] diffHalfMatch(String text1, String text2) {
689 
690         String longtext = text1.length() > text2.length() ? text1 : text2;
691         String shorttext = text1.length() > text2.length() ? text2 : text1;
692         if (longtext.length() < 4 || shorttext.length() * 2 < longtext.length()) {
693             return null;  // Pointless.
694         }
695 
696         // First check if the second quarter is the seed for a half-match.
697         String[] hm1 = this.diffHalfMatchI(longtext, shorttext, (longtext.length() + 3) / 4);
698         // Check again based on the third quarter.
699         String[] hm2 = this.diffHalfMatchI(longtext, shorttext, (longtext.length() + 1) / 2);
700         String[] hm;
701         if (hm1 == null && hm2 == null) {
702             return null;
703         } else if (hm2 == null) {
704             hm = hm1;
705         } else if (hm1 == null) {
706             hm = hm2;
707         } else {
708             // Both matched.  Select the longest.
709             hm = hm1[4].length() > hm2[4].length() ? hm1 : hm2;
710         }
711 
712         // A half-match was found, sort out the return data.
713         if (text1.length() > text2.length()) {
714             return hm;
715         } else {
716             return new String[]{hm[2], hm[3], hm[0], hm[1], hm[4]};
717         }
718     }

721      * Does a substring of shorttext exist within longtext such that the
722      * substring is at least half the length of longtext?
723      * @param longtext Longer string.
724      * @param shorttext Shorter string.
725      * @param i Start index of quarter length substring within longtext.
726      * @return Five element String array, containing the prefix of longtext, the
727      *     suffix of longtext, the prefix of shorttext, the suffix of shorttext
728      *     and the common middle.  Or null if there was no match.
729      */
730     private String[] diffHalfMatchI(String longtext, String shorttext, int i) {
731 
732         // Start with a 1/4 length substring at position i as a seed.
733         String seed = longtext.substring(i, i + longtext.length() / 4);
734         int j = -1;
735         String bestCommon = "";
736         String bestLongtextA = "";
737         String bestLongtextB = "";
738         String bestShorttextA = "";
739         String bestShorttextB = "";
740         while ((j = shorttext.indexOf(seed, j + 1)) != -1) {
741             int prefixLength = this.diffCommonPrefix(longtext.substring(i),
742                     shorttext.substring(j));
743             int suffixLength = this.diffCommonSuffix(longtext.substring(0, i),
744                     shorttext.substring(0, j));
745             if (bestCommon.length() < suffixLength + prefixLength) {

749                 bestLongtextB = longtext.substring(i + prefixLength);
750                 bestShorttextA = shorttext.substring(0, j - suffixLength);
751                 bestShorttextB = shorttext.substring(j + prefixLength);
752             }
753         }
754         if (bestCommon.length() * 2 >= longtext.length()) {
755             return new String[]{bestLongtextA, bestLongtextB,
756                     bestShorttextA, bestShorttextB, bestCommon};
757         } else {
758             return null;
759         }

762     /**
763      * Reduce the number of edits by eliminating semantically trivial equalities.
764      * @param diffs LinkedList of Diff objects.
765      */
766     public void diffCleanupSemantic(LinkedList<Diff> diffs) {
767 
768         if (diffs.isEmpty()) {
769             return;
770         }
771         boolean changes = false;
772         // Synchronized Stack to avoid Exception
773         Stack<Diff> equalities = new Stack<>();  // Stack of qualities.
774         String lastequality = null; // Always equal to equalities.lastElement().text
775         ListIterator<Diff> pointer = diffs.listIterator();
776         // Number of characters that changed prior to the equality.
777         int lengthInsertions1 = 0;
778         int lengthDeletions1 = 0;
779         // Number of characters that changed after the equality.
780         int lengthInsertions2 = 0;
781         int lengthDeletions2 = 0;
782         Diff thisDiff = pointer.next();
783 
784         while (thisDiff != null) {
785             if (thisDiff.getOperation() == Operation.EQUAL) {
786                 // Equality found.
787                 equalities.push(thisDiff);
788                 lengthInsertions1 = lengthInsertions2;
789                 lengthDeletions1 = lengthDeletions2;
790                 lengthInsertions2 = 0;
791                 lengthDeletions2 = 0;
792                 lastequality = thisDiff.getText();
793             } else {
794                 // An insertion or deletion.
795                 if (thisDiff.getOperation() == Operation.INSERT) {
796                     lengthInsertions2 += thisDiff.getText().length();
797                 } else {
798                     lengthDeletions2 += thisDiff.getText().length();
799                 }
800                 // Eliminate an equality that is smaller or equal to the edits on both
801                 // sides of it.
802                 if (
803                     lastequality != null
804                     && lastequality.length() <= Math.max(lengthInsertions1, lengthDeletions1)
805                     && lastequality.length() <= Math.max(lengthInsertions2, lengthDeletions2)
806                 ) {
807                     // Walk back to offending equality.
808                     while (thisDiff != equalities.lastElement()) {
809                         thisDiff = pointer.previous();
810                     }
811                     pointer.next();
812 
813                     // Replace equality with a delete.
814                     pointer.set(new Diff(Operation.DELETE, lastequality));
815                     // Insert a corresponding an insert.
816                     pointer.add(new Diff(Operation.INSERT, lastequality));
817 
818                     equalities.pop();  // Throw away the equality we just deleted.
819                     if (!equalities.empty()) {
820                         // Throw away the previous equality (it needs to be reevaluated).
821                         equalities.pop();
822                     }
823                     if (equalities.empty()) {
824                         // There are no previous equalities, walk back to the start.
825                         while (pointer.hasPrevious()) {
826                             pointer.previous();
827                         }
828                     } else {
829                         // There is a safe equality we can fall back to.
830                         thisDiff = equalities.lastElement();
831                         while (thisDiff != pointer.previous()) {
832                             // Intentionally empty loop.
833                         }
834                     }
835 
836                     lengthInsertions1 = 0;  // Reset the counters.
837                     lengthInsertions2 = 0;
838                     lengthDeletions1 = 0;
839                     lengthDeletions2 = 0;
840                     lastequality = null;
841                     changes = true;
842                 }
843             }
844             thisDiff = pointer.hasNext() ? pointer.next() : null;
845         }
846 
847         // Normalize the diff.
848         if (changes) {
849             this.diffCleanupMerge(diffs);
850         }
851         this.diffCleanupSemanticLossless(diffs);
852 
853         // Find any overlaps between deletions and insertions.
854         // e.g: <del>abcxxx</del><ins>xxxdef</ins>
855         //   -> <del>abc</del>xxx<ins>def</ins>
856         // e.g: <del>xxxabc</del><ins>defxxx</ins>
857         //   -> <ins>def</ins>xxx<del>abc</del>
858         // Only extract an overlap if it is as big as the edit ahead or behind it.
859         pointer = diffs.listIterator();
860         Diff prevDiff = null;
861         thisDiff = null;
862         if (pointer.hasNext()) {
863             prevDiff = pointer.next();
864             if (pointer.hasNext()) {
865                 thisDiff = pointer.next();
866             }
867         }
868 
869         while (thisDiff != null) {
870             if (prevDiff.getOperation() == Operation.DELETE &&
871                     thisDiff.getOperation() == Operation.INSERT) {
872                 String deletion = prevDiff.getText();
873                 String insertion = thisDiff.getText();
874                 int overlapLength1 = this.diffCommonOverlap(deletion, insertion);
875                 int overlapLength2 = this.diffCommonOverlap(insertion, deletion);
876                 if (overlapLength1 >= overlapLength2) {
877                     if (overlapLength1 >= deletion.length() / 2.0 ||
878                             overlapLength1 >= insertion.length() / 2.0) {
879                         // Overlap found.  Insert an equality and trim the surrounding edits.
880                         pointer.previous();
881                         pointer.add(new Diff(Operation.EQUAL,
882                                 insertion.substring(0, overlapLength1)));
883                         prevDiff.setText(deletion.substring(0, deletion.length() - overlapLength1));
884                         thisDiff.setText(insertion.substring(overlapLength1));
885                         // pointer.add inserts the element before the cursor, so there is
886                         // no need to step past the new element.
887                     }
888                 } else {
889                     if (overlapLength2 >= deletion.length() / 2.0 ||
890                             overlapLength2 >= insertion.length() / 2.0) {
891                         // Reverse overlap found.
892                         // Insert an equality and swap and trim the surrounding edits.
893                         pointer.previous();
894                         pointer.add(new Diff(Operation.EQUAL,
895                                 deletion.substring(0, overlapLength2)));
896                         prevDiff.setOperation(Operation.INSERT);
897                         prevDiff.setText(insertion.substring(0, insertion.length() - overlapLength2));
898                         thisDiff.setOperation(Operation.DELETE);
899                         thisDiff.setText(deletion.substring(overlapLength2));
900                         // pointer.add inserts the element before the cursor, so there is
901                         // no need to step past the new element.
902                     }
903                 }
904                 thisDiff = pointer.hasNext() ? pointer.next() : null;
905             }
906             prevDiff = thisDiff;
907             thisDiff = pointer.hasNext() ? pointer.next() : null;
908         }
909     }

911     /**
912      * Look for single edits surrounded on both sides by equalities
913      * which can be shifted sideways to align the edit to a word boundary.
914      * e.g: The c<ins>at c</ins>ame. -> The <ins>cat </ins>came.
915      * @param diffs LinkedList of Diff objects.
916      */
917     public void diffCleanupSemanticLossless(List<Diff> diffs) {
918 
919         StringBuilder equality1 = new StringBuilder();
920         String edit;
921         StringBuilder equality2 = new StringBuilder();
922         String commonString;
923         int commonOffset;
924         int score;
925         int bestScore;
926         String bestEquality1;
927         String bestEdit;
928         String bestEquality2;
929         // Create a new iterator at the start.
930         ListIterator<Diff> pointer = diffs.listIterator();
931         Diff prevDiff = pointer.hasNext() ? pointer.next() : null;
932         Diff thisDiff = pointer.hasNext() ? pointer.next() : null;
933         Diff nextDiff = pointer.hasNext() ? pointer.next() : null;
934 
935         // Intentionally ignore the first and last element (don't need checking).
936         while (nextDiff != null) {
937             if (prevDiff.getOperation() == Operation.EQUAL &&
938                     nextDiff.getOperation() == Operation.EQUAL) {
939                 // This is a single edit surrounded by equalities.
940                 equality1.setLength(0);
941                 equality1.append(prevDiff.getText());
942                 edit = thisDiff.getText();
943                 equality2.setLength(0);
944                 equality2.append(nextDiff.getText());
945 
946                 // First, shift the edit as far left as possible.
947                 commonOffset = this.diffCommonSuffix(equality1.toString(), edit);
948                 if (commonOffset != 0) {
949                     commonString = edit.substring(edit.length() - commonOffset);
950                     String substring = equality1.substring(0, equality1.length() - commonOffset);
951                     equality1.setLength(0);
952                     equality1.append(substring);
953                     edit = commonString + edit.substring(0, edit.length() - commonOffset);
954                     equality2.insert(0, commonString);
955                 }
956 
957                 // Second, step character by character right, looking for the best fit.
958                 bestEquality1 = equality1.toString();
959                 bestEdit = edit;
960                 bestEquality2 = equality2.toString();
961                 bestScore = this.diffCleanupSemanticScore(equality1.toString(), edit)
962                         + this.diffCleanupSemanticScore(edit, equality2.toString());
963                 while (!edit.isEmpty() && equality2.length() != 0
964                         && edit.charAt(0) == equality2.charAt(0)) {
965                     equality1.append(edit.charAt(0));
966                     edit = edit.substring(1) + equality2.charAt(0);
967                     String substring = equality2.substring(1);
968                     equality2.setLength(0);
969                     equality2.append(substring);
970                     score = this.diffCleanupSemanticScore(equality1.toString(), edit)
971                             + this.diffCleanupSemanticScore(edit, equality2.toString());
972                     // The >= encourages trailing rather than leading whitespace on edits.
973                     if (score >= bestScore) {
974                         bestScore = score;
975                         bestEquality1 = equality1.toString();
976                         bestEdit = edit;
977                         bestEquality2 = equality2.toString();
978                     }
979                 }
980 
981                 if (!prevDiff.getText().equals(bestEquality1)) {
982                     // We have an improvement, save it back to the diff.
983                     if (!bestEquality1.isEmpty()) {
984                         prevDiff.setText(bestEquality1);
985                     } else {
986                         pointer.previous(); // Walk past nextDiff.
987                         pointer.previous(); // Walk past thisDiff.
988                         pointer.previous(); // Walk past prevDiff.
989                         pointer.remove(); // Delete prevDiff.
990                         pointer.next(); // Walk past thisDiff.
991                         pointer.next(); // Walk past nextDiff.
992                     }
993                     thisDiff.setText(bestEdit);
994                     if (!bestEquality2.isEmpty()) {
995                         nextDiff.setText(bestEquality2);
996                     } else {
997                         pointer.remove(); // Delete nextDiff.
998                         nextDiff = thisDiff;
999                         thisDiff = prevDiff;
1000                     }
1001                 }
1002             }
1003             prevDiff = thisDiff;
1004             thisDiff = nextDiff;
1005             nextDiff = pointer.hasNext() ? pointer.next() : null;
1006         }
1007     }

1009     /**
1010      * Given two strings, compute a score representing whether the internal
1011      * boundary falls on logical boundaries.
1012      * Scores range from 6 (best) to 0 (worst).
1013      * @param one First string.
1014      * @param two Second string.
1015      * @return The score.
1016      */
1017     private int diffCleanupSemanticScore(String one, String two) {
1018 
1019         if (one.isEmpty() || two.isEmpty()) {
1020             // Edges are the best.
1021             return 6;
1022         }
1023 
1024         // Each port of this function behaves slightly differently due to
1025         // subtle differences in each language's definition of things like
1026         // 'whitespace'.  Since this function's purpose is largely cosmetic,
1027         // the choice has been made to use each language's native features
1028         // rather than force total conformity.
1029         char char1 = one.charAt(one.length() - 1);
1030         char char2 = two.charAt(0);
1031         boolean nonAlphaNumeric1 = !Character.isLetterOrDigit(char1);
1032         boolean nonAlphaNumeric2 = !Character.isLetterOrDigit(char2);
1033         boolean whitespace1 = nonAlphaNumeric1 && Character.isWhitespace(char1);
1034         boolean whitespace2 = nonAlphaNumeric2 && Character.isWhitespace(char2);
1035         boolean lineBreak1 = whitespace1
1036                 && Character.getType(char1) == Character.CONTROL;
1037         boolean lineBreak2 = whitespace2
1038                 && Character.getType(char2) == Character.CONTROL;
1039         boolean blankLine1 = lineBreak1 && DiffMatchPatch.BLANK_LINE_END.matcher(one).find();
1040         boolean blankLine2 = lineBreak2 && DiffMatchPatch.BLANK_LINE_START.matcher(two).find();
1041 
1042         if (blankLine1 || blankLine2) {
1043             // Five points for blank lines.
1044             return 5;
1045         } else if (lineBreak1 || lineBreak2) {
1046             // Four points for line breaks.
1047             return 4;
1048         } else if (nonAlphaNumeric1 && !whitespace1 && whitespace2) {
1049             // Three points for end of sentences.
1050             return 3;
1051         } else if (whitespace1 || whitespace2) {
1052             // Two points for whitespace.
1053             return 2;
1054         } else if (nonAlphaNumeric1 || nonAlphaNumeric2) {
1055             // One point for non-alphanumeric.
1056             return 1;
1057         }
1058         return 0;
1059     }

1178     /**

1180      * Any edit section can move as long as it doesn't cross an equality.
1181      * @param diffs LinkedList of Diff objects.
1182      */
1183     public void diffCleanupMerge(LinkedList<Diff> diffs) {
1184 
1185         diffs.add(new Diff(Operation.EQUAL, ""));  // Add a dummy entry at the end.
1186         ListIterator<Diff> pointer = diffs.listIterator();
1187         int countDelete = 0;
1188         int countInsert = 0;
1189         StringBuilder textDelete = new StringBuilder();
1190         StringBuilder textInsert = new StringBuilder();
1191         Diff thisDiff = pointer.next();
1192         Diff prevEqual = null;
1193         int commonlength;
1194         while (thisDiff != null) {
1195             switch (thisDiff.getOperation()) {
1196             case INSERT:
1197                 countInsert++;
1198                 textInsert.append(thisDiff.getText());
1199                 prevEqual = null;
1200                 break;
1201             case DELETE:
1202                 countDelete++;
1203                 textDelete.append(thisDiff.getText());
1204                 prevEqual = null;
1205                 break;
1206             case EQUAL:
1207                 if (countDelete + countInsert > 1) {
1208 
1209                     boolean bothTypes = countDelete != 0 && countInsert != 0;
1210                     // Delete the offending records.
1211                     pointer.previous();  // Reverse direction.
1212                     while (countDelete-- > 0) {
1213                         pointer.previous();
1214                         pointer.remove();
1215                     }
1216                     while (countInsert-- > 0) {
1217                         pointer.previous();
1218                         pointer.remove();
1219                     }
1220 
1221                     if (bothTypes) {
1222                         // Factor out any common prefixies.
1223                         commonlength = this.diffCommonPrefix(textInsert.toString(), textDelete.toString());
1224                         if (commonlength != 0) {
1225                             if (pointer.hasPrevious()) {
1226                                 thisDiff = pointer.previous();
1227                                 // Previous diff should have been an equality: thisDiff.getOperation() == Operation.EQUAL")
1228                                 thisDiff.setText(thisDiff.getText() + textInsert.substring(0, commonlength));
1229                                 pointer.next();
1230                             } else {
1231                                 pointer.add(new Diff(Operation.EQUAL,
1232                                         textInsert.substring(0, commonlength)));
1233                             }
1234                             String substringIns = textInsert.substring(commonlength);
1235                             textInsert.setLength(0);
1236                             textInsert.append(substringIns);
1237                             String substringDel = textDelete.substring(commonlength);
1238                             textDelete.setLength(0);
1239                             textDelete.append(substringDel);
1240                         }
1241                         // Factor out any common suffixies.
1242                         commonlength = this.diffCommonSuffix(textInsert.toString(), textDelete.toString());
1243                         if (commonlength != 0) {
1244                             thisDiff = pointer.next();
1245                             thisDiff.setText(textInsert.substring(textInsert.length() - commonlength) + thisDiff.getText());
1246                             String substringIns = textInsert.substring(0, textInsert.length() - commonlength);
1247                             textInsert.setLength(0);
1248                             textInsert.append(substringIns);
1249                             String substringDel = textDelete.substring(0, textDelete.length() - commonlength);
1250                             textDelete.setLength(0);
1251                             textDelete.append(substringDel);
1252                             pointer.previous();
1253                         }
1254                     }
1255                     // Insert the merged records.
1256                     if (textDelete.length() != 0) {
1257                         pointer.add(new Diff(Operation.DELETE, textDelete.toString()));
1258                     }
1259                     if (textInsert.length() != 0) {
1260                         pointer.add(new Diff(Operation.INSERT, textInsert.toString()));
1261                     }
1262                     // Step forward to the equality.
1263                     thisDiff = pointer.hasNext() ? pointer.next() : null;
1264                 } else if (prevEqual != null) {
1265                     // Merge this equality with the previous one.
1266                     prevEqual.setText(prevEqual.getText() + thisDiff.getText());
1267                     pointer.remove();
1268                     thisDiff = pointer.previous();
1269                     pointer.next();  // Forward direction
1270                 }
1271                 countInsert = 0;
1272                 countDelete = 0;
1273                 textDelete.setLength(0);
1274                 textInsert.setLength(0);
1275                 prevEqual = thisDiff;
1276                 break;
1277             }
1278             thisDiff = pointer.hasNext() ? pointer.next() : null;
1279         }
1280         if (diffs.getLast().getText().isEmpty()) {
1281             diffs.removeLast();  // Remove the dummy entry at the end.
1282         }
1283 
1284         /*
1285          * Second pass: look for single edits surrounded on both sides by equalities
1286          * which can be shifted sideways to eliminate an equality.
1287          * e.g: A<ins>BA</ins>C -> <ins>AB</ins>AC
1288          */
1289         boolean changes = false;
1290         // Create a new iterator at the start.
1291         // (As opposed to walking the current one back.)
1292         pointer = diffs.listIterator();
1293         Diff prevDiff = pointer.hasNext() ? pointer.next() : null;
1294         thisDiff = pointer.hasNext() ? pointer.next() : null;
1295         Diff nextDiff = pointer.hasNext() ? pointer.next() : null;
1296 
1297         // Intentionally ignore the first and last element (don't need checking).
1298         while (nextDiff != null) {
1299             if (prevDiff.getOperation() == Operation.EQUAL &&
1300                     nextDiff.getOperation() == Operation.EQUAL) {
1301                 // This is a single edit surrounded by equalities.
1302                 if (thisDiff.getText().endsWith(prevDiff.getText())) {
1303                     // Shift the edit over the previous equality.
1304                     thisDiff.setText(prevDiff.getText()
1305                             + thisDiff.getText().substring(0, thisDiff.getText().length()
1306                                     - prevDiff.getText().length()));
1307                     nextDiff.setText(prevDiff.getText() + nextDiff.getText());
1308                     pointer.previous(); // Walk past nextDiff.
1309                     pointer.previous(); // Walk past thisDiff.
1310                     pointer.previous(); // Walk past prevDiff.
1311                     pointer.remove(); // Delete prevDiff.
1312                     pointer.next(); // Walk past thisDiff.
1313                     thisDiff = pointer.next(); // Walk past nextDiff.
1314                     nextDiff = pointer.hasNext() ? pointer.next() : null;
1315                     changes = true;
1316                 } else if (thisDiff.getText().startsWith(nextDiff.getText())) {
1317                     // Shift the edit over the next equality.
1318                     prevDiff.setText(prevDiff.getText() + nextDiff.getText());
1319                     thisDiff.setText(thisDiff.getText().substring(nextDiff.getText().length())
1320                             + nextDiff.getText());
1321                     pointer.remove(); // Delete nextDiff.
1322                     nextDiff = pointer.hasNext() ? pointer.next() : null;
1323                     changes = true;
1324                 }
1325             }
1326             prevDiff = thisDiff;
1327             thisDiff = nextDiff;
1328             nextDiff = pointer.hasNext() ? pointer.next() : null;
1329         }
1330         // If shifts were made, the diff needs reordering and another shift sweep.
1331         if (changes) {
1332             this.diffCleanupMerge(diffs);
1333         }
1334     }

1501     /**
1502      * Given the original text1, and an encoded string which describes the
1503      * operations required to transform text1 into text2, compute the full diff.
1504      * @param text1 Source string for the diff.
1505      * @param delta Delta text.
1506      * @return list of Diff objects or null if invalid.
1507      */
1508     public List<Diff> diffFromDelta(String text1, String delta) {
1509 
1510         List<Diff> diffs = new LinkedList<>();
1511         int pointer = 0;  // Cursor in text1
1512         String[] tokens = delta.split("\t");
1513 
1514         for (String token : tokens) {
1515             if (token.isEmpty()) {
1516                 // Blank tokens are ok (from a trailing \t).
1517                 continue;
1518             }
1519             // Each token begins with a one character parameter which specifies the
1520             // operation of this token (delete, insert, equality).
1521             String param = token.substring(1);
1522             switch (token.charAt(0)) {
1523             case '+':
1524                 // decode would change all "+" to " "
1525                 param = param.replace("+", "%2B");
1526                 try {
1527                     param = URLDecoder.decode(param, StandardCharsets.UTF_8);
1528                 } catch (IllegalArgumentException e) {
1529                     // Malformed URI sequence.
1530                     throw new IllegalArgumentException(
1531                             "Illegal escape in diff_fromDelta: " + param, e);
1532                 }
1533                 diffs.add(new Diff(Operation.INSERT, param));
1534                 break;
1535             case '-':
1536                 // Fall through.
1537             case '=':
1538                 int n;
1539                 try {
1540                     n = Integer.parseInt(param);
1541                 } catch (NumberFormatException e) {
1542                     throw new IllegalArgumentException(
1543                             "Invalid number in diff_fromDelta: " + param, e);
1544                 }
1545                 if (n < 0) {
1546                     throw new IllegalArgumentException(
1547                             "Negative number in diff_fromDelta: " + param);
1548                 }
1549                 String text;
1550                 try {
1551                     int p1 = pointer;
1552                     pointer += n;
1553                     int p2 = pointer;
1554                     text = text1.substring(p1, p2);
1555                 } catch (StringIndexOutOfBoundsException e) {
1556                     throw new IllegalArgumentException("Delta length (" + pointer
1557                             + ") larger than source text length (" + text1.length()
1558                             + ").", e);
1559                 }
1560                 if (token.charAt(0) == '=') {
1561                     diffs.add(new Diff(Operation.EQUAL, text));
1562                 } else {
1563                     diffs.add(new Diff(Operation.DELETE, text));
1564                 }
1565                 break;
1566             default:
1567                 // Anything else is an error.
1568                 throw new IllegalArgumentException(
1569                         "Invalid diff operation in diff_fromDelta: " + token.charAt(0));
1570             }
1571         }
1572         if (pointer != text1.length()) {
1573             throw new IllegalArgumentException("Delta length (" + pointer
1574                     + ") smaller than source text length (" + text1.length() + ").");
1575         }
1576         return diffs;
1577     }

1582     /**
1583      * Locate the best instance of 'pattern' in 'text' near 'loc'.
1584      * Returns -1 if no match found.
1585      * @param text The text to search.
1586      * @param pattern The pattern to search for.
1587      * @param valueLoc The location to search around.
1588      * @return Best match index or -1.
1589      */
1590     public int matchMain(String text, String pattern, int valueLoc) {
1591 
1592         // Check for null inputs.
1593         if (text == null || pattern == null) {
1594             throw new IllegalArgumentException("Null inputs. (match_main)");
1595         }
1596 
1597         int loc = Math.max(0, Math.min(valueLoc, text.length()));
1598         if (text.equals(pattern)) {
1599             // Shortcut (potentially not guaranteed by the algorithm)
1600             return 0;
1601         } else if (text.isEmpty()) {
1602             // Nothing to match.
1603             return -1;
1604         } else if (loc + pattern.length() <= text.length()
1605                 && text.substring(loc, loc + pattern.length()).equals(pattern)) {
1606             // Perfect match at the perfect spot!  (Includes case of null pattern)
1607             return loc;
1608         } else {
1609             // Do a fuzzy compare.
1610             return this.matchBitap(text, pattern, loc);
1611         }
1612     }

1614     /**
1615      * Locate the best instance of 'pattern' in 'text' near 'loc' using the
1616      * Bitap algorithm.  Returns -1 if no match found.
1617      * @param text The text to search.
1618      * @param pattern The pattern to search for.
1619      * @param loc The location to search around.
1620      * @return Best match index or -1.
1621      */
1622     protected int matchBitap(String text, String pattern, int loc) {
1623 
1624         // Initialise the alphabet.
1625         Map<Character, Integer> s = this.matchAlphabet(pattern);
1626 
1627         // Highest score beyond which we give up.
1628         double scoreThreshold = DiffMatchPatch.MATCH_THRESHOLD;
1629         // Is there a nearby exact match? (speedup)
1630         int bestLoc = text.indexOf(pattern, loc);
1631         if (bestLoc != -1) {
1632             scoreThreshold = Math.min(this.matchBitapScore(0, bestLoc, loc, pattern),
1633                     scoreThreshold);
1634             // What about in the other direction? (speedup)
1635             bestLoc = text.lastIndexOf(pattern, loc + pattern.length());
1636             if (bestLoc != -1) {
1637                 scoreThreshold = Math.min(this.matchBitapScore(0, bestLoc, loc, pattern),
1638                         scoreThreshold);
1639             }
1640         }
1641 
1642         // Initialise the bit arrays.
1643         int matchmask = 1 << (pattern.length() - 1);
1644         bestLoc = -1;
1645 
1646         int binMin;
1647         int binMid;
1648         int binMax = pattern.length() + text.length();
1649         // Empty initialization added to appease Java compiler.
1650         int[] lastRd = new int[0];
1651         for (int d = 0; d < pattern.length(); d++) {
1652 
1653             // Scan for the best match; each iteration allows for one more error.
1654             // Run a binary search to determine how far from 'loc' we can stray at
1655             // this error level.
1656             binMin = 0;
1657             binMid = binMax;
1658             while (binMin < binMid) {
1659                 if (this.matchBitapScore(d, loc + binMid, loc, pattern)
1660                         <= scoreThreshold) {
1661                     binMin = binMid;
1662                 } else {
1663                     binMax = binMid;
1664                 }
1665                 binMid = (binMax - binMin) / 2 + binMin;
1666             }
1667             // Use the result from this iteration as the maximum for the next.
1668             binMax = binMid;
1669             int start = Math.max(1, loc - binMid + 1);
1670             int finish = Math.min(loc + binMid, text.length()) + pattern.length();
1671 
1672             int[] rd = new int[finish + 2];
1673             rd[finish + 1] = (1 << d) - 1;
1674             for (int j = finish; j >= start; j--) {
1675 
1676                 int charMatch;
1677                 if (text.length() <= j - 1 || !s.containsKey(text.charAt(j - 1))) {
1678                     // Out of range.
1679                     charMatch = 0;
1680                 } else {
1681                     charMatch = s.get(text.charAt(j - 1));
1682                 }
1683                 if (d == 0) {
1684                     // First pass: exact match.
1685                     rd[j] = ((rd[j + 1] << 1) | 1) & charMatch;
1686                 } else {
1687                     // Subsequent passes : fuzzy match.
1688                     rd[j] = (((rd[j + 1] << 1) | 1) & charMatch)
1689                             | (((lastRd[j + 1] | lastRd[j]) << 1) | 1) | lastRd[j + 1];
1690                 }
1691 
1692                 if ((rd[j] & matchmask) != 0) {
1693                     double score = this.matchBitapScore(d, j - 1, loc, pattern);
1694                     // This match will almost certainly be better than any existing
1695                     // match.  But check anyway.
1696                     if (score <= scoreThreshold) {
1697                         // Told you so.
1698                         scoreThreshold = score;
1699                         bestLoc = j - 1;
1700                         if (bestLoc > loc) {
1701                             // When passing loc, don't exceed our current distance from loc.
1702                             start = Math.max(1, 2 * loc - bestLoc);
1703                         } else {
1704                             // Already passed loc, downhill from here on in.
1705                             break;
1706                         }
1707                     }
1708                 }
1709             }
1710             if (this.matchBitapScore(d + 1, loc, loc, pattern) > scoreThreshold) {
1711                 // No hope for a (better) match at greater error levels.
1712                 break;
1713             }
1714             lastRd = rd;
1715         }
1716         return bestLoc;
1717     }

1720      * Compute and return the score for a match with e errors and x location.
1721      * @param e Number of errors in match.
1722      * @param x Location of match.
1723      * @param loc Expected location of match.
1724      * @param pattern Pattern being sought.
1725      * @return Overall score for match (0.0 = good, 1.0 = bad).
1726      */

1736      * @param pattern The text to encode.

1739     protected Map<Character, Integer> matchAlphabet(String pattern) {

1757     /**
1758      * Increase the context until it is unique,
1759      * but don't let the pattern expand beyond Match_MaxBits.
1760      * @param patch The patch to grow.
1761      * @param text Source text.
1762      */
1763     protected void patchAddContext(Patch patch, String text) {
1764 
1765         if (text.isEmpty()) {
1766             return;
1767         }
1768         String pattern = text.substring(patch.getStart2(), patch.getStart2() + patch.getLength1());
1769         int padding = 0;
1770 
1771         // Look for the first and last matches of pattern in text.  If two different
1772         // matches are found, increase the pattern length.
1773         while (text.indexOf(pattern) != text.lastIndexOf(pattern)
1774                 && pattern.length() < DiffMatchPatch.MATCH_MAX_BITS - DiffMatchPatch.PATCH_MARGIN - DiffMatchPatch.PATCH_MARGIN) {
1775             padding += DiffMatchPatch.PATCH_MARGIN;
1776             pattern = text.substring(Math.max(0, patch.getStart2() - padding),
1777                     Math.min(text.length(), patch.getStart2() + patch.getLength1() + padding));
1778         }
1779         // Add one chunk for good luck.
1780         padding += DiffMatchPatch.PATCH_MARGIN;
1781 
1782         // Add the prefix.
1783         String prefix = text.substring(Math.max(0, patch.getStart2() - padding),
1784                 patch.getStart2());
1785         if (!prefix.isEmpty()) {
1786             patch.getDiffs().addFirst(new Diff(Operation.EQUAL, prefix));
1787         }
1788         // Add the suffix.
1789         String suffix = text.substring(patch.getStart2() + patch.getLength1(),
1790                 Math.min(text.length(), patch.getStart2() + patch.getLength1() + padding));
1791         if (!suffix.isEmpty()) {
1792             patch.getDiffs().addLast(new Diff(Operation.EQUAL, suffix));
1793         }
1794 
1795         // Roll back the start points.
1796         patch.setStart1(patch.getStart1() - prefix.length());
1797         patch.setStart2(patch.getStart2() - prefix.length());
1798         // Extend the lengths.
1799         patch.setLength1(patch.getLength1() + prefix.length() + suffix.length());
1800         patch.setLength2(patch.getLength2() + prefix.length() + suffix.length());
1801     }

1850             throw new IllegalArgumentException("Null inputs. (patch_make)");
1851         }
1852 
1853         List<Patch> patches = new LinkedList<>();
1854         if (diffs.isEmpty()) {
1855             return patches;  // Get rid of the null case.
1856         }
1857         Patch patch = new Patch();
1858         int charCount1 = 0;  // Number of characters into the text1 string.
1859         int charCount2 = 0;  // Number of characters into the text2 string.
1860         // Start with text1 (prepatch_text) and apply the diffs until we arrive at
1861         // text2 (postpatch_text). We recreate the patches one by one to determine
1862         // context info.
1863         String prepatchText = text1;

1866         for (Diff aDiff : diffs) {

1873             switch (aDiff.getOperation()) {
1874             case INSERT:
1875                 patch.getDiffs().add(aDiff);
1876                 patch.setLength2(patch.getLength2() + aDiff.getText().length());
1877                 postpatchText = postpatchText.substring(0, charCount2)
1878                         + aDiff.getText() + postpatchText.substring(charCount2);

1902                     this.patchAddContext(patch, prepatchText);
1903                     patches.add(patch);

1906                     // http://code.google.com/p/google-diff-match-patch/wiki/Unidiff

1908                     // just completed patch.

1916             if (aDiff.getOperation() != Operation.INSERT) {
1917                 charCount1 += aDiff.getText().length();

1919             if (aDiff.getOperation() != Operation.DELETE) {
1920                 charCount2 += aDiff.getText().length();
1921             }
1922         }

1924         if (!patch.getDiffs().isEmpty()) {
1925             this.patchAddContext(patch, prepatchText);

1927         }

1955     /**
1956      * Merge a set of patches onto the text.  Return a patched text, as well
1957      * as an array of true/false values indicating which patches were applied.
1958      * @param valuePatches Array of Patch objects
1959      * @param valueText Old text.
1960      * @return Two element Object array, containing the new text and an array of
1961      *      boolean values.
1962      */
1963     public Object[] patchApply(LinkedList<Patch> valuePatches, String valueText) {
1964 
1965         if (valuePatches.isEmpty()) {
1966             return new Object[]{valueText, new boolean[0]};
1967         }
1968         
1969         // Deep copy the patches so that no changes are made to originals.
1970         LinkedList<Patch> patches = this.patchDeepCopy(valuePatches);
1971 
1972         String nullPadding = this.patchAddPadding(patches);
1973         String text = nullPadding + valueText + nullPadding;
1974         this.patchSplitMax(patches);
1975 
1976         int x = 0;
1977         // delta keeps track of the offset between the expected and actual location
1978         // of the previous patch.  If there are patches expected at positions 10 and
1979         // 20, but the first patch was found at 12, delta is 2 and the second patch
1980         // has an effective expected position of 22.
1981         int delta = 0;
1982         boolean[] results = new boolean[patches.size()];
1983         for (Patch aPatch : patches) {
1984 
1985             int expectedLoc = aPatch.getStart2() + delta;
1986             String text1 = this.diffText1(aPatch.getDiffs());
1987             int startLoc;
1988             int endLoc = -1;
1989 
1990             if (text1.length() > DiffMatchPatch.MATCH_MAX_BITS) {
1991                 // patch_splitMax will only provide an oversized pattern in the case of
1992                 // a monster delete.
1993                 startLoc = this.matchMain(text,
1994                         text1.substring(0, DiffMatchPatch.MATCH_MAX_BITS), expectedLoc);
1995                 if (startLoc != -1) {
1996                     endLoc = this.matchMain(text,
1997                             text1.substring(text1.length() - DiffMatchPatch.MATCH_MAX_BITS),
1998                             expectedLoc + text1.length() - DiffMatchPatch.MATCH_MAX_BITS);
1999                     if (endLoc == -1 || startLoc >= endLoc) {
2000                         // Can't find valid trailing context.  Drop this patch.
2001                         startLoc = -1;
2002                     }
2003                 }
2004             } else {
2005                 startLoc = this.matchMain(text, text1, expectedLoc);
2006             }
2007 
2008             if (startLoc == -1) {
2009                 // No match found.  :(
2010                 results[x] = false;
2011                 // Subtract the delta for this failed patch from subsequent patches.
2012                 delta -= aPatch.getLength2() - aPatch.getLength1();
2013             } else {
2014                 // Found a match.  :)
2015                 results[x] = true;
2016                 delta = startLoc - expectedLoc;
2017                 String text2;
2018                 if (endLoc == -1) {
2019                     text2 = text.substring(startLoc,
2020                             Math.min(startLoc + text1.length(), text.length()));
2021                 } else {
2022                     text2 = text.substring(startLoc,
2023                             Math.min(endLoc + DiffMatchPatch.MATCH_MAX_BITS, text.length()));
2024                 }
2025 
2026                 if (text1.equals(text2)) {
2027                     // Perfect match, just shove the replacement text in.
2028                     text = text.substring(0, startLoc) + this.diffText2(aPatch.getDiffs())
2029                             + text.substring(startLoc + text1.length());
2030                 } else {
2031                     // Imperfect match.  Run a diff to get a framework of equivalent
2032                     // indices.
2033                     List<Diff> diffs = this.diffMain(text1, text2, false);
2034                     if (text1.length() > DiffMatchPatch.MATCH_MAX_BITS
2035                             && this.diffLevenshtein(diffs) / (float) text1.length()
2036                             > DiffMatchPatch.PATCH_DELETE_THRESHOLD) {
2037                         // The end points match, but the content is unacceptably bad.
2038                         results[x] = false;
2039                     } else {
2040                         this.diffCleanupSemanticLossless(diffs);
2041                         int index1 = 0;
2042                         for (Diff aDiff : aPatch.getDiffs()) {
2043                             if (aDiff.getOperation() != Operation.EQUAL) {
2044                                 int index2 = this.diffXIndex(diffs, index1);
2045                                 if (aDiff.getOperation() == Operation.INSERT) {
2046                                     // Insertion
2047                                     text = text.substring(0, startLoc + index2) + aDiff.getText()
2048                                             + text.substring(startLoc + index2);
2049                                 } else if (aDiff.getOperation() == Operation.DELETE) {
2050                                     // Deletion
2051                                     text = text.substring(0, startLoc + index2)
2052                                             + text.substring(startLoc + this.diffXIndex(diffs,
2053                                                     index1 + aDiff.getText().length()));
2054                                 }
2055                             }
2056                             if (aDiff.getOperation() != Operation.DELETE) {
2057                                 index1 += aDiff.getText().length();
2058                             }
2059                         }
2060                     }
2061                 }
2062             }
2063             x++;
2064         }
2065         // Strip the padding off.
2066         text = text.substring(nullPadding.length(), text.length()
2067                 - nullPadding.length());
2068         return new Object[]{text, results};
2069     }

2124             int extraLength = paddingLength - lastDiff.getText().length();
2125             lastDiff.setText(lastDiff.getText() + nullPadding.substring(0, extraLength));
