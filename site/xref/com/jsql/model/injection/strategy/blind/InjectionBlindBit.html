21  /**
22   * A blind attack class using concurrent threads.
23   */
24  public class InjectionBlindBit extends AbstractInjectionMonobit<CallableBlindBit> {
25  
26      /**
27       * Log4j logger sent to view.
28       */
29      private static final Logger LOGGER = LogManager.getRootLogger();
30  
31      private String sourceReferencePage;  // Source code of the TRUE web page (usually ?id=1)
32  
33      /**
34       * List of string differences found in all the FALSE queries, compared
35       * to the reference page. Each FALSE pages should contain
36       * at least one same string, which shouldn't be present in all
37       * the TRUE queries.
38       */
39      private List<Diff> falseDiffs = new ArrayList<>();
40      private List<Diff> trueDiffs = new ArrayList<>();
41  
42      /**
43       * Create blind attack initialization.
44       * If every false diffs are not in true diffs and every true diffs are in
45       * true diffs, then Blind attack is confirmed.
46       */
47      public InjectionBlindBit(InjectionModel injectionModel, BlindOperator blindOperator) {
48          super(injectionModel, blindOperator);
49  
50          List<String> falsys = this.injectionModel.getMediatorVendor().getVendor().instance().getFalsyBit();
51          if (falsys.isEmpty() || this.injectionModel.isStoppedByUser()) {
52              return;
53          }
54          
55          // Call the SQL request which must be TRUE (usually ?id=1)
56          this.sourceReferencePage = this.callUrl(StringUtils.EMPTY, "bit#ref:"+ blindOperator.toString().toLowerCase());
57  
58          // Concurrent calls to the FALSE statements,
59          // it will use inject() from the model
60          ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableGetBlindBitTagFalse");
61          Collection<CallableBlindBit> callablesFalsys = new ArrayList<>();
62          for (String falsy: falsys) {
63              callablesFalsys.add(new CallableBlindBit(
64                  falsy,
65                  injectionModel,
66                  this,
67                  blindOperator,
68                  "bit#falsy"
69              ));
70          }
71          
72          // Delete junk from the results of FALSE statements,
73          // keep only diffs found in each and every FALSE pages.
74          // Allow the user to stop the loop
75          try {
76              List<Future<CallableBlindBit>> futuresFalsys = taskExecutor.invokeAll(callablesFalsys);
77              this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
78              for (Future<CallableBlindBit> futureFalsy: futuresFalsys) {
79                  if (this.injectionModel.isStoppedByUser()) {
80                      return;
81                  }
82                  if (this.falseDiffs.isEmpty()) {
83                      this.falseDiffs = futureFalsy.get().getDiffsWithReference();  // Init diffs
84                  } else {
85                      this.falseDiffs.retainAll(futureFalsy.get().getDiffsWithReference());  // Clean un-matching diffs
86                  }
87              }
88          } catch (ExecutionException e) {
89              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
90          } catch (InterruptedException e) {
91              LOGGER.log(LogLevelUtil.IGNORE, e, e);
92              Thread.currentThread().interrupt();
93          }
94  
95          if (this.injectionModel.isStoppedByUser()) {
96              return;
97          }
98          
99          this.cleanTrueDiffs(injectionModel, blindOperator);
100     }
101 
102     private void cleanTrueDiffs(InjectionModel injectionModel, BlindOperator blindOperator) {
103         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableGetBlindBitTagTrue");
104         Collection<CallableBlindBit> callablesTruthys = new ArrayList<>();
105         List<String> truthys = this.injectionModel.getMediatorVendor().getVendor().instance().getTruthyBit();
106         for (String truthy: truthys) {
107             callablesTruthys.add(new CallableBlindBit(
108                 truthy,
109                 injectionModel,
110                 this,
111                 blindOperator,
112                 "bit#truthy"
113             ));
114         }
115 
116         // Remove TRUE diffs in the FALSE diffs as FALSE statement shouldn't contain any TRUE diff.
117         try {
118             List<Future<CallableBlindBit>> futuresTruthys = taskExecutor.invokeAll(callablesTruthys);
119             this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
120             for (Future<CallableBlindBit> futureTruthy: futuresTruthys) {
121                 if (this.injectionModel.isStoppedByUser()) {
122                     return;
123                 }
124                 if (this.trueDiffs.isEmpty()) {
125                     this.trueDiffs = futureTruthy.get().getDiffsWithReference();  // Init diffs
126                 } else {
127                     this.trueDiffs.retainAll(futureTruthy.get().getDiffsWithReference());  // Clean un-matching diffs
128                 }
129                 this.falseDiffs.removeAll(futureTruthy.get().getDiffsWithReference());
130             }
131         } catch (ExecutionException e) {
132             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
133         } catch (InterruptedException e) {
134             LOGGER.log(LogLevelUtil.IGNORE, e, e);
135             Thread.currentThread().interrupt();
136         }
137     }

139     @Override
140     public CallableBlindBit getCallableBitTest(String sqlQuery, int indexChar, int bit) {
141         return new CallableBlindBit(
142             sqlQuery,
143             indexChar,
144             bit,
145             this.injectionModel,
146             this,
147             this.blindOperator,
148             "bit#" + indexChar + "~" + bit
149         );
150     }
151 
152     @Override
153     public boolean isInjectable() throws StoppedByUserSlidingException {
154         if (this.injectionModel.isStoppedByUser()) {
155             throw new StoppedByUserSlidingException();
156         }
157         var callable = new CallableBlindBit(
158             this.injectionModel.getMediatorVendor().getVendor().instance().sqlBlindConfirm(),
159             this.injectionModel,
160             this,
161             this.blindOperator,
162             "bit#confirm"
163         );
164         try {
165             callable.call();
166         } catch (Exception e) {
167             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
168         }
169         return callable.isTrue()
170             // when insertionChar = true then pages ref == truthy == falsy == confirm => falsy cleaned empty, truthy with opcode EQUAL not reliable
171             && this.trueDiffs.stream().anyMatch(diff -> !Operation.EQUAL.equals(diff.operation))
172             || this.falseDiffs.stream().anyMatch(diff -> !Operation.EQUAL.equals(diff.operation));
173     }

178     }
