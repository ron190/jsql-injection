8   import com.jsql.model.exception.StoppedByUserSlidingException;
9   import com.jsql.model.suspendable.AbstractSuspendable;
10  import org.apache.commons.lang3.StringUtils;
11  
12  import java.util.EnumMap;
13  import java.util.Map;
14  
15  /**
16   * Define a strategy to inject SQL with methods like Error and Time.
17   */
18  public abstract class AbstractStrategy {
19  
20      protected static final String KEY_LOG_CHECKING_STRATEGY = "LOG_CHECKING_STRATEGY";
21      protected static final String KEY_LOG_VULNERABLE = "LOG_VULNERABLE";
22  
23      /**
24       * i.e, 2 in "[..]union select 1,2,[..]", if 2 is found in HTML body.
25       */
26      protected String visibleIndex;  // TODO Should be in Normal
27      
28      /**
29       * HTML body of page successfully responding to
30       * multiple fields selection (select 1,2,3,..).
31       */
32      protected String sourceIndexesFound = StringUtils.EMPTY;
33      
34      /**
35       * True if injection can be used, false otherwise.
36       */
37      protected boolean isApplicable = false;
38  
39      protected final InjectionModel injectionModel;
40      
41      protected AbstractStrategy(InjectionModel injectionModel) {
42          
43          this.injectionModel = injectionModel;
44      }
45  
46      /**
47       * Test if this strategy can be used to inject SQL.
48       */
49      public abstract void checkApplicability() throws JSqlException;
50      
51      /**
52       * Inform the view that this strategy can be used.
53       */
54      protected abstract void allow(int... i);
55      
56      /**
57       * Inform the view that this strategy can't be used.
58       */
59      protected abstract void unallow(int... i);
60      
61      /**
62       * Start the strategy work.
63       * @return Source code
64       */
65      public abstract String inject(String sqlQuery, String startPosition, AbstractSuspendable stoppable, String metadataInjectionProcess) throws StoppedByUserSlidingException;
66      
67      /**
68       * Change model strategy to current applicable strategy only when not already set.
69       * Normal > Stacked > Error > Multibit > Blind > Time
70       */
71      public abstract void activateWhenApplicable();
72      
73      /**
74       * Get number of characters you can obtain from the strategy.
75       */
76      public abstract String getPerformanceLength();
77      
78      /**
79       * Get the injection strategy name.
80       */
81      public abstract String getName();
82      
83      public void markVulnerability(Interaction message, int... indexErrorStrategy) {
84          
85          var request = new Request();
86          request.setMessage(message);
87          
88          Map<Header, Object> msgHeader = new EnumMap<>(Header.class);
89          msgHeader.put(Header.URL, this.injectionModel.getMediatorUtils().getConnectionUtil().getUrlByUser());
90          
91          // Ellipse default to non null array
92          if (indexErrorStrategy.length > 0) {
93              
94              msgHeader.put(Header.INDEX_ERROR_STRATEGY, indexErrorStrategy[0]);
95              msgHeader.put(Header.INJECTION_MODEL, this.injectionModel);
96          }
97  
98          request.setParameters(msgHeader);
99          this.injectionModel.sendToViews(request);
100     }
101     
102     @Override
103     public String toString() {
104         return this.getName();
105     }
106 
107     // Getter and setter
108     
109     public boolean isApplicable() {
110         return this.isApplicable;
111     }
112     
113     public void setApplicable(boolean isApplicable) {
114         this.isApplicable = isApplicable;
115     }
116     
117     public String getVisibleIndex() {
118         return this.visibleIndex;
119     }
120 
121     public void setVisibleIndex(String visibleIndex) {
122         this.visibleIndex = visibleIndex;
123     }
124 
125     public String getSourceIndexesFound() {
126         return this.sourceIndexesFound;
127     }
128 
129     public void setSourceIndexesFound(String sourceIndexesFound) {
130         this.sourceIndexesFound = sourceIndexesFound;
131     }
132 }
