28  /**
29   * Thread unit to test if an administration page exists on the server.
30   * The process can be cancelled by the user.
31   */
32  public class CallableHttpHead implements Callable<CallableHttpHead> {
33      
34      /**
35       * Log4j logger sent to view.
36       */
37      private static final Logger LOGGER = LogManager.getRootLogger();
38      
39      /**
40       * URL to an administration page on the website to get tested.
41       */
42      private final String urlAdminPage;
43      
44      /**
45       * HTTP header response code.
46       */
47      private String responseCodeHttp = StringUtils.EMPTY;
48  
49      private final InjectionModel injectionModel;
50  
51      private final String metadataInjectionProcess;
52      
53      /**
54       * Create a callable to find admin page.
55       * @param urlAdminPage URL of admin page
56       */
57      public CallableHttpHead(String urlAdminPage, InjectionModel injectionModel, String metadataInjectionProcess) {
58          this.urlAdminPage = urlAdminPage;
59          this.injectionModel= injectionModel;
60          this.metadataInjectionProcess= metadataInjectionProcess;
61      }
62  
63      /**
64       * Call URL to an administration page in HEAD mode and send the result back to view.
65       */
66      @Override
67      public CallableHttpHead call() {
68          if (this.injectionModel.getResourceAccess().isSearchAdminStopped()) {
69              return this;
70          }
71          
72          try {
73              var builderHttpRequest = HttpRequest.newBuilder()
74                  .uri(URI.create(this.urlAdminPage))
75                  .method("HEAD", BodyPublishers.noBody())
76                  .timeout(Duration.ofSeconds(4));
77              
78              Stream.of(
79                  this.injectionModel.getMediatorUtils().getParameterUtil().getHeaderFromEntries().split("\\\\r\\\\n")
80              )
81              .map(e -> {
82                  if (e.split(":").length == 2) {
83                      return new SimpleEntry<>(
84                          e.split(":")[0],
85                          e.split(":")[1]
86                      );
87                  } else {
88                      return null;
89                  }
90              })
91              .filter(Objects::nonNull)
92              .forEach(e -> builderHttpRequest.header(e.getKey(), e.getValue()));
93              
94              var httpRequest = builderHttpRequest.build();
95              var httpClient = HttpClient.newBuilder()
96                  .connectTimeout(Duration.ofSeconds(4))
97                  .build();
98              HttpResponse<Void> response = httpClient.send(httpRequest, BodyHandlers.discarding());
99  
100             this.responseCodeHttp = String.valueOf(response.statusCode());
101 
102             Map<Header, Object> msgHeader = new EnumMap<>(Header.class);
103             msgHeader.put(Header.URL, this.urlAdminPage);
104             msgHeader.put(Header.HEADER, ConnectionUtil.getHeadersMap(httpRequest.headers()));
105             msgHeader.put(Header.RESPONSE, ConnectionUtil.getHeadersMap(response));
106             msgHeader.put(Header.METADATA_PROCESS, this.metadataInjectionProcess);
107             
108             var request = new Request();
109             request.setMessage(Interaction.MESSAGE_HEADER);
110             request.setParameters(msgHeader);
111             this.injectionModel.sendToViews(request);
112         } catch (InterruptedException e) {
113             LOGGER.log(LogLevelUtil.IGNORE, e, e);
114             Thread.currentThread().interrupt();
115         } catch (Exception e) {
116             var eMessageImplicit = String.format(
117                 "Problem connecting to %s (implicit reason): %s", 
118                 this.urlAdminPage,
119                 InjectionModel.getImplicitReason(e)
120             );
121             String eMessage = Optional.ofNullable(e.getMessage()).orElse(eMessageImplicit);
122             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, eMessage);
123         }
124         
125         return this;
126     }
127 
128     /**
129      * Check if HTTP response is either 2xx or 3xx, which corresponds to
130      * an acceptable response from the website.
131      * @return true if HTTP code start with 2 or 3
132      */
133     public boolean isHttpResponseOk() {
134         return this.responseCodeHttp.matches("[23]\\d\\d");
135     }
136     
137     
138     // Getters
139     
140     public String getUrl() {
141         return this.urlAdminPage;
142     }
143 }
