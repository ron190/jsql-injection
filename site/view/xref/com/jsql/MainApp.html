1   package com.jsql;

3   import com.jsql.model.InjectionModel;
4   import com.jsql.util.LogLevelUtil;
5   import com.jsql.view.swing.JFrameView;
6   import com.jsql.view.swing.util.MediatorHelper;
7   import com.jsql.view.swing.util.UiUtil;
8   import org.apache.logging.log4j.Level;
9   import org.apache.logging.log4j.LogManager;
10  import org.apache.logging.log4j.Logger;
11  
12  import java.awt.*;
13  
14  /**
15   * Main class of the application and called from the .jar.
16   * This class set the general environment of execution and start the software.
17   */
18  public class MainApp {
19      
20      private static final Logger LOGGER = LogManager.getRootLogger();
21      private static final InjectionModel INJECTION_MODEL;  // required to load preferences first
22      
23      static {
24          System.setProperty("jdk.httpclient.allowRestrictedHeaders", "connection,content-length,expect,host,upgrade");
25  
26          if (GraphicsEnvironment.isHeadless()) {
27              LOGGER.log(Level.ERROR, "Headless runtime not supported, use default Java runtime instead");
28              System.exit(1);
29          }
30  
31          INJECTION_MODEL = new InjectionModel();
32          MainApp.INJECTION_MODEL.getMediatorUtils().getPreferencesUtil().loadSavedPreferences();
33  
34          var nameTheme = MainApp.INJECTION_MODEL.getMediatorUtils().getPreferencesUtil().getThemeFlatLafName();
35          UiUtil.applyTheme(nameTheme);  // required init but not enough, reapplied next
36          MainApp.apply4K();
37      }
38  
39      private MainApp() {
40          // nothing
41      }
42      
43      /**
44       * Application starting point.
45       * @param args CLI parameters (not used)
46       */
47      public static void main(String[] args) {
48          MediatorHelper.register(MainApp.INJECTION_MODEL);
49  
50          MainApp.INJECTION_MODEL.getMediatorUtils().getExceptionUtil().setUncaughtExceptionHandler();
51          MainApp.INJECTION_MODEL.getMediatorUtils().getProxyUtil().initializeProxy();
52          MainApp.INJECTION_MODEL.getMediatorUtils().getAuthenticationUtil().setKerberosCifs();
53  
54          try {
55              var view = new JFrameView(MainApp.INJECTION_MODEL);
56              MediatorHelper.register(view);
57              MainApp.INJECTION_MODEL.subscribe(view.getSubscriber());
58          } catch (HeadlessException e) {
59              MainApp.LOGGER.log(LogLevelUtil.CONSOLE_JAVA, "HeadlessException, command line execution not supported: %s", e);
60          } catch (AWTError e) {
61              // Fix #22668: Assistive Technology not found
62              MainApp.LOGGER.log(LogLevelUtil.CONSOLE_JAVA, String.format(
63                  "Java Access Bridge missing or corrupt, check your access bridge definition in JDK_HOME/jre/lib/accessibility.properties: %s",
64                  e.getMessage()
65              ), e);
66          }
67      }
68  
69      private static void apply4K() {  // required not in UiUtil before frame is set
70          if (MainApp.INJECTION_MODEL.getMediatorUtils().getPreferencesUtil().is4K()) {
71              System.setProperty("sun.java2d.uiScale", "2.5");  // jdk >= 9
72          }
73      }
74  }
