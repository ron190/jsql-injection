51      public TabbedPaneNetworkTab() {
52          this.setName("tabNetwork");
53  
54          I18nViewUtil.addComponentForKey("NETWORK_LINE_PLACEHOLDER_URL", this.textAreaNetworkTabUrl);
55          I18nViewUtil.addComponentForKey("NETWORK_LINE_PLACEHOLDER_RESPONSE", this.textAreaNetworkTabResponse);
56          I18nViewUtil.addComponentForKey("NETWORK_LINE_PLACEHOLDER_SOURCE", this.textAreaNetworkTabSource);
57          I18nViewUtil.addComponentForKey("NETWORK_LINE_PLACEHOLDER_PREVIEW", this.textAreaNetworkTabPreview);
58          I18nViewUtil.addComponentForKey("NETWORK_LINE_PLACEHOLDER_HEADERS", this.textAreaNetworkTabHeader);
59          I18nViewUtil.addComponentForKey("NETWORK_LINE_PLACEHOLDER_REQUEST", this.textAreaNetworkTabParams);
60          Stream.of(
61              new SimpleEntry<>("NETWORK_TAB_URL_LABEL", this.textAreaNetworkTabUrl),
62              new SimpleEntry<>("NETWORK_TAB_HEADERS_LABEL", this.textAreaNetworkTabHeader),
63              new SimpleEntry<>("NETWORK_TAB_PARAMS_LABEL", this.textAreaNetworkTabParams),
64              new SimpleEntry<>("NETWORK_TAB_RESPONSE_LABEL", this.textAreaNetworkTabResponse),
65              new SimpleEntry<>("NETWORK_TAB_SOURCE_LABEL", this.textAreaNetworkTabSource),
66              new SimpleEntry<>("NETWORK_TAB_PREVIEW_LABEL", this.textAreaNetworkTabPreview)
67          )
68          .forEach(entry -> {
69              this.addTab(
70                  I18nUtil.valueByKey(entry.getKey()),
71                  entry.getValue() == this.textAreaNetworkTabSource
72                  ? new RTextScrollPane(entry.getValue(), false)
73                  : new JScrollPane(entry.getValue())
74              );
75              var label = new JLabel(I18nUtil.valueByKey(entry.getKey()));
76              label.setName("label"+ entry.getKey());
77              this.setTabComponentAt(
78                  this.indexOfTab(I18nUtil.valueByKey(entry.getKey())),
79                  label
80              );
81              I18nViewUtil.addComponentForKey(entry.getKey(), label);
82  
83              entry.getValue().setName("text"+ entry.getKey());
84  
85              DefaultCaret caret = (DefaultCaret) entry.getValue().getCaret();
86              caret.setUpdatePolicy(DefaultCaret.NEVER_UPDATE);
87          });
88  
89          this.textAreaNetworkTabSource.setPopupMenu(new JPopupMenuComponent(this.textAreaNetworkTabSource));
90          this.textAreaNetworkTabSource.setEditable(false);
91          this.textAreaNetworkTabSource.getCaret().setBlinkRate(0);
92          this.textAreaNetworkTabSource.addFocusListener(new FocusAdapter() {
93              @Override
94              public void focusGained(FocusEvent focusEvent) {
95                  TabbedPaneNetworkTab.this.textAreaNetworkTabSource.getCaret().setVisible(true);
96                  TabbedPaneNetworkTab.this.textAreaNetworkTabSource.getCaret().setSelectionVisible(true);
97              }
98          });
99          this.textAreaNetworkTabSource.setCursor(Cursor.getPredefinedCursor(Cursor.TEXT_CURSOR));
100         this.textAreaNetworkTabSource.setSyntaxEditingStyle(SyntaxConstants.SYNTAX_STYLE_HTML);
101         this.textAreaNetworkTabSource.setHighlightSecondaryLanguages(true);
102         this.applyTheme();
103 
104         this.textAreaNetworkTabHeader.setLineWrap(true);
105         this.textAreaNetworkTabParams.setLineWrap(true);
106         this.textAreaNetworkTabResponse.setLineWrap(true);
107         this.textAreaNetworkTabUrl.setLineWrap(true);
108         this.textAreaNetworkTabSource.setLineWrap(true);
109         
110         this.textAreaNetworkTabPreview.setEditorKit(new HTMLEditorKitTextPaneWrap());
111         this.textAreaNetworkTabPreview.setContentType("text/html");
112         this.textAreaNetworkTabPreview.setEditable(false);
113         this.textAreaNetworkTabPreview.getCaret().setBlinkRate(0);
114         this.textAreaNetworkTabPreview.addFocusListener(new FocusAdapter() {
115             @Override
116             public void focusGained(FocusEvent focusEvent) {
117                 TabbedPaneNetworkTab.this.textAreaNetworkTabPreview.getCaret().setVisible(true);
118                 TabbedPaneNetworkTab.this.textAreaNetworkTabPreview.getCaret().setSelectionVisible(true);
119             }
120         });
121     }

123     public void changeTextNetwork(HttpHeader networkData) {
124         this.textAreaNetworkTabParams.setText(networkData.getPost());
125         this.textAreaNetworkTabUrl.setText(networkData.getUrl());
126 
127         this.textAreaNetworkTabHeader.setText(StringUtils.EMPTY);
128         if (networkData.getHeader() != null) {
129             for (String key: networkData.getHeader().keySet()) {
130                 this.textAreaNetworkTabHeader.append(key + ": " + networkData.getHeader().get(key));
131                 this.textAreaNetworkTabHeader.append("\n");
132             }
133         }
134 
135         this.textAreaNetworkTabResponse.setText(StringUtils.EMPTY);
136         if (networkData.getResponse() != null) {
137             for (String key: networkData.getResponse().keySet()) {
138                 this.textAreaNetworkTabResponse.append(key + ": " + networkData.getResponse().get(key));
139                 this.textAreaNetworkTabResponse.append("\n");
140             }
141         }
142 
143         // Fix #53736: ArrayIndexOutOfBoundsException on setText()
144         // Fix #54573: NullPointerException on setText()
145         try {
146             this.textAreaNetworkTabSource.setText(
147                 StringUtil.detectUtf8(networkData.getSource())
148                 .replaceAll("#{5,}", "#*")
149                 .trim()
150             );
151         } catch (ArrayIndexOutOfBoundsException | NullPointerException e) {
152             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
153         }
154         
155         this.textAreaNetworkTabPreview.getEditorKit().createDefaultDocument();  // Reset EditorKit to disable previous document effect
156         // Proxy is used by jsoup to display <img> tags
157         // Previous test for 2xx Success and 3xx Redirection was Header only, now get the HTML content
158         // Fix #35352: EmptyStackException on setText()
159         // Fix #39841: RuntimeException on setText()
160         // Fix #42523: ExceptionInInitializerError on clean()
161         try {
162             this.textAreaNetworkTabPreview.setText(
163                 Jsoup.clean(
164                     String.format(
165                         "<html>%s</html>",
166                         StringUtil.detectUtf8(networkData.getSource())
167                     )
168                     .replaceAll("<img[^>]*>",StringUtils.EMPTY)  // avoid loading external resources
169                     .replaceAll("<input[^>]*type=\"?hidden\"?.*>", StringUtils.EMPTY)
170                     .replaceAll(
171                         "<input[^>]*type=\"?(submit|button)\"?.*>",
172                         "<div style=\"background-color:#eeeeee;text-align:center;border:1px solid black;width:100px;\">button</div>"
173                     )
174                     .replaceAll(
175                         "<input[^>]*>",
176                         "<div style=\"text-align:center;border:1px solid black;width:100px;\">input</div>"
177                     ),
178                     Safelist.relaxed().addTags("center", "div", "span").addAttributes(":all", "style")
179                 )
180             );
181         } catch (Exception | ExceptionInInitializerError e) {
182             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
183         }
184     }

196             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
197         }
198         
199         // Fix #41879: ArrayIndexOutOfBoundsException on setText()
200         try {
201             this.textAreaNetworkTabPreview.setText(StringUtils.EMPTY);
202         } catch (ArrayIndexOutOfBoundsException e) {
203             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
204         }
205     }
