11  package com.jsql.view.swing.panel.address;
12  
13  import com.jsql.util.I18nUtil;
14  import com.jsql.view.swing.manager.util.StateButton;
15  import com.jsql.view.swing.text.JToolTipI18n;
16  import com.jsql.view.swing.util.I18nViewUtil;
17  import com.jsql.view.swing.util.UiUtil;
18  
19  import javax.swing.*;
20  import java.awt.*;
21  import java.util.concurrent.atomic.AtomicReference;
22  
23  /**
24   * A button displayed in address.
25   */
26  public class ButtonStart extends JButton {
27      
28      /**
29       * State of current injection.
30       */
31      private StateButton state = StateButton.STARTABLE;
32  
33      private final AtomicReference<JToolTipI18n> tooltipRadio = new AtomicReference<>(
34          new JToolTipI18n(I18nUtil.valueByKey("BUTTON_START_TOOLTIP"))
35      );
36  
37      @Override
38      public JToolTip createToolTip() {
39          tooltipRadio.set(new JToolTipI18n(I18nUtil.valueByKey("BUTTON_START_TOOLTIP")));
40          return tooltipRadio.get();
41      }
42  
43      /**
44       * Create a button in address bar.
45       */
46      public ButtonStart() {
47          this.setName("buttonInUrl");
48          this.setToolTipText(I18nUtil.valueByKey("BUTTON_START_TOOLTIP"));
49          I18nViewUtil.addComponentForKey("BUTTON_START_TOOLTIP", tooltipRadio.get());
50  
51          this.setPreferredSize(new Dimension(18, 16));
52          this.setOpaque(false);
53          this.setContentAreaFilled(false);
54          this.setBorderPainted(false);
55          this.setRolloverEnabled(true);
56          this.setIcons();
57      }
58  
59      private void setIcons() {
60          // required to get correct color at startup instead of blu
61          this.setIcon(UiUtil.ARROW.icon);
62          this.setRolloverIcon(UiUtil.ARROW_HOVER.icon);
63          this.setPressedIcon(UiUtil.ARROW_PRESSED.icon);
64      }
65  
66      /**
67       * Replace button with Stop icon ; user can stop current process.
68       */
69      public void setInjectionReady() {
70          this.state = StateButton.STARTABLE;
71          this.setRolloverEnabled(true);
72          this.setEnabled(true);
73          this.setIcons();
74      }
75  
76      /**
77       * Replace button with Stop icon ; user can stop current process.
78       */
79      public void setInjectionRunning() {
80          this.state = StateButton.STOPPABLE;
81          this.setRolloverEnabled(false);
82          this.setEnabled(true);
83          this.setIcon(UiUtil.CROSS_RED.icon);
84          this.setToolTipText(I18nUtil.valueByKey("BUTTON_STOP_TOOLTIP"));
85      }
86  
87      /**
88       * Replace button with icon loader until injection process
89       * is finished ; user waits the end of process.
90       */
91      public void setInjectionStopping() {
92          this.state = StateButton.STOPPING;
93          this.setRolloverEnabled(false);
94          this.setEnabled(false);
95          this.setIcon(UiUtil.HOURGLASS.icon);
96          this.setToolTipText(I18nUtil.valueByKey("BUTTON_STOPPING_TOOLTIP"));
97      }
98  
99      /**
100      * Return the current state of current process.
101      * @return State of process
102      */
103     public StateButton getState() {
104         return this.state;
105     }
106 }
