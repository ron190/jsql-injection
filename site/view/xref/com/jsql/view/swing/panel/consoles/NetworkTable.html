22      /**
23       * List of HTTP injection requests and responses.
24       */
25      private final transient List<HttpHeader> listHttpHeader = new ArrayList<>();
26  
27      public NetworkTable(TabbedPaneNetworkTab tabbedPaneNetworkTab) {
28          super(0, 4);
29          
30          this.setName("networkTable");
31          this.setComponentPopupMenu(new JPopupMenuTable(this));
32          this.setRowSelectionAllowed(true);
33          this.setSelectionMode(ListSelectionModel.MULTIPLE_INTERVAL_SELECTION);
34          this.setRowHeight(20);
35          this.getTableHeader().setReorderingAllowed(false);
36          
37          this.addMouseListener(new MouseAdapter() {
38              @Override
39              public void mousePressed(MouseEvent e) {
40                  NetworkTable.this.requestFocusInWindow();
41                  
42                  // move selected row and place cursor on focused cell
43                  if (SwingUtilities.isRightMouseButton(e)) {
44                      var point = e.getPoint();
45  
46                      // get the row index that contains that coordinate
47                      var rowNumber = NetworkTable.this.rowAtPoint(point);
48                      var colNumber = NetworkTable.this.columnAtPoint(point);
49                      
50                      // Get the ListSelectionModel of the JTable
51                      DefaultListSelectionModel modelRow = (DefaultListSelectionModel) NetworkTable.this.getSelectionModel();
52                      DefaultListSelectionModel modelColumn = (DefaultListSelectionModel) NetworkTable.this.getColumnModel().getSelectionModel();
53  
54                      NetworkTable.this.setRowSelectionInterval(rowNumber, rowNumber);
55                      modelRow.moveLeadSelectionIndex(rowNumber);
56                      modelColumn.moveLeadSelectionIndex(colNumber);
57                  }
58              }
59          });
60  
61          this.setModel(new DefaultTableModel() {
62              private final String[] columns = {
63                  I18nUtil.valueByKey("NETWORK_TAB_URL_COLUMN"),
64                  String.format("%s (KB)", I18nUtil.valueByKey("NETWORK_TAB_SIZE_COLUMN")),
65                  I18nUtil.valueByKey("SQLENGINE_STRATEGY"),
66                  I18nUtil.valueByKey("SQLENGINE_METADATA")
67              };
68              @Override
69              public int getColumnCount() {
70                  return this.columns.length;
71              }
72              @Override
73              public String getColumnName(int index) {
74                  return this.columns[index];
75              }
76          });
77  
78          this.getColumnModel().getColumn(0).setCellRenderer(new TooltipCellRenderer());
79  
80          DefaultTableCellRenderer centerHorizontalAlignment = new CenterRenderer();
81          this.getColumnModel().getColumn(1).setCellRenderer(centerHorizontalAlignment);
82          this.getColumnModel().getColumn(2).setCellRenderer(centerHorizontalAlignment);
83          this.getColumnModel().getColumn(3).setCellRenderer(new CenterRendererWithIcon());
84          
85          this.getInputMap(JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT).put(KeyStroke.getKeyStroke(KeyEvent.VK_TAB, 0), null);
86          this.getInputMap(JComponent.WHEN_ANCESTOR_OF_FOCUSED_COMPONENT).put(KeyStroke.getKeyStroke(KeyEvent.VK_TAB, InputEvent.SHIFT_DOWN_MASK), null);
87          
88          Set<AWTKeyStroke> forward = new HashSet<>(this.getFocusTraversalKeys(KeyboardFocusManager.FORWARD_TRAVERSAL_KEYS));
89          forward.add(KeyStroke.getKeyStroke("TAB"));  // required to unlock focus
90          this.setFocusTraversalKeys(KeyboardFocusManager.FORWARD_TRAVERSAL_KEYS, forward);
91  
92          Set<AWTKeyStroke> backward = new HashSet<>(this.getFocusTraversalKeys(KeyboardFocusManager.BACKWARD_TRAVERSAL_KEYS));
93          backward.add(KeyStroke.getKeyStroke("shift TAB"));  // required to unlock focus
94          this.setFocusTraversalKeys(KeyboardFocusManager.BACKWARD_TRAVERSAL_KEYS, backward);
95          
96          this.getColumnModel().getColumn(0).setPreferredWidth(300);
97          this.getColumnModel().getColumn(1).setPreferredWidth(20);
98          this.getColumnModel().getColumn(2).setPreferredWidth(50);
99          this.getColumnModel().getColumn(3).setPreferredWidth(70);
100         
101         this.getSelectionModel().addListSelectionListener(event -> {
102             if (!event.getValueIsAdjusting() && this.getSelectedRow() > -1) {  // prevent double event
103                 var httpHeader = this.listHttpHeader.get(this.getSelectedRow());
104                 tabbedPaneNetworkTab.changeTextNetwork(httpHeader);
105             }
106         });
107     }
