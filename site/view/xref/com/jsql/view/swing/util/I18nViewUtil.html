1   package com.jsql.view.swing.util;
2   
3   import com.jsql.util.I18nUtil;
4   import org.apache.commons.lang3.StringUtils;
5   
6   import java.util.*;
7   
8   public class I18nViewUtil {
9   
10      /**
11       * A list of graphical components for each i18n keys in the main properties
12       */
13      private static final Map<String, List<Object>> componentsLocalized = new HashMap<>();
14      
15      // Initialize the list of graphical components
16      static {
17          for (String keyI18n: I18nUtil.getLocaleRoot().keySet()) {
18              I18nViewUtil.componentsLocalized.put(keyI18n, new ArrayList<>());
19          }
20      }
21  
22      private I18nViewUtil() {
23          // Utility class
24      }
25      
26      /**
27       * Return the i18n keys of components whose text is replaced
28       * when the translation changes.
29       * @return a set of key names of a i18n key in the properties
30       */
31      public static Set<String> keys() {
32          return I18nViewUtil.componentsLocalized.keySet();
33      }
34      
35      /**
36       * Get a list of graphical components whose text corresponds
37       * to the i18n key in the properties.
38       * @param key name of a i18n key in the properties
39       * @return a list of graphical components
40       */
41      public static List<Object> componentsByKey(String key) {
42          return I18nViewUtil.componentsLocalized.get(key);
43      }
44      
45      /**
46       * Add a graphical component to those whose text must be changed when
47       * the language changes.
48       * @param key name of a i18n key in the properties
49       * @param component graphical component which will receive the translated text
50       */
51      public static void addComponentForKey(String key, Object component) {
52          I18nViewUtil.componentsLocalized.get(key).add(component);
53      }
54  
55      /**
56       * Return the text corresponding to a i18n key in the properties.
57       * @param key a i18n key in the properties
58       * @return text corresponding to the key
59       */
60      public static String valueByKey(String key) {
61          return I18nViewUtil.isNonUbuntu(I18nUtil.getLocaleDefault())
62              ? I18nViewUtil.formatNonLatin(I18nUtil.valueByKey(key))
63              : I18nUtil.valueByKey(key);
64      }
65  
66      public static String valueByKey(String key, Locale newLocale) {
67          return I18nViewUtil.isNonUbuntu(newLocale)
68              ? I18nViewUtil.valueByKey(key)
69              : I18nUtil.valueByKey(key);
70      }
71  
72      public static boolean isNonUbuntu(Locale locale) {
73          return Locale.forLanguageTag("zh").getLanguage().equals(locale.getLanguage())
74              || Locale.forLanguageTag("ko").getLanguage().equals(locale.getLanguage())
75              || Locale.forLanguageTag("ja").getLanguage().equals(locale.getLanguage());
76      }
77  
78      public static String formatNonLatin(String label) {
79          return I18nViewUtil.formatNonLatin(label, StringUtils.EMPTY);
80      }
81  
82      public static String formatNonLatin(String label, String custom) {
83          return String.format(
84              "<html><span style=\"font-family:'%s';%s\">%s</span></html>",
85              UiUtil.FONT_NAME_MONO_ASIAN,
86              custom,
87              label
88          );
89      }
90  }
