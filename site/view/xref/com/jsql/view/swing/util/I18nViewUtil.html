17      /**
18       * A list of graphical components for each i18n keys in the main properties
19       */
20      private static final Map<String, Set<Object>> componentsLocalized = new HashMap<>();
21      
22      // Initialize the list of graphical components
23      static {
24          for (String keyI18n: I18nUtil.BUNDLE_ROOT.keySet()) {
25              I18nViewUtil.componentsLocalized.put(keyI18n, new HashSet<>());
26          }
27      }
28  
29      private I18nViewUtil() {
30          // Utility class
31      }
32      
33      /**
34       * Return the i18n keys of components whose text is replaced
35       * when the translation changes.
36       * @return a set of key names of an i18n key in the properties
37       */
38      public static Set<String> keys() {
39          return I18nViewUtil.componentsLocalized.keySet();
40      }
41      
42      /**
43       * Get a list of graphical components whose text corresponds
44       * to the i18n key in the properties.
45       * @param key name of an i18n key in the properties
46       * @return set of graphical components
47       */
48      public static Set<Object> componentsByKey(String key) {
49          return I18nViewUtil.componentsLocalized.get(key);
50      }
51  
52      public static void switchI18nComponents() {
53          for (String key: I18nViewUtil.keys()) {
54              String textI18n = I18nViewUtil.valueByKey(key);
55              for (Object componentSwing: I18nViewUtil.componentsByKey(key)) {  // instanceof because no common parent with setText()
56                  if (componentSwing instanceof JTextFieldPlaceholder) {  // Textfield does not need <html> tags for asian fonts
57                      ((JTextFieldPlaceholder) componentSwing).setPlaceholderText(I18nUtil.valueByKey(key));
58                  } else if (componentSwing instanceof JTextAreaPlaceholder) {
59                      ((JTextAreaPlaceholder) componentSwing).setPlaceholderText(I18nUtil.valueByKey(key));
60                  } else if (componentSwing instanceof JTextPanePlaceholder) {
61                      ((JTextPanePlaceholder) componentSwing).setPlaceholderText(I18nUtil.valueByKey(key));
62                  } else if (componentSwing instanceof JPasswordFieldPlaceholder) {
63                      ((JPasswordFieldPlaceholder) componentSwing).setPlaceholderText(I18nUtil.valueByKey(key));
64                  } else if (componentSwing instanceof SyntaxTextArea) {
65                      ((SyntaxTextArea) componentSwing).setPlaceholderText(I18nUtil.valueByKey(key));
66                  } else if (componentSwing instanceof JToolTipI18n) {
67                      ((JToolTipI18n) componentSwing).setText(textI18n);
68                  } else if (componentSwing instanceof JLabel) {
69                      ((JLabel) componentSwing).setText(textI18n);
70                  } else if (componentSwing instanceof JMenuItem) {
71                      ((JMenuItem) componentSwing).setText(textI18n);
72                  } else if (componentSwing instanceof JButton) {
73                      ((JButton) componentSwing).setText(textI18n);
74                  } else if (componentSwing instanceof NodeModelEmpty) {
75                      ((NodeModelEmpty) componentSwing).setText(textI18n);
76                  } else if (componentSwing instanceof DialogAbout) {  // not I18nViewUtil.valueByKey() to avoid html in diag title
77                      ((DialogAbout) componentSwing).setTitle(I18nUtil.valueByKey(key) +" "+ StringUtil.APP_NAME);
78                  } else if (componentSwing instanceof JComboBox) {
79                      ((JComboBox<?>) componentSwing).setToolTipText(textI18n);
80                  } else {
81                      ((JTextComponent) componentSwing).setText(textI18n);
82                  }
83              }
84          }
85      }
86      
87      /**
88       * Add a graphical component to those whose text must be changed when
89       * the language changes.
90       * @param key name of an i18n key in the properties
91       * @param component graphical component which will receive the translated text
92       */
93      public static void addComponentForKey(String key, Object component) {
94          I18nViewUtil.componentsLocalized.get(key).add(component);
95      }
96  
97      /**
98       * Return the text corresponding to an i18n key in the properties.
99       * @param key an i18n key in the properties
100      * @return text corresponding to the key
101      */
102     public static String valueByKey(String key) {
103         return I18nViewUtil.isNonUbuntu(I18nUtil.getCurrentLocale())
104         ? I18nViewUtil.formatNonLatin(I18nUtil.valueByKey(key))
105         : I18nUtil.valueByKey(key);
106     }

108     public static boolean isNonUbuntu(Locale locale) {
109         return Locale.forLanguageTag("zh").getLanguage().equals(locale.getLanguage())
110         || Locale.forLanguageTag("ko").getLanguage().equals(locale.getLanguage())
111         || Locale.forLanguageTag("ja").getLanguage().equals(locale.getLanguage());
112     }
113 
114     public static String formatNonLatin(String label) {
115         return I18nViewUtil.formatNonLatin(label, StringUtils.EMPTY);
116     }
117 
118     public static String formatNonLatin(String label, String custom) {
119         return String.format(
120             "<html><span style=\"font-family:'%s';%s\">%s</span></html>",
121             UiUtil.FONT_NAME_MONO_ASIAN,
122             custom,
123             label
124         );
125     }
