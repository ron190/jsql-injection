23          }
24      }
25      public static boolean preventClose(MouseEvent e, JMenuItem m) {
26          if (e.getID() == MouseEvent.MOUSE_RELEASED && m.contains(e.getPoint())) {
27              m.doClick();
28              m.setArmed(true);
29              return true;
30          }
31          return false;
32      }
