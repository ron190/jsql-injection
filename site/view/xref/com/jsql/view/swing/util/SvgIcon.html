3   import com.formdev.flatlaf.extras.FlatSVGIcon;

5   import javax.swing.*;
6   import java.awt.*;
7   import java.util.Objects;
8   
9   public class SvgIcon {
10      public final String name;
11      public final Color from;
12      public final String toDarkUi;
13      public final Color toDark;
14      public final float scale;
15      public final FlatSVGIcon icon;
16  
17      SvgIcon(String name, float scale) {
18          this(name, Color.BLACK, "ComboBox.foreground", scale);
19      }
20  
21      SvgIcon(String name, int from, float scale) {
22          this(name, new Color(from), "ComboBox.foreground", scale);
23      }
24  
25      SvgIcon(String name, Color from, String toDark, float scale) {
26          this(name, from, toDark, UIManager.getColor(toDark), scale);
27      }
28  
29      SvgIcon(String name, Color from, String toDarkUi, Color toDark, float scale) {
30          this.name = name;
31          this.from = from;
32          this.toDarkUi = toDarkUi;
33          this.toDark = toDark;
34          this.scale = scale;
35          this.icon = createSvgIcon(
36              name,
37              from,
38              toDarkUi != null
39              ? UIManager.getColor(toDarkUi)
40              : toDark,
41              scale
42          );
43      }
44  
45      public FlatSVGIcon createSvgIcon(String name, Color from, Color toDark, float scale) {
46          return new FlatSVGIcon(Objects.requireNonNull(UiUtil.class.getClassLoader().getResource(String.format(
47              name.endsWith(".svg") ? "%s" : "swing/images/icons/%s.svg",
48              name
49          ))))
50          .setColorFilter(new FlatSVGIcon.ColorFilter().add(from, null, toDark))
51          .derive(scale);
52      }
53  
54      public void setColorFilter() {
55          this.icon.setColorFilter(new FlatSVGIcon.ColorFilter().add(
56              this.from,
57              null,
58              this.toDarkUi != null
59              ? UIManager.getColor(this.toDarkUi)
60              : this.toDark
61          ));
62      }
63  }
