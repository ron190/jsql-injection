65      public void createFileTab(String label, String content, String path) {
66  
67          JTextArea fileText = new JPopupTextArea().getProxy();
68          fileText.setText(content);
69          fileText.setFont(new Font(UiUtil.FONT_NAME_MONO_NON_ASIAN, Font.PLAIN, 14));
70          fileText.setCaretPosition(0);
71  
72          this.createTextTab(label, path, fileText);
73          MediatorHelper.tabManagers().addToLists(path, label);
74      }

76      public void createReportTab(String label, String content, String path) {
77  
78          JEditorPane editorPane = new JEditorPane();
79          editorPane.setContentType("text/html");
80          editorPane.setText("<html><span style=\"white-space: nowrap; font-family:'"+ UiUtil.FONT_NAME_MONO_NON_ASIAN +"'\">" + content + "</span></html>");
81          editorPane.setDragEnabled(true);
82          editorPane.setEditable(false);
83          editorPane.setCaretPosition(0);
84  
85          editorPane.setComponentPopupMenu(new JPopupMenuText(editorPane));
86  
87          editorPane.addHyperlinkListener(linkEvent -> {
88              if (HyperlinkEvent.EventType.ACTIVATED.equals(linkEvent.getEventType())) {
89                  try {
90                      Desktop.getDesktop().browse(linkEvent.getURL().toURI());
91                  } catch (IOException | URISyntaxException | UnsupportedOperationException e) {
92                      LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Browsing to Url failed", e);
93                  }
94              }
95          });
96  
97          this.createTextTab(label, path, editorPane);
98      }

100     public void createTextTab(String label, String path, JComponent componentText) {
101 
102         var scroller = new LightScrollPane(1, 0, 0, 0, componentText);
103 
104         this.addTab(label + StringUtils.SPACE, scroller);
105 
106         // Focus on the new tab
107         this.setSelectedComponent(scroller);
108 
109         // Create a custom tab header with close button
110         var header = new TabHeader(label, UiUtil.ICON_FILE_SERVER);
111 
112         this.setToolTipTextAt(this.indexOfComponent(scroller), path);
113 
114         // Apply the custom header to the tab
115         this.setTabComponentAt(this.indexOfComponent(scroller), header);
116     }

118     public void createShell(String url, String path) {
119         try {
120             var terminalID = UUID.randomUUID();
121             var terminal = new ShellWeb(terminalID, url);
122             MediatorHelper.frame().getConsoles().put(terminalID, terminal);
123             
124             LightScrollPane scroller = new LightScrollPaneShell(terminal);
125     
126             this.addTab("Web shell ", scroller);
127     
128             // Focus on the new tab
129             this.setSelectedComponent(scroller);
130     
131             // Create a custom tab header with close button
132             var header = new TabHeader("Web shell", UiUtil.ICON_SHELL_SERVER);
133     
134             this.setToolTipTextAt(
135                 this.indexOfComponent(scroller),
136                 String.format(
137                     "<html><b>URL</b><br>%s<br><b>Path</b><br>%s%s</html>",
138                     url,
139                     path,
140                     MediatorHelper.model().getResourceAccess().filenameWebshell
141                 )
142             );
143     
144             // Apply the custom header to the tab
145             this.setTabComponentAt(this.indexOfComponent(scroller), header);
146     
147             terminal.requestFocusInWindow();
148             
149         } catch (MalformedURLException | URISyntaxException e) {
150             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Incorrect shell Url", e);
151         }
152     }

190     public void createValuesTab(String[][] data, String[] columnNames, AbstractElementDatabase table) {
191         
192         // Create a new table to display the values
193         var newTableJPanel = new PanelTable(data, columnNames);
194         
195         // Create a new tab: add header and table
196         this.addTab(StringUtil.detectUtf8(table.toString()), newTableJPanel);
197         newTableJPanel.setComponentOrientation(ComponentOrientation.getOrientation(I18nUtil.getLocaleDefault()));
198         
199         // Focus on the new tab
200         this.setSelectedComponent(newTableJPanel);
201         
202         // Create a custom tab header with close button
203         var header = new TabHeader(UiStringUtil.detectUtf8Html(table.toString()));
204         
205         this.setToolTipTextAt(
206             this.indexOfComponent(newTableJPanel),
207             String.format(
208                 "<html><b>%s.%s</b><br><i>%s</i></html>",
209                 table.getParent(),
210                 table,
211                 String.join("<br>", Arrays.copyOfRange(columnNames, 2, columnNames.length))
212             )
213         );
214         
215         // Apply the custom header to the tab
216         this.setTabComponentAt(
217             this.indexOfComponent(newTableJPanel),
218             header
219         );
220     }
