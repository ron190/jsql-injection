11          
12          super();
13          this.tabbedPane = tabbedPane;
14          this.setOpaque(false);
15      }
16      
17      @Override
18      protected void paintComponent(Graphics g) {
19          
20          this.tabbedPane.getDropLineRect().ifPresent(rect -> {
21              
22              Graphics2D g2 = (Graphics2D) g.create();
23              var r = SwingUtilities.convertRectangle(this.tabbedPane, rect, this);
24              g2.setComposite(AlphaComposite.getInstance(AlphaComposite.SRC_OVER, .5f));
25              g2.setPaint(Color.RED);
26              g2.fill(r);
27              g2.dispose();
28          });
29      }
30
