51      private void clickArrowButton(String actionKey) {
52          
53          JButton scrollForwardButton = null;
54          JButton scrollBackwardButton = null;
55          
56          for (Component c: this.getComponents()) {
57              if (c instanceof JButton) {
58                  if (scrollForwardButton == null) {
59                      scrollForwardButton = (JButton) c;
60                  } else if (scrollBackwardButton == null) {
61                      scrollBackwardButton = (JButton) c;
62                  }
63              }
64          }
65          
66          JButton button = "scrollTabsForwardAction".equals(actionKey) ? scrollForwardButton : scrollBackwardButton;
67      
68          Optional.ofNullable(button)
69              .filter(JButton::isEnabled)
70              .ifPresent(JButton::doClick);
71      }

73      public void autoScrollTest(Point pt) {
74          
75          Rectangle r = this.getTabAreaBounds();
76          
77          if (isTopBottomTabPlacement(this.getTabPlacement())) {
78              
79              RECT_BACKWARD.setBounds(r.x, r.y, SCROLL_SIZE, r.height);
80              RECT_FORWARD.setBounds(r.x + r.width - SCROLL_SIZE - BUTTON_SIZE, r.y, SCROLL_SIZE + BUTTON_SIZE, r.height);
81              
82          } else {
83              
84              RECT_BACKWARD.setBounds(r.x, r.y, r.width, SCROLL_SIZE);
85              RECT_FORWARD.setBounds(r.x, r.y + r.height - SCROLL_SIZE - BUTTON_SIZE, r.width, SCROLL_SIZE + BUTTON_SIZE);
86          }
87          
88          if (RECT_BACKWARD.contains(pt)) {
89              this.clickArrowButton("scrollTabsBackwardAction");
90          } else if (RECT_FORWARD.contains(pt)) {
91              this.clickArrowButton("scrollTabsForwardAction");
92          }
93      }

109     public DnDDropLocation dropLocationForPointDnD(Point p) {
110         
111         for (var i = 0; i < this.getTabCount(); i++) {
112             if (this.getBoundsAt(i).contains(p)) {
113                 return new DnDDropLocation(p, i);
114             }
115         }
116         
117         if (this.getTabAreaBounds().contains(p)) {
118             return new DnDDropLocation(p, this.getTabCount());
119         }
120         
121         return new DnDDropLocation(p, -1);
122     }

124     public void setDropLocation(TransferHandler.DropLocation location, boolean forDrop) {
125         
126         DnDDropLocation old = this.dropLocation;
127         
128         if (Objects.isNull(location) || !forDrop) {
129             this.dropLocation = new DnDDropLocation(new Point(), -1);
130         } else if (location instanceof DnDDropLocation) {
131             this.dropLocation = (DnDDropLocation) location;
132         }
133         
134         this.firePropertyChange("dropLocation", old, this.dropLocation);
135     }

137     public void exportTab(int dragIndex, JTabbedPane target, int targetIndex) {
138         
139         var cmp = this.getComponentAt(dragIndex);
140         var tab = this.getTabComponentAt(dragIndex);
141         String title = this.getTitleAt(dragIndex);
142         var icon = this.getIconAt(dragIndex);
143         String tip = this.getToolTipTextAt(dragIndex);
144         boolean isEnabled = this.isEnabledAt(dragIndex);
145         
146         this.remove(dragIndex);
147         target.insertTab(title, icon, cmp, tip, targetIndex);
148         target.setEnabledAt(targetIndex, isEnabled);
149 
150         target.setTabComponentAt(targetIndex, tab);
151         target.setSelectedIndex(targetIndex);
152         
153         if (tab instanceof JComponent) {
154             ((JComponent) tab).scrollRectToVisible(tab.getBounds());
155         }
156     }

158     public void convertTab(int prev, int next) {
159         
160         var cmp = this.getComponentAt(prev);
161         var tab = this.getTabComponentAt(prev);
162         String title = this.getTitleAt(prev);
163         var icon = this.getIconAt(prev);
164         String tip = this.getToolTipTextAt(prev);
165         boolean isEnabled = this.isEnabledAt(prev);
166         int tgtindex = prev > next ? next : next - 1;
167         
168         this.remove(prev);
169         this.insertTab(title, icon, cmp, tip, tgtindex);
170         this.setEnabledAt(tgtindex, isEnabled);
171         
172         // When you drag'n'drop a disabled tab, it finishes enabled and selected.
173         // pointed out by dlorde
174         if (isEnabled) {
175             this.setSelectedIndex(tgtindex);
176         }
177         
178         // I have a component in all tabs (jlabel with an X to close the tab) and when i move a tab the component disappear.
179         // pointed out by Daniel Dario Morales Salas
180         this.setTabComponentAt(tgtindex, tab);
181     }

183     public Optional<Rectangle> getDropLineRect() {
184         
185         int index = Optional.ofNullable(this.getDropLocation())
186             .filter(DnDDropLocation::isDroppable)
187             .map(DnDDropLocation::getIndex)
188             .orElse(-1);
189         
190         if (index < 0) {
191             
192             RECT_LINE.setBounds(0, 0, 0, 0);
193             return Optional.empty();
194         }
195         
196         int a = Math.min(index, 1);
197         Rectangle r = this.getBoundsAt(a * (index - 1));
198         
199         if (isTopBottomTabPlacement(this.getTabPlacement())) {
200             RECT_LINE.setBounds(r.x - LINE_WIDTH / 2 + r.width * a, r.y, LINE_WIDTH, r.height);
201         } else {
202             RECT_LINE.setBounds(r.x, r.y - LINE_WIDTH / 2 + r.height * a, r.width, LINE_WIDTH);
203         }
204         
205         return Optional.of(RECT_LINE);
206     }

208     public Rectangle getTabAreaBounds() {
209         
210         Rectangle tabbedRect = this.getBounds();
211         int xx = tabbedRect.x;
212         int yy = tabbedRect.y;
213         
214         Rectangle compRect = Optional.ofNullable(this.getSelectedComponent())
215             .map(Component::getBounds)
216             .orElseGet(Rectangle::new);
217 
218         int tabPlacement = this.getTabPlacement();
219         
220         if (isTopBottomTabPlacement(tabPlacement)) {
221             
222             tabbedRect.height = tabbedRect.height - compRect.height;
223             
224             if (tabPlacement == BOTTOM) {
225                 tabbedRect.y += compRect.y + compRect.height;
226             }
227         } else {
228             
229             tabbedRect.width = tabbedRect.width - compRect.width;
230             
231             if (tabPlacement == RIGHT) {
232                 tabbedRect.x += compRect.x + compRect.width;
233             }
234         }
235         
236         tabbedRect.translate(-xx, -yy);
237         
238         return tabbedRect;
239     }

245     private class Handler extends MouseAdapter implements PropertyChangeListener { // , BeforeDrag
246         
247         private Point startPt;
248         private final int gestureMotionThreshold = DragSource.getDragThreshold();
249 
250         private void repaintDropLocation() {
251             
252             Component c = DnDTabbedPane.this.getRootPane().getGlassPane();
253             
254             if (c instanceof GhostGlassPane) {
255                 
256                 GhostGlassPane glassPane = (GhostGlassPane) c;
257                 glassPane.setTargetTabbedPane(DnDTabbedPane.this);
258                 glassPane.repaint();
259             }
260         }
261         
262         // PropertyChangeListener
263         @Override
264         public void propertyChange(PropertyChangeEvent e) {
265             
266             String propertyName = e.getPropertyName();
267             if ("dropLocation".equals(propertyName)) {
268                 this.repaintDropLocation();
269             }
270         }
271         
272         // MouseListener
273         @Override
274         public void mousePressed(MouseEvent e) {
275             
276             DnDTabbedPane src = (DnDTabbedPane) e.getComponent();
277             boolean isOnlyOneTab = src.getTabCount() <= 1;
278             
279             if (isOnlyOneTab) {
280                 
281                 this.startPt = null;
282                 return;
283             }
284             
285             var tabPt = e.getPoint();
286             int idx = src.indexAtLocation(tabPt.x, tabPt.y);
287             
288             // disabled tab, null component problem.
289             // pointed out by daryl. NullPointerException: i.e. addTab("Tab", null)
290             boolean flag = idx < 0 || !src.isEnabledAt(idx) || Objects.isNull(src.getComponentAt(idx));
291             
292             this.startPt = flag ? null : tabPt;
293         }
294         
295         @Override
296         public void mouseDragged(MouseEvent e) {
297             
298             var tabPt = e.getPoint();
299             
300             if (Objects.nonNull(this.startPt) && this.startPt.distance(tabPt) > this.gestureMotionThreshold) {
301                 
302                 DnDTabbedPane src = (DnDTabbedPane) e.getComponent();
303                 var th = src.getTransferHandler();
304                 DnDTabbedPane.this.dragTabIndex = src.indexAtLocation(tabPt.x, tabPt.y);
305                 
306                 // Unhandled NoClassDefFoundError #56620: Could not initialize class java.awt.dnd.DragSource
307                 th.exportAsDrag(src, e, TransferHandler.MOVE);
308                 
309                 RECT_LINE.setBounds(0, 0, 0, 0);
310                 src.getRootPane().getGlassPane().setVisible(true);
311                 src.setDropLocation(new DnDDropLocation(tabPt, -1), true);
312                 
313                 this.startPt = null;
314             }
315         }
316 
317         @Override
318         public void mouseClicked(MouseEvent e) {
319             
320             var tabPt = e.getPoint();
321             JTabbedPane src = (JTabbedPane) e.getSource();
322             
323             int i = src.indexAtLocation(tabPt.x, tabPt.y);
324             
325             if (-1 < i && e.getButton() == MouseEvent.BUTTON2) {
326                 ActionCloseTabResult.perform(i);
327             }
328         }
329     }
