6   /**
7    * Listener for processing keyboard input.
8    * Used for example to process tampering, sql engine,
9    * or encoding when keyboard input is entered.
10   */
11  public abstract class DocumentListenerEditing implements DocumentListener {
12  
13      public abstract void process();
14      
15      @Override
16      public void changedUpdate(DocumentEvent e) {
17          this.process();
18      }
19      
20      @Override
21      public void removeUpdate(DocumentEvent e) {
22          this.process();
23      }
24      
25      @Override
26      public void insertUpdate(DocumentEvent e) {
27          this.process();
28      }
29  }
