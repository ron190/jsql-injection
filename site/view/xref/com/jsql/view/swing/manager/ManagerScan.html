59      public ManagerScan() {
60          
61          this.setLayout(new BorderLayout());
62  
63          List<ItemList> itemsList = this.getItemList();
64          
65          final DnDList dndListScan = new DnDListScan(itemsList);
66          
67          dndListScan.setTransferHandler(null);
68          dndListScan.setTransferHandler(new ListTransfertHandlerScan());
69          
70          this.listPaths = dndListScan;
71          this.listPaths.setName(ManagerScan.NAME);
72          
73          this.getListPaths().setBorder(BorderFactory.createEmptyBorder(0, 0, LightScrollPane.THUMB_SIZE, 0));
74          this.add(new LightScrollPane(0, 0, 0, 0, dndListScan), BorderLayout.CENTER);
75  
76          JPanel lastLine = this.getLastLinePanel(dndListScan);
77          this.add(lastLine, BorderLayout.SOUTH);
78          
79          dndListScan.addListSelectionListener(e -> {
80              
81              if (dndListScan.getSelectedValue() == null) {
82                  
83                  return;
84              }
85              
86              var beanInjection = ((ItemListScan) dndListScan.getSelectedValue()).getBeanInjection();
87              
88              MediatorHelper.panelAddressBar().getTextFieldAddress().setText(beanInjection.getUrl());
89              MediatorHelper.panelAddressBar().getTextFieldHeader().setText(beanInjection.getHeader());
90              MediatorHelper.panelAddressBar().getTextFieldRequest().setText(beanInjection.getRequest());
91              
92              String requestType = beanInjection.getRequestType();
93              if (requestType != null && !requestType.isEmpty()) {
94                  
95                  MediatorHelper.panelAddressBar().getRadioRequest().setText(requestType);
96                  
97              } else {
98                  
99                  MediatorHelper.panelAddressBar().getRadioRequest().setText("GET");
100             }
101             
102             AbstractMethodInjection method = beanInjection.getMethodInstance();
103             
104             if (method == MediatorHelper.model().getMediatorMethod().getHeader()) {
105                 
106                 MediatorHelper.panelAddressBar().getRadioHeader().setSelected();
107                 
108             } else if (method == MediatorHelper.model().getMediatorMethod().getRequest()) {
109                 
110                 MediatorHelper.panelAddressBar().getRadioRequest().setSelected();
111                 
112             } else {
113                 
114                 MediatorHelper.panelAddressBar().getRadioQueryString().setSelected();
115             }
116         });
117     }

119     private JPanel getLastLinePanel(final DnDList dndListScan) {
120

142     }

144     private List<ItemList> getItemList() {
145         
146         var jsonScan = new StringBuilder();
147         
148         try (
149             var inputStream = UiUtil.class.getClassLoader().getResourceAsStream(UiUtil.INPUT_STREAM_PAGES_SCAN);
150             var inputStreamReader = new InputStreamReader(Objects.requireNonNull(inputStream), StandardCharsets.UTF_8);
151             var reader = new BufferedReader(inputStreamReader)
152         ) {
153             
154             String line;
155             while ((line = reader.readLine()) != null) {
156                 
157                 jsonScan.append(line);
158             }
159             
160         } catch (IOException e) {
161             
162             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
163         }
164         
165         List<ItemList> itemsList = new ArrayList<>();
166         
167         try {
168             var jsonArrayScan = new JSONArray(jsonScan.toString());
169             
170             for (var i = 0 ; i < jsonArrayScan.length() ; i++) {
171                 
172                 var jsonObjectScan = jsonArrayScan.getJSONObject(i);
173                 
174                 var beanInjection = new BeanInjection(
175                     jsonObjectScan.getString("url"),
176                     jsonObjectScan.optString("request"),
177                     jsonObjectScan.optString("header"),
178                     jsonObjectScan.optString("method"),
179                     jsonObjectScan.optString("vendor"),
180                     jsonObjectScan.optString("requestType")
181                 );
182                 
183                 itemsList.add(new ItemListScan(beanInjection));
184             }
185             
186         } catch (JSONException e) {
187             
188             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
189         }
190         
191         return itemsList;
192     }

194     private void initializeRunButton(final DnDList dndListScan) {
195         
196         this.defaultText = "SCAN_RUN_BUTTON_LABEL";
197         this.run = new JButtonStateful(this.defaultText);
198         I18nViewUtil.addComponentForKey("SCAN_RUN_BUTTON_LABEL", this.run);
199         this.run.setToolTipText(I18nUtil.valueByKey("SCAN_RUN_BUTTON_TOOLTIP"));
200 
201         this.run.setContentAreaFilled(false);
202         this.run.setBorder(BorderFactory.createEmptyBorder(4, 8, 4, 8));
203         this.run.setBackground(UiUtil.COLOR_FOCUS_GAINED);
204         
205         this.run.addMouseListener(new FlatButtonMouseAdapter(this.run));
206         
207         this.run.addActionListener(actionEvent -> {
208             
209             if (dndListScan.getSelectedValuesList().isEmpty()) {
210                 
211                 LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select URL(s) to scan");
212                 
213                 return;
214             }
215             
216             new Thread(
217                 () -> {
218                 
219                     if (ManagerScan.this.run.getState() == StateButton.STARTABLE) {
220                         
221                         ManagerScan.this.run.setText(I18nViewUtil.valueByKey("SCAN_RUN_BUTTON_STOP"));
222                         ManagerScan.this.run.setState(StateButton.STOPPABLE);
223                         ManagerScan.this.loader.setVisible(true);
224                         
225                         DefaultListModel<ItemList> listModel = (DefaultListModel<ItemList>) dndListScan.getModel();
226                         for (var i = 0 ; i < listModel.getSize() ; i++) {
227                             
228                             listModel.get(i).reset();
229                         }
230                         
231                         this.scan(dndListScan.getSelectedValuesList());
232                         
233                     } else {
234                         
235                         MediatorHelper.model().getResourceAccess().setScanStopped(true);
236                         MediatorHelper.model().setIsStoppedByUser(true);
237                         ManagerScan.this.run.setEnabled(false);
238                         ManagerScan.this.run.setState(StateButton.STOPPING);
239                     }
240                 },
241                 "ThreadScan"
242             ).start();
243         });
244     }

246     /**
247      * Start fast scan of URLs in sequence and display result.
248      * Unplug any existing view and plug a console-like view in order to
249      * respond appropriately to GUI message with simple text result instead of
250      * build complex graphical components during the multi website injections.
251      * At the end of the scan it plugs again the normal view.
252      * @param urlsItemList contains a list of String URL
253      */
254     public void scan(List<ItemList> urlsItemList) {
255         
256         // Erase everything in the view from a previous injection
257         var requests = new Request();
258         requests.setMessage(Interaction.RESET_INTERFACE);
259         MediatorHelper.model().sendToViews(requests);
260         
261         // wait for ending of ongoing interaction between two injections
262         try {
263             Thread.sleep(500);
264             
265         } catch (InterruptedException e) {
266             
267             LOGGER.log(LogLevelUtil.IGNORE, e, e);
268             Thread.currentThread().interrupt();
269         }
270 
271         // Display result only in console
272         var requestUnsubscribe = new Request();
273         requestUnsubscribe.setMessage(Interaction.UNSUBSCRIBE);
274         MediatorHelper.model().sendToViews(requestUnsubscribe);
275         MediatorHelper.model().subscribe(new ScanListTerminal());
276         
277         MediatorHelper.model().setIsScanning(true);
278         MediatorHelper.model().getResourceAccess().setScanStopped(false);
279         
280         for (ItemList urlItemList: urlsItemList) {
281             
282             var urlItemListScan = (ItemListScan) urlItemList;
283             if (MediatorHelper.model().isStoppedByUser() || MediatorHelper.model().getResourceAccess().isScanStopped()) {
284                 
285                 break;
286             }
287             
288             LOGGER.log(LogLevelUtil.CONSOLE_INFORM, "Scanning {}", urlItemListScan.getBeanInjection().getUrl());
289 
290             Optional<Vendor> vendor = MediatorHelper.model().getMediatorVendor().getVendors()
291                 .stream()
292                 .filter(v -> v.toString().equalsIgnoreCase(urlItemListScan.getBeanInjection().getVendor()))
293                 .findAny();
294             MediatorHelper.model().getMediatorVendor().setVendorByUser(vendor.orElse(MediatorHelper.model().getMediatorVendor().getAuto()));
295 
296             MediatorHelper.model().getMediatorUtils().getParameterUtil().controlInput(
297                 urlItemListScan.getBeanInjection().getUrl(),
298                 urlItemListScan.getBeanInjection().getRequest(),
299                 urlItemListScan.getBeanInjection().getHeader(),
300                 urlItemListScan.getBeanInjection().getMethodInstance(),
301                 urlItemListScan.getBeanInjection().getRequestType(),
302                 true
303             );
304             
305             try {
306                 Thread.sleep(500);
307                 
308             } catch (InterruptedException e) {
309                 
310                 LOGGER.log(LogLevelUtil.IGNORE, e, e);
311                 Thread.currentThread().interrupt();
312             }
313         }
314         
315         // Get back the normal view
316         MediatorHelper.model().sendToViews(requestUnsubscribe);
317         MediatorHelper.model().subscribe(MediatorHelper.frame().getSubscriber());
318         
319         MediatorHelper.model().setIsScanning(false);
320         MediatorHelper.model().setIsStoppedByUser(false);
321         MediatorHelper.model().getResourceAccess().setScanStopped(false);
322 
323         var request = new Request();
324         request.setMessage(Interaction.END_SCAN);
325         MediatorHelper.model().sendToViews(request);
326     }
