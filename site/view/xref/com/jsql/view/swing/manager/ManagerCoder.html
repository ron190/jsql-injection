32  /**
33   * Manager to code/decode string in various methods.
34   */
35  public class ManagerCoder extends JPanel {
36      
37      /**
38       * User input to encode.
39       */
40      private final JTextArea textInput;
41  
42      /**
43       * JTextArea displaying result of encoding/decoding.
44       */
45      private final JTextPane result;
46  
47      /**
48       * Encoding choice by user.
49       */
50      private JLabel menuMethod;
51  
52      private final transient CoderListener actionCoder = new CoderListener(this);
53  
54      public static final String[] HASHES = new String[]{
55          "Adler32", "Crc16", "Crc32", "Crc64", "Md2", "Md4", "Md5", "Sha-1", "Sha-256", "Sha-384", "Sha-512", "Mysql"
56      };
57  
58      public static final String[] HASHES_FOR_EMPTY_TEXT = new String[]{
59          "Md2", "Md4", "Md5", "Sha-1", "Sha-256", "Sha-384", "Sha-512", "Mysql"
60      };
61  
62      private class ChangeMenuListener implements ChangeListener {
63          private final String nameMethod;
64          
65          ChangeMenuListener(String nameMethod) {
66              this.nameMethod = nameMethod;
67          }
68          
69          @Override
70          public void stateChanged(ChangeEvent e) {
71              if (e.getSource() instanceof JMenuItem) {
72                  JMenuItem item = (JMenuItem) e.getSource();
73                  if (item.isSelected() || item.isArmed()) {
74                      ManagerCoder.this.actionCoder.actionPerformed(this.nameMethod);
75                  }
76              }
77          }
78      }
79  
80      /**
81       * Create a panel to encode a string.
82       */
83      public ManagerCoder() {
84          super(new BorderLayout());
85  
86          this.textInput = new JPopupTextArea(new JTextAreaPlaceholder("Type a text to convert")).getProxy();
87          this.textInput.getCaret().setBlinkRate(500);
88          this.textInput.setEditable(true);
89          this.textInput.setLineWrap(true);
90          this.textInput.setName("textInputManagerCoder");
91          this.textInput.getDocument().addDocumentListener(new DocumentListenerEditing() {
92              @Override
93              public void process() {
94                  ManagerCoder.this.actionCoder.actionPerformed();
95              }
96          });
97  
98          JPanel topMixed = this.getTopPanel();
99  
100         this.result = new JPopupTextPane("Result of conversion").getProxy();
101         this.result.setContentType("text/html");
102         this.result.setFont(UIManager.getFont("TextArea.font"));  // required to increase text size
103         this.result.setEditorKit(new HTMLEditorKitTextPaneWrap());
104         this.result.setName("resultManagerCoder");
105         
106         var bottom = new JPanel(new BorderLayout());
107         bottom.add(new JScrollPane(this.result), BorderLayout.CENTER);
108 
109         var divider = new JSplitPane(JSplitPane.VERTICAL_SPLIT);
110         divider.setResizeWeight(0.5);
111         divider.setTopComponent(topMixed);
112         divider.setBottomComponent(bottom);
113         this.add(divider, BorderLayout.CENTER);
114     }
115 
116     private JPanel getTopPanel() {
117         var comboMenubar = this.getLabelMenu();
118 
119         var topMixed = new JPanel(new BorderLayout());
120         topMixed.add(new JScrollPane(this.textInput), BorderLayout.CENTER);
121         topMixed.add(comboMenubar, BorderLayout.SOUTH);
122         return topMixed;
123     }
124 
125     private JLabel getLabelMenu() {
126         Map<String, JMenu> mapMenus = new LinkedHashMap<>();
127         
128         mapMenus.put("Base16", new JMenu("Base16"));
129         mapMenus.put("Base32", new JMenu("Base32"));
130         mapMenus.put("Base58", new JMenu("Base58"));
131         mapMenus.put("Base64", new JMenu("Base64"));
132         mapMenus.put("Hex", new JMenu("Hex"));
133         mapMenus.put("Url", new JMenu("Url"));
134         mapMenus.put("Unicode", new JMenu("Unicode"));
135         
136         var menuHtml = new JMenu("Html");
137         mapMenus.put("Html", menuHtml);
138         mapMenus.put("Base64(zipped)", new JMenu("Base64(zipped)"));
139         mapMenus.put("Hex(zipped)", new JMenu("Hex(zipped)"));
140 
141         var menuEncodeHtmlDecimal = new JMenuItem("Encode to Html (decimal)");
142         menuHtml.add(menuEncodeHtmlDecimal);
143         menuEncodeHtmlDecimal.addActionListener(this.actionCoder);
144         menuEncodeHtmlDecimal.addChangeListener(new ChangeMenuListener("Encode to Html (decimal)"));
145         
146         mapMenus.forEach((key, value) -> {
147             var menuEncode = new JMenuItem("Encode to " + key);
148             menuEncode.addActionListener(this.actionCoder);
149             menuEncode.addChangeListener(new ChangeMenuListener("Encode to " + key));
150             menuEncode.setName("encodeTo" + key);
151 
152             var menuDecode = new JMenuItem("Decode from " + key);
153             menuDecode.addActionListener(this.actionCoder);
154             menuDecode.addChangeListener(new ChangeMenuListener("Decode from " + key));
155             menuDecode.setName("decodeFrom" + key);
156 
157             value.add(menuEncode);
158             value.add(menuDecode);
159             value.setName(key);
160         });
161 
162         mapMenus.put("Hash", new JMenu("Hash"));
163         mapMenus.get("Hash").setName("Hash");
164         
165         Stream.of(ManagerCoder.HASHES).forEach(hash -> {
166             var menuEncode = new JMenuItem("Hash to "+ hash);
167             menuEncode.addActionListener(this.actionCoder);
168             menuEncode.addChangeListener(new ChangeMenuListener("Hash to "+ hash));
169             menuEncode.setName("hashTo"+ hash);
170             mapMenus.get("Hash").add(menuEncode);
171         });
172 
173         JPopupMenu popupMenu = new JPopupMenu();
174         for (JMenu menu: mapMenus.values()) {
175             popupMenu.add(menu);
176         }
177 
178         JLabel labelMenu = new JLabel(UiUtil.ARROW_DOWN.icon, SwingConstants.LEFT);
179         this.menuMethod = labelMenu;
180         labelMenu.setText("Encode to Base64");
181         labelMenu.setName("menuMethodManagerCoder");
182         labelMenu.setBorder(UiUtil.BORDER_5PX);
183         labelMenu.addMouseListener(new MouseAdapter() {
184             @Override
185             public void mousePressed(MouseEvent e) {
186                 Arrays.stream(popupMenu.getComponents()).map(component -> (JMenu) component).forEach(jMenu -> {
187                     jMenu.updateUI();
188                     for (var i = 0 ; i < jMenu.getItemCount() ; i++) {
189                         jMenu.getItem(i).updateUI();
190                     }
191                 });  // required: incorrect when dark/light mode switch
192                 popupMenu.updateUI();  // required: incorrect when dark/light mode switch
193                 popupMenu.show(e.getComponent(), e.getComponent().getX(),e.getComponent().getY() + e.getComponent().getHeight());
194                 popupMenu.setLocation(e.getComponent().getLocationOnScreen().x,e.getComponent().getLocationOnScreen().y + e.getComponent().getHeight());
195             }
196         });
197         return labelMenu;
198     }
