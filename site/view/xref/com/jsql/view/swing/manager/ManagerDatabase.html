36  /**
37   * Manager to code/decode string in various methods.
38   */
39  public class ManagerDatabase extends JPanel {
40      
41      /**
42       * Log4j logger sent to view.
43       */
44      private static final Logger LOGGER = LogManager.getRootLogger();
45  
46      private TreeDatabase tree;
47  
48      /**
49       * Create a panel to encode a string.
50       */
51      public ManagerDatabase() {
52          super(new BorderLayout());
53  
54          // First node in tree
55          AbstractNodeModel nodeModelEmpty = new NodeModelEmpty(I18nViewUtil.valueByKey("DATABASE_EMPTY"));
56          var root = new DefaultMutableTreeNode(nodeModelEmpty);
57          I18nViewUtil.addComponentForKey("DATABASE_EMPTY", nodeModelEmpty);
58  
59          this.tree = new TreeDatabase(root);
60          this.tree.setName("treeDatabases");
61          MediatorHelper.register(this.tree);
62  
63          this.tree.setCellRenderer(new CellRendererNode());
64          this.tree.addMouseListener(this.getTreeMouseListener());
65          this.tree.addKeyListener(this.getTreeKeyListener());
66          this.tree.setCellEditor(new CellEditorNode());
67          this.tree.setEditable(true);  // allows repaint nodes
68          this.tree.setShowsRootHandles(true);
69          this.tree.getSelectionModel().setSelectionMode(TreeSelectionModel.SINGLE_TREE_SELECTION);
70          this.tree.getModel().addTreeModelListener(new TreeModelProgressListener());  // required to repaint progress bar
71  
72          this.add(new JScrollPane(this.tree), BorderLayout.CENTER);
73      }
74  
75      private KeyAdapter getTreeKeyListener() {
76          return new KeyAdapter() {
77              @Override
78              public void keyPressed(KeyEvent e) {
79                  if (e.getKeyCode() == KeyEvent.VK_F2) {
80                      DefaultMutableTreeNode treeNode = (DefaultMutableTreeNode) ManagerDatabase.this.tree.getLastSelectedPathComponent();
81                      if (treeNode != null) {
82                          AbstractNodeModel nodeModel = (AbstractNodeModel) treeNode.getUserObject();
83                          if (nodeModel != null && nodeModel.getPanel() != null && !nodeModel.isRunning()) {
84                              nodeModel.setIsEdited(true);
85                          }
86                      }
87                  }
88              }
89          };
90      }
91  
92      private MouseAdapter getTreeMouseListener() {
93          return new MouseAdapter() {
94              @Override
95              public void mousePressed(MouseEvent event) {
96                  int selRow = ManagerDatabase.this.tree.getRowForLocation(event.getX(), event.getY());
97                  TreePath selPath = ManagerDatabase.this.tree.getPathForLocation(event.getX(), event.getY());
98                  if (selRow != -1 && event.getClickCount() == 2) {
99                      // Fix ArrayIndexOutOfBoundsException on collapsePath()
100                     try {
101                         if (ManagerDatabase.this.tree.isExpanded(selPath)) {
102                             ManagerDatabase.this.tree.collapsePath(selPath);
103                         } else {
104                             ManagerDatabase.this.tree.expandPath(selPath);
105                         }
106                     } catch (ArrayIndexOutOfBoundsException e) {
107                         LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
108                     }
109                 }
110             }
111         };
112     }
113     
114     private class TreeModelProgressListener implements TreeModelListener {
115         @Override
116         public void treeNodesChanged(TreeModelEvent treeModelEvent) {
117             if (treeModelEvent == null) {
118                 return;
119             }
120             ManagerDatabase.this.tree.firePropertyChange(
121                 JTree.ROOT_VISIBLE_PROPERTY,
122                 !ManagerDatabase.this.tree.isRootVisible(),
123                 ManagerDatabase.this.tree.isRootVisible()
124             );
125             ManagerDatabase.this.tree.treeDidChange();
126         }
127         @Override
128         public void treeStructureChanged(TreeModelEvent treeModelEvent) {
129             // Do nothing
130         }
131         @Override
132         public void treeNodesRemoved(TreeModelEvent treeModelEvent) {
133             // Do nothing
134         }
135         @Override
136         public void treeNodesInserted(TreeModelEvent treeModelEvent) {
137             // Do nothing
138         }
139     }
140 }
