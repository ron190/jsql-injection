36  /**
37   * Manager for uploading PHP webshell to the host and send system commands.
38   */
39  public abstract class AbstractManagerShell extends AbstractManagerList {
40      
41      /**
42       * Log4j logger sent to view.
43       */
44      private static final Logger LOGGER = LogManager.getRootLogger();
45      protected final JTextField textfieldUrlShell;
46      
47      /**
48       * Build the manager panel.
49       */
50      protected AbstractManagerShell() {
51          super("swing/list/payload.txt");
52  
53          var tooltip = new AtomicReference<>(new JToolTipI18n(I18nUtil.valueByKey("SHELL_URL_TOOLTIP")));
54          var placeholderResult = new JTextFieldPlaceholder(I18nUtil.valueByKey("SHELL_URL_LABEL")) {
55              @Override
56              public JToolTip createToolTip() {
57                  return tooltip.get();
58              }
59          };
60          this.textfieldUrlShell = new JPopupTextField(placeholderResult).getProxy();
61          I18nViewUtil.addComponentForKey("SHELL_URL_LABEL", this.textfieldUrlShell);
62          I18nViewUtil.addComponentForKey("SHELL_URL_TOOLTIP", tooltip.get());
63          this.textfieldUrlShell.setToolTipText(I18nUtil.valueByKey("SHELL_URL_TOOLTIP"));
64  
65          this.buildRunButton("SHELL_RUN_BUTTON_LABEL", "SHELL_RUN_BUTTON_TOOLTIP");
66          this.run.setEnabled(false);
67          this.buildPrivilege();
68  
69          var southPanel = new JPanel();
70          southPanel.setLayout(new BoxLayout(southPanel, BoxLayout.Y_AXIS));
71          southPanel.add(this.textfieldUrlShell);
72          southPanel.add(this.lastLine);
73          this.add(southPanel, BorderLayout.SOUTH);
74      }
75  
76      protected abstract void createPayload(String pathShell, String urlShell, File fileToUpload) throws JSqlException;
77  
78      protected class ActionExploit implements ActionListener {
79          private final JComboBox<ModelItemType> comboBoxExploitTypes;
80  
81          public ActionExploit(JComboBox<ModelItemType> comboBoxExploitTypes) {
82              this.comboBoxExploitTypes = comboBoxExploitTypes;
83          }
84  
85          @Override
86          public void actionPerformed(ActionEvent evt) {
87              var modelSelectItem = (ModelItemType) this.comboBoxExploitTypes.getSelectedItem();
88              if (ManagerExploit.KEY_UDF_TAB.equals(Objects.requireNonNull(modelSelectItem).getKeyLabel())) {
89                  new SwingWorker<>() {
90                      @Override
91                      protected Object doInBackground() { Thread.currentThread().setName("SwingWorkerExploitUdf");
92                          AbstractManagerShell.this.start(null, null, null);
93                          return null;
94                      }
95                  }.doInBackground();
96                  return;
97              }
98              if (AbstractManagerShell.this.listPaths.getSelectedValuesList().isEmpty()) {
99                  LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select at least one directory in the list");
100                 return;
101             }
102             String urlShell = AbstractManagerShell.this.textfieldUrlShell.getText();
103             if (!urlShell.isEmpty() && !urlShell.matches("(?i)^https?://.*")) {
104                 if (!urlShell.matches("(?i)^\\w+://.*")) {
105                     LOGGER.log(LogLevelUtil.CONSOLE_INFORM, "Undefined shell URL protocol, forcing to [https://]");
106                     urlShell = "https://"+ urlShell;
107                 } else {
108                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Unknown URL protocol");
109                     return;
110                 }
111             }
112             if (StringUtils.isNotEmpty(urlShell)) {
113                 try {
114                     new URI(urlShell);
115                 } catch (URISyntaxException e) {
116                     LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Incorrect URL: %s", e.getMessage()));
117                     return;
118                 }
119             }
120 
121             AtomicReference<File> fileToUpload = new AtomicReference<>();
122             if (ManagerExploit.KEY_UPLOAD_TAB.equals(modelSelectItem.getKeyLabel())) {
123                 fileToUpload.set(AbstractManagerShell.chooseFile());
124                 if (fileToUpload.get() == null) {
125                     return;
126                 }
127             }
128 
129             String urlShellFinal = urlShell;
130             new SwingWorker<>() {
131                 @Override
132                 protected Object doInBackground() { Thread.currentThread().setName("SwingWorkerExploitNonUdf");
133                     AbstractManagerShell.this.horizontalGlue.setVisible(false);
134                     AbstractManagerShell.this.progressBar.setVisible(true);
135                     AbstractManagerShell.this.listPaths.getSelectedValuesList().forEach(remotePathFolder -> {
136                         LOGGER.log(LogLevelUtil.CONSOLE_DEFAULT, String.format("Checking path [%s]...", remotePathFolder));
137                         AbstractManagerShell.this.start(remotePathFolder.toString(), urlShellFinal, fileToUpload.get());
138                     });
139                     AbstractManagerShell.this.endProcess();
140                     return null;
141                 }
142             }.doInBackground();
143         }
144     }
145 
146     private static File chooseFile() {
147         var filechooser = new JFileChooser(MediatorHelper.model().getMediatorUtils().getPreferencesUtil().getPathFile());
148         filechooser.setDialogTitle(I18nUtil.valueByKey("UPLOAD_DIALOG_TEXT"));
149         int returnVal = filechooser.showOpenDialog(MediatorHelper.frame());
150         if (returnVal == JFileChooser.APPROVE_OPTION) {
151             return filechooser.getSelectedFile();
152         }
153         return null;
154     }
155 
156     private void start(String remotePathFolder, String urlShellFinal, File fileToUpload) {
157         try {
158             this.createPayload(remotePathFolder, urlShellFinal, fileToUpload);
159         } catch (JSqlException e) {
160             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Payload creation failure: %s", e.getMessage()));
161         }
162     }
163 }
