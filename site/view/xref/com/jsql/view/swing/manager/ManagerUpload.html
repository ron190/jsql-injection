28  /**
29   * Manager to upload files to the host.
30   */
31  public class ManagerUpload extends AbstractManagerList {
32      
33      /**
34       * Log4j logger sent to view.
35       */
36      private static final Logger LOGGER = LogManager.getRootLogger();
37  
38      /**
39       * Build the manager panel.
40       */
41      public ManagerUpload() {
42          super(UiUtil.PATH_WEB_FOLDERS);
43          this.defaultText = "UPLOAD_RUN_BUTTON_LABEL";
44  
45          final JTextField shellURL = new JPopupTextField(I18nUtil.valueByKey("UPLOAD_URL_LABEL")).getProxy();
46          String urlTooltip = I18nUtil.valueByKey("UPLOAD_URL_TOOLTIP");
47          shellURL.setToolTipText(urlTooltip);
48          this.initializeRunButton(shellURL);
49  
50          this.privilege = new JLabel(I18nUtil.valueByKey("PRIVILEGE_LABEL"), UiUtil.SQUARE.icon, SwingConstants.LEFT);
51          I18nViewUtil.addComponentForKey("PRIVILEGE_LABEL", this.privilege);
52          this.privilege.setToolTipText(I18nUtil.valueByKey("PRIVILEGE_TOOLTIP"));
53          this.progressBar.setVisible(false);
54          var lastLine = new JPanel();
55          lastLine.setLayout(new BoxLayout(lastLine, BoxLayout.X_AXIS));
56          lastLine.add(Box.createRigidArea(new Dimension(5, 0)));
57          lastLine.add(this.privilege);
58          lastLine.add(Box.createHorizontalGlue());
59          lastLine.add(this.run);
60  
61          var southPanel = new JPanel();
62          southPanel.setLayout(new BoxLayout(southPanel, BoxLayout.Y_AXIS));
63          southPanel.add(shellURL);
64          southPanel.add(lastLine);
65          this.add(southPanel, BorderLayout.SOUTH);
66      }
67  
68      private void initializeRunButton(final JTextField shellURL) {
69          this.run = new JButtonStateful(this.defaultText);
70          I18nViewUtil.addComponentForKey(this.defaultText, this.run);
71          this.run.setToolTipText(I18nUtil.valueByKey("UPLOAD_RUN_BUTTON_TOOLTIP"));
72          this.run.setEnabled(false);
73          this.run.addActionListener(actionEvent -> this.initializeRunAction(shellURL));
74      }
75  
76      private void initializeRunAction(final JTextField shellURL) {
77          
78          if (ManagerUpload.this.listPaths.getSelectedValuesList().isEmpty()) {
79              LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Select directory(ies) to upload a file into");
80              return;
81          }
82  
83          final var filechooser = new JFileChooser(MediatorHelper.model().getMediatorUtils().getPreferencesUtil().getPathFile());
84          filechooser.setDialogTitle(I18nUtil.valueByKey("UPLOAD_DIALOG_TEXT"));
85          
86          // Fix #2402: NullPointerException on showOpenDialog()
87          // Fix #40547: ClassCastException on showOpenDialog()
88          try {
89              int returnVal = filechooser.showOpenDialog(MediatorHelper.frame());
90              if (returnVal != JFileChooser.APPROVE_OPTION) {
91                  return;
92              }
93              this.uploadFiles(shellURL, filechooser);
94          } catch (NullPointerException | ClassCastException e) {
95              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
96          }
97      }
98  
99      private void uploadFiles(final JTextField shellURL, final JFileChooser fileChooser) {
100         for (final Object path: ManagerUpload.this.listPaths.getSelectedValuesList()) {
101             new Thread(
102                 () -> {
103                     try {
104                         var file = fileChooser.getSelectedFile();
105                         ManagerUpload.this.progressBar.setVisible(true);
106                         MediatorHelper.model().getResourceAccess().uploadFile(path.toString(), shellURL.getText(), file);
107                     } catch (JSqlException e) {
108                         LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Payload creation error: %s", e.getMessage()));
109                     } catch (IOException e) {
110                         LOGGER.log(LogLevelUtil.CONSOLE_ERROR, String.format("Posting file failed: %s", e.getMessage()), e);
111                     } catch (InterruptedException e) {
112                         LOGGER.log(LogLevelUtil.IGNORE, e, e);
113                         Thread.currentThread().interrupt();
114                     }
115                 },
116                 "ThreadUpload"
117             ).start();
118         }
119     }
