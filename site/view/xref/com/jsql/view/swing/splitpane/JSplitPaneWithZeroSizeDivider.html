26  import com.jsql.view.swing.util.UiUtil;
27  import org.apache.logging.log4j.LogManager;
28  import org.apache.logging.log4j.Logger;

30  import javax.swing.*;
31  import javax.swing.border.Border;
32  import javax.swing.plaf.basic.BasicSplitPaneDivider;
33  import javax.swing.plaf.basic.BasicSplitPaneUI;
34  import java.awt.*;
35  
36  /**
37   * A JSplitPane that uses a 1 pixel thin visible divider,
38   * but a 9 pixel wide transparent drag area.
39   */
40  public class JSplitPaneWithZeroSizeDivider extends JSplitPane {
41      
42      /**
43       * Log4j logger sent to view.
44       */
45      private static final Logger LOGGER = LogManager.getRootLogger();
46      
47      /**
48       * The size of the transparent drag area.
49       */
50      private int dividerDragSize = 9;
51      
52      /**
53       * The offset of the transparent drag area relative to the visible divider line.
54       * Positive offset moves the drag area left/top to the divider line.
55       * If zero then the drag area is right/bottom of the divider line.
56       * Useful values are in the range 0 to {@link #dividerDragSize}.
57       * Default is centered.
58       */
59      private int dividerDragOffset = 0;
60      
61      public JSplitPaneWithZeroSizeDivider(int orientation) {
62          super(orientation, true);
63      }
64      
65      public void disableDragSize() {
66          this.dividerDragSize = 0;
67      }
68      
69      public void enableDragSize() {
70          this.dividerDragSize = 9;
71      }
72  
73      public int getDividerDragSize() {
74          return this.dividerDragSize;
75      }
76  
77      public void setDividerDragSize(int dividerDragSize) {
78          
79          this.dividerDragSize = dividerDragSize;
80          this.revalidate();
81      }
82  
83      public int getDividerDragOffset() {
84          return this.dividerDragOffset;
85      }
86  
87      public void setDividerDragOffset(int dividerDragOffset) {
88          
89          this.dividerDragOffset = dividerDragOffset;
90          this.revalidate();
91      }
92  
93      @Override
94      public void doLayout() {
95          
96          super.doLayout();
97  
98          // increase divider width or height
99          BasicSplitPaneDivider divider = ((BasicSplitPaneUI) this.getUI()).getDivider();
100         Rectangle bounds = divider.getBounds();
101         
102         if (this.orientation == HORIZONTAL_SPLIT) {
103             
104             bounds.x -= this.dividerDragOffset;
105             bounds.width = this.dividerDragSize;
106             
107         } else {
108             
109             bounds.y -= this.dividerDragOffset;
110             bounds.height = this.dividerDragSize;
111         }
112         
113         divider.setBounds(bounds);
114     }
115 
116     @Override
117     public void updateUI() {
118         
119         this.setUI(new SplitPaneWithZeroSizeDividerUI());
120         this.revalidate();
121     }
122 
123     private class SplitPaneWithZeroSizeDividerUI extends BasicSplitPaneUI {
124         
125         @Override
126         public BasicSplitPaneDivider createDefaultDivider() {
127             return new ZeroSizeDivider(this);
128         }
129     }
130 
131     private class ZeroSizeDivider extends BasicSplitPaneDivider {
132         
133         public ZeroSizeDivider(BasicSplitPaneUI ui) {
134             super(ui);
135             super.setBorder(null);
136             this.setBackground(UiUtil.COLOR_COMPONENT_BORDER);
137         }
138 
139         @Override
140         public void setBorder(Border border) {
141             // ignore
142         }
143 
144         @Override
145         public void paint(Graphics g) {
146             
147             g.setColor(this.getBackground());
148             
149             if (this.orientation == HORIZONTAL_SPLIT) {
150                 
151                 // Fix #38925: ClassCastException on drawLine()
152                 try {
153                     g.drawLine(
154                         JSplitPaneWithZeroSizeDivider.this.dividerDragOffset,
155                         0,
156                         JSplitPaneWithZeroSizeDivider.this.dividerDragOffset,
157                         this.getHeight() - 1
158                     );
159                     
160                 } catch (ClassCastException e) {
161                     
162                     LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
163                 }
164                 
165             } else {
166                 
167                 g.drawLine(
168                     0,
169                     JSplitPaneWithZeroSizeDivider.this.dividerDragOffset,
170                     this.getWidth() - 1,
171                     JSplitPaneWithZeroSizeDivider.this.dividerDragOffset
172                 );
173             }
174         }
175 
176         @Override
177         protected void dragDividerTo(int location) {
178             
179             super.dragDividerTo(location + JSplitPaneWithZeroSizeDivider.this.dividerDragOffset);
180         }
181 
182         @Override
183         protected void finishDraggingTo(int location) {
184             
185             super.finishDraggingTo(location + JSplitPaneWithZeroSizeDivider.this.dividerDragOffset);
186         }
187     }
188 }
