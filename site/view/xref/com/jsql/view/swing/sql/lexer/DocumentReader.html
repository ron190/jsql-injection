18  package com.jsql.view.swing.sql.lexer;
19  
20  import com.jsql.util.LogLevelUtil;
21  import org.apache.logging.log4j.LogManager;
22  import org.apache.logging.log4j.Logger;
23  
24  import javax.swing.text.AbstractDocument;
25  import javax.swing.text.BadLocationException;
26  import java.io.Reader;
27  
28  /**
29   * A reader interface for an abstract document.  Since
30   * the syntax highlighting packages only accept Stings and
31   * Readers, this must be used.
32   * Since the close() method does nothing and a seek() method
33   * has been added, this allows us to get some performance
34   * improvements through reuse.  It can be used even after the
35   * lexer explicitly closes it by seeking to the place that
36   * we want to read next, and reseting the lexer.
37   */
38  class DocumentReader extends Reader {
39      
40      /**
41       * Log4j logger sent to view.
42       */
43      private static final Logger LOGGER = LogManager.getRootLogger();
44      
45      /**
46       * Current position in the document. Incremented
47       * whenever a character is read.
48       */
49      private long position = 0;
50      
51      /**
52       * Saved position used in the mark and reset methods.
53       */
54      private long mark = -1;
55      
56      /**
57       * The document that we are working with.
58       */
59      private final AbstractDocument document;
60      
61      /**
62       * Construct a reader on the given document.
63       *
64       * @param document the document to be read.
65       */
66      public DocumentReader(AbstractDocument document) {
67          this.document = document;
68      }
69  
70      /**
71       * Modifying the document while the reader is working is like
72       * pulling the rug out from under the reader.  Alerting the
73       * reader with this method (in a nice thread safe way, this
74       * should not be called at the same time as a read) allows
75       * the reader to compensate.
76       */
77      public void update(int position, int adjustment) {
78          
79          if (position < this.position) {
80              
81              if (this.position < position - adjustment) {
82                  
83                  this.position = position;
84                  
85              } else {
86                  
87                  this.position += adjustment;
88              }
89          }
90      }
91  
92      /**
93       * Has no effect.  This reader can be used even after
94       * it has been closed.
95       */
96      @Override
97      public void close() {
98          // nothing
99      }
100 
101     /**
102      * Save a position for reset.
103      *
104      * @param readAheadLimit ignored.
105      */
106     @Override
107     public void mark(int readAheadLimit) {
108         this.mark = this.position;
109     }
110 
111     /**
112      * This reader support mark and reset.
113      *
114      * @return true
115      */
116     @Override
117     public boolean markSupported() {
118         return true;
119     }
120 
121     /**
122      * Read a single character.
123      *
124      * @return the character or -1 if the end of the document has been reached.
125      */
126     @Override
127     public int read() {
128         
129         if (this.position < this.document.getLength()) {
130             
131             try {
132                 char c = this.document.getText((int)this.position, 1).charAt(0);
133                 this.position++;
134                 
135                 return c;
136                 
137             } catch (BadLocationException e) {
138                 
139                 LOGGER.log(LogLevelUtil.IGNORE, e);
140                 return -1;
141             }
142             
143         } else {
144             
145             return -1;
146         }
147     }
148 
149     /**
150      * Read and fill the buffer.
151      * This method will always fill the buffer unless the end of the document is reached.
152      *
153      * @param cbuf the buffer to fill.
154      * @return the number of characters read or -1 if no more characters are available in the document.
155      */
156     @Override
157     public int read(char[] cbuf) {
158         return this.read(cbuf, 0, cbuf.length);
159     }
160 
161     /**
162      * Read and fill the buffer.
163      * This method will always fill the buffer unless the end of the document is reached.
164      *
165      * @param cbuf the buffer to fill.
166      * @param off offset into the buffer to begin the fill.
167      * @param len maximum number of characters to put in the buffer.
168      * @return the number of characters read or -1 if no more characters are available in the document.
169      */
170     @Override
171     public int read(char[] cbuf, int off, int len) {
172         
173         if (this.position < this.document.getLength()) {
174             
175             int length = len;
176             
177             if (this.position + length >= this.document.getLength()) {
178                 length = this.document.getLength() - (int)this.position;
179             }
180             
181             if (off + length >= cbuf.length) {
182                 length = cbuf.length - off;
183             }
184             
185             try {
186                 String s = this.document.getText((int)this.position, length);
187                 this.position += length;
188                 
189                 for (int i=0; i<length; i++) {
190                     cbuf[off+i] = s.charAt(i);
191                 }
192                 
193                 return length;
194                 
195             } catch (BadLocationException e) {
196                 
197                 LOGGER.log(LogLevelUtil.IGNORE, e);
198                 return -1;
199             }
200         } else {
201             
202             return -1;
203         }
204     }
205 
206     /**
207      * @return true
208      */
209     @Override
210     public boolean ready() {
211         return true;
212     }
213 
214     /**
215      * Reset this reader to the last mark, or the beginning of the document if a mark has not been set.
216      */
217     @Override
218     public void reset() {
219         
220         if (this.mark == -1) {
221             
222             this.position = 0;
223             
224         } else {
225             
226             this.position = this.mark;
227         }
228         
229         this.mark = -1;
230     }
231 
232     /**
233      * Skip characters of input.
234      * This method will always skip the maximum number of characters unless
235      * the end of the file is reached.
236      *
237      * @param n number of characters to skip.
238      * @return the actual number of characters skipped.
239      */
240     @Override
241     public long skip(long n) {
242         
243         if (this.position + n <= this.document.getLength()) {
244             
245             this.position += n;
246             return n;
247             
248         } else {
249             
250             long oldPos = this.position;
251             this.position = this.document.getLength();
252             
253             return this.document.getLength() - oldPos;
254         }
255     }
256 
257     /**
258      * Seek to the given position in the document.
259      *
260      * @param n the offset to which to seek.
261      */
262     public void seek(long n) {
263         
264         if (n <= this.document.getLength()) {
265             
266             this.position = n;
267             
268         } else {
269             
270             this.position = this.document.getLength();
271         }
272     }
273 }
