643     /**
644      * Configure all text components with new coloring and new modelYaml setter.
645      */
646     private static void initializeTextComponents() {
647         
648         SqlEngine.getTextPanes().forEach(SqlEngine::resetLexer);
649         SqlEngine.getTextPanes().forEach(JTextPaneObjectMethod::switchSetterToVendor);
650         SqlEngine.getTextPanes().forEach(textPaneLexer -> textPaneLexer.setText(StringUtils.EMPTY));
651         
652         Stream.of(TextareaWithColor.values())
653         .forEach(entry ->
654             entry.getText().setText(
655                 entry.getText()
656                 .getSupplierGetter()
657                 .get()
658                 .trim()
659             )
660         );
661 
662         SqlEngine.populateTabError();
663     }

665     /**
666      * Dynamically add textPanes to Error tab for current vendor.
667      */
668     private static void populateTabError() {
669         
670         SqlEngine.tabbedPaneError.removeAll();
671         
672         if (SqlEngine.modelYaml.getStrategy().getError() == null) {
673             
674             return;
675         }
676             
677         for (Method methodError: SqlEngine.modelYaml.getStrategy().getError().getMethod()) {
678             
679             var panelError = new JPanel(new BorderLayout());
680             
681             final var refMethodError = new Method[]{ methodError };
682             
683             var textPaneError = new JTextPaneLexer(
684                 refMethodError[0]::setQuery,
685                 refMethodError[0]::getQuery
686             );
687             
688             SqlEngine.resetLexer(textPaneError);
689             textPaneError.switchSetterToVendor();
690             textPaneError.setText(methodError.getQuery().trim());
691             textPaneError.setBorder(SqlEngine.borderRight);
692 
693             panelError.add(new LightScrollPane(1, 0, 1, 0, textPaneError), BorderLayout.CENTER);
694             
695             var panelLimit = new JPanel();
696             panelLimit.setLayout(new BoxLayout(panelLimit, BoxLayout.LINE_AXIS));
697             panelLimit.add(new JLabel(" Overflow limit: "));
698             panelLimit.add(new JTextField(Integer.toString(methodError.getCapacity())));
699             
700             // TODO Integrate Error limit
701             panelError.add(panelLimit, BorderLayout.SOUTH);
702 
703             SqlEngine.tabbedPaneError.addTab(methodError.getName(), panelError);
704             
705             SqlEngine.tabbedPaneError.setTitleAt(
706                 SqlEngine.tabbedPaneError.getTabCount() - 1,
707                 String.format(
708                     "<html><div style=\"text-align:left;width:100px;\">%s</div></html>",
709                     methodError.getName()
710                 )
711             );
712             
713             SqlEngine.textPanesError.add(textPaneError);
714         }
715     }
