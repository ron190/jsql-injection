447     private static JPanel initializeMenuVendor() {
448         var panelMenuVendor = new JPanel();  // required for label on right
449         panelMenuVendor.setLayout(new BorderLayout());
450 
451         JPopupMenu popupMenuVendors = new JPopupMenu();
452         popupMenuVendors.setLayout(UiUtil.getColumnLayout(MediatorHelper.model().getMediatorVendor().getVendors().size()));
453 
454         JLabel labelVendor = new JLabel(MediatorHelper.model().getMediatorVendor().getVendor().toString(), UiUtil.ARROW_DOWN.icon, SwingConstants.LEFT);
455         labelVendor.setBorder(UiUtil.BORDER_5PX);  // required for padding
456         labelVendor.addMouseListener(new MouseAdapter() {
457             @Override
458             public void mousePressed(MouseEvent e) {
459                 Arrays.stream(popupMenuVendors.getComponents()).map(a -> (JComponent) a).forEach(JComponent::updateUI);  // required: incorrect when dark/light mode switch
460                 popupMenuVendors.updateUI();  // required: incorrect when dark/light mode switch
461                 popupMenuVendors.show(e.getComponent(), e.getComponent().getX(),e.getComponent().getY() + e.getComponent().getHeight());
462                 popupMenuVendors.setLocation(e.getComponent().getLocationOnScreen().x,e.getComponent().getLocationOnScreen().y + e.getComponent().getHeight());
463             }
464         });
465 
466         List<Vendor> listVendors = new LinkedList<>(MediatorHelper.model().getMediatorVendor().getVendors());
467         listVendors.removeIf(vendor -> vendor == MediatorHelper.model().getMediatorVendor().getAuto());
468 
469         var groupVendor = new ButtonGroup();
470         for (final Vendor vendor: listVendors) {
471             JMenuItem itemRadioVendor = new JRadioButtonMenuItem(
472                 vendor.toString(),
473                 vendor == MediatorHelper.model().getMediatorVendor().getVendor()
474             );
475             
476             itemRadioVendor.addActionListener(actionEvent -> {
477                 SqlEngine.modelYaml = vendor.instance().getModelYaml();

484         }
485         
486         panelMenuVendor.add(labelVendor, BorderLayout.LINE_END);  // required to set on right
487         return panelMenuVendor;

490     /**
491      * Configure all text components with new coloring and new modelYaml setter.
492      */
493     private static void initializeTextComponents() {
494         SqlEngine.getTextareas().forEach(SqlEngine::reset);
495         SqlEngine.getTextareas().forEach(textPaneLexer -> textPaneLexer.setText(StringUtils.EMPTY));
496 
497         Stream.of(TextareaWithColor.values())
498         .map(textareaWithColor -> textareaWithColor.textarea)
499         .forEach(textArea -> {
500             textArea.setText(textArea.getSupplierGetter().get().trim());
501             textArea.setCaretPosition(0);
502         });
503 
504         SqlEngine.populateTabError();
505         SqlEngine.applyTheme();
506     }

508     public static void applyTheme() {
509         SqlEngine.getTextareas().forEach(UiUtil::applyTheme);
510     }

512     /**
513      * Dynamically add textPanes to Error tab for current vendor.
514      */
515     private static void populateTabError() {
516         SqlEngine.tabsError.removeAll();
517 
518         if (SqlEngine.modelYaml.getStrategy().getError() == null) {
519             return;
520         }
521             
522         for (Method methodError: SqlEngine.modelYaml.getStrategy().getError().getMethod()) {
523             var panelError = new JPanel(new BorderLayout());
524 
525             var textareaError = new JSyntaxTextArea(methodError::setQuery, methodError::getQuery);
526             SqlEngine.reset(textareaError);
527             textareaError.setText(methodError.getQuery().trim());
528             textareaError.setCaretPosition(0);
529             panelError.add(new RTextScrollPane(textareaError, false), BorderLayout.CENTER);
530             
531             var panelLimit = new JPanel();  // TODO Integrate Error limit
532             panelLimit.setLayout(new BoxLayout(panelLimit, BoxLayout.LINE_AXIS));
533             panelLimit.add(new JLabel(" Overflow limit: "));
534             panelLimit.add(new JTextField(Integer.toString(methodError.getCapacity())));
535             panelError.add(panelLimit, BorderLayout.SOUTH);
536 
537             SqlEngine.tabsError.addTab(methodError.getName(), panelError);
538             SqlEngine.tabsError.setTitleAt(
539                 SqlEngine.tabsError.getTabCount() - 1,
540                 String.format(
541                     "<html><div style=\"text-align:left;width:100px;\">%s</div></html>",
542                     methodError.getName()
543                 )
544             );
545             SqlEngine.listTextareasError.add(textareaError);
546         }
547     }

550      * Reset the textarea colorer.
551      * @param textarea which colorer will be reset.
552      */
553     private static void reset(JSyntaxTextArea textarea) {
554         textarea.setDocument(new RSyntaxDocument(SyntaxConstants.SYNTAX_STYLE_SQL));  // required else empty on reopen
555         textarea.getDocument().addDocumentListener(new DocumentListenerEditing() {
556             @Override
557             public void process() {

559             }
560         });
