105     private JMenu initMenuEdit() {
106         var menuEdit = new JMenu(I18nUtil.valueByKey("MENUBAR_EDIT"));
107         I18nViewUtil.addComponentForKey("MENUBAR_EDIT", menuEdit);
108         menuEdit.setMnemonic('E');
109 
110         JMenuItem itemCopy = new JMenuItem(I18nUtil.valueByKey("CONTEXT_MENU_COPY"), 'C');
111         I18nViewUtil.addComponentForKey("CONTEXT_MENU_COPY", itemCopy);
112         itemCopy.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_C, InputEvent.CTRL_DOWN_MASK));
113         itemCopy.addActionListener(actionEvent -> {
114             if (MediatorHelper.tabResults().getSelectedComponent() instanceof PanelTable) {
115                 ((PanelTable) MediatorHelper.tabResults().getSelectedComponent()).copyTable();
116             } else if (MediatorHelper.tabResults().getSelectedComponent() instanceof JScrollPane) {
117                 ((JTextComponent) ((JScrollPane) MediatorHelper.tabResults().getSelectedComponent()).getViewport().getView()).copy();
118             }
119         });
120 
121         JMenuItem itemSelectAll = new JMenuItem(I18nUtil.valueByKey("CONTEXT_MENU_SELECT_ALL"), 'A');
122         I18nViewUtil.addComponentForKey("CONTEXT_MENU_SELECT_ALL", itemSelectAll);
123         itemSelectAll.setAccelerator(KeyStroke.getKeyStroke(KeyEvent.VK_A, InputEvent.CTRL_DOWN_MASK));
124         itemSelectAll.addActionListener(actionEvent -> {
125             if (MediatorHelper.tabResults().getSelectedComponent() instanceof PanelTable) {
126                 ((PanelTable) MediatorHelper.tabResults().getSelectedComponent()).selectTable();
127             } else if (MediatorHelper.tabResults().getSelectedComponent() instanceof JScrollPane) {
128                 // Textarea need focus to select all
129                 ((JScrollPane) MediatorHelper.tabResults().getSelectedComponent()).getViewport().getView().requestFocusInWindow();
130                 ((JTextComponent) ((JScrollPane) MediatorHelper.tabResults().getSelectedComponent()).getViewport().getView()).selectAll();
131             }
132         });
133 
134         menuEdit.add(itemCopy);
135         menuEdit.add(new JSeparator());
136         menuEdit.add(itemSelectAll);
137         return menuEdit;
138     }

158     private JMenu initMenuI18nContribution() {
159         var menuI18nContribution = new JMenu(I18nUtil.valueByKey("MENUBAR_COMMUNITY_HELPTRANSLATE"));
160         I18nViewUtil.addComponentForKey("MENUBAR_COMMUNITY_HELPTRANSLATE", menuI18nContribution);
161 
162         final var dialogTranslate = new DialogTranslate();  // Render the About dialog behind scene
163         AppMenubar.MODELS_ITEM_INTO.forEach(model -> {
164             model.setMenuItem(new JMenuItem(model.getLanguage().getMenuItemLabel(), model.getLanguage().getFlag()));
165             model.getMenuItem().addActionListener(new ActionTranslate(dialogTranslate, model.getLanguage()));
166             menuI18nContribution.add(model.getMenuItem());
167         });
168 
169         var itemIntoOther = new JMenuItem(I18nUtil.valueByKey("MENUBAR_COMMUNITY_ANOTHERLANGUAGE"));
170         I18nViewUtil.addComponentForKey("MENUBAR_COMMUNITY_ANOTHERLANGUAGE", itemIntoOther);
171 
172         AppMenubar.MODELS_ITEM_INTO.stream().filter(model -> model.getLanguage() == Language.AR)
173         .forEach(modelItemTranslate -> modelItemTranslate.getMenuItem().setComponentOrientation(ComponentOrientation.RIGHT_TO_LEFT));
174         AppMenubar.MODELS_ITEM_INTO.stream().filter(model -> model.getLanguage() == Language.FR)
175         .forEach(modelItemTranslate -> modelItemTranslate.getMenuItem().setName("itemIntoFrench"));
176 
177         menuI18nContribution.add(new JSeparator());
178         menuI18nContribution.add(itemIntoOther);
179         itemIntoOther.addActionListener(new ActionTranslate(dialogTranslate, Language.OT));
180 
181         return menuI18nContribution;
182     }

184     private JMenuItem initItemReportIssue() {
185         JMenuItem itemReportIssue = new JMenuItem(I18nUtil.valueByKey("MENUBAR_COMMUNITY_REPORTISSUE"), 'R');
186         I18nViewUtil.addComponentForKey("MENUBAR_COMMUNITY_REPORTISSUE", itemReportIssue);
187 
188         itemReportIssue.addActionListener(actionEvent -> {
189 
190             var panel = new JPanel(new BorderLayout());
191             final JTextArea textarea = new JPopupTextArea(new JTextArea()).getProxy();
192             textarea.setFont(new Font(
193                 UiUtil.FONT_NAME_MONOSPACED,
194                 Font.PLAIN,
195                 UIManager.getDefaults().getFont("TextField.font").getSize()
196             ));
197             textarea.setText(
198                 "## What's the expected behavior?\n\n"
199                 + "## What's the actual behavior?\n\n"
200                 + "## Any other detailed information on the Issue?\n\n"
201                 + "## Steps to reproduce the problem\n\n"
202                 + "  1. ...\n"
203                 + "  2. ...\n\n"
204                 + "## [Community] Request for new feature\n\n"
205             );
206             panel.add(new JLabel("Describe your bug or issue:"), BorderLayout.NORTH);
207             panel.add(new JScrollPane(textarea));
208             panel.setPreferredSize(new Dimension(500, 350));
209             panel.setMinimumSize(new Dimension(500, 350));
210 
211             textarea.addMouseListener(new MouseAdapter() {
212                 @Override
213                 public void mousePressed(MouseEvent e) {
214                     super.mousePressed(e);
215                     textarea.requestFocusInWindow();
216                 }
217             });
218 
219             int result = JOptionPane.showOptionDialog(
220                 MediatorHelper.frame(),
221                 panel,
222                 "Report an issue or a bug",
223                 JOptionPane.OK_CANCEL_OPTION,
224                 JOptionPane.QUESTION_MESSAGE,
225                 null,
226                 new String[] {
227                     "Send",
228                     I18nUtil.valueByKey("LIST_ADD_VALUE_CANCEL")
229                 },
230                 I18nUtil.valueByKey("LIST_ADD_VALUE_CANCEL")
231             );
232 
233             if (StringUtils.isNotEmpty(textarea.getText()) && result == JOptionPane.YES_OPTION) {
234                 MediatorHelper.model().getMediatorUtils().getGitUtil().sendReport(textarea.getText(), ShowOnConsole.YES, "Report");
235             }
236         });
237 
238         return itemReportIssue;
239     }

302     public void switchLocale(Locale oldLocale, Locale newLocale, boolean isStartup) {
303         I18nUtil.setCurrentBundle(newLocale);
304 
305         Stream.of(
306             JTextPaneAppender.ATTRIBUTE_WARN,
307             JTextPaneAppender.ATTRIBUTE_INFORM,
308             JTextPaneAppender.ATTRIBUTE_SUCCESS,
309             JTextPaneAppender.ATTRIBUTE_ALL
310         )
311         .forEach(attribute -> {
312             StyleConstants.setFontFamily(attribute, I18nViewUtil.isNonUbuntu(newLocale) ? UiUtil.FONT_NAME_MONO_ASIAN : UiUtil.FONT_NAME_MONO_NON_ASIAN);
313             StyleConstants.setFontSize(attribute, I18nViewUtil.isNonUbuntu(newLocale) ? UiUtil.FONT_SIZE_MONO_ASIAN : UiUtil.FONT_SIZE_MONO_NON_ASIAN);
314         });
315 
316         MediatorHelper.managerBruteForce().getResult().setFont(I18nViewUtil.isNonUbuntu(newLocale) ? UiUtil.FONT_MONO_ASIAN : UiUtil.FONT_MONO_NON_ASIAN);
317 
318         this.switchNetworkTable();
319         I18nViewUtil.switchI18nComponents();
320         this.switchOrientation(oldLocale, newLocale, isStartup);
321         this.switchMenuItems();  // required to restore proper language orientation
322         
323         MediatorHelper.treeDatabase().reloadNodes();
324 
325         // IllegalArgumentException #92981 on revalidate()
326         try {
327             MediatorHelper.frame().revalidate();  // Fix glitches on Linux
328         } catch (IllegalArgumentException e) {
329             LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
330         }
331     }

333     private void switchOrientation(Locale oldLocale, Locale newLocale, boolean isStartup) {
334         var componentOrientation = ComponentOrientation.getOrientation(I18nUtil.getCurrentLocale());
335         MediatorHelper.frame().applyComponentOrientation(componentOrientation);
336         
337         if (!ComponentOrientation.getOrientation(oldLocale).equals(ComponentOrientation.getOrientation(newLocale))) {
338             
339             JSplitPane splitPaneLeftRight = MediatorHelper.frame().getSplitNS().getSplitEW();
340             var componentLeft = splitPaneLeftRight.getLeftComponent();
341             var componentRight = splitPaneLeftRight.getRightComponent();
342 
343             // Reset components
344             splitPaneLeftRight.setLeftComponent(null);
345             splitPaneLeftRight.setRightComponent(null);
346             splitPaneLeftRight.setLeftComponent(componentRight);
347             splitPaneLeftRight.setRightComponent(componentLeft);
348 
349             if (isStartup) {
350                 // TODO unclear and not working properly when starting as locale arabic
351                 splitPaneLeftRight.setDividerLocation(
352                     splitPaneLeftRight.getDividerLocation()
353                 );
354             } else {  // required as switch to arabic uses reversed location
355                 splitPaneLeftRight.setDividerLocation(
356                     splitPaneLeftRight.getWidth() -
357                     splitPaneLeftRight.getDividerLocation()
358                 );
359             }
360         }
361         
362         MediatorHelper.tabResults().setComponentOrientation(ComponentOrientation.getOrientation(newLocale));
363     }

365     private void switchMenuItems() {
366         Stream.concat(AppMenubar.MODELS_ITEM.stream(), AppMenubar.MODELS_ITEM_INTO.stream())
367         .forEach(model -> model.getMenuItem().setComponentOrientation(
368             model.getLanguage().isRightToLeft()
369             ? ComponentOrientation.RIGHT_TO_LEFT
370             : ComponentOrientation.LEFT_TO_RIGHT
371         ));
372     }

374     private void switchNetworkTable() {
375         JTableHeader header = MediatorHelper.panelConsoles().getNetworkTable().getTableHeader();
376         TableColumnModel columnModel = header.getColumnModel();
377         columnModel.getColumn(0).setHeaderValue(I18nUtil.valueByKey("NETWORK_TAB_URL_COLUMN"));
378         columnModel.getColumn(1).setHeaderValue(I18nUtil.valueByKey("NETWORK_TAB_SIZE_COLUMN") +" (KB)");
379         columnModel.getColumn(2).setHeaderValue(I18nUtil.valueByKey("SQLENGINE_STRATEGY"));
380         columnModel.getColumn(3).setHeaderValue(I18nUtil.valueByKey("SQLENGINE_METADATA"));
381         header.repaint();
382     }
