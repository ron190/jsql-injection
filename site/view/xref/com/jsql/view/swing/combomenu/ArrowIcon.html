1   package com.jsql.view.swing.combomenu;
2   
3   import com.jsql.util.LogLevelUtil;
4   import org.apache.logging.log4j.LogManager;
5   import org.apache.logging.log4j.Logger;
6   
7   import javax.swing.*;
8   import javax.swing.plaf.basic.BasicArrowButton;
9   import java.awt.*;
10  
11  public class ArrowIcon implements Icon, SwingConstants {
12      
13      /**
14       * Log4j logger sent to view.
15       */
16      private static final Logger LOGGER = LogManager.getRootLogger();
17      
18      private static final int DEFAULT_SIZE = 9;
19  
20      private final int size;
21      private final int iconSize;
22      private final int direction;
23      private final boolean isEnabled;
24      private final BasicArrowButton iconRenderer;
25  
26      public ArrowIcon(int direction, boolean isPressedView) {
27          
28          this(DEFAULT_SIZE, direction, isPressedView);
29      }
30  
31      public ArrowIcon(int iconSize, int direction, boolean isEnabled) {
32          
33          this.size = iconSize / 2;
34          this.iconSize = iconSize;
35          this.direction = direction;
36          this.isEnabled = isEnabled;
37          this.iconRenderer = new BasicArrowButton(direction);
38      }
39  
40      @Override
41      public void paintIcon(Component c, Graphics g, int x, int y) {
42          
43          // Fix #4731: ClassCastException on paintTriangle()
44          // Implementation by sun.awt.image
45          try {
46              this.iconRenderer.paintTriangle(g, x, y + 3, this.size, this.direction, this.isEnabled);
47              
48          } catch(ClassCastException e) {
49              
50              LOGGER.log(LogLevelUtil.CONSOLE_JAVA, e, e);
51          }
52      }
53  
54      @Override
55      public int getIconWidth() {
56          
57          switch (this.direction) {
58          
59              case NORTH:
60              case SOUTH: return this.iconSize;
61              case EAST:
62              case WEST:
63              default: return this.size;
64          }
65      }
66  
67      @Override
68      public int getIconHeight() {
69          
70          switch (this.direction) {
71          
72              case NORTH:
73              case SOUTH: return this.size;
74              case EAST:
75              case WEST:
76              default: return this.iconSize;
77          }
78      }
79  }
