3   import com.jsql.model.InjectionModel;
4   import com.jsql.util.GitUtil.ShowOnConsole;
5   import com.jsql.util.I18nUtil;
6   import com.jsql.util.LogLevelUtil;
7   import com.jsql.view.swing.JFrameView;
8   import com.jsql.view.swing.util.MediatorHelper;
9   import org.apache.logging.log4j.Level;
10  import org.apache.logging.log4j.LogManager;
11  import org.apache.logging.log4j.Logger;
12  
13  import java.awt.*;
14  
15  /**
16   * Main class of the application and called from the .jar.
17   * This class set the general environment of execution and start the software.
18   */
19  public class MainApplication {
20      
21      private static final Logger LOGGER = LogManager.getRootLogger();
22      
23      private static final InjectionModel injectionModel;
24      
25      static {
26  
27          System.setProperty("jdk.httpclient.allowRestrictedHeaders", "connection,content-length,expect,host,upgrade");
28  
29          if (GraphicsEnvironment.isHeadless()) {
30              LOGGER.log(
31                  Level.ERROR,
32                  "Headless runtime detected, please install or use default Java runtime instead of headless runtime"
33              );
34              System.exit(1);
35          }
36          
37          injectionModel = new InjectionModel();
38          
39          if (!"true".equals(System.getenv("FROM_CI_PIPELINE"))) {
40              
41              injectionModel.getMediatorUtils().getPreferencesUtil().loadSavedPreferences();
42          }
43          
44          MainApplication.apply4K();
45      }
46      
47      private MainApplication() {
48          // nothing
49      }
50      
51      /**
52       * Application starting point.
53       * @param args CLI parameters (not used)
54       */
55      public static void main(String[] args) {
56          
57          // Initialize MVC
58          MediatorHelper.register(injectionModel);
59          
60          // Configure global environment settings
61          injectionModel.getMediatorUtils().getExceptionUtil().setUncaughtExceptionHandler();
62          injectionModel.getMediatorUtils().getProxyUtil().initializeProxy();
63          injectionModel.getMediatorUtils().getAuthenticationUtil().setKerberosCifs();
64          
65          try {
66              var view = new JFrameView();
67              MediatorHelper.register(view);
68              
69              injectionModel.subscribe(view.getSubscriber());
70              
71          } catch (HeadlessException e) {
72              
73              LOGGER.log(
74                  LogLevelUtil.CONSOLE_JAVA,
75                  String.format(
76                      "HeadlessException, command line execution in jSQL not supported yet: %s",
77                      e.getMessage()
78                  ),
79                  e
80              );
81              return;
82              
83          } catch (AWTError e) {
84              
85              // Fix #22668: Assistive Technology not found
86              LOGGER.log(
87                  LogLevelUtil.CONSOLE_JAVA,
88                  String.format(
89                      "Java Access Bridge missing or corrupt, check your access bridge definition in JDK_HOME/jre/lib/accessibility.properties: %s",
90                      e.getMessage()
91                  ),
92                  e
93              );
94              return;
95          }
96          
97          injectionModel.displayVersion();
98          
99          // Check application status
100         if (!injectionModel.getMediatorUtils().getProxyUtil().isLive(ShowOnConsole.YES)) {
101             return;
102         }
103         
104         if (injectionModel.getMediatorUtils().getPreferencesUtil().isCheckingUpdate()) {
105             
106             injectionModel.getMediatorUtils().getGitUtil().checkUpdate(ShowOnConsole.NO);
107         }
108         
109         I18nUtil.checkCurrentLanguage();
110         injectionModel.getMediatorUtils().getGitUtil().showNews();
111         
112         MainApplication.check4K();
113     }
114     
115     private static void apply4K() {
116         
117         if (injectionModel.getMediatorUtils().getPreferencesUtil().is4K()) {
118             
119             // jdk >= 9
120             System.setProperty("sun.java2d.uiScale", "2.5");
121         }
122     }
123     
124     private static void check4K() {
125         
126         Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
127         int width = (int) screenSize.getWidth();
128         
129         if (width >= 3840 && !injectionModel.getMediatorUtils().getPreferencesUtil().is4K()) {
130             
131             LOGGER.log(LogLevelUtil.CONSOLE_ERROR, "Your screen seems compatible with 4K resolution, enable high-definition mode in Preferences");
132         }
133     }
134 }
