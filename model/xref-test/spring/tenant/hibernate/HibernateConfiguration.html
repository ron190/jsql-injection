12  import java.util.*;
13  
14  @Configuration
15  @EnableTransactionManagement
16  public class HibernateConfiguration {
17  
18      @Bean("hibernateTransactionManager")
19      public PlatformTransactionManager hibernateTransactionManager() {
20          HibernateTransactionManager transactionManager = new HibernateTransactionManager();
21          transactionManager.setSessionFactory(this.sessionFactory().getObject());
22          return transactionManager;
23      }
24  
25      @Bean
26      public LocalSessionFactoryBean sessionFactory() {
27          Properties hibernateProperties = new Properties();
28          hibernateProperties.setProperty(Environment.DIALECT, "org.hibernate.dialect.MySQLDialect");
29          hibernateProperties.put(AvailableSettings.MULTI_TENANT_CONNECTION_PROVIDER, MultiTenantConnectionProviderImpl.class);
30          hibernateProperties.put(AvailableSettings.MULTI_TENANT_IDENTIFIER_RESOLVER, CurrentTenantIdentifierResolverImpl.class);
31          hibernateProperties.setProperty("hibernate.multiTenancy", "DATABASE");
32  
33          LocalSessionFactoryBean sessionFactory = new LocalSessionFactoryBean();
34          sessionFactory.setHibernateProperties(hibernateProperties);
35          return sessionFactory;
36      }
