16      @RequestMapping("/stack")

18          if (name.toLowerCase().contains("union")) {
19              return null;
20          }

22      }

24      @RequestMapping("/stack2")
25      public Greeting endpointStack2(@RequestParam(value="name", defaultValue="World") String name, @RequestParam(value="semicolon", defaultValue="true") String semicolon, @RequestHeader Map<String, String> headers) {
26          Greeting greeting = new Greeting(null);
27          var result = new StringBuilder();
28          Arrays.stream(("select First_Name from Student where '1' = '"+name+"'").split(";")).map(String::trim).forEach(query -> {
29              if ("true".equals(semicolon)) {
30                  query = query +";";
31              }
32              Greeting g = this.getResponse(query, query, true, false, true, false, false, true);
33              if (g != null) result.append(g.getContent());
34          });
35          greeting.setContent(result.toString());
36          return greeting;
37      }
