5   public class Base58 {

7       private static final char[] ALPHABET = "123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz".toCharArray();
8       private static final int BASE_58 = ALPHABET.length;
9       private static final int BASE_256 = 256;
10  
11      private static final int[] INDEXES = new int[128];
12      
13      static {
14          Arrays.fill(INDEXES, -1);
15          
16          for (int i = 0; i < ALPHABET.length; i++) {
17              
18              INDEXES[ALPHABET[i]] = i;
19          }
20      }

26      public static String encode(byte[] input) {
27          
28          if (input.length == 0) {
29              // paying with the same coin
30              return "";
31          }
32  
33          // Make a copy of the input since we are going to modify it.
34          byte[] copyInput = copyOfRange(input, 0, input.length);
35  
36          // Count leading zeroes
37          var zeroCount = 0;
38          while (zeroCount < copyInput.length && copyInput[zeroCount] == 0) {
39              
40              ++zeroCount;
41          }
42  
43          // The actual encoding
44          var temp = new byte[copyInput.length * 2];
45          int j = temp.length;
46  
47          int startAt = zeroCount;
48          while (startAt < copyInput.length) {
49              
50              byte mod = divmod58(copyInput, startAt);
51              if (copyInput[startAt] == 0) {
52                  
53                  ++startAt;
54              }
55  
56              temp[--j] = (byte) ALPHABET[mod];
57          }
58  
59          // Strip extra '1' if any
60          while (j < temp.length && temp[j] == ALPHABET[0]) {
61              ++j;
62          }
63  
64          // Add as many leading '1' as there were leading zeros.
65          while (--zeroCount >= 0) {
66              
67              temp[--j] = (byte) ALPHABET[0];
68          }
69  
70          byte[] output = copyOfRange(temp, j, temp.length);
71          
72          return new String(output);
73      }

75      public static byte[] decode(String input) {
76          
77          if (input.isEmpty()) {
78              
79              // paying with the same coin
80              return new byte[0];
81          }
82  
83          var input58 = new byte[input.length()];
84          
85          // Transform the String to a base58 byte sequence
86          for (var i = 0; i < input.length(); ++i) {
87              
88              var c = input.charAt(i);
89  
90              int digit58 = -1;
91              if (c >= 0 && c < 128) {
92                  
93                  digit58 = INDEXES[c];
94              }
95              
96              if (digit58 < 0) {
97                  
98                  throw new IllegalArgumentException("Not a Base58 input: " + input);
99              }
100 
101             input58[i] = (byte) digit58;
102         }
103 
104         // Count leading zeroes
105         var zeroCount = 0;
106         while (zeroCount < input58.length && input58[zeroCount] == 0) {
107             
108             ++zeroCount;
109         }
110 
111         // The encoding
112         var temp = new byte[input.length()];
113         int j = temp.length;
114 
115         int startAt = zeroCount;
116         while (startAt < input58.length) {
117             
118             byte mod = divmod256(input58, startAt);
119             if (input58[startAt] == 0) {
120                 
121                 ++startAt;
122             }
123 
124             temp[--j] = mod;
125         }
126 
127         // Do no add extra leading zeroes, move j to first non null byte.
128         while (j < temp.length && temp[j] == 0) {
129             ++j;
130         }
131 
132         return copyOfRange(temp, j - zeroCount, temp.length);
133     }

135     private static byte divmod58(byte[] number, int startAt) {
136         
137         var remainder = 0;
138         for (int i = startAt; i < number.length; i++) {
139             
140             int digit256 = number[i] & 0xFF;
141             int temp = remainder * BASE_256 + digit256;
142 
143             number[i] = (byte) (temp / BASE_58);
144 
145             remainder = temp % BASE_58;
146         }
147 
148         return (byte) remainder;
149     }

151     private static byte divmod256(byte[] number58, int startAt) {
152         
153         var remainder = 0;
154         for (int i = startAt; i < number58.length; i++) {
155             
156             int digit58 = number58[i] & 0xFF;
157             int temp = remainder * BASE_58 + digit58;
158 
159             number58[i] = (byte) (temp / BASE_256);
160 
161             remainder = temp % BASE_256;
162         }
163 
164         return (byte) remainder;
165     }

167     private static byte[] copyOfRange(byte[] source, int from, int to) {
168         
169         var range = new byte[to - from];
170         System.arraycopy(source, from, range, 0, range.length);
171 
172         return range;
173     }
